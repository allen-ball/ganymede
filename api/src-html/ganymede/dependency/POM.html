064/**
065 * Target class for YAML representation of POM elements.
066 *
067 * {@bean.info}
068 *
069 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
070 */
071@JsonPropertyOrder({ "localRepository", "interactiveMode", "offline",
072                     "repositories", "dependencies" })
073@Data @NoArgsConstructor @ToString @Log4j2
074public class POM {
075    private static final YAMLFactory YAML_FACTORY =
076        new YAMLFactory()
077        .enable(YAMLGenerator.Feature.WRITE_DOC_START_MARKER);
078    private static final SimpleModule MODULE =
079        new SimpleModule()
080        .addDeserializer(Dependency.class, new DependencyDeserializer())
081        .addSerializer(Dependency.class, new DependencySerializer())
082        .addDeserializer(RemoteRepository.class, new RemoteRepositoryDeserializer())
083        .addSerializer(RemoteRepository.class, new RemoteRepositorySerializer())
084        .addDeserializer(RepositoryPolicy.class, new RepositoryPolicyDeserializer());
085    private static final ObjectMapper OBJECT_MAPPER =
086        new ObjectMapper(YAML_FACTORY)
087        .registerModule(MODULE)
088        .setSerializationInclusion(JsonInclude.Include.NON_EMPTY)
089        .enable(MapperFeature.ACCEPT_CASE_INSENSITIVE_VALUES)
090        .enable(SerializationFeature.INDENT_OUTPUT);
091
092    /**
093     * Static method to get default {@link POM}.
094     *
095     * @return  The parsed {@link POM}.
096     */
097    public static POM getDefault() throws IOException {
098        POM pom = null;
099        var resource = POM.class.getSimpleName() + ".yaml";
100
101        try (var in = POM.class.getResourceAsStream(resource)) {
102            pom = OBJECT_MAPPER.readValue(in, POM.class);
103        }
104
105        return pom;
106    }
107
108    /**
109     * Static method to parse a YAML {@link String} to a {@link POM}.
110     *
111     * @param   yaml            The YAML {@link String}.
112     *
113     * @return  The parsed {@link POM}.
114     */
115    public static POM parse(String yaml) throws Exception {
116        return OBJECT_MAPPER.readValue(yaml, POM.class);
117    }

119    private String localRepository = null;

121    private Boolean offline = null;
122    private Set<RemoteRepository> repositories = new LinkedHashSet<>();
123    private Set<Dependency> dependencies = new LinkedHashSet<>();

125    /**
126     * Method to merge a {@link POM} into {@link.this} {@link POM}.
127     *
128     * @param   that            The {@link POM} to merge.
129     */
130    public void merge(POM that) {
131        update(this, that, POM::getLocalRepository, this::setLocalRepository);
132        update(this, that, POM::getInteractiveMode, this::setInteractiveMode);
133        update(this, that, POM::getOffline, this::setOffline);
134
135        var ids =
136            that.getRepositories().stream()
137            .map(t -> t.getId())
138            .filter(Objects::nonNull)
139            .collect(toSet());
140        var urls =
141            that.getRepositories().stream()
142            .map(t -> t.getUrl())
143            .filter(Objects::nonNull)
144            .collect(toSet());
145
146        this.getRepositories()
147            .removeIf(t -> ids.contains(t.getId()) || urls.contains(t.getUrl()));
148        this.getRepositories().addAll(that.getRepositories());
149
150        var keys =
151            that.getDependencies().stream()
152            .map(Dependency::getArtifact)
153            .map(t -> toVersionlessId(t))
154            .collect(toSet());
155
156        this.getDependencies()
157            .removeIf(t -> keys.contains(toVersionlessId(t.getArtifact())));
158        this.getDependencies().addAll(that.getDependencies());
159    }
160
161    private <T,U> boolean update(T left, T right, Function<T,U> get, Consumer<U> set) {
162        var value = get.apply(right);
163        var modified = (value != null && (! Objects.equals(get.apply(left), value)));
164
165        if (modified) {
166            set.accept(value);
167        }
168
169        return modified;
170    }

172    /**
173     * Method to write {@link.this} {@link POM}'s YAML representation.
174     *
175     * @param   out             The target {@link OutputStream}.
176     *
177     * @throws  IOException     If the {@link POM} cannot be written for any
178     *                          reason.
179     */
180    public void writeTo(OutputStream out) throws IOException {
181        OBJECT_MAPPER.writeValue(out, this);
182    }

184    private static String asText(JsonNode node, String name) {
185        return asText(node, name, null);
186    }

188    private static String asText(JsonNode node, String name, String defaultValue) {
189        return node.has(name) ? node.get(name).asText() : defaultValue;
190    }

192    @ToString
193    private static class DependencyDeserializer extends StdDeserializer<Dependency> {
194        private static final long serialVersionUID = -1L;
195
196        public DependencyDeserializer() { super(Dependency.class); }
197
198        @Override
199        public Dependency deserialize(JsonParser parser,
200                                      DeserializationContext context) throws IOException,
201                                                                             JsonProcessingException {
202            JsonNode node = parser.getCodec().readTree(parser);
203            Artifact artifact = null;
204
205            if (node instanceof ObjectNode) {
206                artifact =
207                    new DefaultArtifact(asText(node, "groupId"),
208                                        asText(node, "artifactId"),
209                                        asText(node, "classifier"),
210                                        asText(node, "type", "jar"),
211                                        asText(node, "version"),
212                                        Map.of(), (File) null);
213            } else {
214                var strings = node.asText().split(":");
215                /*
216                 * groupId:artifactId[:type[:classifier]]:version
217                 */
218                artifact =
219                    new DefaultArtifact(strings[0], strings[1],
220                                        (strings.length > 4) ? strings[3] : null,
221                                        (strings.length > 3) ? strings[2] : "jar",
222                                        (strings.length > 2) ? strings[strings.length - 1] : null,
223                                        Map.of(), (File) null);
224            }
225
226            return new Dependency(artifact, asText(node, "scope", JavaScopes.RUNTIME));
227        }
228    }

230    @ToString
231    private static class DependencySerializer extends StdSerializer<Dependency> {
232        private static final long serialVersionUID = -1L;
233
234        public DependencySerializer() { super(Dependency.class); }
235
236        @Override
237        public void serialize(Dependency value, JsonGenerator generator,
238                              SerializerProvider provider) throws IOException,
239                                                                  JsonProcessingException {
240            var artifact = value.getArtifact();
241            var string =
242                Stream.of(artifact.getGroupId(), artifact.getArtifactId(),
243                          artifact.getExtension(), artifact.getClassifier(),
244                          artifact.getVersion())
245                .filter(Objects::nonNull)
246                .filter(t -> (! t.isBlank()))
247                .collect(joining(":"));
248
249            generator.writeString(string);
250        }
251    }

253    @ToString
254    private static class RemoteRepositoryDeserializer extends StdDeserializer<RemoteRepository> {
255        private static final long serialVersionUID = -1L;
256
257        public RemoteRepositoryDeserializer() { super(RemoteRepository.class); }
258
259        @Override
260        public RemoteRepository deserialize(JsonParser parser,
261                                            DeserializationContext context) throws IOException,
262                                                                                   JsonProcessingException {
263            JsonNode node = parser.getCodec().readTree(parser);
264            var builder =
265                new RemoteRepository.Builder(asText(node, "id"),
266                                             asText(node, "layout", "default"),
267                                             asText(node, "url"));
268
269            if (node.has("releases")) {
270                var policy =
271                    parser.getCodec().treeToValue(node.get("releases"), RepositoryPolicy.class);
272
273                builder.setReleasePolicy(policy);
274            }
275
276            if (node.has("snapshots")) {
277                var policy =
278                    parser.getCodec().treeToValue(node.get("snapshots"), RepositoryPolicy.class);
279
280                builder.setSnapshotPolicy(policy);
281            }
282
283            return builder.build();
284        }
285    }

287    @ToString
288    private static class RemoteRepositorySerializer extends StdSerializer<RemoteRepository> {
289        private static final long serialVersionUID = -1L;
290
291        public RemoteRepositorySerializer() { super(RemoteRepository.class); }
292
293        @Override
294        public void serialize(RemoteRepository value, JsonGenerator generator,
295                              SerializerProvider provider) throws IOException,
296                                                                  JsonProcessingException {
297            var map = new LinkedHashMap<String,Object>();
298
299            map.put("id", value.getId());
300            map.put("layout", value.getContentType());
301            map.put("url", value.getUrl());
302            map.put("releases", value.getPolicy(false));
303            map.put("snapshots", value.getPolicy(true));
304
305            generator.writeObject(map);
306        }
307    }

309    @ToString
310    private static class RepositoryPolicyDeserializer extends StdDeserializer<RepositoryPolicy> {
311        private static final long serialVersionUID = -1L;
312
313        public RepositoryPolicyDeserializer() { super(RepositoryPolicy.class); }
314
315        @Override
316        public RepositoryPolicy deserialize(JsonParser parser,
317                                            DeserializationContext context) throws IOException,
318                                                                                   JsonProcessingException {
319            JsonNode node = parser.getCodec().readTree(parser);
320
321            return new RepositoryPolicy(node.has("enabled") ? node.get("enabled").asBoolean() : true,
322                                        asText(node, "updatePolicy", RepositoryPolicy.UPDATE_POLICY_DAILY),
323                                        asText(node, "checksumPolicy", RepositoryPolicy.CHECKSUM_POLICY_WARN));
324        }
