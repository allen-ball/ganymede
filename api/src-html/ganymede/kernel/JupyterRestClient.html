021import com.fasterxml.jackson.databind.JsonNode;
022import java.net.URI;
023import java.net.http.HttpClient;
024import java.net.http.HttpRequest;
025import java.net.http.HttpResponse;
026import lombok.ToString;
027import lombok.extern.log4j.Log4j2;
028
029import static ganymede.server.Server.OBJECT_MAPPER;
030import static org.springframework.http.HttpHeaders.CONTENT_TYPE;
031import static org.springframework.http.MediaType.APPLICATION_JSON_VALUE;
032
033/**
034 * Ganymede Jupyter Notebook/Lab REST client.  See
035 * {@link.uri https://github.com/jupyter/jupyter/wiki/Jupyter-Notebook-Server-API target=newtab Jupyter Notebook Server API}.
036 *
037 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
038 */
039@Log4j2
040public class JupyterRestClient {
041    private final JsonNode json;
042    private final URI uri;
043    private final String token;
044    private final HttpClient client = HttpClient.newHttpClient();
045
046    /**
047     * Sole constructor.
048     *
049     * @param   json            The {@link JsonNode} describing the server
050     *                          parameters.
051     */
052    public JupyterRestClient(JsonNode json) {
053        this.json = json;
054        this.uri = URI.create(json.get("url").asText());
055        this.token = json.get("token").asText();
056    }
057
058    /**
059     * {@code GET /api/status}
060     */
061    public JsonNode getStatus() throws Exception {
062        var request = builder().uri(uri.resolve("api/status")).GET().build();
063        var response =
064            client.send(request, HttpResponse.BodyHandlers.ofString());
065
066        return OBJECT_MAPPER.readTree(response.body());
067    }
068
069    /**
070     * {@code GET /api/kernels}
071     */
072    public JsonNode getKernels() throws Exception {
073        var request =
074            builder()
075            .uri(uri.resolve("api/kernels"))
076            .GET()
077            .build();
078        var response =
079            client.send(request, HttpResponse.BodyHandlers.ofString());
080
081        return OBJECT_MAPPER.readTree(response.body());
082    }
083
084    /**
085     * {@code GET /api/kernels/(id)}
086     */
087    public JsonNode getKernel(String id) throws Exception {
088        var request =
089            builder()
090            .uri(uri.resolve("api/kernels/").resolve(id))
091            .GET()
092            .build();
093        var response =
094            client.send(request, HttpResponse.BodyHandlers.ofString());
095
096        return OBJECT_MAPPER.readTree(response.body());
097    }
098
099    /**
100     * {@code GET /api/sessions}
101     */
102    public JsonNode getSessions() throws Exception {
103        var request =
104            builder()
105            .uri(uri.resolve("api/sessions"))
106            .GET()
107            .build();
108        var response =
109            client.send(request, HttpResponse.BodyHandlers.ofString());
110
111        return OBJECT_MAPPER.readTree(response.body());
112    }
113
114    /**
115     * {@code GET /api/sessions/(id)}
116     */
117    public JsonNode getSession(String id) throws Exception {
118        var request =
119            builder()
120            .uri(uri.resolve("api/sessions/").resolve(id))
121            .GET()
122            .build();
123        var response =
124            client.send(request, HttpResponse.BodyHandlers.ofString());
125
126        return OBJECT_MAPPER.readTree(response.body());
127    }
128
129    /**
130     * {@code GET /api/terminals}
131     */
132    public JsonNode getTerminals() throws Exception {
133        var request =
134            builder()
135            .uri(uri.resolve("api/terminals"))
136            .GET()
137            .build();
138        var response =
139            client.send(request, HttpResponse.BodyHandlers.ofString());
140
141        return OBJECT_MAPPER.readTree(response.body());
142    }
143
144    /**
145     * {@code GET /api/terminals/(id)}
146     */
147    public JsonNode getTerminal(String id) throws Exception {
148        var request =
149            builder()
150            .uri(uri.resolve("api/terminals/").resolve(id))
151            .GET()
152            .build();
153        var response =
154            client.send(request, HttpResponse.BodyHandlers.ofString());
155
156        return OBJECT_MAPPER.readTree(response.body());
157    }
158
159    private HttpRequest.Builder builder() {
160        var builder =
161            HttpRequest.newBuilder()
162            .header("Authorization", "Token " + token);
163
164        return builder;
165    }
166
167    @Override
168    public String toString() { return json.toPrettyString(); }
169}
