021import com.fasterxml.jackson.databind.JsonNode;
022import java.net.URI;
023import java.net.http.HttpClient;
024import java.net.http.HttpRequest;
025import java.net.http.HttpResponse;
026import java.util.List;
027import java.util.stream.Stream;
028import java.util.stream.StreamSupport;
029import lombok.NoArgsConstructor;
030import lombok.ToString;
031import lombok.extern.log4j.Log4j2;
032
033import static ganymede.server.Server.OBJECT_MAPPER;
034import static java.util.stream.Collectors.toList;
035import static org.springframework.http.HttpHeaders.CONTENT_TYPE;
036import static org.springframework.http.MediaType.APPLICATION_JSON_VALUE;
037
038/**
039 * Ganymede {@link Kernel} REST client.
040 *
041 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
042 */
043@NoArgsConstructor @ToString @Log4j2
044public class KernelRestClient {
045    private static final String TEMPLATE = "http://localhost:%d/";
046
047    private final int port = Integer.decode(System.getProperty(Kernel.PORT_PROPERTY));
048    private final URI uri = URI.create(String.format(TEMPLATE, port));
049    private final HttpClient client = HttpClient.newHttpClient();
050
051    /**
052     * See {@link Kernel#classpath()}.
053     */
054    public List<String> classpath() throws Exception {
055        var request =
056            HttpRequest.newBuilder()
057            .uri(uri.resolve("kernel/classpath"))
058            .header(CONTENT_TYPE, APPLICATION_JSON_VALUE)
059            .GET()
060            .build();
061        var response =
062            client.send(request, HttpResponse.BodyHandlers.ofString());
063        var list =
064            asStream(OBJECT_MAPPER.readTree(response.body()))
065            .map(JsonNode::asText)
066            .collect(toList());
067
068        return list;
069    }
070
071    private <T> Stream<T> asStream(Iterable<T> iterable) {
072        return StreamSupport.stream(iterable.spliterator(), false);
073    }
074
075    /**
076     * See {@link Kernel#display(ObjectNode)}.
077     *
078     * @param   bundle          The MIME bundle {@link JsonNode}.
079     */
080    public void display(JsonNode bundle) throws Exception {
081        var body = bundle.toPrettyString();
082        var request =
083            HttpRequest.newBuilder()
084            .uri(uri.resolve("kernel/display"))
085            .header(CONTENT_TYPE, APPLICATION_JSON_VALUE)
086            .PUT(HttpRequest.BodyPublishers.ofString(body))
087            .build();
088        var response =
089            client.send(request, HttpResponse.BodyHandlers.ofString());
090    }
091
092    /**
093     * See {@link Kernel#print(ObjectNode)}.
094     *
095     * @param   bundle          The MIME bundle {@link JsonNode}.
096     */
097    public void print(JsonNode bundle) throws Exception {
098        var body = bundle.toPrettyString();
099        var request =
100            HttpRequest.newBuilder()
101            .uri(uri.resolve("kernel/print"))
102            .header(CONTENT_TYPE, APPLICATION_JSON_VALUE)
103            .PUT(HttpRequest.BodyPublishers.ofString(body))
104            .build();
105        var response =
106            client.send(request, HttpResponse.BodyHandlers.ofString());
107    }
108}
