021import java.io.StringReader;
022import java.util.Objects;
023import java.util.Properties;
024import lombok.NoArgsConstructor;
025import lombok.ToString;
026import lombok.extern.log4j.Log4j2;
027
028import static lombok.AccessLevel.PROTECTED;
029
030/**
031 * Abstract {@link Properties} {@link ganymede.shell.Magic} base class.
032 *
033 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
034 */
035@NoArgsConstructor(access = PROTECTED) @ToString @Log4j2
036public abstract class AbstractPropertiesMagic extends AbstractMagic {
037
038    /**
039     * Method to convert the cell {@code code} to a {@link Properties}
040     * instance.
041     *
042     * @param   code            The code to parse.
043     *
044     * @return  The {@link Properties}.
045     *
046     * @see #HELPER
047     */
048    protected Properties compile(String code) throws Exception {
049        try (var reader = new StringReader(code)) {
050            var in = new Properties(System.getProperties());
051
052            in.load(reader);
053
054            var out = new Properties(in);
055            var changed = true;
056
057            while (changed) {
058                changed = false;
059
060                for (var key : in.keySet()) {
061                    if (key instanceof String) {
062                        var string = (String) key;
063                        var value = HELPER.replacePlaceholders(in.getProperty(string), out);
064
065                        changed |= (! Objects.equals(value, out.put(string, value)));
066                    }
067                }
068            }
069
070            return out;
071        }
072    }
073}
