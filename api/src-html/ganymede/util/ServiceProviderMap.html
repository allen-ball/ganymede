021import java.util.Comparator;
022import java.util.ServiceConfigurationError;
023import java.util.ServiceLoader;
024import java.util.TreeMap;
025import java.util.function.Predicate;
026
027import static java.util.stream.Collectors.toList;
028
029/**
030 * {@link ServiceLoader Service} {@link java.util.Map}.
031 *
032 * @param       <T>             The service type.
033 *
034 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
035 */
036public class ServiceProviderMap<T> extends TreeMap<Class<? extends T>,T> {
037    private static final long serialVersionUID = 1150156295503292563L;
038
039    /** @serial */ private final Class<T> service;
040    /** @serial */ private final ServiceLoader<T> loader;
041
042    /**
043     * Sole constructor.
044     *
045     * @param   service         The service {@link Class type}.
046     */
047    public ServiceProviderMap(Class<T> service) {
048        super(Comparator.comparing(Class::getName));
049
050        this.service = service;
051        this.loader = ServiceLoader.load(service, service.getClassLoader());
052
053        reload();
054    }
055
056    /**
057     * Method to get the {@link ServiceLoader}'s {@link ClassLoader}.
058     *
059     * @return  The {@link ServiceLoader}'s {@link ClassLoader}.
060     */
061    public ClassLoader getClassLoader() { return service.getClassLoader(); }
062
063    /**
064     * Reload {@link ServiceLoader} and {@link #put(Object,Object)} newly
065     * discovered entries.
066     *
067     * @return  {@link.this}
068     */
069    public ServiceProviderMap<T> reload() { return reload(t -> true); }
070
071    /**
072     * Reload {@link ServiceLoader} and {@link #put(Object,Object)} newly
073     * discovered entries.  A {@link Predicate} may be supplied to test a
074     * provider {@link Class} before attempting to load and might be used to
075     * test that other required {@link Class}es are loaded before attempting
076     * to load.
077     *
078     * @param   predicate       A {@link Predicate} to test whether or not
079     *                          the provider {@link Class} should be
080     *                          loaded.
081     *
082     * @return  {@link.this}
083     */
084    public ServiceProviderMap<T> reload(Predicate<Class<?>> predicate) {
085        loader.reload();
086
087        var providers = loader.stream().collect(toList());
088
089        for (var provider : providers) {
090            try {
091                var type = provider.type();
092
093                if (predicate.test(type)) {
094                    computeIfAbsent(type, k -> provider.get());
095                }
096            } catch (ServiceConfigurationError error) {
097            }
098        }
099
100        return this;
101    }
102}
