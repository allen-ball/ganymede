021import java.lang.reflect.Method;
022import lombok.NoArgsConstructor;
023import lombok.ToString;
024import lombok.extern.log4j.Log4j2;
025import org.springframework.boot.autoconfigure.SpringBootApplication;
026import org.springframework.boot.builder.SpringApplicationBuilder;
027
028import static java.lang.reflect.Modifier.isPublic;
029import static java.lang.reflect.Modifier.isStatic;
030
031/**
032 * {@link Notebook} Spring launcher.
033 *
034 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
035 */
036@SpringBootApplication
037@NoArgsConstructor @ToString
038public class Notebook {
039
040    /**
041     * Static factory method to create an {@link NotebookContext}.  Also
042     * initializes the Spring environment.
043     *
044     * @see SpringApplicationBuilder
045     *
046     * @return  An initialized {@link NotebookContext}.
047     */
048    public static NotebookContext newNotebookContext() {
049        var __ = new NotebookContext();
050
051        try {
052            var type = Notebook.class;
053            var profile = type.getSimpleName().toLowerCase();
054
055            new SpringApplicationBuilder(type)
056                .profiles(profile)
057                .run();
058        } catch (Exception exception) {
059            exception.printStackTrace(System.err);
060        }
061
062        return __;
063    }
064
065    /**
066     * Method to generate the bootstrap code for a new
067     * {@link jdk.jshell.JShell} instance.
068     *
069     * @return  The boostrap code.
070     */
071    public static String bootstrap() {
072        var code =
073            String.format("var __ = %s.newNotebookContext();\n",
074                          Notebook.class.getCanonicalName());
075
076        for (var method : NotebookFunctions.class.getDeclaredMethods()) {
077            var modifiers = method.getModifiers();
078
079            if (isPublic(modifiers) && isStatic(modifiers)) {
080                code += makeWrapperFor(method);
081            }
082        }
083
084        return code;
085    }
086
087    private static String makeWrapperFor(Method method) {
088        var types = method.getGenericParameterTypes();
089        var arguments = new String[types.length];
090        var parameters = new String[types.length];
091
092        for (int i = 0; i < arguments.length; i += 1) {
093            arguments[i] = String.format("argument%d", i);
094        }
095
096        for (int i = 0; i < parameters.length; i += 1) {
097            parameters[i] =
098                String.format("%s %s", types[i].getTypeName(), arguments[i]);
099        }
100
101        var plist = String.join(", ", parameters);
102        var alist = String.join(", ", arguments);
103
104        return String.format("%1$s %2$s(%3$s) { %4$s%5$s.%2$s(%6$s); }\n",
105                             method.getGenericReturnType().getTypeName(),
106                             method.getName(), plist,
107                             Void.TYPE.equals(method.getReturnType()) ? "" : "return ",
108                             method.getDeclaringClass().getName(), alist);
109    }
110}
