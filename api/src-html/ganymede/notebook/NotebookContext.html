021import java.io.IOException;
022import java.io.StreamTokenizer;
023import java.io.StringReader;
024import java.util.LinkedHashSet;
025import java.util.LinkedList;
026import java.util.List;
027import java.util.Map;
028import java.util.concurrent.ConcurrentSkipListMap;
029import javax.script.ScriptContext;
030import javax.script.SimpleBindings;
031import javax.script.SimpleScriptContext;
032import jdk.jshell.JShell;
033import lombok.NoArgsConstructor;
034import lombok.ToString;
035
036import static java.util.stream.Collectors.toCollection;
037import static java.util.stream.Collectors.toMap;
038import static javax.script.ScriptContext.ENGINE_SCOPE;
039import static javax.script.ScriptContext.GLOBAL_SCOPE;
040import static jdk.jshell.Snippet.SubKind.TEMP_VAR_EXPRESSION_SUBKIND;
041
042/**
043 * {@link NotebookContext} for {@link Notebook} {@link ganymede.shell.Shell}
044 * {@link JShell} instance.  Bound to {@code __} in the {@link JShell}
045 * instance.
046 *
047 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
048 */
049@NoArgsConstructor @ToString
050public class NotebookContext {
051
052    /**
053     * Common {@link ScriptContext} supplied to
054     * {@link ganymede.shell.Magic#execute(NotebookContext,String,String)}.
055     */
056    public final ScriptContext context =
057        new SimpleScriptContext() {
058            {
059                setBindings(new SimpleBindings(new ConcurrentSkipListMap<>()), GLOBAL_SCOPE);
060                setBindings(new SimpleBindings(new ConcurrentSkipListMap<>()), ENGINE_SCOPE);
061            }
062        };
063
064    /**
065     * {@link Map} of known bindings' {@link Class types}.
066     */
067    public final Map<String,String> types = new ConcurrentSkipListMap<>();
068
069    /**
070     * {@link List} of accumulated {@code import}s.
071     */
072    public final List<String> imports = new LinkedList<>();
073
074    /**
075     * Method to construct a call to a static
076     * {@link java.lang.reflect.Method}.
077     *
078     * @param   type            The containing {@link Class}.
079     * @param   method          The {@link java.lang.reflect.Method} name.
080     * @param   parameters      The parameter types ({@link Class}es).
081     * @param   arguments       The argument {@link Object} array.
082     *
083     * @return  The {@link Object result}.
084     */
085    public Object invokeStaticMethod(String type, String method,
086                                     Class<?>[] parameters, Object... arguments) {
087        Object object = null;
088
089        try {
090            object =
091                Class.forName(type)
092                .getDeclaredMethod(method, parameters)
093                .invoke(null, arguments);
094        } catch (Throwable throwable) {
095            throwable.printStackTrace(System.err);
096        }
097
098        return object;
099    }
100
101    /**
102     * Method to construct a call to a static
103     * {@link java.lang.reflect.Method} with an empty argument list.
104     *
105     * @param   type            The containing {@link Class}.
106     * @param   method          The {@link java.lang.reflect.Method} name.
107     *
108     * @return  The {@link Object result}.
109     */
110    public Object invokeStaticMethod(String type, String method) {
111        return invokeStaticMethod(type, method,
112                                  new Class<?>[] { }, new Object[] { });
113    }
114
115    /**
116     * Static method used by the {@link ganymede.shell.Shell} to update the
117     * {@link NotebookContext} members.
118     *
119     * @param   jshell          The {@link JShell}.
120     */
121    public static void update(JShell jshell) {
122        var types =
123            jshell.variables()
124            .filter(t -> (! t.subKind().equals(TEMP_VAR_EXPRESSION_SUBKIND)))
125            .filter(t -> (! t.name().equals("__")))
126            .collect(toMap(k -> k.name(), v -> v.typeName()));
127
128        evaluate(jshell, "__.types.clear()");
129
130        for (var entry : types.entrySet()) {
131            evaluate(jshell,
132                     "__.context.getBindings(%1d).put(\"%2$s\", %2$s)",
133                     ENGINE_SCOPE, entry.getKey());
134            evaluate(jshell,
135                     "__.types.put(\"%1$s\", \"%2$s\")",
136                     entry.getKey(), entry.getValue());
137        }
138
139        var imports =
140            jshell.imports()
141            .map(t -> t.source())
142            .map(String::strip)
143            .collect(toCollection(LinkedHashSet::new));
144
145        evaluate(jshell, "__.imports.clear()");
146
147        if (! imports.isEmpty()) {
148            evaluate(jshell,
149                     "java.util.Collections.addAll(__.imports, \"%1$s\".split(\",\"))",
150                     String.join(",", imports));
151        }
152    }
153
154    private static String evaluate(JShell jshell, String expression, Object... argv) {
155        var analyzer = jshell.sourceCodeAnalysis();
156        var info = analyzer.analyzeCompletion(String.format(expression, argv));
157
158        return unescape(jshell.eval(info.source()).get(0).value());
159    }
160
161    /**
162     * Method to unescape a Java-escaped spring literal.
163     *
164     * @param   literal         The {@link String} to unescape.
165     *
166     * @return  The unescaped {@link String}.
167     */
168    public static String unescape(String literal) {
169        var string = literal;
170        /*
171         * https://stackoverflow.com/questions/3537706/how-to-unescape-a-java-string-literal-in-java
172         */
173        if (literal != null) {
174            try (var reader = new StringReader(literal)) {
175                var tokenizer = new StreamTokenizer(reader);
176
177                tokenizer.nextToken();
178
179                if (tokenizer.ttype == '"') {
180                    string = tokenizer.sval;
181                }
182            } catch (IOException exception) {
183            }
184        }
185
186        return string;
187    }
188}
