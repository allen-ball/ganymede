044/**
045 * Jupyter Notebook {@link Server} (base class for kernel implementations).
046 *
047 * {@bean.info}
048 *
049 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
050 */
051@Getter @Setter(PROTECTED) @Log4j2
052public abstract class Server extends ScheduledThreadPoolExecutor {

054    /**
055     * The
056     * {@link.uri https://jupyter-client.readthedocs.io/en/latest/messaging.html#versioning target=newtab Jupyter message specification version}.
057     */
058    protected static final String PROTOCOL_VERSION = "5.3";

060    /**
061     * Common {@link Server} static {@link ObjectMapper} instance.
062     */
063    public static final ObjectMapper OBJECT_MAPPER =
064        new ObjectMapper()
065        .enable(MapperFeature.ACCEPT_CASE_INSENSITIVE_VALUES)
066        .enable(SerializationFeature.INDENT_OUTPUT);

068    @CompileTimeCheck
069    private static final Pattern CONNECTION_FILE_NAME_PATTERN =
070        Pattern.compile("(?i)^(kernel-|)(?<id>[^.]+)[.]json$");

072    private final ZMQ.Context context = ZMQ.context(8);
073    private final ConcurrentSkipListMap<String,Connection> connectionMap =
074        new ConcurrentSkipListMap<>();
075    private final Channel.Heartbeat heartbeat = new Channel.Heartbeat(this);
076    private final Channel.Control control = new Control();
077    private final Channel.IOPub iopub = new Channel.IOPub(this);
078    private final Channel.Stdin stdin = new Stdin();
079    private final Channel.Shell shell = new Shell();
080    private InputStream in = null;
081    private PrintStreamBuffer out = null;
082    private PrintStreamBuffer err = null;
083    private String sessionId = null;
084    protected final AtomicInteger execution_count = new AtomicInteger(0);
085    protected transient Message request = null;

087    /**
088     * Sole constructor.
089     */
090    protected Server() { super(16); }
091
092    /**
093     * Add a connection specified by a {@link Connection} {@link File}.
094     *
095     * @param   path            The path to the {@link Connection}
096     *                          {@link File}.
097     *
098     * @throws  IOException     If the {@link File} cannot be opened or
099     *                          parsed.
100     */
101    public void bind(String path) throws IOException {
102        var file = new File(path);
103        var kernelId =
104            Optional.of(file.getName())
105            .map(CONNECTION_FILE_NAME_PATTERN::matcher)
106            .filter(t -> t.matches())
107            .map(t -> t.group("id"))
108            .orElse(null);
109
110        bind(kernelId, file);
111    }
112
113    /**
114     * Add a connection specified by a {@link Connection} {@link File}.
115     *
116     * @param   kernelId        The kernel ID.
117     * @param   file            The {@link Connection} {@link File}.
118     *
119     * @throws  IOException     If the {@link File} cannot be opened or
120     *                          parsed.
121     */
122    protected void bind(String kernelId, File file) throws IOException {
123        var node = OBJECT_MAPPER.readTree(file);
124        var connection = new Connection(kernelId, (ObjectNode) node);
125
126        connectionMap.put(connection.getId(), connection);
127
128        connection.connect(shell, control, iopub, stdin, heartbeat);
129
130        log.info("Connected to {} {}",
131                 connection.getId(), connection.getNode().toPrettyString());
132    }

134    /**
135     * Method to restart the {@link Server}.  Subclass implementations
136     * should chain to the {@code super}-method.
137     */
138    protected void restart() throws Exception {
139        in = new ByteArrayInputStream(new byte[] { });
140        out = new PrintStreamBuffer();
141        err = new PrintStreamBuffer();
142    }

184    /**
185     * Method to schedule a {@link Message} for publishing.  See
186     * {@link Channel.IOPub#pub(Message)}.
187     *
188     * @param   message     The {@link Message} to send.
189     */
190    protected void pub(Message message) { iopub.pub(message); }
191
192    /**
193     * Method to stamp an outgoing {@link Message}.  Adds
194     * {@link #PROTOCOL_VERSION}, session, and
195     * {@link Message#timestamp() timestamp} if not already specified.
196     *
197     * @param   message         The {@link Message} to stamp.
198     *
199     * @return  The {@link Message}.
200     */
201    protected Message stamp(Message message) {
202        if (message.version() == null) {
203            message.version(PROTOCOL_VERSION);
204        }
205
206        if (message.session() == null) {
207            message.session(getSessionId());
208        }
209
210        return message.timestamp();
211    }

213    @ToString
214    private class Control extends Channel.Control {
215        public Control() { super(Server.this); }
216
217        @Override
218        protected void dispatch(Dispatcher dispatcher, ZMQ.Socket socket, Message message) {
219            if (message.isRequest()) {
220                super.dispatch(dispatcher, socket, message);
221            } else {
222                log.warn("Ignoring non-request {}", message.msg_type());
223            }
224        }
225
226        private void shutdown(Dispatcher dispatcher, Message request, Message reply) throws Exception {
227            var restart = request.content().at("/restart").asBoolean();
228
229            reply.content().put("restart", restart);
230
231            if (restart) {
232                Server.this.restart();
233            } else {
234                Server.this.shutdown();
235            }
236        }
237
238        private void interrupt(Dispatcher dispatcher, Message request, Message reply) throws Exception {
239            Server.this.interrupt();
240        }

242        private void debug(Dispatcher dispatcher, Message request, Message reply) throws Exception {
243            throw new UnsupportedOperationException();
244        }

261    @ToString
262    private class Shell extends Channel.Shell {
263        public Shell() {
264            super(Server.this, Server.this.iopub, Server.this.stdin);
265        }
266
267        @Override
268        protected void dispatch(Dispatcher dispatcher, ZMQ.Socket socket, Message message) {
269            if (message.isRequest()) {
270                try {
271                    Server.this.request = message;
272
273                    super.dispatch(dispatcher, socket, Server.this.request);
274                } finally {
275                    Server.this.request = null;
276                }
277            } else {
278                log.warn("Ignoring non-request {}", message.msg_type());
279            }
280        }

286        private void execute(Dispatcher dispatcher, Message request, Message reply) throws Exception {
287            var cellId = request.metadata().at("/cellId").asText();
288            var code = request.content().at("/code").asText();
289            var silent = request.content().at("/silent").asBoolean();
290            var store_history = request.content().at("/store_history").asBoolean();
291            var user_expressions = request.content().at("/user_expressions");
292            var allow_stdin = request.content().at("/allow_stdin").asBoolean();
293            var stop_on_error = request.content().at("/stop_on_error").asBoolean();
294
295            try {
296                if (! code.isEmpty()) {
297                    if (! silent) {
298                        if (store_history) {
299                            execution_count.incrementAndGet();
300
301                            iopub.pub(request.execute_input(code, execution_count.intValue()));
302                        }
303                    }
304
305                    Server.this.execute(code);
306                }
307            } catch (Throwable throwable) {
308                reply.status(throwable, code);
309            } finally {
310                reply.content().put("execution_count", execution_count.intValue());
311
312                if (reply.content().get("status").asText().equals("ok")) {
313                    reply.content().withArray("payload");
314
315                    var in = request.content().at("/user_expressions");
316                    var iterator = in.fields();
317                    var out = reply.content().with("user_expressions");
318
319                    while (iterator.hasNext()) {
320                        var entry = iterator.next();
321                        var name = entry.getKey();
322                        var expression = entry.getValue().asText();
323
324                        try {
325                            out.put(name, String.valueOf(Server.this.evaluate(expression)));
326                        } catch (Throwable throwable) {
327                            out.set(name, Message.content(throwable, expression));
328                        }
329                    }
330                }
331
332                var stdout = out.toString();
333                var stderr = err.toString();
334
335                out.reset();
336                err.reset();
337
338                if (! silent) {
339                    if (! stdout.isEmpty()) {
340                        iopub.pub(request.stream(Message.stream.stdout, stdout));
341                    }
342
343                    if (! stderr.isEmpty()) {
344                        iopub.pub(request.stream(Message.stream.stderr, stderr));
345                    }
346                }
347            }
348        }

390        private void comm_info(Dispatcher dispatcher, Message request, Message reply) throws Exception {
391            var comms = reply.content().with("comms");
392            /*
393             * Currently unsupported so empty reply.  But remember to *copy*
394             * internal state since output may be culled after satisfying
395             * query.
396             */
397            if (request.content().hasNonNull("target_name")) {
398                var target_name =
399                    request.content().at("/target_name").asText();
400                var iterator = comms.fields();
401
402                while (iterator.hasNext()) {
403                    var entry = iterator.next();
404                    var dict = entry.getValue();
405
406                    if (dict.isObject()) {
407                        ((ObjectNode) dict).retain(target_name);
408
409                        if (dict.isEmpty()) {
410                            iterator.remove();
411                        }
412                    }
413                }
414            }
415        }
