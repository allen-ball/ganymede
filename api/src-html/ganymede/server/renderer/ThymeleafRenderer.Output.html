021import ball.annotation.ServiceProviderFor;
022import com.fasterxml.jackson.databind.node.ObjectNode;
023import ganymede.server.Renderer;
024import java.util.Map;
025import lombok.Data;
026import lombok.NoArgsConstructor;
027import lombok.ToString;
028import org.springframework.core.io.ClassPathResource;
029import org.springframework.util.StreamUtils;
030import org.thymeleaf.TemplateEngine;
031import org.thymeleaf.context.Context;
032import org.thymeleaf.templatemode.TemplateMode;
033import org.thymeleaf.templateresolver.StringTemplateResolver;
034
035import static java.nio.charset.StandardCharsets.UTF_8;
036import static org.springframework.util.MimeTypeUtils.TEXT_HTML_VALUE;
037import static org.springframework.util.MimeTypeUtils.TEXT_PLAIN_VALUE;
038import static org.springframework.util.MimeTypeUtils.TEXT_XML_VALUE;
039
040/**
041 * Thymeleaf template {@link Renderer}.
042 *
043 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
044 */
045@ServiceProviderFor({ Renderer.class })
046@ForClass(ThymeleafRenderer.Output.class)
047@NoArgsConstructor @ToString
048public class ThymeleafRenderer implements Renderer {
049    private static final StringTemplateResolver RESOLVER;
050    private static final TemplateEngine ENGINE;
051
052    static {
053        RESOLVER = new StringTemplateResolver();
054
055        ENGINE = new TemplateEngine();
056        ENGINE.setTemplateResolver(RESOLVER);
057    }
058
059    /**
060     * Method to evaluate a template from a {@link String}.
061     *
062     * @param template      The template {@link String}.
063     * @param mode          The template resolver mode.
064     * @param map           The {@link Context} name-value pairs.
065     *
066     * @return  The {@link Output} to be renderered.
067     */
068    public static Output process(String template, String mode, Map<String,Object> map) {
069        RESOLVER.setTemplateMode(mode);
070
071        return new Output(RESOLVER.getTemplateMode(),
072                          ENGINE.process(template, new Context(null, map)));
073    }
074
075    /**
076     * Method to evaluate a template from a resource.
077     *
078     * @param type          The {@link Class} to search relative for the
079     *                      resource.
080     * @param name          The template resource name.
081     * @param mode          The template resolver mode.
082     * @param map           The {@link Context} name-value pairs.
083     *
084     * @return  The {@link Output} to be renderered.
085     */
086    public static Output process(Class<?> type, String name, String mode, Map<String,Object> map) {
087        return process(getResourceAsString(type, name), mode, map);
088    }
089
090    @Override
091    public void renderTo(ObjectNode bundle, Object object) {
092        var output = (Output) object;
093        var string = output.getOutput();
094        var mimeType = TEXT_PLAIN_VALUE;
095
096        switch (output.getTemplateMode()) {
097        case CSS:
098            mimeType = "text/css";
099            break;
100
101        case HTML:
102        case HTML5:
103        case LEGACYHTML5:
104        case VALIDXHTML:
105        case XHTML:
106            mimeType = TEXT_HTML_VALUE;
107            break;
108
109        case JAVASCRIPT:
110            mimeType = "text/javascript";
111            break;
112
113        case VALIDXML:
114        case XML:
115            mimeType = TEXT_XML_VALUE;
116            break;
117
118        default:
119            break;
120        }
121
122        if (! bundle.with(DATA).has(mimeType)) {
123            bundle.with(DATA).put(mimeType, string);
124        }
125    }
126
127    private static String getResourceAsString(Class<?> type, String name) {
128        try (var in = new ClassPathResource(name, type).getInputStream()) {
129            return StreamUtils.copyToString(in, UTF_8);
130        } catch (Exception exception) {
131            throw new ExceptionInInitializerError(exception);
132        }
133    }
134
135    /**
136     * Customized {@link Output Output} for {@link ThymeleafRenderer}.
137     *
138     * {@bean.info}
139     */
140    @Data
141    public static class Output {
142        private final TemplateMode templateMode;
143        private final String output;
144    }
145}
