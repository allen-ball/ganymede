021import com.fasterxml.jackson.databind.node.ObjectNode;
022import ganymede.server.renderer.ForClass;
023import ganymede.server.renderer.ForClassName;
024import ganymede.util.ServiceProviderMap;
025import java.util.Comparator;
026import java.util.Optional;
027import java.util.TreeMap;
028import lombok.NoArgsConstructor;
029import lombok.ToString;
030
031/**
032 * {@link Renderer} {@link java.util.Map}.
033 *
034 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
035 */
036public class RendererMap extends TreeMap<Class<?>,Renderer> {
037    private static final long serialVersionUID = 5660512908511151101L;
038
039    private static final Comparator<Class<?>> COMPARATOR =
040        new IsAssignableFromOrder().thenComparing(Class::getName);
041
042    /** @serial */
043    private final ServiceProviderMap<Renderer> map =
044        new ServiceProviderMap<>(Renderer.class);
045
046    /**
047     * Sole constructor.
048     */
049    public RendererMap() {
050        super(COMPARATOR);
051
052        reload();
053    }
054
055    /**
056     * Reload the underlying {@link ServiceProviderMap} and add the
057     * corresponding entries.
058     *
059     * @return  {@link.this}
060     */
061    public RendererMap reload() {
062        var iterator = map.reload(this::accept).values().iterator();
063
064        while (iterator.hasNext()) {
065            var renderer = iterator.next();
066
067            try {
068                var key = getRenderType(renderer.getClass());
069
070                computeIfAbsent(key, k -> renderer);
071            } catch (Exception exception) {
072            }
073        }
074
075        return this;
076    }
077
078    private boolean accept(Class<?> provider) {
079        boolean accept = true;
080
081        try {
082            getRenderType(provider);
083        } catch (Exception exception) {
084            accept = false;
085        }
086
087        return accept;
088    }
089
090    private Class<?> getRenderType(Class<?> provider) throws Exception {
091        Class<?> type = null;
092
093        if (provider.isAnnotationPresent(ForClassName.class)) {
094            var name = provider.getAnnotation(ForClassName.class).value();
095
096            type = Class.forName(name, false, map.getClassLoader());
097        } else {
098            type = provider.getAnnotation(ForClass.class).value();
099        }
100
101        return type;
102    }
103
104    /**
105     * Method to render an {@link Object} to a {@code mime-bundle}.
106     *
107     * @param   bundle          The {@code mime-bundle}.
108     * @param   object          The {@link Object} to render.
109     * @param   alternates      Optional alternate representations.
110     */
111    public void renderTo(ObjectNode bundle, Object object, Object... alternates) {
112        var type = (object != null) ? object.getClass() : Object.class;
113
114        reload().find(type).ifPresent(t -> t.renderTo(bundle, object));
115
116        if (alternates != null) {
117            for (var alternate : alternates) {
118                if (alternate != null) {
119                    find(alternate.getClass())
120                        .ifPresent(t -> t.renderTo(bundle, alternate));
121                }
122            }
123        }
124    }
125
126    private Optional<Renderer> find(Class<?> type) {
127        var value =
128            entrySet().stream()
129            .filter(t -> t.getKey().isAssignableFrom(type))
130            .map(t -> t.getValue())
131            .findFirst();
132
133        return value;
134    }
135
136    @NoArgsConstructor @ToString
137    private static class IsAssignableFromOrder implements Comparator<Class<?>> {
138        @Override
139        public int compare(Class<?> left, Class<?> right) {
140            boolean ordered = right.isAssignableFrom(left);
141            boolean reversed = left.isAssignableFrom(right);
142
143            return (ordered ^ reversed) ? (ordered ? -1 : 1) : 0;
144        }
145    }
146}
