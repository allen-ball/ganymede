038import static java.nio.charset.StandardCharsets.UTF_8;
039import static java.util.Collections.disjoint;
040import static javax.script.ScriptContext.ENGINE_SCOPE;

042/**
043 * {@link Scala} {@link Magic}.
044 *
045 * @see Scripted
046 *
047 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
048 */
049@ServiceProviderFor({ Magic.class })
050@Description("Execute code in scala REPL")
051@NoArgsConstructor @ToString @Log4j2
052public class Scala extends AbstractScriptEngineMagic {
053    @Override
054    protected Scripted engine() {
055        if (engine == null) {
056            /*
057             * Can't simply extend AbstractScriptEngineMagic because:
058             *
059             *     new ScriptEngineManager().getEngineByName("scala");
060             *
061             * tickles https://github.com/scala/bug/issues/11754.
062             *
063             * The following workaround is required to use println():
064             *
065             *     Console.withOut(System.out) { println($ctx.greeting) }
066             */
067            try {
068                var manager = new ScriptEngineManager(getClass().getClassLoader());
069                var factory =
070                    (Scripted.Factory)
071                    manager.getEngineFactories().stream()
072                    .filter(t -> (! disjoint(t.getExtensions(), getExtensions())))
073                    .findFirst().orElse(null);
074                var settings = new Settings();
075                /*
076                 * settings.Yreplclassbased().value_$eq(true);
077                 * settings.usejavacp().value_$eq(true);
078                 */
079                context.classpath.stream()
080                    .forEach(t -> settings.classpath().append(t));
081
082                var scripted = Scripted.apply(factory, settings, Scripted.apply$default$3());
083                var bindings = context.context.getBindings(ENGINE_SCOPE);
084
085                scripted.setBindings(bindings, ENGINE_SCOPE);
086
087                if (initialize(scripted)) {
088                    engine = scripted;
089                }
090            } catch (NoClassDefFoundError error) {
091            } catch (Throwable throwable) {
092                throwable.printStackTrace(System.err);
093            }
094        }
095
096        return (Scripted) engine;
097    }
098
099    @Override
100    protected boolean initialize(ScriptEngine engine) {
101        var initialized = true;
102        var scripted = (Scripted) engine;
103        var name = getClass().getSimpleName() + "." + "scala";
104        var resource = new ClassPathResource(name, getClass());
105
106        if (initialized && resource.exists()) {
107            try (var in = new InputStreamReader(resource.getInputStream(), UTF_8)) {
108                scripted.compile(in).eval(context.context);
109            } catch (Throwable throwable) {
110                initialized = false;
111                throwable.printStackTrace(System.err);
112            }
113        }
114
115        if (initialized) {
116            for (var method : NotebookContext.getNotebookFunctions()) {
117                /*
118                 * TBD: Generate Scala wrappers through reflection.
119                 */
120            }
121        }
122
123        return initialized;
124    }
125
126    @Override
127    protected void execute(String code) {
128        var engine = engine();
129
130        if (engine != null) {
131            for (var path : context.classpath) {
132                /*
133                 * TBD: Add paths to Scripted classpath.
134                 */
135            }
136
137            for (var statement : context.imports) {
138                try {
139                    statement =
140                        statement
141                        .replaceAll("[;]", "")
142                        .replaceAll("[\\p{Space}]+", " ")
143                        .replaceAll("import static ", "import ")
144                        .replaceAll("[*]", "_")
145                        .strip();
146                    /*
147                     * TBD: Add imports (depends on successful classpath
148                     * updates).
149                     *
150                     * engine.compile(statement).eval(context.context);
151                     */
152                } catch (Throwable throwable) {
153                    throwable.printStackTrace(System.err);
154                }
155            }
156
157            try {
158                if (! code.isBlank()) {
159                    render(engine.compile(code).eval(context.context));
160                } else {
161                    show();
162                }
163            } catch (Throwable throwable) {
164                throwable.printStackTrace(System.err);
165            }
166        } else {
167            super.execute(code);
168        }
169    }
170}
