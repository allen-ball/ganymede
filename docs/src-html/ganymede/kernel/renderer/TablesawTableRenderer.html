038/**
039 * {@link.uri https://github.com/jtablesaw/tablesaw target=newtab Tablesaw}
040 * {@link Table} {@link Renderer} service provider.
041 *
042 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
043 */
044@ServiceProviderFor({ Renderer.class })
045@ForClassName("tech.tablesaw.api.Table")
046@NoArgsConstructor @ToString
047public class TablesawTableRenderer extends AbstractRenderer {
048    @Override
049    public Optional<TablesawTableRenderer> instance() {
050        return Optional.ofNullable(getRenderType()).map(t -> new Impl());
051    }
052
053    @Override
054    public void renderTo(ObjectNode bundle, Object object) {
055        throw new IllegalStateException();
056    }
057
058    @NoArgsConstructor @ToString
059    private class Impl extends TablesawTableRenderer {
060        @Override
061        public void renderTo(ObjectNode bundle, Object object) {
062            var table = (Table) object;
063/*
064            if (! bundle.with(DATA).has(APPLICATION_JSON_VALUE)) {
065                try {
066                    if (hasWriterForExtension("json")) {
067                        var string = table.write().toString("json");
068                        var node = ObjectMappers.JSON.readTree(string);
069
070                        bundle.with(DATA)
071                            .set(APPLICATION_JSON_VALUE, node);
072                        bundle.with(METADATA)
073                            .with(APPLICATION_JSON_VALUE)
074                            .put("expanded", true);
075                    }
076                } catch (Exception exception) {
077                    exception.printStackTrace(System.err);
078                }
079            }
080*/
081            if (! bundle.with(DATA).has(TEXT_HTML_VALUE)) {
082                try {
083                    if (hasWriterForExtension("html")) {
084                        bundle.with(DATA)
085                            .put(TEXT_HTML_VALUE, table.write().toString("html"));
086                    }
087                } catch (Exception exception) {
088                    exception.printStackTrace(System.err);
089                }
090            }
091
092            if (table != null) {
093                var string = "";
094
095                try (var out = new ByteArrayOutputStream()) {
096                    new DataFramePrinter(Integer.MAX_VALUE, out).print(table);
097
098                    string = new String(out.toByteArray());
099                } catch (Exception exception) {
100                    exception.printStackTrace(System.err);
101                }
102
103                renderers.renderTo(bundle, string);
104            }
105        }
106
107        private boolean hasWriterForExtension(String extension) {
108            return defaultWriterRegistry.getWriterForExtension(extension) != null;
109        }
110    }
111}
