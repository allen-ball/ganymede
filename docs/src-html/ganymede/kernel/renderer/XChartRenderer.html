021import ball.annotation.ServiceProviderFor;
022import com.fasterxml.jackson.databind.node.ObjectNode;
023import ganymede.notebook.AbstractRenderer;
024import ganymede.notebook.ForClassName;
025import ganymede.notebook.Renderer;
026import java.io.IOException;
027import java.util.Optional;
028import lombok.NoArgsConstructor;
029import lombok.ToString;
030import org.knowm.xchart.BitmapEncoder;
031import org.knowm.xchart.internal.chartpart.Chart;
032
033import static org.knowm.xchart.BitmapEncoder.BitmapFormat.PNG;
034
035/**
036 * {@link.uri https://github.com/knowm/XChart target=newtab XChart}
037 * {@link Chart} {@link Renderer} service provider.  See
038 * {@link BitmapEncoder}.
039 *
040 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
041 */
042@ServiceProviderFor({ Renderer.class })
043@ForClassName("org.knowm.xchart.internal.chartpart.Chart")
044@NoArgsConstructor @ToString
045public class XChartRenderer extends AbstractRenderer {
046    @Override
047    public Optional<XChartRenderer> instance() {
048        return Optional.ofNullable(getRenderType()).map(t -> new Impl());
049    }
050
051    @Override
052    public void renderTo(ObjectNode bundle, Object object) {
053        throw new IllegalStateException();
054    }
055
056    @NoArgsConstructor @ToString
057    private class Impl extends XChartRenderer {
058        @Override
059        public void renderTo(ObjectNode bundle, Object object) {
060            try {
061                var bytes = BitmapEncoder.getBitmapBytes((Chart) object, PNG);
062
063                renderers.renderTo(bundle, bytes);
064            } catch (IOException exception) {
065                exception.printStackTrace(System.err);
066            }
067        }
068    }
069}
