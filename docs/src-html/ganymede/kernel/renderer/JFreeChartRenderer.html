021import ball.annotation.ServiceProviderFor;
022import com.fasterxml.jackson.databind.node.ObjectNode;
023import ganymede.notebook.AbstractRenderer;
024import ganymede.notebook.ForClassName;
025import ganymede.notebook.Renderer;
026import java.io.ByteArrayOutputStream;
027import java.io.IOException;
028import java.io.PrintWriter;
029import java.io.StringWriter;
030import java.util.Optional;
031import java.util.UUID;
032import lombok.NoArgsConstructor;
033import lombok.ToString;
034import org.jfree.chart.ChartRenderingInfo;
035import org.jfree.chart.ChartUtils;
036import org.jfree.chart.JFreeChart;
037import org.jfree.chart.entity.StandardEntityCollection;
038
039import static org.springframework.util.MimeTypeUtils.TEXT_HTML_VALUE;
040
041/**
042 * {@link.uri https://github.com/jfree/jfreechart target=newtab JFreeChart}
043 * {@link Renderer} service provider.
044 *
045 * @see JFreeChart
046 * @see ChartUtils
047 *
048 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
049 */
050@ServiceProviderFor({ Renderer.class })
051@ForClassName("org.jfree.chart.JFreeChart")
052@NoArgsConstructor @ToString
053public class JFreeChartRenderer extends AbstractRenderer {
054    @Override
055    public Optional<JFreeChartRenderer> instance() {
056        return Optional.ofNullable(getRenderType()).map(t -> new Impl());
057    }
058
059    @Override
060    public void renderTo(ObjectNode bundle, Object object) {
061        throw new IllegalStateException();
062    }
063
064    @NoArgsConstructor @ToString
065    private class Impl extends JFreeChartRenderer {
066        @Override
067        public Optional<JFreeChartRenderer> instance() {
068            return Optional.of(this);
069        }
070
071        @Override
072        public void renderTo(ObjectNode bundle, Object object) {
073            var chart = (JFreeChart) object;
074            var width = 800;
075            var height = 600;
076
077            try (var out = new ByteArrayOutputStream()) {
078                var info = new ChartRenderingInfo(new StandardEntityCollection());
079
080                ChartUtils.writeChartAsPNG(out, chart, width, height, info);
081
082                var image = out.toByteArray();
083
084                renderers.renderTo(bundle, image);
085
086                if (! bundle.with(DATA).has(TEXT_HTML_VALUE)) {
087                    var html = new StringWriter();
088                    var name = UUID.randomUUID().toString();
089                    var mimeType = bundle.with(METADATA).fieldNames().next();
090                    var base64 = bundle.with(DATA).get(mimeType).asText();
091
092                    try (var writer = new PrintWriter(html)) {
093                        writer.format("<img usemap=\"#%s\" src=\"data:%s;base64,%s\"/>\n", name, mimeType, base64);
094                        ChartUtils.writeImageMap(writer, name, info, false);
095                    }
096
097                    bundle.with(DATA).put(TEXT_HTML_VALUE, html.toString());
098                }
099            } catch (Throwable throwable) {
100                throwable.printStackTrace(System.err);
101            }
102        }
103    }
104}
