021import com.fasterxml.jackson.databind.JsonNode;
022import ball.annotation.ServiceProviderFor;
023import ganymede.notebook.AbstractPropertiesMagic;
024import ganymede.notebook.Description;
025import ganymede.notebook.Magic;
026import lombok.NoArgsConstructor;
027import lombok.ToString;
028import lombok.extern.log4j.Log4j2;
029
030/**
031 * {@link SysProperties} {@link Magic}.
032 *
033 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
034 */
035/* @ServiceProviderFor({ Magic.class }) */
036@Description("Add/Update or print System properties")
037@NoArgsConstructor @ToString @Log4j2
038public class SysProperties extends AbstractPropertiesMagic {
039    @Override
040    public void execute(String line0, String code, JsonNode metadata) throws Exception {
041        if (! code.isBlank()) {
042            var properties = compile(code);
043
044            System.getProperties().putAll(properties);
045        } else {
046            System.getProperties().store(System.out, null);
047        }
048    }
049}
