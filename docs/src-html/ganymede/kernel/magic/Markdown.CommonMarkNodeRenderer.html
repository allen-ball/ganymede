042import static org.springframework.util.MimeTypeUtils.TEXT_HTML_VALUE;
043
044/**
045 * {@link Markdown} {@link Magic}.
046 *
047 * @see Parser
048 * @see HtmlRenderer
049 * @see TextContentRenderer
050 *
051 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
052 */
053@ServiceProviderFor({ Magic.class })
054@Description("Markdown template evaluator")
055@ScriptEngineName("handlebars")
056@NoArgsConstructor @ToString @Log4j2
057public class Markdown extends Handlebars {
058    private static final List<Extension> EXTENSIONS = new ArrayList<>();
059
060    /**
061     * Installed {@link Extension}s.
062     *
063     * {@include #TYPES}
064     */
065    public static final List<Class<? extends Extension>> TYPES =
066        List.of(org.commonmark.ext.autolink.AutolinkExtension.class,
067                org.commonmark.ext.gfm.strikethrough.StrikethroughExtension.class,
068                org.commonmark.ext.gfm.tables.TablesExtension.class,
069                org.commonmark.ext.heading.anchor.HeadingAnchorExtension.class,
070                org.commonmark.ext.image.attributes.ImageAttributesExtension.class,
071                org.commonmark.ext.ins.InsExtension.class,
072                org.commonmark.ext.task.list.items.TaskListItemsExtension.class,
073                org.commonmark.ext.front.matter.YamlFrontMatterExtension.class);
074    static {
075        try {
076            for (Class<? extends Extension> type : TYPES) {
077                var extension = (Extension) type.getMethod("create").invoke(null);
078
079                EXTENSIONS.add(extension);
080            }
081        } catch (Exception exception) {
082            throw new ExceptionInInitializerError(exception);
083        }
084    }
085
086    private final Parser parser = Parser.builder().extensions(EXTENSIONS).build();
087
088    @Override
089    protected void render(Object object) {
090        try {
091            var markdown = (String) object;
092            var node = parser.parse(markdown);
093
094            new KernelRestClient().print(Message.mime_bundle(node));
095        } catch (Exception exception) {
096            exception.printStackTrace(System.err);
097        }
098    }
099
100    /**
101     * CommonMark {@link Node} {@link Renderer} service provider.
102     */
103    @ServiceProviderFor({ Renderer.class })
104    @ForClass(Node.class)
105    @NoArgsConstructor @ToString
106    public static class CommonMarkNodeRenderer extends AbstractRenderer {
107        private HtmlRenderer html = HtmlRenderer.builder().extensions(EXTENSIONS).build();
108        private TextContentRenderer text = TextContentRenderer.builder().extensions(EXTENSIONS).build();
109
110        @Override
111        public void renderTo(ObjectNode bundle, Object object) {
112            var node = (Node) object;
113
114            if (! bundle.with(DATA).has(TEXT_HTML_VALUE)) {
115                bundle.with(DATA).put(TEXT_HTML_VALUE, html.render(node));
116            }
117
118            renderers.renderTo(bundle, text.render(node));
119        }
120    }
121}
