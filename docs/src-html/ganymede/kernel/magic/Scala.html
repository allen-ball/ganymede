058    private Object eval(ScriptEngine engine, ClassPathResource resource) throws IOException, ScriptException {
059        Object object = null;
060
061        try (var in = resource.getInputStream()) {
062            object = eval(engine, in);
063        }
064
065        return object;
066    }
067
068    private Object eval(ScriptEngine engine, InputStream in) throws IOException, ScriptException {
069        Object object = null;
070
071        try (var reader = new InputStreamReader(in, UTF_8)) {
072            var script =
073                engine.getClass()
074                .getMethod("compile", Reader.class)
075                .invoke(engine, reader);
076
077            object =
078                script.getClass()
079                .getMethod("eval", ScriptContext.class)
080                .invoke(script, context.context);
081        } catch (IOException exception) {
082            throw exception;
083        } catch (InvocationTargetException exception) {
084            var cause = exception.getCause();
085
086            if (cause instanceof IOException) {
087                throw (IOException) cause;
088            } else if (cause instanceof ScriptException) {
089                throw (ScriptException) cause;
090            } else {
091                throw new IllegalStateException(exception);
092            }
093        } catch (RuntimeException exception) {
094            throw exception;
095        } catch (Error error) {
096            throw error;
097        } catch (Throwable throwable) {
098            throw new IllegalStateException(throwable);
099        }
100
101        return object;
102    }
103
104    private Object eval(ScriptEngine engine, String code) throws ScriptException {
105        Object object = null;
106
107        try {
108            var script =
109                engine.getClass()
110                .getMethod("compile", String.class)
111                .invoke(engine, code);
112
113            object =
114                script.getClass()
115                .getMethod("eval", ScriptContext.class)
116                .invoke(script, context.context);
117        } catch (InvocationTargetException exception) {
118            var cause = exception.getCause();
119
120            if (cause instanceof ScriptException) {
121                throw (ScriptException) cause;
122            } else {
123                throw new IllegalStateException(exception);
124            }
125        } catch (RuntimeException exception) {
126            throw exception;
127        } catch (Error error) {
128            throw error;
129        } catch (Throwable throwable) {
130            throw new IllegalStateException(throwable);
131        }
132
133        return object;
134    }

136    @Override
137    protected ScriptEngine engine() {
138        if (engine == null) {
139            Class<? extends ScriptEngine> type = null;
140
141            try {
142                type =
143                    Class.forName("scala.tools.nsc.interpreter.Scripted", false, context.getClassLoader())
144                    .asSubclass(ScriptEngine.class);
145            } catch (ClassNotFoundException exception) {
146            }
147
148            isScriptedScriptEngine = (type != null);
149
150            if (isScriptedScriptEngine) {
151                /*
152                 * Can't simply extend AbstractScriptEngineMagic because:
153                 *
154                 *     new ScriptEngineManager().getEngineByName("scala");
155                 *
156                 * tickles https://github.com/scala/bug/issues/11754.
157                 *
158                 * The following workaround is required to use println():
159                 *
160                 *     Console.withOut(System.out) { println($ctx.greeting) }
161                 */
162                try {
163                    var manager = new ScriptEngineManager(context.getClassLoader());
164                    var factory =
165                        manager.getEngineFactories().stream()
166                        .filter(t -> (! disjoint(t.getExtensions(), getExtensions())))
167                        .findFirst().orElse(null);
168                    var settings = new Settings();
169                    /*
170                     * settings.Yreplclassbased().value_$eq(true);
171                     * settings.usejavacp().value_$eq(true);
172                     */
173                    context.classpath.stream()
174                        .forEach(t -> settings.classpath().append(t));
175
176                    var out = type.getMethod("apply$default$3").invoke(null);
177                    var scripted =
178                        type.cast(type.getMethod("apply", ScriptEngineFactory.class, Settings.class, PrintWriter.class)
179                                  .invoke(null, factory, settings, out));
180                    var bindings = context.context.getBindings(ENGINE_SCOPE);
181
182                    scripted.setBindings(bindings, ENGINE_SCOPE);
183
184                    if (initialize(scripted)) {
185                        engine = scripted;
186                    }
187                } catch (NoClassDefFoundError error) {
188                } catch (Throwable throwable) {
189                    throwable.printStackTrace(System.err);
190                }
191            } else {
192                super.engine();
193            }
194        } else {
195            super.engine();
196        }
197
198        return engine;
199    }

201    @Override
202    protected boolean initialize(ScriptEngine engine) {
203        var initialized = true;
204
205        if (isScriptedScriptEngine) {
206            var resource = getInitScript(getExtensions());
207
208            if (initialized && resource != null) {
209                try {
210                    eval(engine, resource);
211                } catch (Throwable throwable) {
212                    initialized = false;
213                    throwable.printStackTrace(System.err);
214                }
215            }
216        } else {
217            initialized = super.initialize(engine);
218        }
219
220        if (initialized) {
221            for (var method : NotebookContext.getNotebookFunctions()) {
222                /*
223                 * TBD: Generate Scala wrappers through reflection.
224                 */
225            }
226        }
227
228        return initialized;
229    }

231    @Override
232    protected void execute(String code) {
233        var engine = engine();
234
235        if (engine != null) {
236            for (var path : context.classpath) {
237                /*
238                 * TBD: Add paths to ScriptEngine classpath.
239                 */
240            }
241
242            for (var statement : context.imports) {
243                try {
244                    statement =
245                        statement
246                        .replaceAll("[;]", "")
247                        .replaceAll("[\\p{Space}]+", " ")
248                        .replaceAll("import static ", "import ")
249                        .replaceAll("[*]", "_")
250                        .strip();
251                    /*
252                     * TBD: Add imports (depends on successful classpath
253                     * updates).
254                     *
255                     * eval(engine, statement);
256                     */
257                } catch (Throwable throwable) {
258                    throwable.printStackTrace(System.err);
259                }
260            }
261
262            if (isScriptedScriptEngine && (! code.isBlank())) {
263                try {
264                    render(eval(engine, code));
265                } catch (Throwable throwable) {
266                    throwable.printStackTrace(System.err);
267                }
268            } else {
269                super.execute(code);
270            }
271        } else {
272            super.execute(code);
273        }
274    }
