021import com.fasterxml.jackson.databind.JsonNode;
022import ball.annotation.ServiceProviderFor;
023import ganymede.notebook.AbstractPropertiesMagic;
024import ganymede.notebook.Description;
025import ganymede.notebook.Magic;
026import ganymede.notebook.MagicNames;
027import lombok.Data;
028import lombok.NoArgsConstructor;
029import lombok.ToString;
030import lombok.extern.log4j.Log4j2;
031import org.apache.spark.SparkConf;
032import picocli.CommandLine.Command;
033import picocli.CommandLine.Option;
034import picocli.CommandLine.ParameterException;
035import picocli.CommandLine.Parameters;
036
037import static org.apache.spark.sql.SparkSession.builder;
038import static org.apache.spark.sql.SparkSession.getDefaultSession;
039import static org.apache.spark.sql.SparkSession.setActiveSession;
040import static org.apache.spark.sql.SparkSession.setDefaultSession;
041
042/**
043 * {@link SparkSession} {@link Magic}.
044 *
045 * @see org.apache.spark.sql.SparkSession
046 *
047 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
048 */
049@ServiceProviderFor({ Magic.class })
050@MagicNames({ "spark-session" })
051@Description("Configure and start a Spark session")
052@NoArgsConstructor @ToString @Log4j2
053public class SparkSession extends AbstractPropertiesMagic {
054    @Override
055    public void execute(String line0, String code, JsonNode metadata) throws Exception {
056        try {
057            var argv = Magic.getCellMagicCommand(line0);
058            var arguments = new Arguments();
059
060            parse(argv, arguments);
061
062            var config = new SparkConf();
063
064            compile(code).entrySet().stream()
065                .forEach(t -> config.set(t.getKey().toString(), String.valueOf(t.getValue())));
066
067            if (arguments.getMaster() != null) {
068                config.setMaster(arguments.getMaster());
069            }
070
071            if (arguments.getAppName() != null) {
072                config.setAppName(arguments.getAppName());
073            }
074
075            var builder = builder().config(config);
076
077            if (arguments.isEnableHiveIfAvailable()) {
078                try {
079                    builder.enableHiveSupport();
080                } catch (Exception exception) {
081                    System.err.println(exception.getMessage());
082                }
083            }
084
085            var session = builder.getOrCreate();
086
087            if (getDefaultSession().isEmpty()) {
088                setDefaultSession(session);
089            }
090
091            setActiveSession(session);
092
093            context.display(session);
094        } catch (ParameterException exception) {
095            System.err.println(exception.getMessage());
096            System.err.println();
097            exception.getCommandLine().usage(System.err);
098        } catch (NoClassDefFoundError error) {
099            System.err.format("Apache Spark not found on classpath: %s:\n", error);
100        } catch (Throwable throwable) {
101            throwable.printStackTrace(System.err);
102        }
103    }
104
105    @Override
106    public String getUsage() { return getUsage(new Arguments()); }
107
108    @Command @Data
109    private class Arguments {
110        @Parameters(description = { "Spark master" }, index = "0", arity = "0..1")
111        private String master = null;
112
113        @Parameters(description = { "Spark appName" }, index = "1", arity = "0..1", defaultValue = "root")
114        private String appName = null;
115
116        @Option(names = { "--no-enable-hive-if-available" }, negatable = true,
117                description = { "Enable Hive if available.  true by default" })
118        private boolean enableHiveIfAvailable = true;
119    }
120}
