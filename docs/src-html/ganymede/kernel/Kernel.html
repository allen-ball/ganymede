077     * The name of the {@link System} property containing the

082    @Value("${JPY_PARENT_PID:#{-1}}")
083    private long jpy_parent_pid = -1;
084
085    @Value("${runtime-dir:#{null}}")
086    private String runtime_dir = null;
087
088    @Value("${connection-file:#{null}}")
089    private String connection_file = null;
090
091    @Value("${spark-home:#{null}}")
092    private String spark_home = null;
093
094    @Value("${hive-home:#{null}}")
095    private String hive_home = null;
096
097    @Value("${kernel.version}")
098    private String kernel_version = null;
099
100    private final Shell shell = new Shell(this);
101    private ApplicationContext context = null;
102    private int port = -1;
103    private JupyterRestClient client = null;
104    private final ObjectNode kernel_info_reply_content;
105

107        try (var in = getClass().getResourceAsStream("kernel_info_reply.yaml")) {
108            kernel_info_reply_content = (ObjectNode) YAML_OBJECT_MAPPER.readTree(in).with("content");
109            kernel_info_reply_content.put("protocol_version", PROTOCOL_VERSION.toString());
110        } catch (Exception exception) {
111            throw new ExceptionInInitializerError(exception);

122    @PostConstruct
123    public void init() throws Exception {
124        if (runtime_dir != null) {
125            var glob = String.format("{nb,jp}server-%d.json", jpy_parent_pid);
126
127            try (var stream = Files.newDirectoryStream(Paths.get(runtime_dir), glob)) {
128                var path = stream.iterator().next();
129
130                client = new JupyterRestClient(JSON_OBJECT_MAPPER.readTree(path.toFile()));
131            } catch (NoSuchElementException exception) {
132                log.warn("{}: No match found for '{}'", runtime_dir, glob);
133            } catch (Exception exception) {
134                log.warn("{}: {}", runtime_dir, exception);
135            }
136        }
137
138        if (spark_home != null) {
139            var parent = Paths.get(spark_home, "jars").toFile();
140
141            shell.addKnownDependenciesToClasspath(parent);
142        }
143
144        if (hive_home != null) {
145            var parent = Paths.get(hive_home, "lib").toFile();
146
147            shell.addKnownDependenciesToClasspath(parent);
148        }
149
150        restart();
151    }
152
153    @PreDestroy
154    public void destroy() { super.shutdown(); }
155
156    /**
157     * REST method to retrieve the current {@link Shell#classpath()}.
158     */
159    @RequestMapping(method = { GET }, value = { "kernel/classpath" })
160    public ResponseEntity<String> classpath() throws Exception {
161        var json = JSON_OBJECT_MAPPER.writeValueAsString(shell.classpath());
162
163        return new ResponseEntity<>(json, HttpStatus.OK);
164    }
165
166    /**
167     * REST method to display MIME bundles from a sub-process.  See
168     * {@link KernelRestClient#display(JsonNode)}.
169     *
170     * @param   bundle          The MIME bundle {@link ObjectNode}.
171     */
172    @RequestMapping(method = { PUT }, value = { "kernel/display" })
173    public ResponseEntity<String> display(@RequestBody ObjectNode bundle) {
174        var request = this.request;
175
176        if (request != null) {
177            var silent = request.content().at("/silent").asBoolean();
178
179            if (! silent) {
180                pub(request.display_data(bundle));
181            }
182        }
183
184        return new ResponseEntity<>(HttpStatus.OK);
185    }
186
187    /**
188     * REST method to print MIME bundles from a sub-process.  See
189     * {@link KernelRestClient#print(JsonNode)}.
190     *
191     * @param   bundle          The MIME bundle {@link ObjectNode}.
192     */
193    @RequestMapping(method = { PUT }, value = { "kernel/print" })
194    public ResponseEntity<String> print(@RequestBody ObjectNode bundle) {
195        var request = this.request;
196
197        if (request != null) {
198            var silent = request.content().at("/silent").asBoolean();
199
200            if (! silent) {
201                pub(request.execute_result(execution_count.intValue(), bundle));
202            }
203        }
204
205        return new ResponseEntity<>(HttpStatus.OK);
206    }
207
208    @Override

210        this.context = context;
211    }
212
213    @EventListener({ ServletWebServerInitializedEvent.class })
214    public void onApplicationEvent(ServletWebServerInitializedEvent event) {
215        port = event.getWebServer().getPort();
216    }
217
218    @EventListener({ ContextClosedEvent.class })
219    public void onApplicationEvent(ContextClosedEvent event) {
220        super.shutdown();
221    }
222
223    @Override
224    protected void bind(String kernelId, File file) throws IOException {
225        super.bind(kernelId, file);
226
227        var node = (ObjectNode) JSON_OBJECT_MAPPER.readTree(file);
228
229        node.put("pid", ProcessHandle.current().pid());
230
231        if (port > 0) {
232            node.put("port", port);
233        }
234
235        JSON_OBJECT_MAPPER.writeValue(file, node);
236    }
237
238    @Override
239    protected void restart() throws Exception {
240        super.restart();
241
242        shell.restart(getIn(), getOut(), getErr());
243
244        setSessionId(UUID.randomUUID().toString());
245    }
246
247    @Override
248    protected ObjectNode getKernelInfo() { return kernel_info_reply_content; }
249
250    @Override
251    protected void execute(String code) throws Exception {
252        shell.execute(code);
253    }
254
255    @Override
256    protected String evaluate(String expression) throws Exception {
257        return shell.evaluate(expression);
258    }
259
260    @Override
261    protected Message.completeness isComplete(String code) throws Exception {
262        return shell.isComplete(code);
263    }
264
265    @Override
266    protected void interrupt() {
267        var shell = this.shell;
268
269        if (shell != null) {
270            shell.stop();
271        }
272    }
273
274    @Override
275    public void shutdown() { SpringApplication.exit(context, () -> 0); }
276
277    @Override
278    public void run(ApplicationArguments arguments) throws Exception {
279        try {
280            if (connection_file != null) {
281                bind(connection_file);
282            }
283        } catch (Exception exception) {
284            log.warn("{}", exception);
285        }
286    }
