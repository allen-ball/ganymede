021import com.fasterxml.jackson.databind.node.ObjectNode;
022import ganymede.server.Message;
023import ganymede.server.Server;
024import ganymede.shell.Shell;
025import ganymede.util.ObjectMappers;
026import java.io.File;
027import java.io.IOException;
028import java.nio.file.Paths;
029import java.util.UUID;
030import javax.annotation.PostConstruct;
031import javax.annotation.PreDestroy;
032import lombok.NoArgsConstructor;
033import lombok.ToString;
034import lombok.extern.log4j.Log4j2;
035import org.springframework.beans.factory.annotation.Value;
036import org.springframework.boot.ApplicationArguments;
037import org.springframework.boot.ApplicationRunner;
038import org.springframework.boot.SpringApplication;
039import org.springframework.boot.autoconfigure.SpringBootApplication;
040import org.springframework.boot.web.servlet.context.ServletWebServerInitializedEvent;
041import org.springframework.context.ApplicationContext;
042import org.springframework.context.ApplicationContextAware;
043import org.springframework.context.event.ContextClosedEvent;
044import org.springframework.context.event.EventListener;
045import org.springframework.http.HttpStatus;
046import org.springframework.http.ResponseEntity;
047import org.springframework.web.bind.annotation.RequestBody;
048import org.springframework.web.bind.annotation.RequestMapping;
049import org.springframework.web.bind.annotation.RestController;
050
051import static org.springframework.http.MediaType.APPLICATION_JSON_VALUE;
052import static org.springframework.web.bind.annotation.RequestMethod.GET;
053import static org.springframework.web.bind.annotation.RequestMethod.PUT;
054
055/**
056 * Ganymede Jupyter {@link Kernel}.
057 *
058 * {@injected.fields}
059 *
060 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
061 */
062@SpringBootApplication
063@RestController
064@RequestMapping(value = { "/" },
065                consumes = APPLICATION_JSON_VALUE,
066                produces = APPLICATION_JSON_VALUE)
067@NoArgsConstructor @ToString @Log4j2
068public class Kernel extends Server implements ApplicationContextAware,
069                                              ApplicationRunner {
070
071    /**
072     * The name of the {@link System} property containing the
073     * {@link Kernel}'s REST server port.
074     */
075    public static final String PORT_PROPERTY = "kernel.port";
076
077    @Value("${connection-file:#{null}}")
078    private String connection_file = null;
079
080    @Value("${spark-home:#{null}}")
081    private String spark_home = null;
082
083    @Value("${hive-home:#{null}}")
084    private String hive_home = null;
085
086    @Value("${kernel.version}")
087    private String kernel_version = null;
088
089    private final Shell shell = new Shell(this);
090    private ApplicationContext context = null;
091    private int port = -1;
092    private final ObjectNode kernel_info_reply_content;
093
094    {
095        try (var in = getClass().getResourceAsStream("kernel_info_reply.yaml")) {
096            kernel_info_reply_content = (ObjectNode) ObjectMappers.YAML.readTree(in).with("content");
097            kernel_info_reply_content.put("protocol_version", PROTOCOL_VERSION.toString());
098        } catch (Exception exception) {
099            throw new ExceptionInInitializerError(exception);
100        }
101    }
102
103    /**
104     * Method to get the {@link Kernel} REST server port.
105     *
106     * @return  The port.
107     */
108    public int getPort() { return port; }
109
110    @PostConstruct
111    public void init() throws Exception {
112        if (spark_home != null) {
113            var parent = Paths.get(spark_home, "jars").toFile();
114
115            shell.addKnownDependenciesToClasspath(parent);
116        }
117
118        if (hive_home != null) {
119            var parent = Paths.get(hive_home, "lib").toFile();
120
121            shell.addKnownDependenciesToClasspath(parent);
122        }
123
124        restart();
125    }
126
127    @PreDestroy
128    public void destroy() { super.shutdown(); }
129
130    /**
131     * REST method to retrieve the current {@link Shell#classpath()}.
132     */
133    @RequestMapping(method = { GET }, value = { "kernel/classpath" })
134    public ResponseEntity<String> classpath() throws Exception {
135        var json = ObjectMappers.JSON.writeValueAsString(shell.classpath());
136
137        return new ResponseEntity<>(json, HttpStatus.OK);
138    }
139
140    /**
141     * REST method to display MIME bundles from a sub-process.  See
142     * {@link KernelRestClient#display(JsonNode)}.
143     *
144     * @param   bundle          The MIME bundle {@link ObjectNode}.
145     */
146    @RequestMapping(method = { PUT }, value = { "kernel/display" })
147    public ResponseEntity<String> display(@RequestBody ObjectNode bundle) {
148        var request = this.request;
149
150        if (request != null) {
151            var silent = request.content().at("/silent").asBoolean();
152
153            if (! silent) {
154                pub(request.display_data(bundle));
155            }
156        }
157
158        return new ResponseEntity<>(HttpStatus.OK);
159    }
160
161    /**
162     * REST method to print MIME bundles from a sub-process.  See
163     * {@link KernelRestClient#print(JsonNode)}.
164     *
165     * @param   bundle          The MIME bundle {@link ObjectNode}.
166     */
167    @RequestMapping(method = { PUT }, value = { "kernel/print" })
168    public ResponseEntity<String> print(@RequestBody ObjectNode bundle) {
169        var request = this.request;
170
171        if (request != null) {
172            var silent = request.content().at("/silent").asBoolean();
173
174            if (! silent) {
175                pub(request.execute_result(execution_count.intValue(), bundle));
176            }
177        }
178
179        return new ResponseEntity<>(HttpStatus.OK);
180    }
181
182    @Override
183    public void setApplicationContext(ApplicationContext context) {
184        this.context = context;
185    }
186
187    @EventListener({ ServletWebServerInitializedEvent.class })
188    public void onApplicationEvent(ServletWebServerInitializedEvent event) {
189        port = event.getWebServer().getPort();
190    }
191
192    @EventListener({ ContextClosedEvent.class })
193    public void onApplicationEvent(ContextClosedEvent event) {
194        super.shutdown();
195    }
196
197    @Override
198    protected void bind(File file) throws IOException {
199        super.bind(file);
200
201        var node = (ObjectNode) ObjectMappers.JSON.readTree(file);
202
203        node.put("pid", ProcessHandle.current().pid());
204
205        if (port > 0) {
206            node.put("port", port);
207        }
208
209        ObjectMappers.JSON.writeValue(file, node);
210    }
211
212    @Override
213    protected void restart() throws Exception {
214        super.restart();
215
216        shell.restart(getIn(), getOut(), getErr());
217
218        setSessionId(UUID.randomUUID().toString());
219    }
220
221    @Override
222    protected ObjectNode getKernelInfo() { return kernel_info_reply_content; }
223
224    @Override
225    protected void execute(String code) throws Exception {
226        shell.execute(code);
227    }
228
229    @Override
230    protected String evaluate(String expression) throws Exception {
231        return shell.evaluate(expression);
232    }
233
234    @Override
235    protected Message.completeness isComplete(String code) throws Exception {
236        return shell.isComplete(code);
237    }
238
239    @Override
240    protected void interrupt() {
241        var shell = this.shell;
242
243        if (shell != null) {
244            shell.stop();
245        }
246    }
247
248    @Override
249    public void shutdown() { SpringApplication.exit(context, () -> 0); }
250
251    @Override
252    public void run(ApplicationArguments arguments) throws Exception {
253        try {
254            if (connection_file != null) {
255                bind(connection_file);
256            }
257        } catch (Exception exception) {
258            log.warn("{}", exception);
259        }
260    }
261}
