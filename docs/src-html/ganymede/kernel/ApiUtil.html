001package ganymede.kernel;
002
003import org.springframework.web.context.request.NativeWebRequest;
004
005import javax.servlet.http.HttpServletResponse;
006import java.io.IOException;
007
008public class ApiUtil {
009    public static void setExampleResponse(NativeWebRequest req, String contentType, String example) {
010        try {
011            HttpServletResponse res = req.getNativeResponse(HttpServletResponse.class);
012            res.setCharacterEncoding("UTF-8");
013            res.addHeader("Content-Type", contentType);
014            res.getWriter().print(example);
015        } catch (IOException e) {
016            throw new RuntimeException(e);
017        }
018    }
019}
