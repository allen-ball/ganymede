021import com.fasterxml.jackson.databind.JsonNode;
022import ganymede.kernel.client.api.DefaultApi;
023import java.util.List;
024import java.util.Map;
025import java.util.UUID;
026import lombok.ToString;
027import lombok.extern.log4j.Log4j2;
028
029/**
030 * Ganymede {@link ganymede.kernel.Kernel} REST client.
031 *
032 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
033 */
034@ToString @Log4j2
035public class KernelRestClient extends ApiClient {
036
037    /**
038     * The name of the {@link System} property containing the
039     * {@link ganymede.kernel.Kernel}'s REST server port.
040     */
041    public static final String PORT_PROPERTY = "kernel.port";
042
043    {
044        setPort(Integer.decode(System.getProperty(PORT_PROPERTY)));
045        setRequestInterceptor(t -> t.header("Accept", "application/json"));
046    }
047
048    /**
049     * See {@link DefaultApi#kernelId()}.
050     */
051    public UUID kernelId() throws Exception {
052        return new DefaultApi(this).kernelId();
053    }
054
055    /**
056     * See {@link DefaultApi#getExecuteRequest()}.
057     */
058    public JsonNode getExecuteRequest() throws Exception {
059        return new DefaultApi(this).getExecuteRequest();
060    }
061
062    /**
063     * See {@link DefaultApi#display(JsonNode)}.
064     *
065     * @param   bundle          The MIME bundle {@link JsonNode}.
066     */
067    public void display(JsonNode bundle) throws Exception {
068        new DefaultApi(this).display(bundle);
069    }
070
071    /**
072     * See {@link DefaultApi#print(JsonNode)}.
073     *
074     * @param   bundle          The MIME bundle {@link JsonNode}.
075     */
076    public void print(JsonNode bundle) throws Exception {
077        new DefaultApi(this).print(bundle);
078    }
079
080    /**
081     * See {@link DefaultApi#classpath()}.
082     */
083    public List<String> classpath() throws Exception {
084        return new DefaultApi(this).classpath();
085    }
086
087    /**
088     * See {@link DefaultApi#imports()}.
089     */
090    public List<String> imports() throws Exception {
091        return new DefaultApi(this).imports();
092    }
093
094    /**
095     * See {@link DefaultApi#variables()}.
096     */
097    public Map<String,String> variables() throws Exception {
098        return new DefaultApi(this).variables();
099    }
100}
