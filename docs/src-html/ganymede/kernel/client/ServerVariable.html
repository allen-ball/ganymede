001package ganymede.kernel.client;
002
003import java.util.HashSet;
004
005/**
006 * Representing a Server Variable for server URL template substitution.
007 */
008public class ServerVariable {
009  public String description;
010  public String defaultValue;
011  public HashSet<String> enumValues = null;
012
013  /**
014   * @param description A description for the server variable.
015   * @param defaultValue The default value to use for substitution.
016   * @param enumValues An enumeration of string values to be used if the
017   *     substitution options are from a limited set.
018   */
019  public ServerVariable(String description, String defaultValue,
020                        HashSet<String> enumValues) {
021    this.description = description;
022    this.defaultValue = defaultValue;
023    this.enumValues = enumValues;
024  }
025}
