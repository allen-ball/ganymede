021import java.io.Reader;
022import java.util.Scanner;
023import java.util.stream.Stream;
024import javax.script.Bindings;
025import javax.script.ScriptContext;
026import javax.script.ScriptException;
027import javax.script.SimpleBindings;
028import lombok.NoArgsConstructor;
029import lombok.ToString;
030import lombok.extern.log4j.Log4j2;
031import picocli.CommandLine;
032import picocli.CommandLine.ParseResult;
033
034import static javax.script.ScriptContext.ENGINE_SCOPE;
035import static lombok.AccessLevel.PROTECTED;
036
037/**
038 * Abstract {@link javax.script.ScriptEngine} base class.
039 *
040 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
041 */
042@NoArgsConstructor(access = PROTECTED) @ToString @Log4j2
043public abstract class AbstractScriptEngine extends javax.script.AbstractScriptEngine {
044
045    /**
046     * Method to parse {@link ScriptContext} {@link #ARGV ARGV} value to an
047     * annotated command {@link Object}.
048     *
049     * @param   context         The {@link ScriptContext}.
050     * @param   command         The annotated command {@link Object}.
051     *
052     * @return  The {@link ParseResult}.
053     *
054     * @see CommandLine
055     * @see CommandLine.Command
056     * @see CommandLine.Option
057     * @see CommandLine.Parameters
058     */
059    protected <T> ParseResult parse(ScriptContext context, T command) {
060        var bindings = context.getBindings(ENGINE_SCOPE);
061        var argv = (String[]) bindings.get(ARGV);
062
063        if (argv == null) {
064            argv = new String[] { };
065        }
066
067        var name = (String) null;
068
069        if (argv.length > 0) {
070            name = argv[0];
071            argv = Stream.of(argv).skip(1).toArray(String[]::new);
072        }
073
074        var line =
075            new CommandLine(command)
076            .setCommandName(name)
077            .setCaseInsensitiveEnumValuesAllowed(true)
078            .setUnmatchedArgumentsAllowed(false);
079        var result = line.parseArgs(argv);
080
081        return result;
082    }
083
084    @Override
085    public abstract String eval(String script, ScriptContext context) throws ScriptException;
086
087    @Override
088    public String eval(Reader reader, ScriptContext context) throws ScriptException {
089        try (var scanner = new Scanner(reader)) {
090            return eval(scanner.useDelimiter("\\A").next(), context);
091        }
092    }
093
094    @Override
095    public Bindings createBindings() { return new SimpleBindings(); }
096}
