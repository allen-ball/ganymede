021import freemarker.cache.StringTemplateLoader;
022import freemarker.template.Configuration;
023import freemarker.template.Version;
024import java.io.StringWriter;
025import java.util.Objects;
026import javax.script.ScriptContext;
027import javax.script.ScriptException;
028import lombok.Data;
029import lombok.Getter;
030import lombok.ToString;
031import lombok.extern.log4j.Log4j2;
032import picocli.CommandLine.Command;
033import picocli.CommandLine.ParameterException;
034
035import static javax.script.ScriptContext.ENGINE_SCOPE;
036
037/**
038 * FreeMarker {@link javax.script.ScriptEngine}.
039 *
040 * {@bean.info}
041 *
042 * @see Configuration
043 * @see StringTemplateLoader
044 *
045 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
046 */
047@Getter @ToString @Log4j2
048public class FreeMarkerScriptEngine extends AbstractScriptEngine {
049    private final FreeMarkerScriptEngineFactory factory;
050    private final Configuration configuration;
051    private final StringTemplateLoader loader = new StringTemplateLoader();
052
053    /**
054     * Sole constructor.
055     */
056    protected FreeMarkerScriptEngine(FreeMarkerScriptEngineFactory factory) {
057        this.factory = factory;
058
059        var version = new Version(factory.getEngineVersion());
060
061        configuration = new Configuration(version);
062        configuration.setTemplateLoader(loader);
063    }
064
065    @Override
066    public String eval(String script, ScriptContext context) throws ScriptException {
067        var out = new StringWriter();
068
069        try {
070            var arguments = new Arguments();
071            var result = parse(context, arguments);
072            var name = String.class.getName() + "@" + Integer.toString(Objects.hashCode(script));
073
074            loader.putTemplate(name, script);
075
076            var template = configuration.getTemplate(name);
077            var bindings = context.getBindings(ENGINE_SCOPE);
078
079            template.process(bindings, out);
080        } catch (ParameterException exception) {
081            System.err.println(exception.getMessage());
082            System.err.println();
083            exception.getCommandLine().usage(System.err);
084        } catch (Exception exception) {
085            exception.printStackTrace(System.err);
086        }
087
088        return out.toString();
089    }
090
091    @Command @Data
092    private class Arguments { }
093}
