021import freemarker.cache.StringTemplateLoader;
022import freemarker.template.Configuration;
023import freemarker.template.Version;
024import java.io.StringWriter;
025import java.util.Objects;
026import javax.script.ScriptContext;
027import javax.script.ScriptException;
028import lombok.Getter;
029import lombok.ToString;
030import lombok.extern.log4j.Log4j2;
031
032import static javax.script.ScriptContext.ENGINE_SCOPE;
033
034/**
035 * FreeMarker {@link javax.script.ScriptEngine}.
036 *
037 * {@bean.info}
038 *
039 * @see Configuration
040 * @see StringTemplateLoader
041 *
042 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
043 */
044@Getter @ToString @Log4j2
045public class FreeMarkerScriptEngine extends AbstractTemplateScriptEngine {
046    private final FreeMarkerScriptEngineFactory factory;
047    private final Configuration configuration;
048    private final StringTemplateLoader loader = new StringTemplateLoader();
049
050    /**
051     * Sole constructor.
052     */
053    protected FreeMarkerScriptEngine(FreeMarkerScriptEngineFactory factory) {
054        this.factory = factory;
055
056        var version = new Version(factory.getEngineVersion());
057
058        configuration = new Configuration(version);
059        configuration.setTemplateLoader(loader);
060    }
061
062    @Override
063    public String eval(String script, ScriptContext context) throws ScriptException {
064        var bindings = context.getBindings(ENGINE_SCOPE);
065        var out = new StringWriter();
066
067        try {
068            var name = String.class.getName() + "@" + Integer.toString(Objects.hashCode(script));
069
070            loader.putTemplate(name, script);
071
072            var template = configuration.getTemplate(name);
073
074            template.process(bindings, out);
075        } catch (Exception exception) {
076            exception.printStackTrace(System.err);
077        }
078
079        return out.toString();
080    }
081}
