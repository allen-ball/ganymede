021import java.util.Map;
022import java.util.TreeMap;
023import javax.script.ScriptEngine;
024import lombok.NoArgsConstructor;
025import lombok.ToString;
026import lombok.extern.log4j.Log4j2;
027
028import static lombok.AccessLevel.PROTECTED;
029
030/**
031 * Abstract {@link javax.script.ScriptEngineFactory} base class.
032 *
033 * {@bean.info}
034 *
035 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
036 */
037@NoArgsConstructor(access = PROTECTED)  @ToString @Log4j2
038public abstract class AbstractScriptEngineFactory implements AnnotatedScriptEngineFactory {
039    private Map<String,String> parameters = null;
040
041    @Override
042    public String getEngineName() {
043        return getParameters().get(ScriptEngine.NAME);
044    }
045
046    @Override
047    public String getEngineVersion() {
048        return getParameters().get(ScriptEngine.ENGINE_VERSION);
049    }
050
051    @Override
052    public String getLanguageName() {
053        return getParameters().get(ScriptEngine.LANGUAGE);
054    }
055
056    @Override
057    public String getLanguageVersion() {
058        return getParameters().get(ScriptEngine.LANGUAGE_VERSION);
059    }
060
061    @Override
062    public Map<String,String> getParameters() {
063        if (parameters == null) {
064            parameters = new TreeMap<>();
065            parameters.putAll(AnnotatedScriptEngineFactory.super.getParameters());
066        }
067
068        return parameters;
069    }
070
071    @Override
072    public String getParameter(String name) {
073        return getParameters().get(name);
074    }
075
076    @Override
077    public String getMethodCallSyntax(String object, String method, String... argv) {
078        throw new UnsupportedOperationException();
079    }
080
081    @Override
082    public String getOutputStatement(String toDisplay) {
083        throw new UnsupportedOperationException();
084    }
085
086    @Override
087    public String getProgram(String... statements) {
088        throw new UnsupportedOperationException();
089    }
090}
