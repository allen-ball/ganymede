021import java.util.List;
022import java.util.Map;
023import java.util.Objects;
024import java.util.stream.Stream;
025import javax.script.ScriptEngineFactory;
026
027import static java.util.stream.Collectors.toMap;
028
029/**
030 * {@link ScriptEngineFactory} annotation services.
031 *
032 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
033 */
034public interface AnnotatedScriptEngineFactory extends ScriptEngineFactory {
035    @Override
036    default List<String> getNames() {
037        var annotation = getClass().getAnnotation(Names.class);
038        var value = (annotation != null) ? annotation.value() : null;
039
040        return (value != null) ? List.of(value) : List.of();
041    }
042
043    @Override
044    default List<String> getExtensions() {
045        var annotation = getClass().getAnnotation(Extensions.class);
046        var value = (annotation != null) ? annotation.value() : null;
047
048        return (value != null) ? List.of(value) : List.of();
049    }
050
051    @Override
052    default List<String> getMimeTypes() {
053        var annotation = getClass().getAnnotation(MimeTypes.class);
054        var value = (annotation != null) ? annotation.value() : null;
055
056        return (value != null) ? List.of(value) : List.of();
057    }
058
059    default Map<String,String> getParameters() {
060        var annotation = getClass().getAnnotation(Parameters.class);
061        var map =
062            Stream.of(annotation)
063            .filter(Objects::nonNull)
064            .flatMap(t -> Stream.of(t.value()))
065            .collect(toMap(k -> k.name(), v -> v.value()));
066
067        return map;
068    }
069}
