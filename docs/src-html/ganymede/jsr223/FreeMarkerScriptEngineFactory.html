021import ball.annotation.ServiceProviderFor;
022import freemarker.template.Configuration;
023import java.util.Map;
024import javax.script.ScriptEngine;
025import javax.script.ScriptEngineFactory;
026import lombok.NoArgsConstructor;
027import lombok.ToString;
028import lombok.extern.log4j.Log4j2;
029
030/**
031 * FreeMarker {@link ScriptEngineFactory}.
032 *
033 * {@bean.info}
034 *
035 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
036 */
037@ServiceProviderFor({ ScriptEngineFactory.class })
038@Names({ "freemarker" })
039@Parameter(name = ScriptEngine.NAME, value = "freemarker")
040@Parameter(name = ScriptEngine.ENGINE, value = "freemarker")
041@Parameter(name = ScriptEngine.LANGUAGE, value = "freemarker")
042@Extensions({ "ftl" })
043@NoArgsConstructor @ToString @Log4j2
044public class FreeMarkerScriptEngineFactory extends AbstractScriptEngineFactory {
045    @Override
046    public Map<String,String> getParameters() {
047        if (parameters == null) {
048            parameters = super.getParameters();
049
050            var version = Configuration.VERSION_2_3_31.toString();
051
052            parameters.put(ScriptEngine.ENGINE_VERSION, version);
053            parameters.put(ScriptEngine.LANGUAGE_VERSION, version);
054        }
055
056        return parameters;
057    }
058
059    @Override
060    public FreeMarkerScriptEngine getScriptEngine() {
061        FreeMarkerScriptEngine engine = null;
062
063        try {
064            engine = new FreeMarkerScriptEngine(this);
065        } catch (Throwable throwable) {
066            log.warn("{}", throwable);
067        }
068
069        return engine;
070    }
071}
