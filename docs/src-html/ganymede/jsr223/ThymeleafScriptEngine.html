021import java.io.Reader;
022import java.util.Scanner;
023import javax.script.AbstractScriptEngine;
024import javax.script.Bindings;
025import javax.script.ScriptContext;
026import javax.script.ScriptEngine;
027import javax.script.ScriptException;
028import javax.script.SimpleBindings;
029import lombok.Getter;
030import lombok.RequiredArgsConstructor;
031import lombok.ToString;
032import lombok.extern.log4j.Log4j2;
033import org.thymeleaf.TemplateEngine;
034import org.thymeleaf.context.Context;
035import org.thymeleaf.templatemode.TemplateMode;
036import org.thymeleaf.templateresolver.StringTemplateResolver;
037
038import static javax.script.ScriptContext.ENGINE_SCOPE;
039import static lombok.AccessLevel.PROTECTED;
040
041/**
042 * Thymeleaf {@link ScriptEngine}.
043 *
044 * {@bean.info}
045 *
046 * @see TemplateEngine
047 * @see StringTemplateResolver
048 *
049 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
050 */
051@RequiredArgsConstructor(access = PROTECTED) @Getter @ToString @Log4j2
052public class ThymeleafScriptEngine extends AbstractScriptEngine {
053    private final ThymeleafScriptEngineFactory factory;
054    private final StringTemplateResolver resolver = new StringTemplateResolver();
055
056    @Override
057    public String eval(String script, ScriptContext context) throws ScriptException {
058        var bindings = context.getBindings(ENGINE_SCOPE);
059        var mode = StringTemplateResolver.DEFAULT_TEMPLATE_MODE;
060        var argv = (String[]) bindings.get(ScriptEngine.ARGV);
061
062        if (argv != null && argv.length > 1) {
063            mode = TemplateMode.valueOf(argv[1].toUpperCase());
064        }
065
066        resolver.setTemplateMode(mode);
067
068        var engine = new TemplateEngine();
069
070        engine.setTemplateResolver(resolver);
071
072        return engine.process(script, new Context(null, bindings));
073    }
074
075    @Override
076    public String eval(Reader reader, ScriptContext context) throws ScriptException {
077        try (var scanner = new Scanner(reader)) {
078            return eval(scanner.useDelimiter("\\A").next(), context);
079        }
080    }
081
082    @Override
083    public Bindings createBindings() { return new SimpleBindings(); }
084}
