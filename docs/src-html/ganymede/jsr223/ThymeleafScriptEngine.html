021import java.util.stream.Stream;
022import javax.script.ScriptContext;
023import javax.script.ScriptException;
024import lombok.Data;
025import lombok.Getter;
026import lombok.RequiredArgsConstructor;
027import lombok.ToString;
028import lombok.extern.log4j.Log4j2;
029import org.thymeleaf.TemplateEngine;
030import org.thymeleaf.context.Context;
031import org.thymeleaf.extras.java8time.dialect.Java8TimeDialect;
032import org.thymeleaf.templatemode.TemplateMode;
033import org.thymeleaf.templateresolver.StringTemplateResolver;
034import picocli.CommandLine.Command;
035import picocli.CommandLine.ParameterException;
036import picocli.CommandLine.Parameters;
037
038import static javax.script.ScriptContext.ENGINE_SCOPE;
039import static lombok.AccessLevel.PROTECTED;
040
041/**
042 * Thymeleaf {@link javax.script.ScriptEngine}.
043 *
044 * {@bean.info}
045 *
046 * @see TemplateEngine
047 * @see StringTemplateResolver
048 * @see Java8TimeDialect
049 *
050 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
051 */
052@RequiredArgsConstructor(access = PROTECTED) @Getter @ToString @Log4j2
053public class ThymeleafScriptEngine extends AbstractScriptEngine {
054    private final ThymeleafScriptEngineFactory factory;
055    private final StringTemplateResolver resolver = new StringTemplateResolver();
056
057    @Override
058    public String eval(String script, ScriptContext context) throws ScriptException {
059        var out = "";
060
061        try {
062            var arguments = new Arguments();
063            var result = parse(context, arguments);
064
065            resolver.setTemplateMode(arguments.getMode());
066
067            var engine = new TemplateEngine();
068
069            engine.setTemplateResolver(resolver);
070            engine.addDialect(new Java8TimeDialect());
071
072            var bindings = context.getBindings(ENGINE_SCOPE);
073
074            out = engine.process(script, new Context(null, bindings));
075        } catch (ParameterException exception) {
076            System.err.println(exception.getMessage());
077            System.err.println();
078            exception.getCommandLine().usage(System.err);
079        } catch (Exception exception) {
080            exception.printStackTrace(System.err);
081        }
082
083        return out;
084    }
085
086    @Command @Data
087    private class Arguments {
088        @Parameters(index = "0", arity = "0..1", defaultValue = "HTML",
089                    description = { "One of: ${COMPLETION-CANDIDATES} (default: ${DEFAULT-VALUE})" })
090        private TemplateMode mode = StringTemplateResolver.DEFAULT_TEMPLATE_MODE;
091    }
092}
