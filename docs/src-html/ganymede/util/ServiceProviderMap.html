021import java.util.Comparator;
022import java.util.Objects;
023import java.util.ServiceConfigurationError;
024import java.util.ServiceLoader;
025import java.util.TreeMap;
026import java.util.function.Function;
027
028/**
029 * {@link ServiceLoader Service} {@link java.util.Map}.
030 *
031 * @param       <T>             The service type.
032 *
033 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
034 */
035public class ServiceProviderMap<T> extends TreeMap<Class<? extends T>,T> {
036    private static final long serialVersionUID = 5423712518719767597L;
037
038    private static final Comparator<Class<?>> COMPARATOR = Comparator.comparing(Class::getName);
039
040    /** @serial */ private final Class<T> service;
041    /** @serial */ private final ServiceLoader<T> loader;
042    /** @serial */ private Function<ServiceLoader.Provider<T>,T> compute;
043
044    /**
045     * Sole constructor.
046     *
047     * @param   service         The service {@link Class type}.
048     * @param   compute         The {@link Function} to obtain a value
049     *                          instance from a
050     *                          {@link ServiceLoader.Provider}.
051     */
052    public ServiceProviderMap(Class<T> service, Function<ServiceLoader.Provider<T>,T> compute) {
053        super(COMPARATOR);
054
055        this.service = service;
056        this.loader = ServiceLoader.load(service, service.getClassLoader());
057        this.compute = Objects.requireNonNullElse(compute, this::compute);
058    }
059
060    private T compute(ServiceLoader.Provider<T> provider) {
061        return provider.get();
062    }
063
064    /**
065     * Method to get the {@link ServiceLoader}'s {@link ClassLoader}.
066     *
067     * @return  The {@link ServiceLoader}'s {@link ClassLoader}.
068     */
069    public ClassLoader getClassLoader() { return service.getClassLoader(); }
070
071    /**
072     * Reload {@link ServiceLoader} and {@link #put(Object,Object)} newly
073     * discovered entries.
074     *
075     * @return  {@link.this}
076     */
077    public ServiceProviderMap<T> reload() {
078        loader.reload();
079
080        var iterator = loader.stream().iterator();
081
082        while (iterator.hasNext()) {
083            var provider = iterator.next();
084
085            try {
086                computeIfAbsent(provider.type(), k -> compute.apply(provider));
087            } catch (ServiceConfigurationError error) {
088            }
089        }
090
091        return this;
092    }
093}
