021import java.util.Properties;
022import java.util.regex.Pattern;
023import lombok.NoArgsConstructor;
024import lombok.ToString;
025import org.springframework.core.io.ClassPathResource;
026
027import static org.apache.commons.lang3.StringUtils.isNotBlank;
028
029/**
030 * {@link Javadoc} utilities.
031 *
032 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
033 */
034@NoArgsConstructor @ToString
035public class Javadoc {
036    @ToString.Exclude
037    private final Properties properties = new Properties();
038
039    {
040        try (var in = new ClassPathResource("javadoc-map.properties").getInputStream()) {
041            properties.load(in);
042        } catch (Exception exception) {
043            throw new ExceptionInInitializerError(exception);
044        }
045    }
046
047    /**
048     * Get a Javadoc link href from a canonical class name.
049     *
050     * @param   name            The canonical class name.
051     *
052     * @return  The URL (as a {@link String}) or {@code null} if the package
053     *          is not known.
054     */
055    public String href(String name) {
056        String url = null;
057
058        if (name != null) {
059            var offset = name.lastIndexOf(".");
060
061            if (offset > 0) {
062                url = href(name.substring(0, offset), name.substring(offset + 1));
063            }
064        }
065
066        return url;
067    }
068
069    /**
070     * Get a Javadoc link href from a possibly simple class name and a value
071     * ({@link Object}).
072     *
073     * @param   name            The (possibly simple) class name.
074     * @param   value           The {@link Object}.
075     *
076     * @return  The URL (as a {@link String}) or {@code null} if the package
077     *          cannot be determined.
078     */
079    public String href(String name, Object value) {
080        return href(name, (value != null) ? value.getClass() : null);
081    }
082
083    /**
084     * Get a Javadoc link href from a possibly simple class name and a type
085     * ({@link Class}).
086     *
087     * @param   name            The (possibly simple) class name.
088     * @param   type            The {@link Class}.
089     *
090     * @return  The URL (as a {@link String}) or {@code null} if the package
091     *          cannot be determined.
092     */
093    public String href(String name, Class<?> type) {
094        String url = null;
095
096        if (name != null) {
097            if (name.lastIndexOf(".") != -1) {
098                url = href(name);
099            } else {
100                if (type != null) {
101                    if (name.equals(type.getSimpleName())) {
102                        url = href(type.getPackage().getName(), name);
103                    } else {
104                        url = href(name, type.getSuperclass());
105
106                        if (url == null) {
107                            for (Class<?> supertype : type.getInterfaces()) {
108                                url = href(name, supertype);
109
110                                if (url != null) {
111                                    break;
112                                }
113                            }
114                        }
115                    }
116                }
117            }
118        }
119
120        return url;
121    }
122
123    private String href(String pkg, String name) {
124        String url = null;
125
126        if (isNotBlank(pkg) && isNotBlank(name)) {
127            url = properties.getProperty(pkg);
128
129            if (url != null) {
130                var module = properties.getProperty(pkg + "-module");
131
132                if (module != null) {
133                    url += module + "/";
134                }
135
136                url += String.join("/", pkg.split(Pattern.quote(".")));
137                url += "/" + name + ".html" + "?is-external=true";
138            }
139        }
140
141        return url;
142    }
143}
