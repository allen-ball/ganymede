043/**
044 * {@link Javadoc} utilities.
045 *
046 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
047 */
048@NoArgsConstructor @ToString
049public class Javadoc implements HTMLTemplates, XalanConstants {
050    @ToString.Exclude private final Properties properties = new Properties();
051    @ToString.Exclude private final Transformer transformer;
052    @ToString.Exclude private final FluentDocument document;
053

056            properties.load(in);
057
058            transformer = TransformerFactory.newInstance().newTransformer();
059            transformer.setOutputProperty(OMIT_XML_DECLARATION, YES);
060            transformer.setOutputProperty(INDENT, NO);
061
062            document =
063                FluentDocumentBuilderFactory.newInstance()
064                .newDocumentBuilder()
065                .newDocument();

068                             element("head",
069                                     element("meta",
070                                             attr("charset", "utf-8"))),
071                             element("body")));
072        } catch (Exception exception) {
073            throw new ExceptionInInitializerError(exception);
074        }
075    }
076
077    /**
078     * Create am {@code <a/>} element from a possibly simple class name and
079     * an implementation type ({@link Class}).
080     *
081     * @param   name            The (possibly simple) class name.
082     * @param   type            The {@link Class}.
083     *
084     * @return  {@code <a/>} {@link Element} (serialized to {@link String})
085     */
086    public String a(String name, Class<?> type) {
087        var writer = new StringWriter();
088        var href = href(name, type);
089
090        if (name != null || href != null) {
091            var node = a(href, name);
092
093            ((Element) node).setAttribute("target", "_newtab");
094
095            try {
096                transformer.transform(new DOMSource(node), new StreamResult(writer));
097            } catch (RuntimeException exception) {
098                throw exception;
099            } catch (Error error) {
100                throw error;
101            } catch (Exception exception) {
102                throw new RuntimeException(exception);
103            }
104        }
105
106        return writer.toString();
107    }
108
109    /**
110     * Get a Javadoc link href from a canonical class name.
111     *
112     * @param   name            The canonical class name.
113     *
114     * @return  The URL (as a {@link String}) or {@code null} if the package
115     *          is not known.
116     */
117    public URI href(String name) {
118        URI uri = null;
119
120        if (name != null) {
121            var offset = name.lastIndexOf(".");
122
123            if (offset > 0) {
124                uri = href(name.substring(0, offset), name.substring(offset + 1));
125            }
126        }
127
128        return uri;
129    }
130
131    /**
132     * Get a Javadoc link href from a possibly simple class name and a value
133     * ({@link Object}).
134     *
135     * @param   name            The (possibly simple) class name.
136     * @param   value           The {@link Object}.
137     *
138     * @return  The URL (as a {@link URI}) or {@code null} if the package
139     *          cannot be determined.
140     */
141    public URI href(String name, Object value) {
142        return href(name, (value != null) ? value.getClass() : null);
143    }
144
145    /**
146     * Get a Javadoc link href from a possibly simple class name and an
147     * implementation type ({@link Class}).
148     *
149     * @param   name            The (possibly simple) class name.
150     * @param   type            The {@link Class}.
151     *
152     * @return  The URL (as a {@link URI}) or {@code null} if the package
153     *          cannot be determined.
154     */
155    public URI href(String name, Class<?> type) {
156        URI uri = null;
157
158        if (name != null) {
159            if (name.lastIndexOf(".") != -1) {
160                uri = href(name);
161            } else {
162                if (type != null) {
163                    if (name.equals(type.getSimpleName())) {
164                        uri = href(type.getPackage().getName(), name);
165                    } else {
166                        uri = href(name, type.getSuperclass());
167
168                        if (uri == null) {
169                            for (Class<?> supertype : type.getInterfaces()) {
170                                uri = href(name, supertype);
171
172                                if (uri != null) {
173                                    break;
174                                }
175                            }
176                        }
177                    }
178                }
179            }
180        }
181
182        return uri;
183    }
184
185    private URI href(String pkg, String name) {
186        String string = null;
187
188        if (isNotBlank(pkg) && isNotBlank(name)) {
189            string = properties.getProperty(pkg);
190
191            if (string != null) {
192                var module = properties.getProperty(pkg + "-module");
193
194                if (module != null) {
195                    string += module + "/";
196                }
197
198                string += String.join("/", pkg.split(Pattern.quote(".")));
199                string += "/" + name + ".html" + "?is-external=true";
200            }
201        }
202
203        return (string != null) ? URI.create(string) : null;
204    }
205
206    @Override
207    public FluentDocument document() { return document; }
208}
