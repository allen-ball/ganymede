047/**

050 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}

053public class Javadoc implements HTMLTemplates, XalanConstants {
054    private static final Map<String,Class<?>> PRIMITIVES =
055        Stream.of(Boolean.TYPE, Byte.TYPE, Character.TYPE, Double.TYPE,
056                  Float.TYPE, Integer.TYPE, Long.TYPE, Short.TYPE, Void.TYPE)
057        .collect(toMap(k -> k.getName(), v -> v));
058
059    @ToString.Exclude private final Properties properties = new Properties();
060    @ToString.Exclude private final Transformer transformer;
061    @ToString.Exclude private final FluentDocument document;
062
063    {
064        try (var in = new ClassPathResource("javadoc-map.properties").getInputStream()) {
065            properties.load(in);
066
067            transformer = TransformerFactory.newInstance().newTransformer();
068            transformer.setOutputProperty(OMIT_XML_DECLARATION, YES);
069            transformer.setOutputProperty(INDENT, NO);
070
071            document =
072                FluentDocumentBuilderFactory.newInstance()
073                .newDocumentBuilder()
074                .newDocument();
075            document
076                .add(element("html",
077                             element("head",
078                                     element("meta",
079                                             attr("charset", "utf-8"))),
080                             element("body")));
081        } catch (Exception exception) {
082            throw new ExceptionInInitializerError(exception);
083        }
084    }

086    /**
087     * Create an {@code <a/>} element from a possibly simple class name and
088     * an implementation type ({@link Class}).
089     *
090     * @param   name            The (possibly simple) class name.
091     * @param   type            The {@link Class}.
092     *
093     * @return  {@code <a/>} {@link Element} (serialized to {@link String})
094     */
095    public String link(String name, Class<?> type) {
096        var writer = new StringWriter();
097        var declaredType = type;
098
099        if (name == null) {
100            if (type != null) {
101                var dimensions = "";
102
103                while (type.isArray()) {
104                    dimensions += "[]";
105                    type = type.getComponentType();
106                }
107
108                name = getCanonicalName(type);
109                name += dimensions;
110            }
111        }
112
113        if (name != null) {
114            var substrings = name.split("\\[|<", 2);
115            var suffix = name.substring(substrings[0].length());
116
117            name = substrings[0];
118
119            declaredType = typeOf(name, List.of("java.lang"));
120
121            if (declaredType == null) {
122                if (type != null) {
123                    while (type.isArray()) {
124                        type = type.getComponentType();
125                    }
126
127                    declaredType = declaredTypeOf(name, type);
128                }
129            }
130
131            if (declaredType != null) {
132                name = getCanonicalName(declaredType);
133            }
134
135            name += suffix;
136        }
137
138        if (name != null) {
139            var node = code(name);
140            var href = href(declaredType);
141
142            if (href != null) {
143                node = a(href, node);
144                ((Element) node).setAttribute("target", "_newtab");
145            }
146
147            try {
148                transformer.transform(new DOMSource(node), new StreamResult(writer));
149            } catch (RuntimeException exception) {
150                throw exception;
151            } catch (Error error) {
152                throw error;
153            } catch (Throwable throwable) {
154                throw new RuntimeException(throwable);
155            }
156        }
157
158        return writer.toString();
159    }

161    private Class<?> typeOf(String name, List<String> packages) {
162        var type = PRIMITIVES.get(name);
163
164        if (type == null) {
165            try {
166                type = Class.forName(name);
167            } catch (Exception exception) {
168            }
169        }
170
171        if (type == null) {
172            for (var pkg : packages) {
173                try {
174                    type = Class.forName(pkg + "." + type);
175                    break;
176                } catch (Exception exception) {
177                    continue;
178                }
179            }
180        }
181
182        return type;
183    }
184
185    private Class<?> declaredTypeOf(String name, Class<?> type) {
186        Class<?> declaredType = null;
187
188        if (type != null) {
189            if (type.isPrimitive()
190                || type.getName().equals(name)
191                || type.getName().endsWith("." + name)
192                || getCanonicalName(type).equals(name)
193                || getCanonicalName(type).endsWith("." + name)) {
194                declaredType = type;
195            } else {
196                declaredType = declaredTypeOf(name, type.getSuperclass());
197
198                if (declaredType == null) {
199                    for (Class<?> supertype : type.getInterfaces()) {
200                        declaredType = declaredTypeOf(name, supertype);
201
202                        if (declaredType != null) {
203                            break;
204                        }
205                    }
206                }
207            }
208        }
209
210        return declaredType;
211    }
212
213    private URI href(Class<?> type) {
214        String uri = null;
215
216        if (type != null && (! type.isPrimitive())) {
217            var pkg = type.getPackageName();
218
219            uri = properties.getProperty(pkg);
220
221            if (uri != null) {
222                var module = properties.getProperty(pkg + "-module");
223
224                if (module != null) {
225                    uri += module + "/";
226                }
227
228                uri += String.join("/", pkg.split(Pattern.quote(".")));
229
230                var name = type.getSimpleName();
231
232                while (type.getEnclosingClass() != null) {
233                    type = type.getEnclosingClass();
234                    name = type.getSimpleName() + "." + name;
235                }
236
237                uri += "/" + name + ".html";
238                uri += "?is-external=true";
239            }
240        }
241
242        return (uri != null) ? URI.create(uri) : null;
243    }
244
245    private String getCanonicalName(Class<?> type) {
246        var name = type.getName();
247
248        return Objects.requireNonNullElse(type.getCanonicalName(), name);
249    }
250
251    private String getPackageQualifiedName(Class<?> type) {
252        var name = getCanonicalName(type);
253
254        if (! type.isPrimitive()) {
255            var prefix = type.getPackageName() + ".";
256
257            if (name.startsWith(prefix)) {
258                name = name.substring(prefix.length());
259            }
260        }
261
262        return name;
263    }
264
265    @Override
266    public FluentDocument document() { return document; }
