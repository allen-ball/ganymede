021import java.io.File;
022import java.util.Comparator;
023import java.util.TreeMap;
024import java.util.stream.Stream;
025
026/**
027 * Path to Property {@link java.util.Map}.
028 *
029 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
030 */
031public class PathPropertyMap extends TreeMap<String,String> {
032    private static final long serialVersionUID = 3186771870028981288L;
033
034    private static final String SEPARATOR = System.getProperty("path.separator");
035
036    /**
037     * Sole constructor.
038     */
039    public PathPropertyMap() {
040        super(Comparator
041              .comparingInt(String::length).reversed()
042              .thenComparing(String::toString, Comparator.naturalOrder()));
043
044        Stream.of(System.getenv(), System.getProperties())
045            .flatMap(t -> t.entrySet().stream())
046            .filter(t -> (! t.getKey().toString().endsWith(".path")))
047            .filter(t -> t.getValue() != null)
048            .filter(t -> t.getValue().toString().length() > 1)
049            .filter(t -> (! t.getValue().toString().contains(SEPARATOR)))
050            .filter(t -> new File(t.getValue().toString()).isAbsolute())
051            .forEach(t -> put(t.getValue().toString(), "${" + t.getKey().toString() + "}"));
052    }
053
054    /**
055     * Method to shorten a {@link File} path by substituting placeholder
056     * references from the {@link PathPropertyMap}.
057     *
058     * @param   file            The {@link File}.
059     *
060     * @return  The possibly shortened {@link File} path.
061     */
062    public String shorten(File file) {
063        return (file != null) ? shorten(file.getAbsolutePath()) : null;
064    }
065
066    /**
067     * Method to shorten a {@link String} by substituting placeholder
068     * references for paths from the {@link PathPropertyMap}.
069     *
070     * @param   string          The candidate {@link String} to be
071     *                          shortened.
072     *
073     * @return  The possibly shortened {@link String} with substitutions.
074     */
075    public String shorten(String string) {
076        var shortened = string;
077
078        for (var key : keySet()) {
079            if (shortened.startsWith(key)) {
080                shortened = shortened.replace(key, get(key));
081            }
082        }
083
084        return shortened;
085    }
086}
