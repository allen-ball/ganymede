061/**
062 * Target class for YAML representation of POM elements.
063 *
064 * {@bean.info}
065 *
066 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
067 */
068@JsonPropertyOrder({ "localRepository", "interactiveMode", "offline", "repositories", "dependencies" })
069@Data @NoArgsConstructor @ToString @Log4j2
070public class POM {
071    private static final SimpleModule MODULE =
072        new SimpleModule()
073        .addDeserializer(Dependency.class, new DependencyDeserializer())
074        .addSerializer(Dependency.class, new DependencySerializer())
075        .addDeserializer(RemoteRepository.class, new RemoteRepositoryDeserializer())
076        .addSerializer(RemoteRepository.class, new RemoteRepositorySerializer())
077        .addDeserializer(RepositoryPolicy.class, new RepositoryPolicyDeserializer());
078    private static final ObjectMapper YAML_OBJECT_MAPPER =
079        ObjectMappers.YAML
080        .registerModule(MODULE)
081        .setSerializationInclusion(JsonInclude.Include.NON_EMPTY);
082
083    /**
084     * Static method to get default {@link POM}.
085     *
086     * @return  The parsed {@link POM}.
087     */
088    public static POM getDefault() throws IOException {
089        POM pom = null;
090        var resource = POM.class.getSimpleName() + ".yaml";
091
092        try (var in = POM.class.getResourceAsStream(resource)) {
093            pom = YAML_OBJECT_MAPPER.readValue(in, POM.class);
094        }
095
096        return pom;
097    }
098
099    /**
100     * Static method to parse a YAML {@link String} to a {@link POM}.
101     *
102     * @param   yaml            The YAML {@link String}.
103     *
104     * @return  The parsed {@link POM}.
105     */
106    public static POM parse(String yaml) throws Exception {
107        return YAML_OBJECT_MAPPER.readValue(yaml, POM.class);
108    }

110    private String localRepository = null;
111    private Boolean interactiveMode = null;
112    private Boolean offline = null;
113    private Set<RemoteRepository> repositories = new LinkedHashSet<>();
114    private Set<Dependency> dependencies = new LinkedHashSet<>();
115
116    /**
117     * Method to merge a {@link POM} into {@link.this} {@link POM}.
118     *
119     * @param   that            The {@link POM} to merge.
120     */
121    public void merge(POM that) {
122        update(this, that, POM::getLocalRepository, this::setLocalRepository);
123        update(this, that, POM::getInteractiveMode, this::setInteractiveMode);
124        update(this, that, POM::getOffline, this::setOffline);
125
126        var ids =
127            that.getRepositories().stream()
128            .map(t -> t.getId())
129            .filter(Objects::nonNull)
130            .collect(toSet());
131        var urls =
132            that.getRepositories().stream()
133            .map(t -> t.getUrl())
134            .filter(Objects::nonNull)
135            .collect(toSet());
136
137        this.getRepositories().removeIf(t -> ids.contains(t.getId()) || urls.contains(t.getUrl()));
138        this.getRepositories().addAll(that.getRepositories());
139
140        var keys =
141            that.getDependencies().stream()
142            .map(Dependency::getArtifact)
143            .map(t -> toVersionlessId(t))
144            .collect(toSet());
145
146        this.getDependencies().removeIf(t -> keys.contains(toVersionlessId(t.getArtifact())));
147        this.getDependencies().addAll(that.getDependencies());
148    }
149
150    private <T,U> boolean update(T left, T right, Function<T,U> get, Consumer<U> set) {
151        var value = get.apply(right);
152        var modified = (value != null && (! Objects.equals(get.apply(left), value)));
153
154        if (modified) {
155            set.accept(value);
156        }
157
158        return modified;
159    }

161    /**
162     * Method to write {@link.this} {@link POM}'s YAML representation.
163     *
164     * @param   out             The target {@link OutputStream}.
165     *
166     * @throws  IOException     If the {@link POM} cannot be written for any
167     *                          reason.
168     */
169    public void writeTo(OutputStream out) throws IOException {
170        YAML_OBJECT_MAPPER.writeValue(out, this);
171    }

173    private static String asText(JsonNode node, String name) {
174        return asText(node, name, null);
175    }
176
177    private static String asText(JsonNode node, String name, String defaultValue) {
178        return node.has(name) ? node.get(name).asText() : defaultValue;
179    }

181    @ToString
182    private static class DependencyDeserializer extends StdDeserializer<Dependency> {
183        private static final long serialVersionUID = -1L;
184
185        public DependencyDeserializer() { super(Dependency.class); }
186
187        @Override
188        public Dependency deserialize(JsonParser parser, DeserializationContext context) throws IOException, JsonProcessingException {
189            JsonNode node = parser.getCodec().readTree(parser);
190            Artifact artifact = null;
191
192            if (node instanceof ObjectNode) {
193                artifact =
194                    new DefaultArtifact(asText(node, "groupId"),
195                                        asText(node, "artifactId"),
196                                        asText(node, "classifier"),
197                                        asText(node, "type", "jar"),
198                                        asText(node, "version"),
199                                        Map.of(), (File) null);
200            } else {
201                var strings = node.asText().split(":");
202                /*
203                 * groupId:artifactId[:type[:classifier]]:version
204                 */
205                artifact =
206                    new DefaultArtifact(strings[0], strings[1],
207                                        (strings.length > 4) ? strings[3] : null,
208                                        (strings.length > 3) ? strings[2] : "jar",
209                                        (strings.length > 2) ? strings[strings.length - 1] : null,
210                                        Map.of(), (File) null);
211            }
212
213            return new Dependency(artifact, asText(node, "scope", JavaScopes.RUNTIME));
214        }
215    }

217    @ToString
218    private static class DependencySerializer extends StdSerializer<Dependency> {
219        private static final long serialVersionUID = -1L;
220
221        public DependencySerializer() { super(Dependency.class); }
222
223        @Override
224        public void serialize(Dependency value, JsonGenerator generator, SerializerProvider provider) throws IOException, JsonProcessingException {
225            var artifact = value.getArtifact();
226            var string =
227                Stream.of(artifact.getGroupId(), artifact.getArtifactId(),
228                          artifact.getExtension(), artifact.getClassifier(),
229                          artifact.getVersion())
230                .filter(Objects::nonNull)
231                .filter(t -> (! t.isBlank()))
232                .collect(joining(":"));
233
234            generator.writeString(string);
235        }
236    }

238    @ToString
239    private static class RemoteRepositoryDeserializer extends StdDeserializer<RemoteRepository> {
240        private static final long serialVersionUID = -1L;
241
242        public RemoteRepositoryDeserializer() { super(RemoteRepository.class); }
243
244        @Override
245        public RemoteRepository deserialize(JsonParser parser, DeserializationContext context) throws IOException, JsonProcessingException {
246            JsonNode node = parser.getCodec().readTree(parser);
247            var builder =
248                new RemoteRepository.Builder(asText(node, "id"),
249                                             asText(node, "layout", "default"),
250                                             asText(node, "url"));
251
252            if (node.has("releases")) {
253                var policy = parser.getCodec().treeToValue(node.get("releases"), RepositoryPolicy.class);
254
255                builder.setReleasePolicy(policy);
256            }
257
258            if (node.has("snapshots")) {
259                var policy = parser.getCodec().treeToValue(node.get("snapshots"), RepositoryPolicy.class);
260
261                builder.setSnapshotPolicy(policy);
262            }
263
264            return builder.build();
265        }
266    }

268    @ToString
269    private static class RemoteRepositorySerializer extends StdSerializer<RemoteRepository> {
270        private static final long serialVersionUID = -1L;
271
272        public RemoteRepositorySerializer() { super(RemoteRepository.class); }
273
274        @Override
275        public void serialize(RemoteRepository value, JsonGenerator generator, SerializerProvider provider) throws IOException, JsonProcessingException {
276            var map = new LinkedHashMap<String,Object>();
277
278            map.put("id", value.getId());
279            map.put("layout", value.getContentType());
280            map.put("url", value.getUrl());
281            map.put("releases", value.getPolicy(false));
282            map.put("snapshots", value.getPolicy(true));
283
284            generator.writeObject(map);
285        }
286    }

288    @ToString
289    private static class RepositoryPolicyDeserializer extends StdDeserializer<RepositoryPolicy> {
290        private static final long serialVersionUID = -1L;
291
292        public RepositoryPolicyDeserializer() { super(RepositoryPolicy.class); }
293
294        @Override
295        public RepositoryPolicy deserialize(JsonParser parser, DeserializationContext context) throws IOException, JsonProcessingException {
296            JsonNode node = parser.getCodec().readTree(parser);
297
298            return new RepositoryPolicy(node.has("enabled") ? node.get("enabled").asBoolean() : true,
299                                        asText(node, "updatePolicy", RepositoryPolicy.UPDATE_POLICY_DAILY),
300                                        asText(node, "checksumPolicy", RepositoryPolicy.CHECKSUM_POLICY_WARN));
301        }
302    }
