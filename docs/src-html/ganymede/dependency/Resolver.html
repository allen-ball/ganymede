147    /**
148     * Method to add a {@link File}(s) to the {@link #classpath()}.  May
149     * quietly ignore requests to add {@link File}s that violate internal
150     * heuristics (e.g.,
151     * {@link.uri http://www.slf4j.org/codes.html#multiple_bindings target=newtab multiple SLF4J bindings}).
152     *
153     * @param   files           The {@link File}(s).
154     *
155     * @return  The {@link List} of {@link File}s added to the
156     *          {@link #classpath()}.
157     */
158    public List<File> addToClasspath(File... files) {
159        var list = new LinkedList<File>();
160
161        for (var file : files) {
162            file = file.getAbsoluteFile();
163
164            var artifacts =
165                getShadedArtifactSet(file).stream()
166                .map(repository::resolve)
167                .collect(toSet());
168
169            if (! artifacts.isEmpty()) {
170                var array = artifacts.toArray(new Artifact[] { });
171
172                list.addAll(addToClasspath(array));
173            } else {
174                if (classpath.add(file)) {
175                    list.add(file);
176                }
177            }
178        }
179
180        return list;
181    }

183    /**
184     * Method to add a {@link Artifact}(s) to the {@link #classpath()}.
185     * May quietly ignore requests to add {@link Artifact}s that violate
186     * internal heuristics (e.g.,
187     * {@link.uri http://www.slf4j.org/codes.html#multiple_bindings target=newtab multiple SLF4J bindings}).
188     *
189     * @param   artifacts       The {@link Artifact}(s).
190     *
191     * @return  The {@link List} of {@link File}s added to the
192     *          {@link #classpath()}.
193     */
194    public List<File> addToClasspath(Artifact... artifacts) {
195        var list = new LinkedList<File>();
196        var resolved =
197            Stream.of(artifacts)
198            .map(repository::resolve)
199            .collect(toList());
200
201        for (var artifact : resolved) {
202            var ignore = false;
203            var id = ArtifactIdUtils.toVersionlessId(artifact);
204
205            if (! ignore) {
206                ignore |= LOGGING_IGNORE.contains(id);
207            }
208
209            if (! ignore) {
210                if (SLF4J_BINDINGS.contains(id)) {
211                    ignore |=
212                        repository.getArtifactsOn(classpath)
213                        .map(ArtifactIdUtils::toVersionlessId)
214                        .anyMatch(t -> SLF4J_BINDINGS.contains(t));
215                }
216            }
217
218            if (! ignore) {
219                if (JCL_BRIDGES.contains(id)) {
220                    ignore |=
221                        repository.getArtifactsOn(classpath)
222                        .map(ArtifactIdUtils::toVersionlessId)
223                        .anyMatch(t -> JCL_BRIDGES.contains(t));
224                }
225            }
226
227            if (! ignore) {
228                if (! classpath.contains(artifact.getFile())) {
229                    var installed =
230                        repository.getArtifactsOn(classpath)
231                        .filter(t -> ArtifactIdUtils.equalsVersionlessId(artifact, t))
232                        .findFirst().orElse(null);
233
234                    if (installed == null) {
235                        if (classpath.add(artifact.getFile())) {
236                            list.add(artifact.getFile());
237                        }
238                    } else {
239                        log.debug("Ignored resolved artifact {}", artifact);
240                        log.debug("    for {} @ {}",
241                                  () -> installed.getVersion(),
242                                  () -> pathMap.shorten(installed.getFile()));
243                    }
244                }
245            }
246        }
247
248        return list;
249    }

251    /**
252     * Method to add known dependencies found within a parent directory to
253     * the {@link #classpath()}.  An artifact
254     * {@code artifactId-version.packaging} is a known artifact if there is
255     * {@code META-INF/artifactId-version.packaging.dependencies} file on
256     * the classpath which contains the artifact's dependencies.
257     *
258     * @param   parent          The parent {@link File} to analyze.
259     *
260     * @return  The {@link List} of {@link File}s added to the
261     *          {@link #classpath()}.
262     */
263    public List<File> addKnownDependenciesToClasspath(File parent) {
264        var list = new LinkedList<File>();
265
266        try (var stream = Files.newDirectoryStream(parent.toPath(), "*.jar")) {
267            for (var entry : stream) {
268                var resource = String.format(DEPENDENCIES_FORMAT, entry.getFileName());
269
270                try (var in = getClass().getResourceAsStream(resource)) {
271                    if (in != null) {
272                        var artifacts =
273                            parse(in)
274                            .map(t -> t.setFile(new File(parent,
275                                                         t.getArtifactId()
276                                                         + "-" + t.getVersion()
277                                                         + "." + t.getExtension())
278                                                .getAbsoluteFile()))
279                            .filter(t -> t.getFile().exists())
280                            .toArray(Artifact[]::new);
281
282                        list.addAll(addToClasspath(artifacts));
283                    }
284                } catch (Exception exception) {
285                    log.warn("Could not read {}", resource, exception);
286                }
287            }
288        } catch (Exception exception) {
289            log.warn("{}: {}", parent, exception);
290        }
291
292        return list;
293    }

295    /**
296     * Merge the argument {@link POM} and resolve any dependencies.
297     *
298     * @param   shell           The {@link Shell}.
299     * @param   out             The {@code stdout} {@link PrintStream}.
300     * @param   err             The {@code stderr} {@link PrintStream}.
301     * @param   pom             The {@link POM} to merge.
302     *
303     * @return  The {@link List} of {@link File}s added to the
304     *          {@link #classpath()}.
305     */
306    public List<File> resolve(Shell shell, PrintStream out, PrintStream err, POM pom) {
307        var files = new ArrayList<File>();
308
309        pom().merge(pom);
310
311        var session = session(out, err);
312        var repositories = pom().getRepositories().stream().collect(toList());
313        var scope = JavaScopes.RUNTIME;
314        var filter = DependencyFilterUtils.classpathFilter(scope);
315        var iterator = pom().getDependencies().iterator();
316
317        while (iterator.hasNext()) {
318            var dependency = iterator.next();
319
320            try {
321                var request = new DependencyRequest(new CollectRequest(dependency, repositories), filter);
322
323                for (var result : system.resolveDependencies(session, request).getArtifactResults()) {
324                    if (result.isResolved()) {
325                        var artifact = repository.resolve(result.getArtifact());
326
327                        if (! classpath.contains(artifact.getFile())) {
328                            var installed =
329                                repository.getArtifactsOn(classpath)
330                                .filter(t -> ArtifactIdUtils.equalsVersionlessId(artifact, t))
331                                .findFirst().orElse(null);
332
333                            if (installed == null) {
334                                files.addAll(addToClasspath(artifact));
335                            } else {
336                                log.debug("Ignored resolved artifact {}", artifact);
337                                log.debug("    for {} @ {}",
338                                          () -> installed.getVersion(),
339                                          () -> pathMap.shorten(installed.getFile()));
340                            }
341                        }
342                    }
343
344                    if (result.isMissing()) {
345                        result.getExceptions().stream()
346                            .forEach(t -> err.println(t.getMessage()));
347                    }
348                }
349                continue;
350            } catch (DependencyResolutionException exception) {
351                err.println(exception.getMessage());
352            } catch (Exception exception) {
353                exception.printStackTrace(err);
354            }
355
356            iterator.remove();
357        }
358
359        return files;
360    }

362    private RepositorySystemSession session(PrintStream out, PrintStream err) {
363        var session = MavenRepositorySystemUtils.newSession();
364        var properties = new LinkedHashMap<Object,Object>();
365
366        properties.put(ConfigurationProperties.USER_AGENT,
367                       String.format("Ganymede/(Java %s; %s)",
368                                     System.getProperty("java.version"),
369                                     System.getProperty("os.version")));
370        /*
371         * properties.putAll((Map<?,?>) project.getProperties());
372         */
373        /*
374         * ${settings.servers}
375         * AntRepoSys: processServerConfiguration(properties);
376         */
377        session.setConfigProperties(properties);
378        session.setOffline(Objects.requireNonNullElse(pom().getOffline(), false));
379        /*
380         * AntRepoSys: session.setUserProperties(project.getUserProperties());
381         */
382        /*
383         * ${settings.servers}, ${settings.mirrors}
384         * session.setProxySelector(getProxySelector());
385         * session.setMirrorSelector(getMirrorSelector());
386         * session.setAuthenticationSelector(getAuthSelector());
387         */
388        session.setCache(new DefaultRepositoryCache());
389
390        var path =
391            Stream.of(pom().getLocalRepository(), System.getProperty("maven.repo.local"))
392            .filter(Objects::nonNull)
393            .map(Paths::get)
394            .findFirst().orElse(Paths.get(System.getProperty("user.home"), ".m2", "repository"));
395        var local = new LocalRepository(path.toFile());
396        var manager = system.newLocalRepositoryManager(session, local);
397
398        session.setLocalRepositoryManager(manager);
399        session.setWorkspaceReader(repository);
400        session.setRepositoryListener(new RepositoryListener(out, err));
401        session.setTransferListener(new TransferListener(out, err));
402
403        return session;
404    }

406    @NoArgsConstructor
407    private class RepositoryImpl extends TreeMap<String,Artifact> implements WorkspaceReader {
408        private static final long serialVersionUID = -1L;
409
410        public Set<Artifact> resolve(Set<Artifact> set) {
411            var resolved =
412                set.stream()
413                .map(this::resolve)
414                .collect(toCollection(LinkedHashSet::new));
415
416            return resolved;
417        }
418
419        public Artifact resolve(Artifact artifact) {
420            if (artifact.getFile() == null) {
421                throw new IllegalArgumentException();
422            }
423
424            return computeIfAbsent(ArtifactIdUtils.toId(artifact), k -> artifact);
425        }
426
427        public Stream<Artifact> getArtifactsOn(Set<File> classpath) {
428            return values().stream().filter(t -> classpath.contains(t.getFile()));
429        }
430
431        @Override
432        public WorkspaceRepository getRepository() {
433            return new WorkspaceRepository(getClass().getPackage().getName());
434        }
435
436        @Override
437        public File findArtifact(Artifact artifact) {
438            var id = ArtifactIdUtils.toId(artifact);
439            var file =
440                values().stream()
441                .filter(t -> Objects.equals(id, ArtifactIdUtils.toId(t)))
442                .map(Artifact::getFile)
443                .findFirst().orElse(null);
444
445            return file;
446        }
447
448        @Override
449        public List<String> findVersions(Artifact artifact) {
450            var list =
451                values().stream()
452                .filter(t -> ArtifactIdUtils.equalsVersionlessId(artifact, t))
453                .map(Artifact::getVersion)
454                .collect(toList());
455
456            return list;
457        }
458    }
