135    /**
136     * Method to add a {@link File}(s) to the {@link #classpath()}.  May
137     * quietly ignore requests to add {@link File}s that violate internal
138     * heuristics (e.g.,
139     * {@link.uri http://www.slf4j.org/codes.html#multiple_bindings target=newtab multiple SLF4J bindings}).
140     *
141     * @param   files           The {@link File}(s).
142     *
143     * @return  The {@link List} of {@link File}s added to the
144     *          {@link #classpath()}.
145     */
146    public List<File> addToClasspath(File... files) {
147        var list = new LinkedList<File>();
148
149        for (var file : files) {
150            file = file.getAbsoluteFile();
151
152            var artifacts =
153                getShadedArtifactSet(file).stream()
154                .map(repository::resolve)
155                .collect(toSet());
156
157            if (! artifacts.isEmpty()) {
158                var array = artifacts.toArray(new Artifact[] { });
159
160                list.addAll(addToClasspath(array));
161            } else {
162                if (classpath.add(file)) {
163                    list.add(file);
164                }
165            }
166        }
167
168        return list;
169    }
170
171    /**
172     * Method to add a {@link Artifact}(s) to the {@link #classpath()}.
173     * May quietly ignore requests to add {@link Artifact}s that violate
174     * internal heuristics (e.g.,
175     * {@link.uri http://www.slf4j.org/codes.html#multiple_bindings target=newtab multiple SLF4J bindings}).
176     *
177     * @param   artifacts       The {@link Artifact}(s).
178     *
179     * @return  The {@link List} of {@link File}s added to the
180     *          {@link #classpath()}.
181     */
182    public List<File> addToClasspath(Artifact... artifacts) {
183        var list = new LinkedList<File>();
184        var resolved =
185            Stream.of(artifacts)
186            .map(repository::resolve)
187            .collect(toList());
188
189        for (var artifact : resolved) {
190            var ignore = false;
191            var id = ArtifactIdUtils.toVersionlessId(artifact);
192
193            if (! ignore) {
194                ignore |= LOGGING_IGNORE.contains(id);
195            }
196
197            if (! ignore) {
198                if (SLF4J_BINDINGS.contains(id)) {
199                    ignore |=
200                        repository.getArtifactsOn(classpath)
201                        .map(ArtifactIdUtils::toVersionlessId)
202                        .anyMatch(t -> SLF4J_BINDINGS.contains(t));
203                }
204            }
205
206            if (! ignore) {
207                if (JCL_BRIDGES.contains(id)) {
208                    ignore |=
209                        repository.getArtifactsOn(classpath)
210                        .map(ArtifactIdUtils::toVersionlessId)
211                        .anyMatch(t -> JCL_BRIDGES.contains(t));
212                }
213            }
214
215            if (! ignore) {
216                if (! classpath.contains(artifact.getFile())) {
217                    var installed =
218                        repository.getArtifactsOn(classpath)
219                        .filter(t -> ArtifactIdUtils.equalsVersionlessId(artifact, t))
220                        .findFirst().orElse(null);
221
222                    if (installed == null) {
223                        if (classpath.add(artifact.getFile())) {
224                            list.add(artifact.getFile());
225                        }
226                    } else {
227                        log.debug("Ignored resolved artifact {}", artifact);
228                        log.debug("    for {} @ {}",
229                                  () -> installed.getVersion(),
230                                  () -> pathMap.shorten(installed.getFile()));
231                    }
232                }
233            }
234        }
235
236        return list;
237    }

239    /**
240     * Method to add known dependencies found within a parent directory to
241     * the {@link #classpath()}.  An artifact
242     * {@code artifactId-version.packaging} is a known artifact if there is
243     * {@code META-INF/artifactId-version.packaging.dependencies} file on
244     * the classpath which contains the artifact's dependencies.
245     *
246     * @param   parent          The parent {@link File} to analyze.
247     *
248     * @return  The {@link List} of {@link File}s added to the
249     *          {@link #classpath()}.
250     */
251    public List<File> addKnownDependenciesToClasspath(File parent) {
252        var list = new LinkedList<File>();
253
254        try (var stream = Files.newDirectoryStream(parent.toPath(), "*.jar")) {
255            for (var entry : stream) {
256                var resource = String.format(DEPENDENCIES_FORMAT, entry.getFileName());
257
258                try (var in = getClass().getResourceAsStream(resource)) {
259                    if (in != null) {
260                        var artifacts =
261                            parse(in)
262                            .map(t -> t.setFile(new File(parent,
263                                                         t.getArtifactId()
264                                                         + "-" + t.getVersion()
265                                                         + "." + t.getExtension())
266                                                .getAbsoluteFile()))
267                            .filter(t -> t.getFile().exists())
268                            .toArray(Artifact[]::new);
269
270                        list.addAll(addToClasspath(artifacts));
271                    }
272                } catch (Exception exception) {
273                    log.warn("Could not read {}", resource, exception);
274                }
275            }
276        } catch (Exception exception) {
277            log.warn("{}: {}", parent, exception);
278        }
279
280        return list;
281    }

283    /**
284     * Merge the argument {@link POM} and resolve any dependencies.
285     *
286     * @param   shell           The {@link Shell}.
287     * @param   out             The {@code stdout} {@link PrintStream}.
288     * @param   err             The {@code stderr} {@link PrintStream}.
289     * @param   pom             The {@link POM} to merge.
290     *
291     * @return  The {@link List} of {@link File}s added to the
292     *          {@link #classpath()}.
293     */
294    public List<File> resolve(Shell shell, PrintStream out, PrintStream err, POM pom) {
295        var files = new ArrayList<File>();
296
297        pom().merge(pom);
298
299        var system = system();
300        var session = session();
301        var repositories = pom().getRepositories().stream().collect(toList());
302        var scope = JavaScopes.RUNTIME;
303        var filter = DependencyFilterUtils.classpathFilter(scope);
304        var iterator = pom().getDependencies().iterator();
305
306        while (iterator.hasNext()) {
307            var dependency = iterator.next();
308
309            try {
310                var request = new DependencyRequest(new CollectRequest(dependency, repositories), filter);
311
312                for (var result :
313                         system.resolveDependencies(session, request)
314                         .getArtifactResults()) {
315                    if (result.isResolved()) {
316                        var artifact = repository.resolve(result.getArtifact());
317
318                        if (! classpath.contains(artifact.getFile())) {
319                            var installed =
320                                repository.getArtifactsOn(classpath)
321                                .filter(t -> ArtifactIdUtils.equalsVersionlessId(artifact, t))
322                                .findFirst().orElse(null);
323
324                            if (installed == null) {
325                                files.addAll(addToClasspath(artifact));
326                            } else {
327                                log.debug("Ignored resolved artifact {}", artifact);
328                                log.debug("    for {} @ {}",
329                                          () -> installed.getVersion(),
330                                          () -> pathMap.shorten(installed.getFile()));
331                            }
332                        }
333                    }
334
335                    if (result.isMissing()) {
336                        result.getExceptions().stream()
337                            .forEach(t -> err.println(t.getMessage()));
338                    }
339                }
340                continue;
341            } catch (DependencyResolutionException exception) {
342                err.println(exception.getMessage());
343            } catch (Exception exception) {
344                exception.printStackTrace(err);
345            }
346
347            iterator.remove();
348        }
349
350        return files;
351    }

353    @Synchronized
354    private RepositorySystem system() {
355        if (system == null) {
356            var locator = MavenRepositorySystemUtils.newServiceLocator();
357
358            locator.addService(RepositoryConnectorFactory.class, BasicRepositoryConnectorFactory.class);
359            locator.addService(TransporterFactory.class, FileTransporterFactory.class);
360            locator.addService(TransporterFactory.class, HttpTransporterFactory.class);
361            locator.addService(TransporterFactory.class, ClasspathTransporterFactory.class);
362
363            system = locator.getService(RepositorySystem.class);
364        }
365
366        return system;
367    }

369    @Synchronized
370    private RepositorySystemSession session() {
371        var session = MavenRepositorySystemUtils.newSession();
372        var properties = new LinkedHashMap<Object,Object>();
373
374        properties.put(ConfigurationProperties.USER_AGENT,
375                       String.format("Ganymede/(Java %s; %s)",
376                                     System.getProperty("java.version"),
377                                     System.getProperty("os.version")));
378        /*
379         * properties.putAll((Map<?,?>) project.getProperties());
380         */
381        /*
382         * ${settings.servers}
383         * AntRepoSys: processServerConfiguration(properties);
384         */
385        session.setConfigProperties(properties);
386        session.setOffline(Objects.requireNonNullElse(pom().getOffline(), false));
387        /*
388         * AntRepoSys: session.setUserProperties(project.getUserProperties());
389         */
390        /*
391         * ${settings.servers}, ${settings.mirrors}
392         * session.setProxySelector(getProxySelector());
393         * session.setMirrorSelector(getMirrorSelector());
394         * session.setAuthenticationSelector(getAuthSelector());
395         */
396        session.setCache(new DefaultRepositoryCache());
397        session.setLocalRepositoryManager(getLocalRepositoryManager(session));
398        session.setRepositoryListener(new RepositoryListener());
399        session.setTransferListener(new TransferListener());
400        session.setWorkspaceReader(repository);
401
402        return session;
403    }

405    private LocalRepositoryManager getLocalRepositoryManager(RepositorySystemSession session) {
406        var path =
407            Stream.of(pom().getLocalRepository(),
408                      System.getProperty("maven.repo.local"))
409            .filter(Objects::nonNull)
410            .map(Paths::get)
411            .findFirst()
412            .orElse(Paths.get(System.getProperty("user.home"), ".m2", "repository"));
413        var repository = new LocalRepository(path.toFile());
414
415        return system().newLocalRepositoryManager(session, repository);
416    }

418    @NoArgsConstructor
419    private class RepositoryImpl extends TreeMap<String,Artifact> implements WorkspaceReader {
420        private static final long serialVersionUID = -1L;
421
422        public Set<Artifact> resolve(Set<Artifact> set) {
423            var resolved =
424                set.stream()
425                .map(this::resolve)
426                .collect(toCollection(LinkedHashSet::new));
427
428            return resolved;
429        }
430
431        public Artifact resolve(Artifact artifact) {
432            if (artifact.getFile() == null) {
433                throw new IllegalArgumentException();
434            }
435
436            return computeIfAbsent(ArtifactIdUtils.toId(artifact), k -> artifact);
437        }
438
439        public Stream<Artifact> getArtifactsOn(Set<File> classpath) {
440            return values().stream().filter(t -> classpath.contains(t.getFile()));
441        }
442
443        @Override
444        public WorkspaceRepository getRepository() {
445            return new WorkspaceRepository(getClass().getPackage().getName());
446        }
447
448        @Override
449        public File findArtifact(Artifact artifact) {
450            var id = ArtifactIdUtils.toId(artifact);
451            var file =
452                values().stream()
453                .filter(t -> Objects.equals(id, ArtifactIdUtils.toId(t)))
454                .map(Artifact::getFile)
455                .findFirst().orElse(null);
456
457            return file;
458        }
459
460        @Override
461        public List<String> findVersions(Artifact artifact) {
462            var list =
463                values().stream()
464                .filter(t -> ArtifactIdUtils.equalsVersionlessId(artifact, t))
465                .map(Artifact::getVersion)
466                .collect(toList());
467
468            return list;
469        }
470    }
