134    /**
135     * Method to add a {@link File}(s) to the {@link #classpath()}.  May
136     * quietly ignore requests to add {@link File}s that violate internal
137     * heuristics (e.g.,
138     * {@link.uri http://www.slf4j.org/codes.html#multiple_bindings target=newtab multiple SLF4J bindings}).
139     *
140     * @param   files           The {@link File}(s).
141     *
142     * @return  The {@link List} of {@link File}s added to the
143     *          {@link #classpath()}.
144     */
145    public List<File> addToClasspath(File... files) {
146        var list = new LinkedList<File>();
147
148        for (var file : files) {
149            file = file.getAbsoluteFile();
150
151            var artifacts =
152                getShadedArtifactSet(file).stream()
153                .map(repository::resolve)
154                .collect(toSet());
155
156            if (! artifacts.isEmpty()) {
157                var array = artifacts.toArray(new Artifact[] { });
158
159                list.addAll(addToClasspath(array));
160            } else {
161                if (classpath.add(file)) {
162                    list.add(file);
163                }
164            }
165        }
166
167        return list;
168    }
169
170    /**
171     * Method to add a {@link Artifact}(s) to the {@link #classpath()}.
172     * May quietly ignore requests to add {@link Artifact}s that violate
173     * internal heuristics (e.g.,
174     * {@link.uri http://www.slf4j.org/codes.html#multiple_bindings target=newtab multiple SLF4J bindings}).
175     *
176     * @param   artifacts       The {@link Artifact}(s).
177     *
178     * @return  The {@link List} of {@link File}s added to the
179     *          {@link #classpath()}.
180     */
181    public List<File> addToClasspath(Artifact... artifacts) {
182        var list = new LinkedList<File>();
183        var resolved =
184            Stream.of(artifacts)
185            .map(repository::resolve)
186            .collect(toList());
187
188        for (var artifact : resolved) {
189            var ignore = false;
190            var id = ArtifactIdUtils.toVersionlessId(artifact);
191
192            if (! ignore) {
193                ignore |= LOGGING_IGNORE.contains(id);
194            }
195
196            if (! ignore) {
197                if (SLF4J_BINDINGS.contains(id)) {
198                    ignore |=
199                        repository.getArtifactsOn(classpath)
200                        .map(ArtifactIdUtils::toVersionlessId)
201                        .anyMatch(t -> SLF4J_BINDINGS.contains(t));
202                }
203            }
204
205            if (! ignore) {
206                if (JCL_BRIDGES.contains(id)) {
207                    ignore |=
208                        repository.getArtifactsOn(classpath)
209                        .map(ArtifactIdUtils::toVersionlessId)
210                        .anyMatch(t -> JCL_BRIDGES.contains(t));
211                }
212            }
213
214            if (! ignore) {
215                if (! classpath.contains(artifact.getFile())) {
216                    var installed =
217                        repository.getArtifactsOn(classpath)
218                        .filter(t -> ArtifactIdUtils.equalsVersionlessId(artifact, t))
219                        .findFirst().orElse(null);
220
221                    if (installed == null) {
222                        if (classpath.add(artifact.getFile())) {
223                            list.add(artifact.getFile());
224                        }
225                    } else {
226                        log.debug("Ignored resolved artifact {}", artifact);
227                        log.debug("    for {} @ {}",
228                                  installed.getVersion(), installed.getFile());
229                    }
230                }
231            }
232        }
233
234        return list;
235    }

237    /**
238     * Method to add known dependencies found within a parent directory to
239     * the {@link #classpath()}.  An artifact
240     * {@code artifactId-version.packaging} is a known artifact if there is
241     * {@code META-INF/artifactId-version.packaging.dependencies} file on
242     * the classpath which contains the artifact's dependencies.
243     *
244     * @param   parent          The parent {@link File} to analyze.
245     *
246     * @return  The {@link List} of {@link File}s added to the
247     *          {@link #classpath()}.
248     */
249    public List<File> addKnownDependenciesToClasspath(File parent) {
250        var list = new LinkedList<File>();
251
252        try (var stream = Files.newDirectoryStream(parent.toPath(), "*.jar")) {
253            for (var entry : stream) {
254                var resource = String.format(DEPENDENCIES_FORMAT, entry.getFileName());
255
256                try (var in = getClass().getResourceAsStream(resource)) {
257                    if (in != null) {
258                        var artifacts =
259                            parse(in)
260                            .map(t -> t.setFile(new File(parent,
261                                                         t.getArtifactId()
262                                                         + "-" + t.getVersion()
263                                                         + "." + t.getExtension())
264                                                .getAbsoluteFile()))
265                            .filter(t -> t.getFile().exists())
266                            .toArray(Artifact[]::new);
267
268                        list.addAll(addToClasspath(artifacts));
269                    }
270                } catch (Exception exception) {
271                    log.warn("Could not read {}", resource, exception);
272                }
273            }
274        } catch (Exception exception) {
275            log.warn("{}: {}", parent, exception);
276        }
277
278        return list;
279    }

281    /**
282     * Merge the argument {@link POM} and resolve any dependencies.
283     *
284     * @param   shell           The {@link Shell}.
285     * @param   out             The {@code stdout} {@link PrintStream}.
286     * @param   err             The {@code stderr} {@link PrintStream}.
287     * @param   pom             The {@link POM} to merge.
288     *
289     * @return  The {@link List} of {@link File}s added to the
290     *          {@link #classpath()}.
291     */
292    public List<File> resolve(Shell shell, PrintStream out, PrintStream err, POM pom) {
293        var files = new ArrayList<File>();
294
295        pom().merge(pom);
296
297        var system = system();
298        var session = session();
299        var repositories = pom().getRepositories().stream().collect(toList());
300        var scope = JavaScopes.RUNTIME;
301        var filter = DependencyFilterUtils.classpathFilter(scope);
302        var iterator = pom().getDependencies().iterator();
303
304        while (iterator.hasNext()) {
305            var dependency = iterator.next();
306
307            try {
308                var request = new DependencyRequest(new CollectRequest(dependency, repositories), filter);
309
310                for (var result :
311                         system.resolveDependencies(session, request)
312                         .getArtifactResults()) {
313                    if (result.isResolved()) {
314                        var artifact = repository.resolve(result.getArtifact());
315
316                        if (! classpath.contains(artifact.getFile())) {
317                            var installed =
318                                repository.getArtifactsOn(classpath)
319                                .filter(t -> ArtifactIdUtils.equalsVersionlessId(artifact, t))
320                                .findFirst().orElse(null);
321
322                            if (installed == null) {
323                                files.addAll(addToClasspath(artifact));
324                            } else {
325                                log.debug("Ignored resolved artifact {}", artifact);
326                                log.debug("    for {} @ {}",
327                                          installed.getVersion(), installed.getFile());
328                            }
329                        }
330                    }
331
332                    if (result.isMissing()) {
333                        result.getExceptions().stream()
334                            .forEach(t -> err.println(t.getMessage()));
335                    }
336                }
337                continue;
338            } catch (DependencyResolutionException exception) {
339                err.println(exception.getMessage());
340            } catch (Exception exception) {
341                exception.printStackTrace(err);
342            }
343
344            iterator.remove();
345        }
346
347        return files;
348    }

350    @Synchronized
351    private RepositorySystem system() {
352        if (system == null) {
353            var locator = MavenRepositorySystemUtils.newServiceLocator();
354
355            locator.addService(RepositoryConnectorFactory.class, BasicRepositoryConnectorFactory.class);
356            locator.addService(TransporterFactory.class, FileTransporterFactory.class);
357            locator.addService(TransporterFactory.class, HttpTransporterFactory.class);
358            locator.addService(TransporterFactory.class, ClasspathTransporterFactory.class);
359            /* locator.setServices(Logger.class, log); */
360            locator
361                .setErrorHandler(new DefaultServiceLocator.ErrorHandler() {
362                        @Override
363                        public void serviceCreationFailed(Class<?> type, Class<?> impl, Throwable exception) {
364                            log.error("Service creation failed for {} with implementation {}",
365                                      type, impl, exception);
366                        }
367                });
368
369            system = locator.getService(RepositorySystem.class);
370        }
371
372        return system;
373    }

375    @Synchronized
376    private RepositorySystemSession session() {
377        var session = MavenRepositorySystemUtils.newSession();
378        var properties = new LinkedHashMap<Object,Object>();
379
380        properties.put(ConfigurationProperties.USER_AGENT,
381                       String.format("Ganymede/(Java %s; %s)",
382                                     System.getProperty("java.version"),
383                                     System.getProperty("os.version")));
384        /*
385         * properties.putAll((Map<?,?>) project.getProperties());
386         */
387        /*
388         * ${settings.servers}
389         * AntRepoSys: processServerConfiguration(properties);
390         */
391        session.setConfigProperties(properties);
392        session.setOffline(Objects.requireNonNullElse(pom().getOffline(), false));
393        /*
394         * AntRepoSys: session.setUserProperties(project.getUserProperties());
395         */
396        /*
397         * ${settings.servers}, ${settings.mirrors}
398         * session.setProxySelector(getProxySelector());
399         * session.setMirrorSelector(getMirrorSelector());
400         * session.setAuthenticationSelector(getAuthSelector());
401         */
402        session.setCache(new DefaultRepositoryCache());
403        session.setLocalRepositoryManager(getLocalRepositoryManager(session));
404        session.setRepositoryListener(new RepositoryListener());
405        session.setTransferListener(new TransferListener());
406        session.setWorkspaceReader(repository);
407
408        return session;
409    }

411    private LocalRepositoryManager getLocalRepositoryManager(RepositorySystemSession session) {
412        var path =
413            Stream.of(pom().getLocalRepository(),
414                      System.getProperty("maven.repo.local"))
415            .filter(Objects::nonNull)
416            .map(Paths::get)
417            .findFirst()
418            .orElse(Paths.get(System.getProperty("user.home"), ".m2", "repository"));
419        var repository = new LocalRepository(path.toFile());
420
421        return system().newLocalRepositoryManager(session, repository);
422    }

424    @NoArgsConstructor
425    private class RepositoryImpl extends TreeMap<String,Artifact> implements WorkspaceReader {
426        private static final long serialVersionUID = -1L;
427
428        public Set<Artifact> resolve(Set<Artifact> set) {
429            var resolved =
430                set.stream()
431                .map(this::resolve)
432                .collect(toCollection(LinkedHashSet::new));
433
434            return resolved;
435        }
436
437        public Artifact resolve(Artifact artifact) {
438            if (artifact.getFile() == null) {
439                throw new IllegalArgumentException();
440            }
441
442            return computeIfAbsent(ArtifactIdUtils.toId(artifact), k -> artifact);
443        }
444
445        public Stream<Artifact> getArtifactsOn(Set<File> classpath) {
446            return values().stream().filter(t -> classpath.contains(t.getFile()));
447        }
448
449        @Override
450        public WorkspaceRepository getRepository() {
451            return new WorkspaceRepository(getClass().getPackage().getName());
452        }
453
454        @Override
455        public File findArtifact(Artifact artifact) {
456            var id = ArtifactIdUtils.toId(artifact);
457            var file =
458                values().stream()
459                .filter(t -> Objects.equals(id, ArtifactIdUtils.toId(t)))
460                .map(Artifact::getFile)
461                .findFirst().orElse(null);
462
463            return file;
464        }
465
466        @Override
467        public List<String> findVersions(Artifact artifact) {
468            var list =
469                values().stream()
470                .filter(t -> ArtifactIdUtils.equalsVersionlessId(artifact, t))
471                .map(Artifact::getVersion)
472                .collect(toList());
473
474            return list;
475        }
476    }
