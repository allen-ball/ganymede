133    /**
134     * Method to add a {@link File}(s) to the {@link #classpath()}.  May
135     * quietly ignore requests to add {@link File}s that violate internal
136     * heuristics (e.g.,
137     * {@link.uri http://www.slf4j.org/codes.html#multiple_bindings target=newtab multiple SLF4J bindings}).
138     *
139     * @param   files           The {@link File}(s).
140     *
141     * @return  The {@link List} of {@link File}s added to the
142     *          {@link #classpath()}.
143     */
144    public List<File> addToClasspath(File... files) {
145        var list = new LinkedList<File>();
146
147        for (var file : files) {
148            file = file.getAbsoluteFile();
149
150            var artifacts =
151                getShadedArtifactSet(file).stream()
152                .map(repository::resolve)
153                .collect(toSet());
154
155            if (! artifacts.isEmpty()) {
156                var array = artifacts.toArray(new Artifact[] { });
157
158                list.addAll(addToClasspath(array));
159            } else {
160                if (classpath.add(file)) {
161                    list.add(file);
162                }
163            }
164        }
165
166        return list;
167    }
168
169    /**
170     * Method to add a {@link Artifact}(s) to the {@link #classpath()}.
171     * May quietly ignore requests to add {@link Artifact}s that violate
172     * internal heuristics (e.g.,
173     * {@link.uri http://www.slf4j.org/codes.html#multiple_bindings target=newtab multiple SLF4J bindings}).
174     *
175     * @param   artifacts       The {@link Artifact}(s).
176     *
177     * @return  The {@link List} of {@link File}s added to the
178     *          {@link #classpath()}.
179     */
180    public List<File> addToClasspath(Artifact... artifacts) {
181        var list = new LinkedList<File>();
182        var resolved =
183            Stream.of(artifacts)
184            .map(repository::resolve)
185            .collect(toList());
186
187        for (var artifact : resolved) {
188            var ignore = false;
189            var id = ArtifactIdUtils.toVersionlessId(artifact);
190
191            if (! ignore) {
192                ignore |= LOGGING_IGNORE.contains(id);
193            }
194
195            if (! ignore) {
196                if (SLF4J_BINDINGS.contains(id)) {
197                    ignore |=
198                        repository.getArtifactsOn(classpath)
199                        .map(ArtifactIdUtils::toVersionlessId)
200                        .anyMatch(t -> SLF4J_BINDINGS.contains(t));
201                }
202            }
203
204            if (! ignore) {
205                if (JCL_BRIDGES.contains(id)) {
206                    ignore |=
207                        repository.getArtifactsOn(classpath)
208                        .map(ArtifactIdUtils::toVersionlessId)
209                        .anyMatch(t -> JCL_BRIDGES.contains(t));
210                }
211            }
212
213            if (! ignore) {
214                if (! classpath.contains(artifact.getFile())) {
215                    var installed =
216                        repository.getArtifactsOn(classpath)
217                        .filter(t -> ArtifactIdUtils.equalsVersionlessId(artifact, t))
218                        .findFirst().orElse(null);
219
220                    if (installed == null) {
221                        if (classpath.add(artifact.getFile())) {
222                            list.add(artifact.getFile());
223                        }
224                    } else {
225                        log.debug("Ignored resolved artifact {}", artifact);
226                        log.debug("    for {} @ {}",
227                                  installed.getVersion(), installed.getFile());
228                    }
229                }
230            }
231        }
232
233        return list;
234    }

236    /**
237     * Method to add known dependencies found within a parent directory to
238     * the {@link #classpath()}.  An artifact
239     * {@code artifactId-version.packaging} is a known artifact if there is
240     * {@code META-INF/artifactId-version.packaging.dependencies} file on
241     * the classpath which contains the artifact's dependencies.
242     *
243     * @param   parent          The parent {@link File} to analyze.
244     *
245     * @return  The {@link List} of {@link File}s added to the
246     *          {@link #classpath()}.
247     */
248    public List<File> addKnownDependenciesToClasspath(File parent) {
249        var list = new LinkedList<File>();
250
251        try (var stream = Files.newDirectoryStream(parent.toPath(), "*.jar")) {
252            for (var entry : stream) {
253                var resource = String.format(DEPENDENCIES_FORMAT, entry.getFileName());
254
255                try (var in = getClass().getResourceAsStream(resource)) {
256                    if (in != null) {
257                        var artifacts =
258                            parse(in)
259                            .map(t -> t.setFile(new File(parent,
260                                                         t.getArtifactId()
261                                                         + "-" + t.getVersion()
262                                                         + "." + t.getExtension())
263                                                .getAbsoluteFile()))
264                            .filter(t -> t.getFile().exists())
265                            .toArray(Artifact[]::new);
266
267                        list.addAll(addToClasspath(artifacts));
268                    }
269                } catch (Exception exception) {
270                    log.warn("Could not read {}", resource, exception);
271                }
272            }
273        } catch (Exception exception) {
274            log.warn("{}: {}", parent, exception);
275        }
276
277        return list;
278    }

280    /**
281     * Merge the argument {@link POM} and resolve any dependencies.
282     *
283     * @param   shell           The {@link Shell}.
284     * @param   out             The {@code stdout} {@link PrintStream}.
285     * @param   err             The {@code stderr} {@link PrintStream}.
286     * @param   pom             The {@link POM} to merge.
287     *
288     * @return  The {@link List} of {@link File}s added to the
289     *          {@link #classpath()}.
290     */
291    public List<File> resolve(Shell shell, PrintStream out, PrintStream err, POM pom) {
292        var files = new ArrayList<File>();
293
294        pom().merge(pom);
295
296        var system = system();
297        var session = session();
298        var repositories = pom().getRepositories().stream().collect(toList());
299        var scope = JavaScopes.RUNTIME;
300        var filter = DependencyFilterUtils.classpathFilter(scope);
301        var iterator = pom().getDependencies().iterator();
302
303        while (iterator.hasNext()) {
304            var dependency = iterator.next();
305
306            try {
307                var request = new DependencyRequest(new CollectRequest(dependency, repositories), filter);
308
309                for (var result :
310                         system.resolveDependencies(session, request)
311                         .getArtifactResults()) {
312                    if (result.isResolved()) {
313                        var artifact = repository.resolve(result.getArtifact());
314
315                        if (! classpath.contains(artifact.getFile())) {
316                            var installed =
317                                repository.getArtifactsOn(classpath)
318                                .filter(t -> ArtifactIdUtils.equalsVersionlessId(artifact, t))
319                                .findFirst().orElse(null);
320
321                            if (installed == null) {
322                                files.addAll(addToClasspath(artifact));
323                            } else {
324                                log.debug("Ignored resolved artifact {}", artifact);
325                                log.debug("    for {} @ {}",
326                                          installed.getVersion(), installed.getFile());
327                            }
328                        }
329                    }
330
331                    if (result.isMissing()) {
332                        result.getExceptions().stream()
333                            .forEach(t -> err.println(t.getMessage()));
334                    }
335                }
336                continue;
337            } catch (DependencyResolutionException exception) {
338                err.println(exception.getMessage());
339            } catch (Exception exception) {
340                exception.printStackTrace(err);
341            }
342
343            iterator.remove();
344        }
345
346        return files;
347    }

349    @Synchronized
350    private RepositorySystem system() {
351        if (system == null) {
352            var locator = MavenRepositorySystemUtils.newServiceLocator();
353
354            locator.addService(RepositoryConnectorFactory.class, BasicRepositoryConnectorFactory.class);
355            locator.addService(TransporterFactory.class, FileTransporterFactory.class);
356            locator.addService(TransporterFactory.class, HttpTransporterFactory.class);
357            locator.addService(TransporterFactory.class, ClasspathTransporterFactory.class);
358
359            system = locator.getService(RepositorySystem.class);
360        }
361
362        return system;
363    }

365    @Synchronized
366    private RepositorySystemSession session() {
367        var session = MavenRepositorySystemUtils.newSession();
368        var properties = new LinkedHashMap<Object,Object>();
369
370        properties.put(ConfigurationProperties.USER_AGENT,
371                       String.format("Ganymede/(Java %s; %s)",
372                                     System.getProperty("java.version"),
373                                     System.getProperty("os.version")));
374        /*
375         * properties.putAll((Map<?,?>) project.getProperties());
376         */
377        /*
378         * ${settings.servers}
379         * AntRepoSys: processServerConfiguration(properties);
380         */
381        session.setConfigProperties(properties);
382        session.setOffline(Objects.requireNonNullElse(pom().getOffline(), false));
383        /*
384         * AntRepoSys: session.setUserProperties(project.getUserProperties());
385         */
386        /*
387         * ${settings.servers}, ${settings.mirrors}
388         * session.setProxySelector(getProxySelector());
389         * session.setMirrorSelector(getMirrorSelector());
390         * session.setAuthenticationSelector(getAuthSelector());
391         */
392        session.setCache(new DefaultRepositoryCache());
393        session.setLocalRepositoryManager(getLocalRepositoryManager(session));
394        session.setRepositoryListener(new RepositoryListener());
395        session.setTransferListener(new TransferListener());
396        session.setWorkspaceReader(repository);
397
398        return session;
399    }

401    private LocalRepositoryManager getLocalRepositoryManager(RepositorySystemSession session) {
402        var path =
403            Stream.of(pom().getLocalRepository(),
404                      System.getProperty("maven.repo.local"))
405            .filter(Objects::nonNull)
406            .map(Paths::get)
407            .findFirst()
408            .orElse(Paths.get(System.getProperty("user.home"), ".m2", "repository"));
409        var repository = new LocalRepository(path.toFile());
410
411        return system().newLocalRepositoryManager(session, repository);
412    }

414    @NoArgsConstructor
415    private class RepositoryImpl extends TreeMap<String,Artifact> implements WorkspaceReader {
416        private static final long serialVersionUID = -1L;
417
418        public Set<Artifact> resolve(Set<Artifact> set) {
419            var resolved =
420                set.stream()
421                .map(this::resolve)
422                .collect(toCollection(LinkedHashSet::new));
423
424            return resolved;
425        }
426
427        public Artifact resolve(Artifact artifact) {
428            if (artifact.getFile() == null) {
429                throw new IllegalArgumentException();
430            }
431
432            return computeIfAbsent(ArtifactIdUtils.toId(artifact), k -> artifact);
433        }
434
435        public Stream<Artifact> getArtifactsOn(Set<File> classpath) {
436            return values().stream().filter(t -> classpath.contains(t.getFile()));
437        }
438
439        @Override
440        public WorkspaceRepository getRepository() {
441            return new WorkspaceRepository(getClass().getPackage().getName());
442        }
443
444        @Override
445        public File findArtifact(Artifact artifact) {
446            var id = ArtifactIdUtils.toId(artifact);
447            var file =
448                values().stream()
449                .filter(t -> Objects.equals(id, ArtifactIdUtils.toId(t)))
450                .map(Artifact::getFile)
451                .findFirst().orElse(null);
452
453            return file;
454        }
455
456        @Override
457        public List<String> findVersions(Artifact artifact) {
458            var list =
459                values().stream()
460                .filter(t -> ArtifactIdUtils.equalsVersionlessId(artifact, t))
461                .map(Artifact::getVersion)
462                .collect(toList());
463
464            return list;
465        }
466    }
