138    /**
139     * Method to add a {@link File}(s) to the {@link #classpath()}.  May
140     * quietly ignore requests to add {@link File}s that violate internal
141     * heuristics (e.g.,
142     * {@link.uri http://www.slf4j.org/codes.html#multiple_bindings target=newtab multiple SLF4J bindings}).
143     *
144     * @param   files           The {@link File}(s).
145     *
146     * @return  The {@link List} of {@link File}s added to the
147     *          {@link #classpath()}.
148     */
149    public List<File> addToClasspath(File... files) {
150        var list = new LinkedList<File>();
151
152        for (var file : files) {
153            file = file.getAbsoluteFile();
154
155            var artifacts =
156                getShadedArtifactSet(file).stream()
157                .map(repository::resolve)
158                .collect(toSet());
159
160            if (! artifacts.isEmpty()) {
161                var array = artifacts.toArray(new Artifact[] { });
162
163                list.addAll(addToClasspath(array));
164            } else {
165                if (classpath.add(file)) {
166                    list.add(file);
167                }
168            }
169        }
170
171        return list;
172    }
173
174    /**
175     * Method to add a {@link Artifact}(s) to the {@link #classpath()}.
176     * May quietly ignore requests to add {@link Artifact}s that violate
177     * internal heuristics (e.g.,
178     * {@link.uri http://www.slf4j.org/codes.html#multiple_bindings target=newtab multiple SLF4J bindings}).
179     *
180     * @param   artifacts       The {@link Artifact}(s).
181     *
182     * @return  The {@link List} of {@link File}s added to the
183     *          {@link #classpath()}.
184     */
185    public List<File> addToClasspath(Artifact... artifacts) {
186        var list = new LinkedList<File>();
187        var resolved =
188            Stream.of(artifacts)
189            .map(repository::resolve)
190            .collect(toList());
191
192        for (var artifact : resolved) {
193            var ignore = false;
194            var id = ArtifactIdUtils.toVersionlessId(artifact);
195
196            if (! ignore) {
197                ignore |= LOGGING_IGNORE.contains(id);
198            }
199
200            if (! ignore) {
201                if (SLF4J_BINDINGS.contains(id)) {
202                    ignore |=
203                        repository.getArtifactsOn(classpath)
204                        .map(ArtifactIdUtils::toVersionlessId)
205                        .anyMatch(t -> SLF4J_BINDINGS.contains(t));
206                }
207            }
208
209            if (! ignore) {
210                if (JCL_BRIDGES.contains(id)) {
211                    ignore |=
212                        repository.getArtifactsOn(classpath)
213                        .map(ArtifactIdUtils::toVersionlessId)
214                        .anyMatch(t -> JCL_BRIDGES.contains(t));
215                }
216            }
217
218            if (! ignore) {
219                if (! classpath.contains(artifact.getFile())) {
220                    var installed =
221                        repository.getArtifactsOn(classpath)
222                        .filter(t -> ArtifactIdUtils.equalsVersionlessId(artifact, t))
223                        .findFirst().orElse(null);
224
225                    if (installed == null) {
226                        if (classpath.add(artifact.getFile())) {
227                            list.add(artifact.getFile());
228                        }
229                    } else {
230                        log.debug("Ignored resolved artifact {}", artifact);
231                        log.debug("    for {} @ {}",
232                                  () -> installed.getVersion(),
233                                  () -> pathMap.shorten(installed.getFile()));
234                    }
235                }
236            }
237        }
238
239        return list;
240    }

242    /**
243     * Method to add known dependencies found within a parent directory to
244     * the {@link #classpath()}.  An artifact
245     * {@code artifactId-version.packaging} is a known artifact if there is
246     * {@code META-INF/artifactId-version.packaging.dependencies} file on
247     * the classpath which contains the artifact's dependencies.
248     *
249     * @param   parent          The parent {@link File} to analyze.
250     *
251     * @return  The {@link List} of {@link File}s added to the
252     *          {@link #classpath()}.
253     */
254    public List<File> addKnownDependenciesToClasspath(File parent) {
255        var list = new LinkedList<File>();
256
257        try (var stream = Files.newDirectoryStream(parent.toPath(), "*.jar")) {
258            for (var entry : stream) {
259                var resource = String.format(DEPENDENCIES_FORMAT, entry.getFileName());
260
261                try (var in = getClass().getResourceAsStream(resource)) {
262                    if (in != null) {
263                        var artifacts =
264                            parse(in)
265                            .map(t -> t.setFile(new File(parent,
266                                                         t.getArtifactId()
267                                                         + "-" + t.getVersion()
268                                                         + "." + t.getExtension())
269                                                .getAbsoluteFile()))
270                            .filter(t -> t.getFile().exists())
271                            .toArray(Artifact[]::new);
272
273                        list.addAll(addToClasspath(artifacts));
274                    }
275                } catch (Exception exception) {
276                    log.warn("Could not read {}", resource, exception);
277                }
278            }
279        } catch (Exception exception) {
280            log.warn("{}: {}", parent, exception);
281        }
282
283        return list;
284    }

286    /**
287     * Merge the argument {@link POM} and resolve any dependencies.
288     *
289     * @param   shell           The {@link Shell}.
290     * @param   out             The {@code stdout} {@link PrintStream}.
291     * @param   err             The {@code stderr} {@link PrintStream}.
292     * @param   pom             The {@link POM} to merge.
293     *
294     * @return  The {@link List} of {@link File}s added to the
295     *          {@link #classpath()}.
296     */
297    public List<File> resolve(Shell shell, PrintStream out, PrintStream err, POM pom) {
298        var files = new ArrayList<File>();
299
300        pom().merge(pom);
301
302        var system = system();
303        var session = session();
304        var repositories = pom().getRepositories().stream().collect(toList());
305        var scope = JavaScopes.RUNTIME;
306        var filter = DependencyFilterUtils.classpathFilter(scope);
307        var iterator = pom().getDependencies().iterator();
308
309        while (iterator.hasNext()) {
310            var dependency = iterator.next();
311
312            try {
313                var request = new DependencyRequest(new CollectRequest(dependency, repositories), filter);
314
315                for (var result :
316                         system.resolveDependencies(session, request)
317                         .getArtifactResults()) {
318                    if (result.isResolved()) {
319                        var artifact = repository.resolve(result.getArtifact());
320
321                        if (! classpath.contains(artifact.getFile())) {
322                            var installed =
323                                repository.getArtifactsOn(classpath)
324                                .filter(t -> ArtifactIdUtils.equalsVersionlessId(artifact, t))
325                                .findFirst().orElse(null);
326
327                            if (installed == null) {
328                                files.addAll(addToClasspath(artifact));
329                            } else {
330                                log.debug("Ignored resolved artifact {}", artifact);
331                                log.debug("    for {} @ {}",
332                                          () -> installed.getVersion(),
333                                          () -> pathMap.shorten(installed.getFile()));
334                            }
335                        }
336                    }
337
338                    if (result.isMissing()) {
339                        result.getExceptions().stream()
340                            .forEach(t -> err.println(t.getMessage()));
341                    }
342                }
343                continue;
344            } catch (DependencyResolutionException exception) {
345                err.println(exception.getMessage());
346            } catch (Exception exception) {
347                exception.printStackTrace(err);
348            }
349
350            iterator.remove();
351        }
352
353        return files;
354    }

356    @Synchronized
357    private RepositorySystem system() {
358        if (system == null) {
359            var locator = MavenRepositorySystemUtils.newServiceLocator();
360
361            locator.addService(RepositoryConnectorFactory.class, BasicRepositoryConnectorFactory.class);
362            locator.addService(TransporterFactory.class, FileTransporterFactory.class);
363            locator.addService(TransporterFactory.class, HttpTransporterFactory.class);
364            locator.addService(TransporterFactory.class, ClasspathTransporterFactory.class);
365
366            system = locator.getService(RepositorySystem.class);
367        }
368
369        return system;
370    }

372    @Synchronized
373    private RepositorySystemSession session() {
374        var session = MavenRepositorySystemUtils.newSession();
375        var properties = new LinkedHashMap<Object,Object>();
376
377        properties.put(ConfigurationProperties.USER_AGENT,
378                       String.format("Ganymede/(Java %s; %s)",
379                                     System.getProperty("java.version"),
380                                     System.getProperty("os.version")));
381        /*
382         * properties.putAll((Map<?,?>) project.getProperties());
383         */
384        /*
385         * ${settings.servers}
386         * AntRepoSys: processServerConfiguration(properties);
387         */
388        session.setConfigProperties(properties);
389        session.setOffline(Objects.requireNonNullElse(pom().getOffline(), false));
390        /*
391         * AntRepoSys: session.setUserProperties(project.getUserProperties());
392         */
393        /*
394         * ${settings.servers}, ${settings.mirrors}
395         * session.setProxySelector(getProxySelector());
396         * session.setMirrorSelector(getMirrorSelector());
397         * session.setAuthenticationSelector(getAuthSelector());
398         */
399        session.setCache(new DefaultRepositoryCache());
400        session.setLocalRepositoryManager(getLocalRepositoryManager(session));
401        session.setRepositoryListener(new RepositoryListener());
402        session.setTransferListener(new TransferListener());
403        session.setWorkspaceReader(repository);
404
405        return session;
406    }

408    private LocalRepositoryManager getLocalRepositoryManager(RepositorySystemSession session) {
409        var path =
410            Stream.of(pom().getLocalRepository(),
411                      System.getProperty("maven.repo.local"))
412            .filter(Objects::nonNull)
413            .map(Paths::get)
414            .findFirst()
415            .orElse(Paths.get(System.getProperty("user.home"), ".m2", "repository"));
416        var repository = new LocalRepository(path.toFile());
417
418        return system().newLocalRepositoryManager(session, repository);
419    }

421    @NoArgsConstructor
422    private class RepositoryImpl extends TreeMap<String,Artifact> implements WorkspaceReader {
423        private static final long serialVersionUID = -1L;
424
425        public Set<Artifact> resolve(Set<Artifact> set) {
426            var resolved =
427                set.stream()
428                .map(this::resolve)
429                .collect(toCollection(LinkedHashSet::new));
430
431            return resolved;
432        }
433
434        public Artifact resolve(Artifact artifact) {
435            if (artifact.getFile() == null) {
436                throw new IllegalArgumentException();
437            }
438
439            return computeIfAbsent(ArtifactIdUtils.toId(artifact), k -> artifact);
440        }
441
442        public Stream<Artifact> getArtifactsOn(Set<File> classpath) {
443            return values().stream().filter(t -> classpath.contains(t.getFile()));
444        }
445
446        @Override
447        public WorkspaceRepository getRepository() {
448            return new WorkspaceRepository(getClass().getPackage().getName());
449        }
450
451        @Override
452        public File findArtifact(Artifact artifact) {
453            var id = ArtifactIdUtils.toId(artifact);
454            var file =
455                values().stream()
456                .filter(t -> Objects.equals(id, ArtifactIdUtils.toId(t)))
457                .map(Artifact::getFile)
458                .findFirst().orElse(null);
459
460            return file;
461        }
462
463        @Override
464        public List<String> findVersions(Artifact artifact) {
465            var list =
466                values().stream()
467                .filter(t -> ArtifactIdUtils.equalsVersionlessId(artifact, t))
468                .map(Artifact::getVersion)
469                .collect(toList());
470
471            return list;
472        }
473    }
