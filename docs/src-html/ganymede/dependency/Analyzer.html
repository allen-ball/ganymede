021import ball.annotation.CompileTimeCheck;
022import java.io.BufferedReader;
023import java.io.File;
024import java.io.IOException;
025import java.io.InputStream;
026import java.io.InputStreamReader;
027import java.util.LinkedHashSet;
028import java.util.List;
029import java.util.Map;
030import java.util.Objects;
031import java.util.Properties;
032import java.util.Set;
033import java.util.jar.Attributes;
034import java.util.jar.JarEntry;
035import java.util.jar.JarFile;
036import java.util.regex.Matcher;
037import java.util.regex.Pattern;
038import java.util.stream.Stream;
039import lombok.NoArgsConstructor;
040import lombok.ToString;
041import lombok.extern.log4j.Log4j2;
042import org.eclipse.aether.artifact.Artifact;
043import org.eclipse.aether.artifact.DefaultArtifact;
044
045import static java.nio.charset.StandardCharsets.UTF_8;
046import static java.util.stream.Collectors.toList;
047
048/**
049 * Dependency {@link Analyzer}.
050 *
051 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
052 */
053@NoArgsConstructor @ToString @Log4j2
054public class Analyzer {
055    private static final String SHADED_DEPENDENCIES = "META-INF/shaded.dependencies";
056
057    @CompileTimeCheck
058    private static final Pattern POM_PROPERTIES =
059        Pattern.compile("(?i)^META-INF/maven/(?<g>[^/]+)/(?<a>[^/]+)/pom.properties$");
060
061    /**
062     * Get the Maven {@link Artifact}s the argument JAR includes.  Returns
063     * an empty {@link Set} if the {@link File} is not a JAR or if the
064     * analysis cannot identify any {@link Artifact}s.
065     *
066     * @param   file            The {@link File} ({@link JarFile}).
067     *
068     * @return  The {@link Set} of implemented {@link Artifact}s.
069     */
070    public Set<Artifact> getShadedArtifactSet(File file) {
071        var set = new LinkedHashSet<Artifact>();
072
073        if (! file.isDirectory()) {
074            try (var jar = new JarFile(file)) {
075                var manifest = jar.getManifest();
076                var entry = jar.getJarEntry(SHADED_DEPENDENCIES);
077
078                if (entry != null) {
079                    try (var in = jar.getInputStream(entry)) {
080                        parse(in)
081                            .map(t -> t.setFile(file))
082                            .forEach(set::add);
083                    }
084                } else {
085                    var attributes = (manifest != null) ? manifest.getMainAttributes() : null;
086                    var list =
087                        jar.stream()
088                        .map(JarEntry::getName)
089                        .map(POM_PROPERTIES::matcher)
090                        .filter(Matcher::matches)
091                        .collect(toList());
092
093                    for (var matcher : list) {
094                        try (var in = jar.getInputStream(jar.getJarEntry(matcher.group()))) {
095                            var properties = new Properties();
096
097                            properties.load(in);
098
099                            if (Objects.equals(matcher.group("g"), properties.get("groupId"))
100                                && Objects.equals(matcher.group("a"), properties.get("artifactId"))) {
101                                var artifact =
102                                    artifact(matcher.group("g"), matcher.group("a"),
103                                             properties.getProperty("version"), file);
104
105                                set.add(artifact);
106                            } else {
107                                log.warn("{} does not specify {}:{}",
108                                         jar.getName() + "!/" + matcher.group(),
109                                         matcher.group("g"), matcher.group("a"));
110                            }
111                        } catch (IOException exception) {
112                            log.warn("{}", jar.getName() + "!/" + matcher.group(), exception);
113                        }
114                    }
115
116                    if (set.isEmpty()) {
117                        if (attributes != null) {
118                            var artifact = getArtifactFrom(attributes, file);
119
120                            if (artifact != null) {
121                                set.add(artifact);
122                            }
123                        }
124                    }
125                }
126            } catch (Exception exception) {
127                log.warn("{}", file, exception);
128            }
129        }
130
131        return set;
132    }
133
134    /**
135     * Method to convert an {@link InputStream} to a {@link Stream} of
136     * {@link DefaultArtifact}s.
137     *
138     * @param   in              The {@link InputStream}.
139     *
140     * @return  The {@link Stream} of {@link DefaultArtifact}s.
141     */
142    protected Stream<DefaultArtifact> parse(InputStream in) {
143        var stream =
144            new BufferedReader(new InputStreamReader(in, UTF_8))
145            .lines()
146            .map(String::strip)
147            .map(t -> t.split(":"))
148            .filter(t -> t.length > 3)
149            .map(t -> new DefaultArtifact(t[0], t[1], null, t[2], t[3]));
150
151        return stream;
152    }
153
154    private static final List<List<String>> CANDIDATES =
155        List.of(List.of("Bundle-SymbolicName", "Bundle-Name", "Bundle-Version"),
156                List.of("Implementation-Vendor-Id", "Implementation-Title", "Implementation-Version"),
157                List.of("Implementation-Vendor", "Implementation-Title", "Implementation-Version"));
158
159    private DefaultArtifact getArtifactFrom(Attributes attributes, File file) {
160        var coordinates =
161            CANDIDATES.stream()
162            .map(t -> t.stream()
163                       .map(attributes::getValue)
164                       .filter(Objects::nonNull)
165                       .map(Object::toString)
166                       .collect(toList()))
167            .filter(t -> t.size() == 3)
168            .findFirst().orElse(null);
169
170        return (coordinates != null) ? artifact(coordinates, file) : null;
171    }
172
173    private DefaultArtifact artifact(String g, String a, String v, File file) {
174        return new DefaultArtifact(g, a, null, "jar", v, Map.of(), file);
175    }
176
177    private DefaultArtifact artifact(List<String> coordinates, File file) {
178        return artifact(coordinates.get(0), coordinates.get(1), coordinates.get(2), file);
179    }
180}
