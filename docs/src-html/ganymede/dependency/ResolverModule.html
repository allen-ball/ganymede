021import com.google.inject.AbstractModule;
022import com.google.inject.Provides;
023import com.google.inject.name.Names;
024import java.util.Map;
025import java.util.Set;
026import javax.inject.Named;
027import javax.inject.Singleton;
028import lombok.NoArgsConstructor;
029import lombok.ToString;
030import lombok.extern.log4j.Log4j2;
031import org.apache.maven.model.building.DefaultModelBuilderFactory;
032import org.apache.maven.model.building.ModelBuilder;
033import org.apache.maven.repository.internal.DefaultArtifactDescriptorReader;
034import org.apache.maven.repository.internal.DefaultVersionRangeResolver;
035import org.apache.maven.repository.internal.DefaultVersionResolver;
036import org.apache.maven.repository.internal.SnapshotMetadataGeneratorFactory;
037import org.apache.maven.repository.internal.VersionsMetadataGeneratorFactory;
038import org.eclipse.aether.connector.basic.BasicRepositoryConnectorFactory;
039import org.eclipse.aether.impl.ArtifactDescriptorReader;
040import org.eclipse.aether.impl.MetadataGeneratorFactory;
041import org.eclipse.aether.impl.VersionRangeResolver;
042import org.eclipse.aether.impl.VersionResolver;
043import org.eclipse.aether.impl.guice.AetherModule;
044import org.eclipse.aether.impl.guice.AetherModule;
045import org.eclipse.aether.spi.connector.RepositoryConnectorFactory;
046import org.eclipse.aether.spi.connector.transport.TransporterFactory;
047import org.eclipse.aether.transport.file.FileTransporterFactory;
048import org.eclipse.aether.transport.http.ChecksumExtractor;
049import org.eclipse.aether.transport.http.HttpTransporterFactory;
050
051/**
052 * {@link com.google.inject.Guice} module for {@link Resolver}.
053 *
054 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
055 */
056@NoArgsConstructor @ToString @Log4j2
057public class ResolverModule extends AbstractModule {
058    @Override
059    protected void configure() {
060        install(new AetherModule());
061
062        bind(ArtifactDescriptorReader.class)
063            .to(DefaultArtifactDescriptorReader.class)
064            .in(Singleton.class);
065        bind(VersionResolver.class)
066            .to(DefaultVersionResolver.class)
067            .in(Singleton.class);
068        bind(VersionRangeResolver.class)
069            .to(DefaultVersionRangeResolver.class)
070            .in(Singleton.class);
071        bind(MetadataGeneratorFactory.class)
072            .annotatedWith(Names.named("snapshot"))
073            .to(SnapshotMetadataGeneratorFactory.class)
074            .in(Singleton.class);
075        bind(MetadataGeneratorFactory.class)
076            .annotatedWith(Names.named("versions"))
077            .to(VersionsMetadataGeneratorFactory.class)
078            .in(Singleton.class);
079        bind(RepositoryConnectorFactory.class)
080            .annotatedWith(Names.named("basic"))
081            .to(BasicRepositoryConnectorFactory.class);
082        bind(TransporterFactory.class)
083            .annotatedWith(Names.named("file"))
084            .to(FileTransporterFactory.class);
085        bind(TransporterFactory.class)
086            .annotatedWith(Names.named("http"))
087            .to(HttpTransporterFactory.class);
088    }
089
090    @Provides @Singleton
091    public Map<String,ChecksumExtractor> provideChecksumExtractors() { return Map.of(); }
092
093    @Provides @Singleton
094    public Set<RepositoryConnectorFactory> provideRepositoryConnectorFactories(@Named("basic") RepositoryConnectorFactory basic)
095    {
096        return Set.of(basic);
097    }
098
099    @Provides @Singleton
100    public Set<TransporterFactory> provideTransporterFactories(@Named("file") TransporterFactory file,
101                                                               @Named("http") TransporterFactory http) {
102        return Set.of(file, http);
103    }
104
105    @Provides @Singleton
106    public Set<MetadataGeneratorFactory> provideMetadataGeneratorFactories(@Named("snapshot") MetadataGeneratorFactory snapshot,
107                                                                           @Named("versions") MetadataGeneratorFactory versions) {
108        return Set.of(snapshot, versions);
109    }
110
111    @Provides
112    public ModelBuilder provideModelBuilder() { return new DefaultModelBuilderFactory().newInstance(); }
113}
