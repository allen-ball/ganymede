021import com.fasterxml.jackson.databind.JsonNode;
022import java.io.IOException;
023import java.nio.file.Files;
024import java.nio.file.Paths;
025import java.util.LinkedHashMap;
026import java.util.List;
027import java.util.function.Function;
028import java.util.stream.Stream;
029import lombok.NoArgsConstructor;
030import lombok.ToString;
031import lombok.extern.log4j.Log4j2;
032import org.springframework.beans.factory.annotation.Value;
033import org.springframework.boot.ApplicationArguments;
034import org.springframework.boot.ApplicationRunner;
035import org.springframework.boot.autoconfigure.SpringBootApplication;
036import org.springframework.boot.system.ApplicationHome;
037
038import static ganymede.server.Server.OBJECT_MAPPER;
039import static java.nio.charset.StandardCharsets.UTF_8;
040import static java.util.stream.Collectors.joining;
041import static org.springframework.util.FileCopyUtils.copy;
042import static org.springframework.util.FileCopyUtils.copyToByteArray;
043import static org.springframework.util.FileSystemUtils.deleteRecursively;
044
045/**
046 * Ganymede Jupyter {@link ganymede.kernel.Kernel} {@link Install}er.
047 *
048 * {@injected.fields}
049 *
050 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
051 */
052@SpringBootApplication
053@NoArgsConstructor @ToString @Log4j2
054public class Install implements ApplicationRunner {
055    @Value("${id-prefix:}")
056    private String id_prefix = null;
057    @Value("${id:ganymede-${project.version}-java-${java.specification.version}}")
058    private String id = null;
059    @Value("${id-suffix:}")
060    private String id_suffix = null;
061    @Value("${display-name-prefix:}")
062    private String display_name_prefix = null;
063    @Value("${display-name:Ganymede ${project.version} (Java ${java.specification.version})}")
064    private String display_name = null;
065    @Value("${display-name-suffix:}")
066    private String display_name_suffix = null;
067    @Value("${env:}")
068    private List<String> envvars = null;
069    @Value("${copy-jar:true}")
070    private boolean copy_jar = true;
071
072    @Override
073    public void run(ApplicationArguments arguments) throws Exception {
074        id =
075            Stream.of(id_prefix, id, id_suffix)
076            .map(String::strip)
077            .filter(t -> (! t.isEmpty()))
078            .collect(joining("-"));
079
080        display_name =
081            Stream.of(display_name_prefix, display_name, display_name_suffix)
082            .map(String::strip)
083            .filter(t -> (! t.isEmpty()))
084            .collect(joining(" "));
085
086        var sysPrefix = false;
087
088        for (var argument : arguments.getSourceArgs()) {
089            if (argument.equals("--sys-prefix")) {
090                sysPrefix = true;
091            } else if (argument.equals("--user")) {
092                sysPrefix = false;
093            }
094        }
095
096        var sysProperties = new LinkedHashMap<String,Object>();
097        var tmp = Files.createTempDirectory(getClass().getPackage().getName() + "-");
098
099        try {
100            /*
101             * java
102             */
103            var java = ProcessHandle.current().info().command().orElse("java");
104            /*
105             * kernel.jar
106             */
107            var jarPath = new ApplicationHome(getClass()).getSource().toPath();
108            /*
109             * which python
110             */
111            var python = which("python");
112            /*
113             * which jupyter
114             */
115            var jupyter = which("jupyter");
116            /*
117             * jupyter --paths --json
118             */
119            var paths = getOutputAsJson(jupyter, "--paths", "--json");
120            /*
121             * jupyter --runtime-dir
122             */
123            var runtime_dir = getOutputAsString(jupyter, "--runtime-dir");
124            /*
125             * Maven local repository
126             */
127            if (sysPrefix) {
128                var repository =
129                    Paths.get(paths.at("/data").get(1).asText(), "repository");
130
131                sysProperties.put("maven.repo.local", repository);
132
133                try {
134                    Files.createDirectories(repository);
135                } catch (Exception exception) {
136                    log.warn("{}: Could not create", repository, exception);
137                }
138            }
139            /*
140             * kernelspec
141             */
142            var kernelspec = tmp.resolve(id);
143
144            Files.createDirectory(kernelspec);
145            /*
146             * kernel.json
147             */
148            var kernel = OBJECT_MAPPER.createObjectNode();
149            var argv = kernel.withArray("argv");
150            var jar = jarPath.toAbsolutePath().toString();
151
152            if (copy_jar) {
153                var name = "kernel.jar";
154
155                copy(jarPath.toFile(), kernelspec.resolve(name).toFile());
156
157                var prefix = paths.at("/data").get(sysPrefix ? 1 : 0).asText();
158
159                jar = Paths.get(prefix, "kernels", id, name).toString();
160            }
161
162            Stream.of(Stream.of(java, "--illegal-access=permit",
163                                "--add-opens", "java.base/jdk.internal.misc=ALL-UNNAMED"),
164                      sysProperties.entrySet().stream().map(t -> "-D" + t),
165                      Stream.of("-jar", jar,
166                                "--runtime-dir=" + runtime_dir,
167                                "--connection-file={connection_file}"))
168                .flatMap(Function.identity())
169                .map(Object::toString)
170                .forEach(argv::add);
171
172            kernel.put("display_name", display_name);
173
174            var env = kernel.with("env");
175
176            for (var envvar : envvars) {
177                var pair = envvar.split("=", 2);
178
179                env.put(pair[0], (pair.length > 1) ? pair[1] : "");
180            }
181
182            kernel.put("interrupt_mode", "message");
183            kernel.put("language", "java");
184
185            OBJECT_MAPPER.writeValue(kernelspec.resolve("kernel.json").toFile(), kernel);
186            /*
187             * logo-16x16.png and logo-32x32.png
188             */
189            for (var png : List.of("16x16.png", "32x32.png")) {
190                var from = "/static/images/ball-java-jar-" + png;
191                var to = kernelspec.resolve("logo-" + png).toFile();
192
193                try (var in = getClass().getResourceAsStream(from)) {
194                    copy(copyToByteArray(in), to);
195                } catch (Exception exception) {
196                    log.warn("Could not copy resource {} to {}",
197                             from, to, exception);
198                }
199            }
200            /*
201             * jupyter kernelspec install <kernelspec>
202             */
203            new ProcessBuilder(jupyter, "kernelspec", "install", kernelspec.toString(),
204                               sysPrefix ? "--sys-prefix" : "--user",
205                               "--replace")
206                .inheritIO()
207                .start().waitFor();
208        } catch (Exception exception) {
209            log.fatal("{}", exception.getMessage(), exception);
210        } finally {
211            deleteRecursively(tmp);
212        }
213    }
214
215    private String which(String command) throws Exception {
216        return getOutputAsString("which", command);
217    }
218
219    private String getOutputAsString(String... argv) throws Exception {
220        var bytes = new byte[] { };
221        var process =
222            new ProcessBuilder(argv)
223            .inheritIO()
224            .redirectOutput(ProcessBuilder.Redirect.PIPE)
225            .redirectError(ProcessBuilder.Redirect.DISCARD)
226            .start();
227
228        try (var in = process.getInputStream()) {
229            bytes = in.readAllBytes();
230
231            var status = process.waitFor();
232
233            if (status != 0) {
234                throw new IOException("Cannot read output of " + List.of(argv));
235            }
236        }
237
238        return new String(bytes, UTF_8).trim();
239    }
240
241    private JsonNode getOutputAsJson(String... argv) throws Exception {
242        var node = OBJECT_MAPPER.nullNode();
243        var process =
244            new ProcessBuilder(argv)
245            .inheritIO()
246            .redirectOutput(ProcessBuilder.Redirect.PIPE)
247            .redirectError(ProcessBuilder.Redirect.DISCARD)
248            .start();
249
250        try (var in = process.getInputStream()) {
251            node = OBJECT_MAPPER.readTree(in);
252
253            var status = process.waitFor();
254
255            if (status != 0) {
256                throw new IOException("Cannot read output of " + List.of(argv));
257            }
258        }
259
260        return node;
261    }
