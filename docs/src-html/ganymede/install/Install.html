021import com.fasterxml.jackson.databind.JsonNode;
022import java.io.IOException;
023import java.nio.file.Files;
024import java.nio.file.Paths;
025import java.util.LinkedHashMap;
026import java.util.List;
027import java.util.stream.Stream;
028import lombok.NoArgsConstructor;
029import lombok.ToString;
030import lombok.extern.log4j.Log4j2;
031import org.springframework.beans.factory.annotation.Value;
032import org.springframework.boot.ApplicationArguments;
033import org.springframework.boot.ApplicationRunner;
034import org.springframework.boot.autoconfigure.SpringBootApplication;
035import org.springframework.boot.system.ApplicationHome;
036
037import static ganymede.server.Server.OBJECT_MAPPER;
038import static java.nio.charset.StandardCharsets.UTF_8;
039import static java.util.stream.Collectors.joining;
040import static org.springframework.util.FileCopyUtils.copy;
041import static org.springframework.util.FileCopyUtils.copyToByteArray;
042import static org.springframework.util.FileSystemUtils.deleteRecursively;
043
044/**
045 * Ganymede Jupyter {@link ganymede.kernel.Kernel} {@link Install}er.
046 *
047 * {@injected.fields}
048 *
049 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
050 */
051@SpringBootApplication
052@NoArgsConstructor @ToString @Log4j2
053public class Install implements ApplicationRunner {
054    @Value("${id-prefix:}")
055    private String id_prefix = null;
056    @Value("${id:ganymede-${project.version}-java-${java.specification.version}}")
057    private String id = null;
058    @Value("${id-suffix:}")
059    private String id_suffix = null;
060    @Value("${display-name-prefix:}")
061    private String display_name_prefix = null;
062    @Value("${display-name:Ganymede ${project.version} (Java ${java.specification.version})}")
063    private String display_name = null;
064    @Value("${display-name-suffix:}")
065    private String display_name_suffix = null;
066    @Value("${env:}")
067    private List<String> envvars = null;
068    @Value("${copy-jar:true}")
069    private boolean copy_jar = true;
070
071    @Override
072    public void run(ApplicationArguments arguments) throws Exception {
073        id =
074            Stream.of(id_prefix, id, id_suffix)
075            .map(String::strip)
076            .filter(t -> (! t.isEmpty()))
077            .collect(joining("-"));
078
079        display_name =
080            Stream.of(display_name_prefix, display_name, display_name_suffix)
081            .map(String::strip)
082            .filter(t -> (! t.isEmpty()))
083            .collect(joining(" "));
084
085        var sysPrefix = false;
086
087        for (var argument : arguments.getSourceArgs()) {
088            if (argument.equals("--sys-prefix")) {
089                sysPrefix = true;
090            } else if (argument.equals("--user")) {
091                sysPrefix = false;
092            }
093        }
094
095        var sysProperties = new LinkedHashMap<String,Object>();
096        var tmp = Files.createTempDirectory(getClass().getPackage().getName() + "-");
097
098        try {
099            /*
100             * java
101             */
102            var java = ProcessHandle.current().info().command().orElse("java");
103            /*
104             * kernel.jar
105             */
106            var jarPath = new ApplicationHome(getClass()).getSource().toPath();
107            /*
108             * which python
109             */
110            var python = which("python");
111            /*
112             * which jupyter
113             */
114            var jupyter = which("jupyter");
115            /*
116             * jupyter --paths --json
117             */
118            var paths = getOutputAsJson(jupyter, "--paths", "--json");
119            /*
120             * jupyter --runtime-dir
121             */
122            var runtime_dir = getOutputAsString(jupyter, "--runtime-dir");
123            /*
124             * Maven local repository
125             */
126            if (sysPrefix) {
127                var repository =
128                    Paths.get(paths.at("/data").get(1).asText(), "repository");
129
130                sysProperties.put("maven.repo.local", repository);
131
132                try {
133                    Files.createDirectories(repository);
134                } catch (Exception exception) {
135                    log.warn("{}: Could not create", repository, exception);
136                }
137            }
138            /*
139             * kernelspec
140             */
141            var kernelspec = tmp.resolve(id);
142
143            Files.createDirectory(kernelspec);
144            /*
145             * kernel.json
146             */
147            var kernel = OBJECT_MAPPER.createObjectNode();
148            var argv = kernel.withArray("argv");
149            var jar = jarPath.toAbsolutePath().toString();
150
151            if (copy_jar) {
152                var name = "kernel.jar";
153
154                copy(jarPath.toFile(), kernelspec.resolve(name).toFile());
155
156                var prefix = paths.at("/data").get(sysPrefix ? 1 : 0).asText();
157
158                jar = Paths.get(prefix, "kernels", id, name).toString();
159            }
160
161            argv.add(java);
162            sysProperties.entrySet().stream()
163                .map(t -> "-D" + t)
164                .forEach(argv::add);
165            Stream.of("-jar", jar,
166                      "--runtime-dir=" + runtime_dir,
167                      "--connection-file={connection_file}")
168                .map(Object::toString)
169                .forEach(argv::add);
170
171            kernel.put("display_name", display_name);
172
173            var env = kernel.with("env");
174
175            for (var envvar : envvars) {
176                var pair = envvar.split("=", 2);
177
178                env.put(pair[0], (pair.length > 1) ? pair[1] : "");
179            }
180
181            kernel.put("interrupt_mode", "message");
182            kernel.put("language", "java");
183
184            OBJECT_MAPPER.writeValue(kernelspec.resolve("kernel.json").toFile(), kernel);
185            /*
186             * logo-16x16.png and logo-32x32.png
187             */
188            for (var png : List.of("16x16.png", "32x32.png")) {
189                var from = "/static/images/ball-java-jar-" + png;
190                var to = kernelspec.resolve("logo-" + png).toFile();
191
192                try (var in = getClass().getResourceAsStream(from)) {
193                    copy(copyToByteArray(in), to);
194                } catch (Exception exception) {
195                    log.warn("Could not copy resource {} to {}",
196                             from, to, exception);
197                }
198            }
199            /*
200             * jupyter kernelspec install <kernelspec>
201             */
202            new ProcessBuilder(jupyter, "kernelspec", "install", kernelspec.toString(),
203                               sysPrefix ? "--sys-prefix" : "--user",
204                               "--replace")
205                .inheritIO()
206                .start().waitFor();
207        } catch (Exception exception) {
208            log.fatal("{}", exception.getMessage(), exception);
209        } finally {
210            deleteRecursively(tmp);
211        }
212    }
213
214    private String which(String command) throws Exception {
215        return getOutputAsString("which", command);
216    }
217
218    private String getOutputAsString(String... argv) throws Exception {
219        var bytes = new byte[] { };
220        var process =
221            new ProcessBuilder(argv)
222            .inheritIO()
223            .redirectOutput(ProcessBuilder.Redirect.PIPE)
224            .redirectError(ProcessBuilder.Redirect.DISCARD)
225            .start();
226
227        try (var in = process.getInputStream()) {
228            bytes = in.readAllBytes();
229
230            var status = process.waitFor();
231
232            if (status != 0) {
233                throw new IOException("Cannot read output of " + List.of(argv));
234            }
235        }
236
237        return new String(bytes, UTF_8).trim();
238    }
239
240    private JsonNode getOutputAsJson(String... argv) throws Exception {
241        var node = OBJECT_MAPPER.nullNode();
242        var process =
243            new ProcessBuilder(argv)
244            .inheritIO()
245            .redirectOutput(ProcessBuilder.Redirect.PIPE)
246            .redirectError(ProcessBuilder.Redirect.DISCARD)
247            .start();
248
249        try (var in = process.getInputStream()) {
250            node = OBJECT_MAPPER.readTree(in);
251
252            var status = process.waitFor();
253
254            if (status != 0) {
255                throw new IOException("Cannot read output of " + List.of(argv));
256            }
257        }
258
259        return node;
260    }
261}
