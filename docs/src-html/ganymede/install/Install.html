021import com.fasterxml.jackson.databind.JsonNode;
022import com.fasterxml.jackson.databind.node.JsonNodeFactory;
023import com.fasterxml.jackson.databind.node.ObjectNode;

026import java.io.IOException;
027import java.nio.file.Files;
028import java.nio.file.Paths;
029import java.util.LinkedHashMap;
030import java.util.List;
031import java.util.function.Function;
032import java.util.stream.Stream;

034import lombok.NoArgsConstructor;
035import lombok.ToString;

046import static java.nio.charset.StandardCharsets.UTF_8;
047import static java.util.stream.Collectors.joining;
048import static org.springframework.util.FileCopyUtils.copy;
049import static org.springframework.util.FileCopyUtils.copyToByteArray;
050import static org.springframework.util.FileSystemUtils.deleteRecursively;

052/**
053 * Ganymede Jupyter {@link ganymede.kernel.Kernel} {@link Install}er.
054 *
055 * {@injected.fields}
056 *
057 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
058 */
059@SpringBootApplication
060@Command
061@NoArgsConstructor @ToString @Log4j2
062public class Install implements ApplicationRunner {
063    @Option(description = { "Install Ganymede kernel" }, names = { "-i" })
064    @Value("${install:#{null}}")
065    private Boolean install = null;
066
067    @Value("${id-prefix}")
068    private String id_prefix = null;
069
070    @Value("${id}")
071    private String id = null;
072
073    @Value("${id-suffix}")
074    private String id_suffix = null;
075
076    @Value("${display-name-prefix}")
077    private String display_name_prefix = null;
078
079    @Value("${display-name}")
080    private String display_name = null;
081
082    @Value("${display-name-suffix}")
083    private String display_name_suffix = null;
084
085    @Value("${env:}")
086    private List<String> envvars = null;
087
088    @Value("${copy-jar:true}")
089    private boolean copy_jar = true;
090
091    @Override
092    public void run(ApplicationArguments arguments) throws Exception {
093        new CommandLine(this)
094            .setCaseInsensitiveEnumValuesAllowed(true)
095            .parseArgs(arguments.getNonOptionArgs().toArray(new String [] { }));
096
097        id =
098            Stream.of(id_prefix, id, id_suffix)
099            .map(String::strip)
100            .filter(t -> (! t.isEmpty()))
101            .collect(joining("-"));
102
103        display_name =
104            Stream.of(display_name_prefix, display_name, display_name_suffix)
105            .map(String::strip)
106            .filter(t -> (! t.isEmpty()))
107            .collect(joining(" "));
108
109        var sys_prefix = false;
110
111        for (var argument : arguments.getSourceArgs()) {
112            if (argument.equals("--sys-prefix")) {
113                sys_prefix = true;
114            } else if (argument.equals("--user")) {
115                sys_prefix = false;
116            }
117        }
118
119        var sysProperties = new LinkedHashMap<String,Object>();
120        var tmp = Files.createTempDirectory(getClass().getPackage().getName() + "-");
121
122        try {
123            /*
124             * java
125             */
126            var java = ProcessHandle.current().info().command().orElse("java");
127            /*
128             * kernel.jar
129             */
130            var jarPath = new ApplicationHome(getClass()).getSource().toPath();
131            /*
132             * which python
133             */
134            var python = which("python");
135            /*
136             * python -c 'import sys; print(sys.prefix)'
137             */
138            var sys_prefix_dir = getOutputAsString(python, "-c", "import sys; print(sys.prefix)");
139            /*
140             * which jupyter
141             */
142            var jupyter = which("jupyter");
143            /*
144             * jupyter --paths --json
145             */
146            var paths = getOutputAsJson(jupyter, "--paths", "--json");
147            var config_path =
148                StreamSupport.stream(paths.get("config").spliterator(), false)
149                .map(JsonNode::asText)
150                .collect(joining(File.pathSeparator));
151            /*
152             * jupyter --config-dir
153             */
154            var config_dir = getOutputAsString(jupyter, "--config-dir");
155            /*
156             * jupyter --data-dir
157             */
158            var data_dir = getOutputAsString(jupyter, "--data-dir");
159            /*
160             * jupyter --runtime-dir
161             */
162            var runtime_dir = getOutputAsString(jupyter, "--runtime-dir");
163            /*
164             * Maven local repository
165             */
166            if (sys_prefix) {
167                var repository = Paths.get(data_dir, "repository");
168
169                sysProperties.put("maven.repo.local", repository);
170
171                try {
172                    Files.createDirectories(repository);
173                } catch (Exception exception) {
174                    log.warn("{}: Could not create", repository, exception);
175                }
176            }
177            /*
178             * kernelspec
179             */
180            var kernelspec = tmp.resolve(id);
181
182            Files.createDirectory(kernelspec);
183            /*
184             * kernel.json
185             */
186            var kernel = new ObjectNode(JsonNodeFactory.instance);
187            var argv = kernel.withArray("argv");
188            var jar = jarPath.toAbsolutePath().toString();
189
190            if (copy_jar) {
191                var name = "kernel.jar";
192
193                copy(jarPath.toFile(), kernelspec.resolve(name).toFile());
194
195                var target = Paths.get(data_dir, "kernels").toString();
196
197                if (sys_prefix) {
198                    target = Paths.get(sys_prefix_dir, "share/jupyter/kernels").toString();
199                }
200
201                jar = Paths.get(target, id, name).toString();
202            }
203
204            Stream.of(Stream.of(java,
205                                "--add-opens", "java.base/jdk.internal.misc=ALL-UNNAMED",
206                                "--illegal-access=permit",
207                                "-Djava.awt.headless=true",
208                                "-Djdk.disableLastUsageTracking=true"),
209                      sysProperties.entrySet().stream().map(t -> "-D" + t),
210                      Stream.of("-jar", jar, "-f", "{connection_file}"))
211                .flatMap(Function.identity())
212                .map(Object::toString)
213                .forEach(argv::add);
214
215            kernel.put("display_name", display_name);
216
217            var env = kernel.with("env");
218
219            env.put("JUPYTER_CONFIG_DIR", config_dir);
220            env.put("JUPYTER_CONFIG_PATH", config_path);
221            env.put("JUPYTER_DATA_DIR", data_dir);
222            env.put("JUPYTER_RUNTIME_DIR", runtime_dir);
223
224            for (var envvar : envvars) {
225                var pair = envvar.split("=", 2);
226
227                env.put(pair[0], (pair.length > 1) ? pair[1] : "");
228            }
229
230            kernel.put("interrupt_mode", "message");
231            kernel.put("language", "java");
232
233            ObjectMappers.JSON.writeValue(kernelspec.resolve("kernel.json").toFile(), kernel);
234            /*
235             * logo-16x16.png and logo-32x32.png
236             */
237            for (var png : List.of("16x16.png", "32x32.png")) {
238                var from = "/static/images/ball-java-jar-" + png;
239                var to = kernelspec.resolve("logo-" + png).toFile();
240
241                try (var in = getClass().getResourceAsStream(from)) {
242                    copy(copyToByteArray(in), to);
243                } catch (Exception exception) {
244                    log.warn("Could not copy resource {} to {}", from, to, exception);
245                }
246            }
247            /*
248             * jupyter kernelspec install <kernelspec>
249             */
250            new ProcessBuilder(jupyter, "kernelspec", "install", kernelspec.toString(),
251                               sys_prefix ? "--sys-prefix" : "--user",
252                               "--replace")
253                .inheritIO()
254                .start().waitFor();
255        } catch (Exception exception) {
256            log.fatal("{}", exception.getMessage(), exception);
257        } finally {
258            deleteRecursively(tmp);
259        }
260    }
261
262    private String which(String command) throws Exception {
263        return getOutputAsString("which", command);
264    }
265
266    private String getOutputAsString(String... argv) throws Exception {
267        var bytes = new byte[] { };
268        var process =
269            new ProcessBuilder(argv)
270            .inheritIO()
271            .redirectOutput(ProcessBuilder.Redirect.PIPE)
272            .redirectError(ProcessBuilder.Redirect.DISCARD)
273            .start();
274
275        try (var in = process.getInputStream()) {
276            bytes = in.readAllBytes();
277
278            var status = process.waitFor();
279
280            if (status != 0) {
281                throw new IOException("Cannot read output of " + List.of(argv));
282            }
283        }
284
285        return new String(bytes, UTF_8).trim();
286    }
287
288    private JsonNode getOutputAsJson(String... argv) throws Exception {
289        var node = ObjectMappers.JSON.nullNode();
290        var process =
291            new ProcessBuilder(argv)
292            .inheritIO()
293            .redirectOutput(ProcessBuilder.Redirect.PIPE)
294            .redirectError(ProcessBuilder.Redirect.DISCARD)
295            .start();
296
297        try (var in = process.getInputStream()) {
298            node = ObjectMappers.JSON.readTree(in);
299
300            var status = process.waitFor();
301
302            if (status != 0) {
303                throw new IOException("Cannot read output of " + List.of(argv));
304            }
305        }
306
307        return node;
308    }
