021import com.fasterxml.jackson.databind.JsonNode;
022import ganymede.util.ObjectMappers;
023import java.io.File;
024import java.io.IOException;
025import java.net.URI;
026import java.net.http.HttpClient;
027import java.net.http.HttpRequest;
028import java.net.http.HttpResponse;
029import lombok.ToString;
030import lombok.extern.log4j.Log4j2;
031
032/**
033 * Ganymede Jupyter Notebook/Lab REST client.  See
034 * {@link.uri https://github.com/jupyter/jupyter/wiki/Jupyter-Notebook-Server-API target=newtab Jupyter Notebook Server API}.
035 *
036 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
037 */
038@Log4j2
039public class JupyterRestClient {
040    private final JsonNode json;
041    private final URI uri;
042    private final String token;
043    private final HttpClient client = HttpClient.newHttpClient();
044
045    /**
046     * {@link File} constructor.
047     *
048     * @param   file            The {@link File} containing the server
049     *                          parameters.
050     *
051     * @throws  IOException     If the {@link File} cannot be opened or
052     *                          parsed.
053     */
054    public JupyterRestClient(File file) throws IOException {
055        this(ObjectMappers.JSON.readTree(file));
056    }
057
058    /**
059     * {@link JsonNode} constructor.
060     *
061     * @param   json            The {@link JsonNode} describing the server
062     *                          parameters.
063     */
064    public JupyterRestClient(JsonNode json) {
065        this.json = json;
066        this.uri = URI.create(json.get("url").asText());
067        this.token = json.get("token").asText();
068    }
069
070    /**
071     * {@code GET /api/status}
072     */
073    public JsonNode getStatus() throws Exception {
074        var request = builder().uri(uri.resolve("api/status")).GET().build();
075        var response =
076            client.send(request, HttpResponse.BodyHandlers.ofString());
077
078        return ObjectMappers.JSON.readTree(response.body());
079    }
080
081    /**
082     * {@code GET /api/kernels}
083     */
084    public JsonNode getKernels() throws Exception {
085        var request =
086            builder()
087            .uri(uri.resolve("api/kernels"))
088            .GET()
089            .build();
090        var response =
091            client.send(request, HttpResponse.BodyHandlers.ofString());
092
093        return ObjectMappers.JSON.readTree(response.body());
094    }
095
096    /**
097     * {@code GET /api/kernels/(id)}
098     */
099    public JsonNode getKernel(String id) throws Exception {
100        var request =
101            builder()
102            .uri(uri.resolve("api/kernels/").resolve(id))
103            .GET()
104            .build();
105        var response =
106            client.send(request, HttpResponse.BodyHandlers.ofString());
107
108        return ObjectMappers.JSON.readTree(response.body());
109    }
110
111    /**
112     * {@code GET /api/sessions}
113     */
114    public JsonNode getSessions() throws Exception {
115        var request =
116            builder()
117            .uri(uri.resolve("api/sessions"))
118            .GET()
119            .build();
120        var response =
121            client.send(request, HttpResponse.BodyHandlers.ofString());
122
123        return ObjectMappers.JSON.readTree(response.body());
124    }
125
126    /**
127     * {@code GET /api/sessions/(id)}
128     */
129    public JsonNode getSession(String id) throws Exception {
130        var request =
131            builder()
132            .uri(uri.resolve("api/sessions/").resolve(id))
133            .GET()
134            .build();
135        var response =
136            client.send(request, HttpResponse.BodyHandlers.ofString());
137
138        return ObjectMappers.JSON.readTree(response.body());
139    }
140
141    /**
142     * {@code GET /api/terminals}
143     */
144    public JsonNode getTerminals() throws Exception {
145        var request =
146            builder()
147            .uri(uri.resolve("api/terminals"))
148            .GET()
149            .build();
150        var response =
151            client.send(request, HttpResponse.BodyHandlers.ofString());
152
153        return ObjectMappers.JSON.readTree(response.body());
154    }
155
156    /**
157     * {@code GET /api/terminals/(id)}
158     */
159    public JsonNode getTerminal(String id) throws Exception {
160        var request =
161            builder()
162            .uri(uri.resolve("api/terminals/").resolve(id))
163            .GET()
164            .build();
165        var response =
166            client.send(request, HttpResponse.BodyHandlers.ofString());
167
168        return ObjectMappers.JSON.readTree(response.body());
169    }
170
171    private HttpRequest.Builder builder() {
172        var builder =
173            HttpRequest.newBuilder()
174            .header("Authorization", "Token " + token);
175
176        return builder;
177    }
178
179    @Override
180    public String toString() { return json.toPrettyString(); }
181}
