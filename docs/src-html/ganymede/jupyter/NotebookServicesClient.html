021import com.fasterxml.jackson.databind.JsonNode;
022import ganymede.jupyter.notebook.ApiClient;
023import ganymede.jupyter.notebook.ApiException;
024import ganymede.jupyter.notebook.JSON;
025import ganymede.jupyter.notebook.api.KernelsApi;
026import ganymede.jupyter.notebook.api.SessionsApi;
027import ganymede.jupyter.notebook.api.TerminalsApi;
028import ganymede.jupyter.notebook.model.Kernel;
029import ganymede.jupyter.notebook.model.Session;
030import ganymede.jupyter.notebook.model.Terminal;
031import java.io.File;
032import java.io.FileNotFoundException;
033import java.io.IOException;
034import java.net.URI;
035import java.nio.file.Files;
036import java.nio.file.Paths;
037import java.util.List;
038import java.util.NoSuchElementException;
039import java.util.UUID;
040import lombok.ToString;
041import lombok.extern.log4j.Log4j2;
042
043/**
044 * Ganymede Jupyter Notebook/Lab REST client.  See
045 * {@link.uri https://github.com/jupyter/jupyter/wiki/Jupyter-Notebook-Server-API target=newtab Jupyter Notebook Server API}.
046 *
047 * {@bean.info}
048 *
049 * @see ApiClient
050 *
051 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
052 */
053@ToString @Log4j2
054public class NotebookServicesClient extends ApiClient {
055    private static final String JUPYTER_RUNTIME_DIR = "JUPYTER_RUNTIME_DIR";
056    private static final String JPY_PARENT_PID = "JPY_PARENT_PID";
057    private static final String GLOB_FORMAT = "{nb,jp}server-%d.json";
058
059    private static File getFileFromEnv() throws IOException {
060        var directory = Paths.get(System.getenv().get("JUPYTER_RUNTIME_DIR"));
061        var pid = Long.parseLong(System.getenv().get("JPY_PARENT_PID"));
062        var glob = String.format(GLOB_FORMAT, pid);
063
064        try (var stream = Files.newDirectoryStream(directory, glob)) {
065            var path = stream.iterator().next();
066
067            return path.toFile();
068        } catch (NoSuchElementException exception) {
069            throw new FileNotFoundException(glob);
070        }
071    }
072
073    /**
074     * No-argument constructor.  {@link File} is specified by the
075     * {@code JUPYTER_RUNTIME_DIR} and {@code JPY_PARENT_PID} environment
076     * variables.
077     *
078     * @throws  IOException     If the {@link File} cannot be opened or
079     *                          parsed.
080     */
081    public NotebookServicesClient() throws IOException {
082        this(getFileFromEnv());
083    }
084
085    /**
086     * {@link File} constructor.
087     *
088     * @param   file            The {@link File} containing the server
089     *                          parameters.
090     *
091     * @throws  IOException     If the {@link File} cannot be opened or
092     *                          parsed.
093     */
094    public NotebookServicesClient(File file) throws IOException {
095        this(JSON.getDefault().getMapper().readTree(file));
096    }
097
098    /**
099     * {@link JsonNode} constructor.
100     *
101     * @param   node            The {@link JsonNode} describing the server
102     *                          parameters.
103     */
104    public NotebookServicesClient(JsonNode node) {
105        super();
106
107        var url = URI.create(node.get("url").asText());
108
109        setScheme(url.getScheme());
110        setHost(url.getHost());
111        setPort(url.getPort());
112
113        var token = node.get("token").asText();
114
115        setRequestInterceptor(t -> t.header("Authorization", "Token " + token));
116    }
117
118    /**
119     * See {@link KernelsApi#apiKernelsGet()}.
120     *
121     * @return  The {@link List} of {@link Kernel}s.
122     */
123    public List<Kernel> getKernelList() throws ApiException {
124        return new KernelsApi(this).apiKernelsGet();
125    }
126
127    /**
128     * See {@link SessionsApi#apiSessionsGet()}.
129     *
130     * @return  The {@link List} of {@link Session}s.
131     */
132    public List<Session> getSessionList() throws ApiException {
133        return new SessionsApi(this).apiSessionsGet();
134    }
135
136    /**
137     * See {@link TerminalsApi#apiTerminalsGet()}.
138     *
139     * @return  The {@link List} of {@link Terminal}s.
140     */
141    public List<Terminal> getTerminalList() throws ApiException {
142        return new TerminalsApi(this).apiTerminalsGet();
143    }
144}
