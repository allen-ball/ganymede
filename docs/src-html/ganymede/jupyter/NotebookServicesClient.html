046/**
047 * Ganymede Jupyter Notebook/Lab REST client.  See
048 * {@link.uri https://github.com/jupyter/jupyter/wiki/Jupyter-Notebook-Server-API target=newtab Jupyter Notebook Server API}.
049 *
050 * {@bean.info}
051 *
052 * @see ApiClient
053 *
054 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
055 */
056@ToString @Log4j2
057public class NotebookServicesClient extends ApiClient {
058    private static final String JUPYTER_RUNTIME_DIR = "JUPYTER_RUNTIME_DIR";
059    private static final String JPY_PARENT_PID = "JPY_PARENT_PID";
060    private static final String GLOB_FORMAT = "{nb,jp}server-%d.json";
061
062    private static File getFileFromEnv() throws IOException {
063        var directory = Paths.get(System.getenv().get("JUPYTER_RUNTIME_DIR"));
064        var pid = Long.parseLong(System.getenv().get("JPY_PARENT_PID"));
065        var glob = String.format(GLOB_FORMAT, pid);
066
067        try (var stream = Files.newDirectoryStream(directory, glob)) {
068            var path = stream.iterator().next();
069
070            return path.toFile();
071        } catch (NoSuchElementException exception) {
072            throw new FileNotFoundException(glob);
073        }
074    }
075
076    /**
077     * No-argument constructor.  {@link File} is specified by the
078     * {@code JUPYTER_RUNTIME_DIR} and {@code JPY_PARENT_PID} environment
079     * variables.
080     *
081     * @throws  IOException     If the {@link File} cannot be opened or
082     *                          parsed.
083     */
084    public NotebookServicesClient() throws IOException {
085        this(getFileFromEnv());
086    }
087
088    /**
089     * {@link File} constructor.
090     *
091     * @param   file            The {@link File} containing the server
092     *                          parameters.
093     *
094     * @throws  IOException     If the {@link File} cannot be opened or
095     *                          parsed.
096     */
097    public NotebookServicesClient(File file) throws IOException {
098        this(JSON.getDefault().getMapper().readTree(file));
099    }
100
101    /**
102     * {@link JsonNode} constructor.
103     *
104     * @param   node            The {@link JsonNode} describing the server
105     *                          parameters.
106     */
107    public NotebookServicesClient(JsonNode node) {
108        super();
109
110        if (node.has("url")) {
111            var url = URI.create(node.get("url").asText());
112
113            setScheme(url.getScheme());
114            setHost(url.getHost());
115            setPort(url.getPort());
116        }
117
118        if (node.has("token")) {
119            setRequestInterceptor(t -> t.header("Authorization", "Token " + node.get("token").asText()));
120        }
121    }
122
123    /**
124     * See {@link KernelsApi#apiKernelsGet()}.
125     *
126     * @return  The {@link List} of {@link Kernel}s.
127     */
128    public List<Kernel> getKernelList() throws ApiException {
129        return new KernelsApi(this).apiKernelsGet();
130    }
131
132    /**
133     * See {@link KernelsApi#apiKernelsKernelIdGet(UUID)}.
134     *
135     * @param   id              The kernel {@link UUID}.
136     *
137     * @return  The {@link Kernel} matching the argument ID.
138     */
139    public Kernel getKernel(UUID id) throws ApiException {
140        return new KernelsApi(this).apiKernelsKernelIdGet(id);
141    }
142
143    /**
144     * See {@link KernelspecsApi#apiKernelspecsGet()}.
145     *
146     * @return  The {@link Map} of {@link KernelSpec}s.
147     */
148    public Map<String,KernelSpec> getKernelSpecMap() throws ApiException {
149        return new KernelspecsApi(this).apiKernelspecsGet().getKernelspecs();
150    }
151
152    /**
153     * See {@link SessionsApi#apiSessionsGet()}.
154     *
155     * @return  The {@link List} of {@link Session}s.
156     */
157    public List<Session> getSessionList() throws ApiException {
158        return new SessionsApi(this).apiSessionsGet();
159    }
160
161    /**
162     * See {@link SessionsApi#apiSessionsSessionGet(UUID)}.
163     *
164     * @param   id              The session {@link UUID}.
165     *
166     * @return  The {@link Session} matching the argument ID.
167     */
168    public Session getSession(UUID id) throws ApiException {
169        return new SessionsApi(this).apiSessionsSessionGet(id);
170    }
171
172    /**
173     * See {@link TerminalsApi#apiTerminalsGet()}.
174     *
175     * @return  The {@link List} of {@link Terminal}s.
176     */
177    public List<Terminal> getTerminalList() throws ApiException {
178        return new TerminalsApi(this).apiTerminalsGet();
179    }
180
181    /**
182     * See {@link TerminalsApi#apiTerminalsTerminalIdGet(String)}.
183     *
184     * @param   id              The terminal ID.
185     *
186     * @return  The {@link Terminal} matching the argument ID.
187     */
188    public Terminal getTerminal(String id) throws ApiException {
189        return new TerminalsApi(this).apiTerminalsTerminalIdGet(id);
190    }
191}
