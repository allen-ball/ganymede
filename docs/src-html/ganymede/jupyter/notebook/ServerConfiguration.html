001package ganymede.jupyter.notebook;
002
003import java.util.Map;
004
005/**
006 * Representing a Server configuration.
007 */
008public class ServerConfiguration {
009  public String URL;
010  public String description;
011  public Map<String, ServerVariable> variables;
012
013  /**
014   * @param URL A URL to the target host.
015   * @param description A description of the host designated by the URL.
016   * @param variables A map between a variable name and its value. The value is
017   *     used for substitution in the server's URL template.
018   */
019  public ServerConfiguration(String URL, String description,
020                             Map<String, ServerVariable> variables) {
021    this.URL = URL;
022    this.description = description;
023    this.variables = variables;
024  }
025
026  /**
027   * Format URL template using given variables.
028   *
029   * @param variables A map between a variable name and its value.
030   * @return Formatted URL.
031   */
032  public String URL(Map<String, String> variables) {
033    String url = this.URL;
034
035    // go through variables and replace placeholders
036    for (Map.Entry<String, ServerVariable> variable :
037         this.variables.entrySet()) {
038      String name = variable.getKey();
039      ServerVariable serverVariable = variable.getValue();
040      String value = serverVariable.defaultValue;
041
042      if (variables != null && variables.containsKey(name)) {
043        value = variables.get(name);
044        if (serverVariable.enumValues.size() > 0 &&
045            !serverVariable.enumValues.contains(value)) {
046          throw new RuntimeException("The variable " + name +
047                                     " in the server URL has invalid value " +
048                                     value + ".");
049        }
050      }
051      url = url.replaceAll("\\{" + name + "\\}", value);
052    }
053    return url;
054  }
055
056  /**
057   * Format URL template using default server variables.
058   *
059   * @return Formatted URL.
060   */
061  public String URL() { return URL(null); }
062}
