021import com.fasterxml.jackson.databind.node.ObjectNode;
022import java.util.Base64;
023
024import static ganymede.server.Server.OBJECT_MAPPER;
025
026/**
027 * {@link Message#mime_bundle(Object,Object...)} output {@link Renderer}.
028 *
029 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
030 */
031public interface Renderer {
032    public static final String DATA = "data";
033    public static final String METADATA = "metadata";
034
035    /**
036     * {@link Base64.Encoder} instance.
037     */
038    public static final Base64.Encoder BASE64_ENCODER = Base64.getEncoder();
039
040    /**
041     * Method to render an {@link Object} to a {@code mime-bundle}.
042     *
043     * @param   bundle          The {@link Message} {@code mime-bundle}.
044     * @param   object          The {@link Object} to render.
045     */
046    public void renderTo(ObjectNode bundle, Object object);
047
048    /**
049     * Static {@link RendererMap} instance used by
050     * {@link #render(Object,Object...)}.
051     */
052    public static RendererMap MAP = new RendererMap();
053
054    /**
055     * Method to render an {@link Object} to an
056     * {@link Message#execute_result(int,ObjectNode)}.
057     *
058     * @param   object          The {@link Object} to encode.
059     * @param   alternates      Optional alternate representations.
060     *
061     * @return  The {@link Message} {@code mime-bundle}.
062     */
063    public static ObjectNode render(Object object, Object... alternates) {
064        var bundle = OBJECT_MAPPER.createObjectNode();
065
066        MAP.renderTo(bundle, object, alternates);
067
068        return bundle;
069    }
070}
