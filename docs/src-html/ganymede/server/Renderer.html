021import com.fasterxml.jackson.databind.node.ObjectNode;
022import ganymede.server.renderer.ForClass;
023import ganymede.server.renderer.ForClassName;
024import java.util.Base64;
025import java.util.Optional;
026
027/**
028 * {@link Message#mime_bundle(Object,Object...)} output {@link Renderer}.
029 *
030 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
031 */
032public interface Renderer {
033    public static final String DATA = "data";
034    public static final String METADATA = "metadata";
035
036    /**
037     * Singleton {@link RendererMap} instance.
038     */
039    public static final RendererMap MAP = new RendererMap();
040
041    /**
042     * {@link Base64.Encoder} instance.
043     */
044    public static final Base64.Encoder BASE64_ENCODER = Base64.getEncoder();
045
046    /**
047     * Method to get a configured instance of this {@link Renderer}.
048     * Subclasses may override to return an instance that requires
049     * additional resources and/or configuration.  Default implementaion
050     * returns {@link.this}.
051     *
052     * @return  An {@link Optional} containing the configured instance.
053     */
054    public default Optional<? extends Renderer> instance() {
055        return Optional.of(this);
056    }
057
058    /**
059     * Default method to analyze {@link.this} instance for {@link ForClass}
060     * and {@link ForClassName} annotations.  May return {@code null} if a
061     * {@link Class} cannot be loaded.
062     *
063     * @return  The {@link Class type} the argument renders (may be
064     *          {@code null}).
065     */
066    public default Class<?> getRenderType() {
067        return getRenderType(getClass());
068    }
069
070    /**
071     * Method to render an {@link Object} to a {@code mime-bundle}.
072     *
073     * @param   bundle          The {@link Message} {@code mime-bundle}.
074     * @param   object          The {@link Object} to render.
075     */
076    public void renderTo(ObjectNode bundle, Object object);
077
078    /**
079     * Static method to analyze a class for {@link ForClass} and
080     * {@link ForClassName} annotations.  May return {@code null} if a
081     * {@link Class} cannot be loaded.
082     *
083     * @param   type            The {@link Class} to analyze.
084     *
085     * @return  The {@link Class type} the argument renders (may be
086     *          {@code null}).
087     */
088    public static Class<?> getRenderType(Class<?> type) {
089        Class<?> renderType = null;
090
091        if (type != null) {
092            try {
093                if (type.isAnnotationPresent(ForClassName.class)) {
094                    var name = type.getAnnotation(ForClassName.class).value();
095
096                    renderType = Class.forName(name, false, type.getClassLoader());
097                } else {
098                    renderType = type.getAnnotation(ForClass.class).value();
099                }
100            } catch (Throwable throwable) {
101            }
102
103            if (renderType == null) {
104                var superType = type.getSuperclass();
105
106                if (superType != null && Renderer.class.isAssignableFrom(superType)) {
107                    renderType = getRenderType(superType);
108                }
109            }
110        }
111
112        return renderType;
113    }
114}
