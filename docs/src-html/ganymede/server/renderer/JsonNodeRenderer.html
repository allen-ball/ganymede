021import ball.annotation.ServiceProviderFor;
022import com.fasterxml.jackson.databind.JsonNode;
023import com.fasterxml.jackson.databind.node.ObjectNode;
024import ganymede.server.Renderer;
025import lombok.NoArgsConstructor;
026import lombok.ToString;
027
028import static org.springframework.util.MimeTypeUtils.APPLICATION_JSON_VALUE;
029
030/**
031 * {@link JsonNode} {@link Renderer} service provider.
032 *
033 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
034 */
035@ServiceProviderFor({ Renderer.class })
036@ForClass(JsonNode.class)
037@NoArgsConstructor @ToString
038public class JsonNodeRenderer implements Renderer {
039    @Override
040    public void renderTo(ObjectNode bundle, Object object) {
041        var node = (JsonNode) object;
042
043        if (! bundle.with(DATA).has(APPLICATION_JSON_VALUE)) {
044            bundle.with(DATA)
045                .set(APPLICATION_JSON_VALUE, node);
046            bundle.with(METADATA).with(APPLICATION_JSON_VALUE)
047                .put("expanded", true);
048        }
049
050        if (node != null) {
051            MAP.renderTo(bundle, node.toPrettyString());
052        }
053    }
054}
