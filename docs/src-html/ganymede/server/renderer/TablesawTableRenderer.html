037/**
038 * {@link.uri https://github.com/jtablesaw/tablesaw target=newtab Tablesaw}
039 * {@link Table} {@link Renderer} service provider.
040 *
041 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
042 */
043@ServiceProviderFor({ Renderer.class })
044@ForClassName("tech.tablesaw.api.Table")
045@NoArgsConstructor @ToString
046public class TablesawTableRenderer implements Renderer {
047    @Override
048    public Optional<TablesawTableRenderer> instance() {
049        return Optional.ofNullable(getRenderType()).map(t -> new Impl());
050    }
051
052    @Override
053    public void renderTo(ObjectNode bundle, Object object) {
054        throw new IllegalStateException();
055    }
056
057    @NoArgsConstructor @ToString
058    private class Impl extends TablesawTableRenderer {
059        @Override
060        public void renderTo(ObjectNode bundle, Object object) {
061            var table = (Table) object;
062/*
063            if (! bundle.with(DATA).has(APPLICATION_JSON_VALUE)) {
064                try {
065                    if (hasWriterForExtension("json")) {
066                        var string = table.write().toString("json");
067                        var node = JSON_OBJECT_MAPPER.readTree(string);
068
069                        bundle.with(DATA)
070                            .set(APPLICATION_JSON_VALUE, node);
071                        bundle.with(METADATA)
072                            .with(APPLICATION_JSON_VALUE)
073                            .put("expanded", true);
074                    }
075                } catch (Exception exception) {
076                    exception.printStackTrace(System.err);
077                }
078            }
079*/
080            if (! bundle.with(DATA).has(TEXT_HTML_VALUE)) {
081                try {
082                    if (hasWriterForExtension("html")) {
083                        bundle.with(DATA)
084                            .put(TEXT_HTML_VALUE, table.write().toString("html"));
085                    }
086                } catch (Exception exception) {
087                    exception.printStackTrace(System.err);
088                }
089            }
090
091            if (table != null) {
092                var string = "";
093
094                try (var out = new ByteArrayOutputStream()) {
095                    new DataFramePrinter(Integer.MAX_VALUE, out).print(table);
096
097                    string = new String(out.toByteArray());
098                } catch (Exception exception) {
099                    exception.printStackTrace(System.err);
100                }
101
102                MAP.renderTo(bundle, string);
103            }
104        }
105
106        private boolean hasWriterForExtension(String extension) {
107            return defaultWriterRegistry.getWriterForExtension(extension) != null;
108        }
109    }
110}
