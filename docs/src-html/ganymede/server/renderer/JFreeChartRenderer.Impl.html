021import ball.annotation.ServiceProviderFor;
022import com.fasterxml.jackson.databind.node.ObjectNode;
023import ganymede.server.Renderer;
024import java.io.ByteArrayOutputStream;
025import java.io.IOException;
026import java.io.PrintWriter;
027import java.io.StringWriter;
028import java.util.Optional;
029import java.util.UUID;
030import lombok.NoArgsConstructor;
031import lombok.ToString;
032import org.jfree.chart.ChartRenderingInfo;
033import org.jfree.chart.ChartUtils;
034import org.jfree.chart.JFreeChart;
035import org.jfree.chart.entity.StandardEntityCollection;
036
037import static org.springframework.util.MimeTypeUtils.TEXT_HTML_VALUE;
038
039/**
040 * {@link.uri https://github.com/jfree/jfreechart target=newtab JFreeChart}
041 * {@link Renderer} service provider.
042 *
043 * @see JFreeChart
044 * @see ChartUtils
045 *
046 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
047 */
048@ServiceProviderFor({ Renderer.class })
049@ForClassName("org.jfree.chart.JFreeChart")
050@NoArgsConstructor @ToString
051public class JFreeChartRenderer implements Renderer {
052    @Override
053    public Optional<JFreeChartRenderer> instance() {
054        return Optional.ofNullable(getRenderType()).map(t -> new Impl());
055    }
056
057    @Override
058    public void renderTo(ObjectNode bundle, Object object) {
059        throw new IllegalStateException();
060    }
061
062    @NoArgsConstructor @ToString
063    public class Impl extends JFreeChartRenderer {
064        @Override
065        public Optional<JFreeChartRenderer> instance() {
066            return Optional.of(this);
067        }
068
069        @Override
070        public void renderTo(ObjectNode bundle, Object object) {
071            var chart = (JFreeChart) object;
072            var width = 800;
073            var height = 600;
074
075            try (var out = new ByteArrayOutputStream()) {
076                var info = new ChartRenderingInfo(new StandardEntityCollection());
077
078                ChartUtils.writeChartAsPNG(out, chart, width, height, info);
079
080                var image = out.toByteArray();
081
082                MAP.renderTo(bundle, image);
083
084                if (! bundle.with(DATA).has(TEXT_HTML_VALUE)) {
085                    var html = new StringWriter();
086                    var name = UUID.randomUUID().toString();
087                    var mimeType = bundle.with(METADATA).fieldNames().next();
088                    var base64 = bundle.with(DATA).get(mimeType).asText();
089
090                    try (var writer = new PrintWriter(html)) {
091                        writer.format("<img usemap=\"#%s\" src=\"data:%s;base64,%s\"/>\n",
092                                      name, mimeType, base64);
093                        ChartUtils.writeImageMap(writer, name, info, false);
094                    }
095
096                    bundle.with(DATA).put(TEXT_HTML_VALUE, html.toString());
097                }
098            } catch (Throwable throwable) {
099                throwable.printStackTrace(System.err);
100            }
101        }
102    }
103}
