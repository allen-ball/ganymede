021import ball.annotation.ServiceProviderFor;
022import com.fasterxml.jackson.databind.node.ObjectNode;
023import ganymede.server.Renderer;
024import java.util.Map;
025import javax.swing.table.TableModel;
026import lombok.NoArgsConstructor;
027import lombok.ToString;
028
029/**
030 * {@link TableModel} {@link Renderer} service provider.
031 *
032 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
033 */
034@ServiceProviderFor({ Renderer.class })
035@ForClass(TableModel.class)
036@NoArgsConstructor @ToString
037public class TableModelRenderer implements Renderer {
038    @Override
039    public void renderTo(ObjectNode bundle, Object object) {
040        var resource = getClass().getSimpleName() + ".html";
041        var map = Map.<String,Object>of("model", (TableModel) object);
042        var output = ThymeleafRenderer.process(getClass(), resource, "html", map);
043
044        MAP.renderTo(bundle, output);
045    }
046}
