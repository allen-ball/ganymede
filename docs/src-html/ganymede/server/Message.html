352    /**
353     * Method to serialize a {@link Message}.
354     *
355     * @param   digester        The {@link HMACDigester} (may be
356     *                          {@code null}).
357     *
358     * @return  The {@link List} of serialized frames.
359     */
360    public List<byte[]> serialize(HMACDigester digester) {
361        var frames = new LinkedList<byte[]>();
362
363        frames.addAll(envelope());
364        frames.add(DELIMITER_BYTES);
365
366        var header = serialize(header());
367        var parentHeader = serialize(parentHeader());
368        var metadata = serialize(metadata());
369        var content = serialize(content());
370
371        var digest = "";
372
373        if (digester != null) {
374            digest = digester.digest(header, parentHeader, metadata, content);
375        }
376
377        Collections.addAll(frames,
378                           digest.getBytes(ZMQ.CHARSET),
379                           header, parentHeader, metadata, content);
380
381        frames.addAll(buffers());
382
383        return frames;
384    }
385
386    private byte[] serialize(JsonNode node) {
387        var string = "{}";
388
389        try {
390            string = ObjectMappers.JSON.writeValueAsString(node);
391        } catch (Exception exception) {
392            log.warn("{}", exception);
393        }
394
395        return string.getBytes(ZMQ.CHARSET);
396    }

398    /**
399     * Method to receive a {@link Message} on a {@link ZMQ.Socket}.
400     *
401     * param    connection      The {@link Connection}.
402     * @param   socket          The {@link ZMQ.Socket}.
403     * @param   frame           The first message frame.
404     */
405    public static Message receive(Connection connection, ZMQ.Socket socket, byte[] frame) {
406        var envelope = new LinkedList<byte[]>();
407
408        while (! DELIMITER_ZDATA.equals(frame)) {
409            envelope.add(frame);
410
411            frame = recv(socket);
412        }
413
414        var signature = recv(socket);
415        var header = recv(socket);
416        var parentHeader = recv(socket);
417        var metadata = recv(socket);
418        var content = recv(socket);
419        var buffers = new LinkedList<byte[]>();
420
421        while (socket.hasReceiveMore()) {
422            buffers.add(recv(socket));
423        }
424
425        var digester = connection.getDigester();
426
427        if (digester != null) {
428            if (! digester.verify(new String(signature, ZMQ.CHARSET),
429                                  header, parentHeader, metadata, content)) {
430                throw new SecurityException("Invalid signature");
431            }
432        }
433
434        var message = new Message(connection);
435
436        message.envelope().addAll(envelope);
437        message.header().setAll(deserialize(header));
438        message.parentHeader().setAll(deserialize(parentHeader));
439        message.metadata().setAll(deserialize(metadata));
440        message.content().setAll(deserialize(content));
441        message.buffers().addAll(buffers);
442
443        return message;
444    }
445
446    private static byte[] recv(ZMQ.Socket socket) {
447        return Objects.requireNonNull(socket.recv(ZMQ.DONTWAIT));
448    }

450    private static ObjectNode deserialize(byte[] bytes) {
451        ObjectNode value = null;
452
453        try {
454            value = (ObjectNode) ObjectMappers.JSON.readTree(new String(bytes, ZMQ.CHARSET));
455        } catch (Exception exception) {
456            log.warn("{}", exception);
457        }
458
459        return value;
460    }

462    /**
463     * Method to create a "standard" error node.  If both arguments are
464     * {@code null} then an {@link ObjectNode} with {@code status = "ok"} is
465     * created.
466     *
467     * @param   throwable       The {@link Throwable} source of the error.
468     * @param   evalue          The expression value.
469     *
470     * @return  The corresponding {@link ObjectNode}.
471     */
472    public static ObjectNode content(Throwable throwable, String evalue) {
473        var node = new ObjectNode(JsonNodeFactory.instance);
474
475        if (throwable != null || evalue != null) {
476            node.put("status", "error");
477            node.put("ename", throwable.getClass().getCanonicalName());
478            node.put("evalue", evalue);
479
480            var array = node.putArray("traceback");
481
482            if (throwable != null) {
483                var buffer = new PrintStreamBuffer();
484
485                throwable.printStackTrace(buffer);
486                Stream.of(buffer.toString().split("\\R"))
487                    .forEach(t -> array.add(t));
488            }
489        } else {
490            node.put("status", "ok");
491        }
492
493        return node;
494    }

509    private static String getCallingMethodName(int skip) {
510        String name =
511            WALKER.walk(t -> t.map(StackWalker.StackFrame::getMethodName)
512                                  .skip(skip)
513                                  .findFirst())
514            .get();

519    /**
520     * Method to send a {@link Message}.
521     *
522     * param    connection      The {@link Connection}.
523     * @param   socket          The {@link ZMQ.Socket}.
524     */
525    public void send(Connection connection, ZMQ.Socket socket) {
526        var list = serialize(connection.getDigester());
527        var iterator = list.iterator();
528
529        while (iterator.hasNext()) {
530            var frame = iterator.next();
531
532            socket.send(frame, iterator.hasNext() ? ZMQ.SNDMORE : 0);
533        }
534    }

536    @Override
537    public String toString() {
538        var string =
539            Stream.of(envelope().stream().map(ZData::new),
540                      Stream.of(DELIMITER_STRING, "DIGEST"),
541                      Stream.of(header(), parentHeader(), metadata(), content())
542                      .map(t -> t.toPrettyString()),
543                      buffers().stream().map(ZData::new))
544            .flatMap(t -> t)
545            .map(Object::toString)
546            .collect(joining("\n"));
547
548        return string;
549    }

564    private static abstract class Child extends Message {
565        protected Child(String msg_type, Message request) {
566            super(null);
567
568            if (msg_type != null) {
569                msg_type(msg_type);
570            }
571
572            if (request != null) {
573                parentHeader().setAll(request.header());
574            }
575        }
576    }
577
578    private static class Reply extends Child {
579        public Reply(Message request) {
580            super(request.getMessageTypeAction() + "_reply", request);
581
582            if (request.isRequest()) {
583                envelope().addAll(request.envelope());
584                content().setAll(content(null, null));
585                buffers().addAll(request.buffers());
586            } else {
587                throw new IllegalStateException("Source message is not a request");
588            }
589        }
590    }
591
592    private static class Pub extends Child {
593        public Pub(String msg_type, Message request) {
594            super(msg_type, request);
595
596            envelope().clear();
597
598            var topic = msg_type;
599
600            if (request != null && request.connection != null) {
601                var kernelId = request.connection.getKernelId();
602
603                if (kernelId != null) {
604                    topic = String.format("kernel.%s.%s", kernelId, msg_type);
605                }
606            }
607
608            if (topic != null) {
609                envelope().add(topic.getBytes(ZMQ.CHARSET));
610            }
611        }
612    }
