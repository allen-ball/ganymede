021import ball.annotation.CompileTimeCheck;
022import com.fasterxml.jackson.databind.node.ObjectNode;
023import java.io.File;
024import java.io.IOException;
025import java.util.regex.Pattern;
026import lombok.Data;
027import lombok.NonNull;
028import lombok.extern.log4j.Log4j2;
029
030import static ganymede.server.Server.JSON_OBJECT_MAPPER;
031
032/**
033 * Jupyter {@link Connection}.  See
034 * "{@link.uri https://jupyter-client.readthedocs.io/en/stable/kernels.html#connection-files target=newtab Connection files}".
035 *
036 * {@bean.info}
037 *
038 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
039 */
040@Data @Log4j2
041public class Connection {
042
043    /**
044     * {@link Connection} file name {@link Pattern}.
045     * Provides named group "kernelId".
046     */
047    @CompileTimeCheck
048    public static final Pattern FILE_NAME_PATTERN =
049        Pattern.compile("(?i)^(kernel-|)(?<kernelId>[^.]+)[.]json$");
050
051    @NonNull private final String kernelId;
052    @NonNull private final ObjectNode node;
053    @NonNull private final HMACDigester digester;
054
055    /**
056     * {@link File} constructor.
057     *
058     * @param   id              The kernel ID.
059     * @param   file            The {@link File} describing the
060     *                          {@link Connection}.
061     */
062    public Connection(String id, File file) throws IOException {
063        this(id, (ObjectNode) JSON_OBJECT_MAPPER.readTree(file));
064    }
065
066    /**
067     * {@link ObjectNode} constructor.
068     *
069     * @param   kernelId        The kernel ID.
070     * @param   node            The {@link ObjectNode} describing the
071     *                          {@link Connection}.
072     */
073    public Connection(String kernelId, ObjectNode node) {
074        this.kernelId = kernelId;
075        this.node = node;
076        this.digester = new HMACDigesterImpl();
077    }
078
079    /**
080     * Method to connect a kernel's {@link Channel}s.
081     *
082     * @param   shell           The {@link Channel.Shell Shell}
083     *                          {@link Channel}.
084     * @param   control         The {@link Channel.Control Control}
085     *                          {@link Channel}.
086     * @param   iopub           The {@link Channel.IOPub IOPub}
087     *                          {@link Channel}.
088     * @param   stdin           The {@link Channel.Stdin Stdin}
089     *                          {@link Channel}.
090     * @param   heartbeat       The {@link Channel.Heartbeat Heartbeat}
091     *                          {@link Channel}.
092     */
093    public void connect(Channel.Shell shell, Channel.Control control,
094                        Channel.IOPub iopub, Channel.Stdin stdin,
095                        Channel.Heartbeat heartbeat) {
096        var digester = getDigester();
097
098        shell.connect(this, getAddress("shell_port"));
099        control.connect(this, getAddress("control_port"));
100        iopub.connect(this, getAddress("iopub_port"));
101        stdin.connect(this, getAddress("stdin_port"));
102        heartbeat.connect(this, getAddress("hb_port"));
103    }
104
105    /**
106     * Method to get the address corresponding to port name (JSON field).
107     *
108     * @param   portName        The JSON field specifying the desired port.
109     *
110     * @return  The address (as a {@link String}).
111     */
112    public String getAddress(String portName) {
113        String address =
114            String.format("%s://%s:%d",
115                          node.at("/transport").asText(),
116                          node.at("/ip").asText(),
117                          node.at("/" + portName).asInt());
118
119        return address;
120    }
121
122    private class HMACDigesterImpl extends HMACDigester {
123        public HMACDigesterImpl() {
124            super(node.at("/signature_scheme").asText(),
125                  node.at("/key").asText());
126        }
127    }
128}
