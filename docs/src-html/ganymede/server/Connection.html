021import com.fasterxml.jackson.databind.node.ObjectNode;
022import lombok.Data;
023import lombok.NonNull;
024import lombok.extern.log4j.Log4j2;
025
026/**
027 * Jupyter {@link Connection}.  See
028 * "{@link.uri https://jupyter-client.readthedocs.io/en/stable/kernels.html#connection-files target=newtab Connection files}".
029 *
030 * {@bean.info}
031 *
032 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
033 */
034@Data @Log4j2
035public class Connection {
036    @NonNull private final String id;
037    @NonNull private final ObjectNode node;
038    @NonNull private final HMACDigester digester;
039
040    /**
041     * Sole constructor.
042     *
043     * @param   id              The kernel ID.
044     * @param   node            The {@link ObjectNode} describing the
045     *                          {@link Connection}.
046     */
047    public Connection(String id, ObjectNode node) {
048        this.id = id;
049        this.node = node;
050        this.digester = new HMACDigesterImpl();
051    }
052
053    /**
054     * Method to connect a kernel's {@link Channel}s.
055     *
056     * @param   shell           The {@link Channel.Shell Shell}
057     *                          {@link Channel}.
058     * @param   control         The {@link Channel.Control Control}
059     *                          {@link Channel}.
060     * @param   iopub           The {@link Channel.IOPub IOPub}
061     *                          {@link Channel}.
062     * @param   stdin           The {@link Channel.Stdin Stdin}
063     *                          {@link Channel}.
064     * @param   heartbeat       The {@link Channel.Heartbeat Heartbeat}
065     *                          {@link Channel}.
066     */
067    public void connect(Channel.Shell shell, Channel.Control control,
068                        Channel.IOPub iopub, Channel.Stdin stdin,
069                        Channel.Heartbeat heartbeat) {
070        var digester = getDigester();
071
072        shell.connect(this, getAddress("shell_port"));
073        control.connect(this, getAddress("control_port"));
074        iopub.connect(this, getAddress("iopub_port"));
075        stdin.connect(this, getAddress("stdin_port"));
076        heartbeat.connect(this, getAddress("hb_port"));
077    }
078
079    private String getAddress(String portName) {
080        String address =
081            String.format("%s://%s:%d",
082                          node.at("/transport").asText(),
083                          node.at("/ip").asText(),
084                          node.at("/" + portName).asInt());
085
086        return address;
087    }
088
089    private class HMACDigesterImpl extends HMACDigester {
090        public HMACDigesterImpl() {
091            super(node.at("/signature_scheme").asText(),
092                  node.at("/key").asText());
093        }
094    }
095}
