355    /**
356     * Method to serialize a {@link Message}.
357     *
358     * @param   digester        The {@link HMACDigester} (may be
359     *                          {@code null}).
360     *
361     * @return  The {@link List} of serialized frames.
362     */
363    public List<byte[]> serialize(HMACDigester digester) {
364        var frames = new LinkedList<byte[]>();
365
366        frames.addAll(envelope());
367        frames.add(DELIMITER_BYTES);
368
369        var header = serialize(header());
370        var parentHeader = serialize(parentHeader());
371        var metadata = serialize(metadata());
372        var content = serialize(content());
373
374        var digest = "";
375
376        if (digester != null) {
377            digest = digester.digest(header, parentHeader, metadata, content);
378        }
379
380        Collections.addAll(frames,
381                           digest.getBytes(ZMQ.CHARSET),
382                           header, parentHeader, metadata, content);
383
384        frames.addAll(buffers());
385
386        return frames;
387    }
388
389    private byte[] serialize(JsonNode node) {
390        var string = "{}";
391
392        try {
393            string = ObjectMappers.JSON.writeValueAsString(node);
394        } catch (Exception exception) {
395            log.warn("{}", exception);
396        }
397
398        return string.getBytes(ZMQ.CHARSET);
399    }

401    /**
402     * Method to receive a {@link Message} on a {@link ZMQ.Socket}.
403     *
404     * param    connection      The {@link Connection}.
405     * @param   socket          The {@link ZMQ.Socket}.
406     * @param   frame           The first message frame.
407     */
408    public static Message receive(Connection connection, ZMQ.Socket socket, byte[] frame) {
409        var envelope = new LinkedList<byte[]>();
410
411        while (! DELIMITER_ZDATA.equals(frame)) {
412            envelope.add(frame);
413
414            frame = recv(socket);
415        }
416
417        var signature = recv(socket);
418        var header = recv(socket);
419        var parentHeader = recv(socket);
420        var metadata = recv(socket);
421        var content = recv(socket);
422        var buffers = new LinkedList<byte[]>();
423
424        while (socket.hasReceiveMore()) {
425            buffers.add(recv(socket));
426        }
427
428        var digester = connection.getDigester();
429
430        if (digester != null) {
431            if (! digester.verify(new String(signature, ZMQ.CHARSET), header, parentHeader, metadata, content)) {
432                throw new SecurityException("Invalid signature");
433            }
434        }
435
436        var message = new Message(connection);
437
438        message.envelope().addAll(envelope);
439        message.header().setAll(deserialize(header));
440        message.parentHeader().setAll(deserialize(parentHeader));
441        message.metadata().setAll(deserialize(metadata));
442        message.content().setAll(deserialize(content));
443        message.buffers().addAll(buffers);
444
445        return message;
446    }
447
448    private static byte[] recv(ZMQ.Socket socket) {
449        return Objects.requireNonNull(socket.recv(ZMQ.DONTWAIT));
450    }

452    private static ObjectNode deserialize(byte[] bytes) {
453        ObjectNode value = null;
454
455        try {
456            value = (ObjectNode) ObjectMappers.JSON.readTree(new String(bytes, ZMQ.CHARSET));
457        } catch (Exception exception) {
458            log.warn("{}", exception);
459        }
460
461        return value;
462    }

464    /**
465     * Method to create a "standard" error node.  If both arguments are
466     * {@code null} then an {@link ObjectNode} with {@code status = "ok"} is
467     * created.
468     *
469     * @param   throwable       The {@link Throwable} source of the error.
470     * @param   evalue          The expression value.
471     *
472     * @return  The corresponding {@link ObjectNode}.
473     */
474    public static ObjectNode content(Throwable throwable, String evalue) {
475        var node = new ObjectNode(JsonNodeFactory.instance);
476
477        if (throwable != null || evalue != null) {
478            node.put("status", "error");
479            node.put("ename", throwable.getClass().getCanonicalName());
480            node.put("evalue", evalue);
481
482            var array = node.putArray("traceback");
483
484            if (throwable != null) {
485                var buffer = new PrintStreamBuffer();
486
487                throwable.printStackTrace(buffer);
488                Stream.of(buffer.toString().split("\\R"))
489                    .forEach(t -> array.add(t));
490            }
491        } else {
492            node.put("status", "ok");
493        }
494
495        return node;
496    }

511    private static String getCallingMethodName(int skip) {
512        String name =
513            WALKER.walk(t -> t.map(StackWalker.StackFrame::getMethodName)
514                                  .skip(skip)
515                                  .findFirst())
516            .get();
517
518        return name;
519    }

521    /**
522     * Method to send a {@link Message}.
523     *
524     * param    connection      The {@link Connection}.
525     * @param   socket          The {@link ZMQ.Socket}.
526     */
527    public void send(Connection connection, ZMQ.Socket socket) {
528        var list = serialize(connection.getDigester());
529        var iterator = list.iterator();
530
531        while (iterator.hasNext()) {
532            var frame = iterator.next();
533
534            socket.send(frame, iterator.hasNext() ? ZMQ.SNDMORE : 0);
535        }
536    }

556    @Override
557    public String toString() {
558        var string =
559            Stream.of(envelope().stream().map(ZData::new),
560                      Stream.of(DELIMITER_STRING, "DIGEST"),
561                      Stream.of(header(), parentHeader(), metadata(), content())
562                      .map(t -> t.toPrettyString()),
563                      buffers().stream().map(ZData::new))
564            .flatMap(t -> t)
565            .map(Object::toString)
566            .collect(joining("\n"));
567
568        return string;
569    }

584    private static abstract class Child extends Message {
585        protected Child(String msg_type, Message request) {
586            super(null);
587
588            if (msg_type != null) {
589                msg_type(msg_type);
590            }
591
592            if (request != null) {
593                parentHeader().setAll(request.header());
594            }
595        }
596    }
597
598    private static class Reply extends Child {
599        public Reply(Message request) {
600            super(request.getMessageTypeAction() + "_reply", request);
601
602            if (request.isRequest()) {
603                envelope().addAll(request.envelope());
604                content().setAll(content(null, null));
605                buffers().addAll(request.buffers());
606            } else {
607                throw new IllegalStateException("Source message is not a request");
608            }
609        }
610    }
611
612    private static class Pub extends Child {
613        public Pub(String msg_type, Message request) {
614            super(msg_type, request);
615
616            envelope().clear();
617
618            var topic = msg_type;
619
620            if (request != null && request.connection != null) {
621                var kernelId = request.connection.getKernelId();
622
623                if (kernelId != null) {
624                    topic = String.format("kernel.%s.%s", kernelId, msg_type);
625                }
626            }
627
628            if (topic != null) {
629                envelope().add(topic.getBytes(ZMQ.CHARSET));
630            }
631        }
632    }
