021import java.math.BigInteger;
022import java.util.stream.Stream;
023import javax.crypto.Mac;
024import javax.crypto.spec.SecretKeySpec;
025import lombok.Data;
026import lombok.extern.log4j.Log4j2;
027import org.zeromq.ZMQ;
028
029/**
030 * HMAC message digester.  See discussion in
031 * {@link.uri https://jupyter-client.readthedocs.io/en/latest/messaging.html#the-wire-protocol target=newtab The Wire Protocol}.
032 *
033 * {@bean.info}
034 *
035 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
036 */
037@Data @Log4j2
038public class HMACDigester {
039    private final Mac mac;
040
041    /**
042     * Sole constructor.
043     *
044     * @param   scheme          The signature scheme.
045     * @param   key             See key.
046     */
047    public HMACDigester(String scheme, String key) {
048        Mac mac = null;
049
050        if (key != null && (! key.isEmpty())) {
051            try {
052                mac = Mac.getInstance(scheme.replaceAll("-", ""));
053                mac.init(new SecretKeySpec(key.getBytes(ZMQ.CHARSET), scheme));
054            } catch (Exception exception) {
055                throw new ExceptionInInitializerError(exception);
056            }
057        }
058
059        this.mac = mac;
060    }
061
062    /**
063     * Method to calculate a digest for message frames.  See
064     * {@link Mac#update(byte[])} and {@link Mac#doFinal()}.
065     *
066     * @param   frames          The {@code byte[]} frames of the message to
067     *                          digest.
068     *
069     * @return  The digest {@link String}.
070     */
071    public String digest(byte[]... frames) {
072        String digest = "";
073
074        if (mac != null) {
075            synchronized (mac) {
076                Stream.of(frames).forEach(mac::update);
077
078                var bytes = mac.doFinal();
079
080                digest = new BigInteger(1, bytes).toString(16);
081            }
082
083            var length = 2 * mac.getMacLength();
084
085            while (digest.length() < length) {
086                digest = "0" + digest;
087            }
088        }
089
090        return digest;
091    }
092
093    /**
094     * Method to verify a digest for message frames.
095     *
096     * @param   digest          The digest to verify.
097     * @param   frames          The {@code byte[]} frames of the message to
098     *                          digest.
099     *
100     * @return  {@code true} if the argument digest matches the one
101     *          calculated; {@code false} otherwise.
102     */
103    public boolean verify(String digest, byte[]... frames) {
104        return digest.equals(digest(frames));
105    }
106}
