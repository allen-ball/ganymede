021import java.util.concurrent.BlockingQueue;
022import java.util.concurrent.SynchronousQueue;
023import lombok.Data;
024import lombok.NonNull;
025import lombok.extern.log4j.Log4j2;
026import org.zeromq.ZMQ;
027
028import static java.util.concurrent.TimeUnit.MILLISECONDS;
029
030/**
031 * Jupyter {@link ZMQ.Socket} {@link Dispatcher}.  All {@link ZMQ.Socket}
032 * creation and manipulation calls happens in the {@link #run()} method.
033 * See {@link.uri https://zguide.zeromq.org/ target=newtab Ã˜MQ - The Guide},
034 * {@link.uri https://zguide.zeromq.org/docs/chapter3/ target=newtab Chapter 3}.
035 *
036 * {@bean.info}
037 *
038 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
039 */
040@Data @Log4j2
041public class Dispatcher implements Runnable {
042    @NonNull private final Channel channel;
043    @NonNull private final Connection connection;
044    @NonNull private final String address;
045    private final BlockingQueue<Message> outgoing = new SynchronousQueue<>();
046
047    /**
048     * Callback method to dispatch a received message.  Default
049     * implementation calls
050     * {@link Channel#dispatch(Dispatcher,ZMQ.Socket,byte[])}.
051     *
052     * @param   socket          The {@link ZMQ.Socket}.
053     * @param   frame           The first message frame.
054     */
055    protected void dispatch(ZMQ.Socket socket, byte[] frame) {
056        getChannel().dispatch(this, socket, frame);
057    }
058
059    /**
060     * Callback method to dispatch a {@link Message}.  Default
061     * implementation calls
062     * {@link Channel.Protocol#dispatch(Dispatcher,ZMQ.Socket,Message)}.
063     *
064     * @param   socket          The {@link ZMQ.Socket}.
065     * @param   message         The {@link Message}.
066     */
067    protected void dispatch(ZMQ.Socket socket, Message message) {
068        ((Channel.Protocol) getChannel()).dispatch(this, socket, message);
069    }
070
071    /**
072     * Method to schedule a message for publishing.
073     *
074     * @param   message         The message to send.
075     */
076    public void pub(Message message) {
077        var type = getChannel().getSocketType();
078
079        switch (type) {
080        case PUB:
081            try {
082                outgoing.put(message);
083            } catch (InterruptedException exception) {
084                log.warn("{}", exception);
085            }
086            break;
087
088        default:
089            throw new IllegalStateException("Unsupported SocketType: " + type);
090        }
091    }
092
093    @Override
094    public void run() {
095        var server = getChannel().getServer();
096        var context = server.getContext();
097        var digester = getConnection().getDigester();
098        var type = getChannel().getSocketType();
099
100        while (! server.isTerminating()) {
101            try (var socket = context.socket(type)) {
102                if (socket.bind(getAddress())) {
103                    log.info("Bound {} {}", type, address);
104                } else {
105                    log.warn("Could not bind to {}", address);
106                }
107
108                switch (type) {
109                case REP:
110                case ROUTER:
111                    try (var poller = context.poller(1)) {
112                        poller.register(socket, ZMQ.Poller.POLLIN);
113
114                        while (! server.isTerminating()) {
115                            int events = poller.poll(100);
116
117                            if (events > 0 && poller.pollin(0)) {
118                                var message = socket.recv();
119
120                                if (message != null) {
121                                    dispatch(socket, message);
122                                }
123                            }
124                        }
125                    }
126                    break;
127
128                case PUB:
129                    while (! server.isTerminating()) {
130                        var message = outgoing.poll(100, MILLISECONDS);
131
132                        if (message != null) {
133                            dispatch(socket, message);
134                        }
135                    }
136                    break;
137
138                default:
139                    throw new IllegalStateException("Unsupported SocketType: " + type);
140                }
141            } catch (Exception exception) {
142                log.warn("{}", exception);
143            }
144        }
145    }
146}
