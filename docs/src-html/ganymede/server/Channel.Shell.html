021import java.lang.reflect.InvocationTargetException;
022import java.lang.reflect.Method;
023import java.util.Queue;
024import java.util.concurrent.ConcurrentLinkedQueue;
025import lombok.Data;
026import lombok.NonNull;
027import lombok.ToString;
028import lombok.extern.log4j.Log4j2;
029import org.zeromq.SocketType;
030import org.zeromq.ZMQ;

032/**
033 * {@link Server} {@link Channel}.
034 *
035 * {@bean.info}
036 *
037 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
038 */
039@Data @Log4j2
040public abstract class Channel {
041    @NonNull private final Server server;
042    @NonNull private final SocketType socketType;
043    private final Queue<Dispatcher> dispatcherQueue = new ConcurrentLinkedQueue<>();
044
045    /**
046     * Method to schedule creation of and binding to a {@link ZMQ.Socket}
047     * for this address.
048     *
049     * @param   connection      The kernel {@link Connection}.
050     * @param   address         The address of the {@link ZMQ.Socket} to be
051     *                          created.
052     */
053    public void connect(Connection connection, String address) {
054        Dispatcher dispatcher = new Dispatcher(this, connection, address);
055
056        getDispatcherQueue().add(dispatcher);
057        getServer().submit(dispatcher);
058
059        getServer()
060            .setCorePoolSize(Math.max(getServer().getActiveCount() + 4,
061                                      getServer().getCorePoolSize()));
062    }
063
064    /**
065     * Callback method to receive and dispatch a {@link Message}.  This
066     * method is called on the same thread that the {@link ZMQ.Socket} was
067     * created on and the implementation may call {@link ZMQ.Socket} methods
068     * (including {@code send()}).
069     *
070     * @param   dispatcher      The {@link Dispatcher}.
071     * @param   socket          The {@link ZMQ.Socket}.
072     * @param   frame           The first message frame.
073     */
074    protected abstract void dispatch(Dispatcher dispatcher, ZMQ.Socket socket, byte[] frame);
075
076    /**
077     * Callback method to {@link Server#stamp(Message) stamp} and dispatch a
078     * {@link Message}.  This method is called on the same thread that the
079     * {@link ZMQ.Socket} was created on and the implementation may call
080     * {@link ZMQ.Socket} methods (including {@code send()}).
081     *
082     * @param   dispatcher      The {@link Dispatcher}.
083     * @param   socket          The {@link ZMQ.Socket}.
084     * @param   message         The {@link Message}.
085     */
086    protected void send(Dispatcher dispatcher, ZMQ.Socket socket, Message message) {
087        getServer().stamp(message);
088
089        log.debug("{}\n{}", dispatcher.getAddress(), message);
090
091        message.send(dispatcher.getConnection(), socket);
092    }
093
094    /**
095     * Standard
096     * {@link.uri https://jupyter-client.readthedocs.io/en/latest/messaging.html#heartbeat-for-kernels Heartbeat}
097     * {@link Channel}.
098     *
099     * {@bean.info}
100     */
101    @ToString @Log4j2
102    public static class Heartbeat extends Channel {
103
104        /**
105         * Sole constructor.
106         *
107         * @param  server       The {@link Server}.
108         */
109        public Heartbeat(Server server) { super(server, SocketType.REP); }
110
111        @Override
112        protected void dispatch(Dispatcher dispatcher, ZMQ.Socket socket, byte[] frame) {
113            socket.send(frame);
114        }
115    }
116
117    /**
118     * Jupyter protocol {@link Channel} abstract base class.
119     *
120     * {@bean.info}
121     */
122    @ToString @Log4j2
123    public static abstract class Protocol extends Channel {
124
125        /**
126         * Sole constructor.
127         *
128         * @param  server       The {@link Server}.
129         * @param  type         The {@link SocketType} for created
130         *                      {@link ZMQ.Socket}s.
131         */
132        public Protocol(Server server, SocketType type) { super(server, type); }
133
134        /**
135         * Callback method to dispatch a {@link Message}.  Default
136         * implementation looks for a declared method
137         * {@code action(Dispatcher,ZMQ.Socket,Message)}.
138         *
139         * @param  dispatcher   The {@link Dispatcher}.
140         * @param  socket       The {@link ZMQ.Socket}.
141         * @param  message      The {@link Message}.
142         */
143        protected abstract void dispatch(Dispatcher dispatcher, ZMQ.Socket socket, Message message);
144
145        @Override
146        protected void dispatch(Dispatcher dispatcher, ZMQ.Socket socket, byte[] frame) {
147            try {
148                var message = Message.receive(dispatcher.getConnection(), socket, frame);
149
150                log.debug("{}\n{}", dispatcher.getAddress(), message);
151
152                dispatch(dispatcher, socket, message);
153            } catch (Exception exception) {
154                log.warn("{}", exception);
155            }
156        }
157    }
158
159    /**
160     * {@link Control Control} {@link Channel} abstract base class.  The
161     * default implementation of
162     * {@link #dispatch(Dispatcher,ZMQ.Socket,Message)} constructs a
163     * {@link Message reply} skeleton, executes a declared method of the
164     * form {@code action(Dispatcher,Message,Message) throws Exception},
165     * catches any {@link Exception} and updates the reply as necessary, and
166     * sends the reply.
167     *
168     * {@bean.info}
169     */
170    @ToString @Log4j2
171    public static abstract class Control extends Protocol {
172        private interface PROTOTYPE {
173            public void action(Dispatcher dispatcher, Message request, Message reply) throws Exception;
174        }
175
176        private static final Method PROTOTYPE;
177
178        static {
179            PROTOTYPE = PROTOTYPE.class.getDeclaredMethods()[0];
180            PROTOTYPE.setAccessible(true);
181        }
182
183        /**
184         * Sole constructor.
185         *
186         * @param  server       The {@link Server}.
187         */
188        protected Control(Server server) { super(server, SocketType.ROUTER); }
189
190        @Override
191        protected void dispatch(Dispatcher dispatcher, ZMQ.Socket socket, Message message) {
192            var action = message.getMessageTypeAction();
193
194            if (action != null) {
195                var reply = message.reply();
196
197                try {
198                    var method = getClass().getDeclaredMethod(action, PROTOTYPE.getParameterTypes());
199
200                    method.setAccessible(true);
201                    method.invoke(this, dispatcher, message, reply);
202                } catch (Throwable throwable) {
203                    if (throwable instanceof InvocationTargetException) {
204                        if (throwable.getCause() != null) {
205                            throwable = throwable.getCause();
206                        }
207                    }
208
209                    reply.status(throwable);
210                } finally {
211                    if (reply != null) {
212                        send(dispatcher, socket, reply);
213                    }
214                }
215            } else {
216                log.warn("Could not determine action from {}", message.header());
217            }
218        }
219    }

221    /**
222     * {@link IOPub IOPub} {@link Channel}.
223     *
224     * {@bean.info}
225     */
226    @ToString @Log4j2
227    public static class IOPub extends Protocol {
228
229        /**
230         * Sole constructor.
231         *
232         * @param  server       The {@link Server}.
233         */
234        public IOPub(Server server) { super(server, SocketType.PUB); }
235
236        /**
237         * Method to schedule a {@link Message} for publishing.
238         *
239         * @param   message     The {@link Message} to send.
240         */
241        public void pub(Message message) {
242            getDispatcherQueue().forEach(t -> t.pub(message));
243        }
244
245        @Override
246        protected void dispatch(Dispatcher dispatcher, ZMQ.Socket socket, byte[] frame) {
247            throw new UnsupportedOperationException();
248        }
249
250        @Override
251        protected void dispatch(Dispatcher dispatcher, ZMQ.Socket socket, Message message) {
252            send(dispatcher, socket, message);
253        }
254    }

272    /**
273     * {@link Shell Shell} {@link Channel} abstract base class.
274     *
275     * {@bean.info}
276     */
277    @ToString @Log4j2
278    public static abstract class Shell extends Control {
279        private final IOPub iopub;
280        private final Stdin stdin;
281
282        /**
283         * Sole constructor.
284         *
285         * @param  server       The {@link Server}.
286         * @param  iopub        The associated {@link IOPub IOPub}
287         *                      {@link Channel}.
288         * @param  stdin        The associated {@link Stdin Stdin}
289         *                      {@link Channel}.
290         */
291        protected Shell(Server server, IOPub iopub, Stdin stdin) {
292            super(server);
293
294            this.iopub = iopub;
295            this.stdin = stdin;
296        }
297
298        @Override
299        public void connect(Connection connection, String address) {
300            boolean isStarting = getDispatcherQueue().isEmpty();
301
302            super.connect(connection, address);
303
304            if (isStarting) {
305                iopub.pub(Message.status(Message.status.starting, null));
306                iopub.pub(Message.status(Message.status.idle, null));
307            }
308        }
309
310        @Override
311        protected void dispatch(Dispatcher dispatcher, ZMQ.Socket socket, Message message) {
312            try {
313                iopub.pub(message.status(Message.status.busy));
314
315                super.dispatch(dispatcher, socket, message);
316            } finally {
317                iopub.pub(message.status(Message.status.idle));
318            }
319        }
320    }
