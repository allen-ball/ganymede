102    /**
103     * Add a connection specified by a {@link Connection} {@link File}.
104     *
105     * @param   file            The {@link Connection} {@link File}.
106     *
107     * @throws  IOException     If the {@link File} cannot be opened or
108     *                          parsed.
109     */
110    protected void bind(File file) throws IOException {
111        var connection = Connection.parse(file);
112        var starting = (getKernelId() == null);
113
114        setKernelId(connection.getKernelId());
115
116        log.info("Kernel {}", getKernelId());
117
118        connection.connect(shell, control, iopub, stdin, heartbeat);
119
120        log.info("Connected to {}", connection.getNode().toPrettyString());
121
122        if (starting) {
123            iopub.pub(Message.status(Message.status.starting, null));
124        }
125    }

127    /**
128     * Method to restart the {@link Server}.  Subclass implementations
129     * should chain to the {@code super}-method.
130     */
131    protected void restart() throws Exception {
132        in = new ByteArrayInputStream(new byte[] { });
133        out = new PrintStreamBuffer();
134        err = new PrintStreamBuffer();
135    }

183
184    /**
185     * Method to stamp an outgoing {@link Message}.  Adds
186     * {@link #PROTOCOL_VERSION}, session, and
187     * {@link Message#timestamp() timestamp} if not already specified.
188     *
189     * @param   message         The {@link Message} to stamp.
190     *
191     * @return  The {@link Message}.
192     */
193    protected Message stamp(Message message) {
194        if (message.version() == null) {
195            message.version(PROTOCOL_VERSION.toString());
196        }
197
198        if (message.session() == null) {
199            message.session(getKernelSessionId().toString());
200        }
201
202        return message.timestamp();
203    }

215            }

218        private void shutdown(Dispatcher dispatcher, Message request, Message reply) throws Exception {
219            var restart = request.content().at("/restart").asBoolean();
220
221            reply.content().put("restart", restart);
222
223            if (restart) {
224                Server.this.restart();
225            } else {
226                Server.this.shutdown();
227            }
228        }

232        }

253    @ToString
254    private class Shell extends Channel.Shell {
255        public Shell() {
256            super(Server.this, Server.this.iopub, Server.this.stdin);
257        }
258
259        @Override
260        protected void dispatch(Dispatcher dispatcher, ZMQ.Socket socket, Message message) {
261            if (message.isRequest()) {
262                try {
263                    Server.this.request = message;
264
265                    super.dispatch(dispatcher, socket, Server.this.request);
266                } finally {
267                    Server.this.request = null;
268                }
269            } else {
270                log.warn("Ignoring non-request {}", message.msg_type());
271            }
272        }

278        private void execute(Dispatcher dispatcher, Message request, Message reply) throws Exception {
279            var code = request.content().at("/code").asText();
280            var silent = request.content().at("/silent").asBoolean();
281            var store_history = request.content().at("/store_history").asBoolean();
282            var user_expressions = request.content().at("/user_expressions");
283            var allow_stdin = request.content().at("/allow_stdin").asBoolean();
284            var stop_on_error = request.content().at("/stop_on_error").asBoolean();
285
286            try {
287                if (! code.isEmpty()) {
288                    if (! silent) {
289                        if (store_history) {
290                            execution_count.incrementAndGet();
291
292                            iopub.pub(request.execute_input(code, execution_count.intValue()));
293                        }
294                    }
295                    /*
296                     * jupyter lab populates execute_request metadata.  E.g.,
297                     *
298                     * metadata: {
299                     *   "deletedCells" : [ ],
300                     *   "recordTiming" : false,
301                     *   "cellId" : "4cf407d2"
302                     * }
303                     */
304                    Server.this.execute(code);
305                }
306            } catch (Throwable throwable) {
307                reply.status(throwable, code);
308            } finally {
309                reply.content().put("execution_count", execution_count.intValue());
310
311                if (reply.content().get("status").asText().equals("ok")) {
312                    reply.content().withArray("payload");
313
314                    var in = request.content().at("/user_expressions");
315                    var iterator = in.fields();
316                    var out = reply.content().with("user_expressions");
317
318                    while (iterator.hasNext()) {
319                        var entry = iterator.next();
320                        var name = entry.getKey();
321                        var expression = entry.getValue().asText();
322
323                        try {
324                            out.put(name, String.valueOf(Server.this.evaluate(expression)));
325                        } catch (Throwable throwable) {
326                            out.set(name, Message.content(throwable, expression));
327                        }
328                    }
329                }
330
331                var stdout = out.toString();
332                var stderr = err.toString();
333
334                out.reset();
335                err.reset();
336
337                if (! silent) {
338                    if (! stdout.isEmpty()) {
339                        iopub.pub(request.stream(Message.stream.stdout, stdout));
340                    }
341
342                    if (! stderr.isEmpty()) {
343                        iopub.pub(request.stream(Message.stream.stderr, stderr));
344                    }
345                }
346            }
347        }

389        private void comm_info(Dispatcher dispatcher, Message request, Message reply) throws Exception {
390            var comms = reply.content().with("comms");
391            /*
392             * Currently unsupported so empty reply.  But remember to *copy*
393             * internal state since output may be culled after satisfying
394             * query.
395             */
396            if (request.content().hasNonNull("target_name")) {
397                var target_name =
398                    request.content().at("/target_name").asText();
399                var iterator = comms.fields();
400
401                while (iterator.hasNext()) {
402                    var entry = iterator.next();
403                    var dict = entry.getValue();
404
405                    if (dict.isObject()) {
406                        ((ObjectNode) dict).retain(target_name);
407
408                        if (dict.isEmpty()) {
409                            iterator.remove();
410                        }
411                    }
412                }
413            }
414        }
