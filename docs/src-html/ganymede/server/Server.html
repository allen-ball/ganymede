090    /**

092     *
093     * @param   file            The {@link Connection} {@link File}.
094     *
095     * @throws  IOException     If the {@link File} cannot be opened or
096     *                          parsed.
097     */
098    protected void bind(File file) throws IOException {
099        var connection = Connection.parse(file);
100
101        setKernelId(connection.getKernelId());
102
103        log.info("Kernel {}", getKernelId());
104
105        connection.connect(shell, control, iopub, stdin, heartbeat);
106
107        log.info("Connected to {}", connection.getNode().toPrettyString());
108    }

110    /**

112     * should chain to the {@code super}-method.

114    protected void restart() throws Exception {

116        out = new PrintStreamBuffer();
117        err = new PrintStreamBuffer();

168    /**
169     * Method to stamp an outgoing {@link Message}.  Adds
170     * {@link #PROTOCOL_VERSION}, session, and
171     * {@link Message#timestamp() timestamp} if not already specified.
172     *
173     * @param   message         The {@link Message} to stamp.
174     *
175     * @return  The {@link Message}.
176     */
177    protected Message stamp(Message message) {
178        if (message.version() == null) {
179            message.version(PROTOCOL_VERSION.toString());
180        }
181
182        if (message.session() == null) {
183            message.session(getKernelSessionId().toString());
184        }
185
186        return message.timestamp();
187    }

189    @ToString
190    private class Control extends Channel.Control {
191        public Control() { super(Server.this); }

193        @Override
194        protected void dispatch(Dispatcher dispatcher, ZMQ.Socket socket, Message message) {
195            if (message.isRequest()) {
196                super.dispatch(dispatcher, socket, message);
197            } else {
198                log.warn("Ignoring non-request {}", message.msg_type());
199            }
200        }

202        private void shutdown(Dispatcher dispatcher, Message request, Message reply) throws Exception {
203            var restart = request.content().at("/restart").asBoolean();
204
205            reply.content().put("restart", restart);
206
207            if (restart) {
208                Server.this.restart();
209            } else {
210                Server.this.shutdown();
211            }
212        }

215            Server.this.interrupt();
216        }

237    @ToString
238    private class Shell extends Channel.Shell {
239        public Shell() {
240            super(Server.this, Server.this.iopub, Server.this.stdin);
241        }
242
243        @Override
244        protected void dispatch(Dispatcher dispatcher, ZMQ.Socket socket, Message message) {
245            if (message.isRequest()) {
246                try {
247                    Server.this.request = message;
248
249                    super.dispatch(dispatcher, socket, Server.this.request);
250                } finally {
251                    Server.this.request = null;
252                }
253            } else {
254                log.warn("Ignoring non-request {}", message.msg_type());
255            }
256        }

258        private void kernel_info(Dispatcher dispatcher, Message request, Message reply) throws Exception {
259            reply.content().setAll(getKernelInfo());
260        }
261
262        private void execute(Dispatcher dispatcher, Message request, Message reply) throws Exception {
263            /*
264             * jupyter lab populates execute_request metadata.  E.g.,
265             *
266             * metadata: {
267             *   "deletedCells" : [ ],
268             *   "recordTiming" : false,
269             *   "cellId" : "4cf407d2"
270             * }
271             */
272            var code = request.content().at("/code").asText();
273            var silent = request.content().at("/silent").asBoolean();
274            var store_history = request.content().at("/store_history").asBoolean();
275            var user_expressions = request.content().at("/user_expressions");
276            var allow_stdin = request.content().at("/allow_stdin").asBoolean();
277            var stop_on_error = request.content().at("/stop_on_error").asBoolean();
278
279            try {
280                if (! code.isEmpty()) {
281                    if (! silent) {
282                        if (store_history) {
283                            execution_count.incrementAndGet();
284
285                            iopub.pub(request.execute_input(code, execution_count.intValue()));
286                        }
287                    }
288
289                    Server.this.execute(code);
290                }
291            } catch (Throwable throwable) {
292                reply.status(throwable, code);
293            } finally {
294                reply.content().put("execution_count", execution_count.intValue());
295
296                if (reply.content().get("status").asText().equals("ok")) {
297                    reply.content().withArray("payload");
298
299                    var in = request.content().at("/user_expressions");
300                    var iterator = in.fields();
301                    var out = reply.content().with("user_expressions");
302
303                    while (iterator.hasNext()) {
304                        var entry = iterator.next();
305                        var name = entry.getKey();
306                        var expression = entry.getValue().asText();
307
308                        try {
309                            out.put(name, String.valueOf(Server.this.evaluate(expression)));
310                        } catch (Throwable throwable) {
311                            out.set(name, Message.content(throwable, expression));
312                        }
313                    }
314                }
315
316                var stdout = out.toString();
317                var stderr = err.toString();
318
319                out.reset();
320                err.reset();
321
322                if (! silent) {
323                    if (! stdout.isEmpty()) {
324                        iopub.pub(request.stream(Message.stream.stdout, stdout));
325                    }
326
327                    if (! stderr.isEmpty()) {
328                        iopub.pub(request.stream(Message.stream.stderr, stderr));
329                    }
330                }
331            }
332        }

374        private void comm_info(Dispatcher dispatcher, Message request, Message reply) throws Exception {
375            var comms = reply.content().with("comms");
376            /*
377             * Currently unsupported so empty reply.  But remember to *copy*
378             * internal state since output may be culled after satisfying
379             * query.
380             */
381            if (request.content().hasNonNull("target_name")) {
382                var target_name =
383                    request.content().at("/target_name").asText();
384                var iterator = comms.fields();
385
386                while (iterator.hasNext()) {
387                    var entry = iterator.next();
388                    var dict = entry.getValue();
389
390                    if (dict.isObject()) {
391                        ((ObjectNode) dict).retain(target_name);
392
393                        if (dict.isEmpty()) {
394                            iterator.remove();
395                        }
396                    }
397                }
398            }
399        }
