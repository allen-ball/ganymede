098    /**
099     * Method to start a {@link Shell}.
100     *
101     * @param   in              The {@code in} {@link InputStream}.
102     * @param   out             The {@code out} {@link PrintStream}.
103     * @param   err             The {@code err} {@link PrintStream}.
104     */
105    @Synchronized
106    public void start(InputStream in, PrintStream out, PrintStream err) {
107        this.in = in;
108        this.out = out;
109        this.err = err;
110
111        magics.clear();
112        magics.reload();
113
114        Stream.of(java.getMagicNames()).forEach(t -> magics.put(t, java));
115    }

117    /**
118     * Method to restart a {@link Shell}.
119     *
120     * @param   in              The {@code in} {@link InputStream}.
121     * @param   out             The {@code out} {@link PrintStream}.
122     * @param   err             The {@code err} {@link PrintStream}.
123     */
124    @Synchronized
125    public void restart(InputStream in, PrintStream out, PrintStream err) {
126        /* stop(); */
127        close();
128
129        if (! kernel().isTerminating()) {
130            start(in, out, err);
131        }
132    }

135     * Method to close (terminate) a {@link Shell}.
136     */
137    @Override
138    @Synchronized
139    public void close() {
140        Stream.of(java.getMagicNames()).forEach(t -> magics.remove(t));
141
142        try (var jshell = this.jshell) {
143            this.jshell = null;
144
145            if (jshell != null) {
146                restarts.incrementAndGet();
147            }
148        }
149    }

178    public void resolve(POM pom) {
179        for (var file : resolver().resolve(this, out, err, pom)) {
180            var jshell = this.jshell;
181
182            if (jshell != null) {
183                jshell.addToClasspath(file.toString());
184            } else {
185                break;
186            }
187        }

243        var imports = Set.<String>of();
244        var jshell = this.jshell;
245
246        if (jshell != null) {
247            imports =
248                jshell.imports()
249                .map(t -> t.source())

257    /**
258     * Method to get the current {@link Map} of defined variables to their
259     * type definitions.
260     *
261     * @return  The {@link Map} of defined variables and their types as
262     *          {@link String}s.
263     */
264    public Map<String,String> variables() {
265        var variables = Map.<String,String>of();
266        var jshell = this.jshell;
267
268        if (jshell != null) {
269            variables =
270                jshell.variables()
271                .filter(t -> (! t.subKind().equals(TEMP_VAR_EXPRESSION_SUBKIND)))
272                .collect(toMap(k -> k.name(), v -> v.typeName()));
273        }
274
275        return variables;
276    }

278    /**
279     * Accessor to the {@link JShell} instance (created and initialized on
280     * first call).
281     *
282     * @return  The {@link JShell} instance.
283     */
284    @Synchronized
285    public JShell jshell() {
286        if (jshell == null) {
287            var options = new ArrayList<String>();
288            var definitions =
289                Stream.of(ProcessHandle.current().info().arguments())
290                .flatMap(Optional::stream)
291                .flatMap(Stream::of)
292                .takeWhile(t -> (! Objects.equals(t, "-jar")))
293                .filter(t -> t.startsWith("-D"))
294                .toArray(String[]::new);
295
296            Collections.addAll(options, definitions);
297            Collections.addAll(options, VMOPTIONS);
298            options.add("-D" + Map.entry(PORT_PROPERTY, kernel.getPort()));
299
300            if (! kernel.isTerminating()) {
301                try {
302                    jshell =
303                        JShell.builder()
304                        .remoteVMOptions(options.toArray(new String[] { }))
305                        .in(in).out(out).err(err).build();
306
307                    resolver().classpath()
308                        .forEach(t -> jshell.addToClasspath(t.toString()));
309
310                    var logIn =
311                        IoBuilder.forLogger(log)
312                        .setLevel(WARN)
313                        .filter(InputStream.nullInputStream())
314                        .buildInputStream();
315                    var logOut =
316                        IoBuilder.forLogger(log)
317                        .setLevel(WARN)
318                        .buildPrintStream();
319
320                    java.execute(jshell, logIn, logOut, logOut, NotebookContext.bootstrap());
321                } catch (Exception exception) {
322                    log.warn("{}", exception, exception);
323                }
324            }
325        }
326
327        return jshell;
328    }

330    /**
331     * Method to execute code (typically a cell's contents).
332     *
333     * @param   code            The code to execute.
334     */
335    public void execute(String code) {
336        try {
337            NotebookContext.preExecute(this);
338
339            var application = new Magic.Application(code);
340            var name = application.getMagicName();
341            var magic = (name != null) ? magics.reload().get(name) : java;
342
343            if (magic != null) {
344                application.apply(this, magic, in, out, err);
345            } else {
346                application.apply(this);
347            }
348        } catch (Exception exception) {
349            exception.printStackTrace(err);
350        } finally {
351            NotebookContext.postExecute(this);
352        }
353    }

355    /**
356     * Method to evaluate an expression.
357     *
358     * @param   expression      The code to evaluate.
359     *
360     * @return  The result of evaluating the expression.
361     */
362    @Synchronized
363    public String evaluate(String expression) throws Exception {
364        var result = "";
365        var jshell = jshell();
366
367        if (jshell != null) {
368            var analyzer = jshell.sourceCodeAnalysis();
369            var info = analyzer.analyzeCompletion(expression);
370
371            if (! info.completeness().isComplete()) {
372                throw new IllegalArgumentException(expression);
373            }
374
375            result = unescape(jshell.eval(info.source()).get(0).value());
376        }
377
378        return result;
379    }

381    /**
382     * Method to determine code's {@link Message.completeness completeness}.
383     *
384     * @param   code            The code to execute.
385     *
386     * @return  The code's {@link Message.completeness completeness}.
387     */
388    public Message.completeness isComplete(String code) {
389        var completeness = Message.completeness.unknown;
390        var application = new Magic.Application(code);
391        var name = application.getMagicName();
392
393        if (name == null || magics.containsKey(name)) {
394            var magic = (name != null) ? magics.get(name) : java;
395
396            completeness = magic.isComplete(application.getLine0(), application.getCode());
397        } else {
398            completeness = Message.completeness.invalid;
399        }
400
401        return completeness;
402    }

424    @MagicNames({ "java" })
425    @Description("Execute code in Java REPL")
426    @NoArgsConstructor @ToString
427    private class Java implements AnnotatedMagic {
428        @Override
429        public void execute(Shell shell,
430                            InputStream in, PrintStream out, PrintStream err,
431                            String line0, String code) throws Exception {
432            if (! code.isBlank()) {
433                execute(shell.jshell(), in, out, err, code);
434            } else {
435                if (line0 != null) {
436                    AnnotatedMagic.super.execute(shell, in, out, err, line0, code);
437                }
438            }
439        }
440
441        @Override
442        public Message.completeness isComplete(String line0, String code) {
443            var completeness = Message.completeness.unknown;
444
445            if (! code.isBlank()) {
446                var jshell = Shell.this.jshell;
447
448                if (jshell != null) {
449                    try {
450                        if (! parse(jshell, code).isEmpty()) {
451                            completeness = Message.completeness.complete;
452                        } else {
453                            completeness = Message.completeness.incomplete;
454                        }
455                    } catch (ParseException exception) {
456                        completeness = Message.completeness.incomplete;
457                    }
458                }
459            } else {
460                completeness = Message.completeness.incomplete;
461            }
462
463            return completeness;
464        }
465
466        private SortedMap<Integer,SourceCodeAnalysis.CompletionInfo> parse(JShell jshell, String code) throws ParseException {
467            var map = new TreeMap<Integer,SourceCodeAnalysis.CompletionInfo>();
468            var analyzer = jshell.sourceCodeAnalysis();
469            var offset = 0;
470            var remaining = code;
471
472            while (! remaining.isEmpty()) {
473                var value = analyzer.analyzeCompletion(remaining);
474
475                switch (value.completeness()) {
476                case DEFINITELY_INCOMPLETE:
477                    throw new IncompleteParseException(offset);
478                    /* break; */
479
480                case UNKNOWN:
481                    throw new UnknownParseException(offset);
482                    /* break; */
483
484                default:
485                    map.put(offset, value);
486
487                    offset += value.source().length();
488                    remaining = remaining.substring(value.source().length());
489                    break;
490                }
491
492                if (remaining.isBlank()) {
493                    break;
494                }
495            }
496
497            return map;
498        }

500        @Override
501        public void configure(NotebookContext context) {
502            throw new IllegalStateException();
503        }
504
505        @Override
506        public void execute(String line0, String code) throws Exception {
507            throw new IllegalStateException();
508        }
509
510        protected void execute(JShell jshell,
511                               InputStream in, PrintStream out, PrintStream err,
512                               String code) {
513            try {
514                var iterator = parse(jshell, code).entrySet().iterator();
515
516                while (iterator.hasNext()) {
517                    var entry = iterator.next();
518                    var errored = false;
519                    var info = entry.getValue();
520
521                    switch (info.completeness()) {
522                    case EMPTY:
523                        break;
524
525                    case COMPLETE:
526                    case COMPLETE_WITH_SEMI:
527                    case CONSIDERED_INCOMPLETE:
528                        var events = jshell.eval(info.source());
529
530                        for (var event : events) {
531                            if (! event.status().equals(REJECTED)) {
532                                if (event.exception() != null) {
533                                    errored |= true;
534
535                                    event.exception().printStackTrace(err);
536                                }
537                            } else {
538                                errored |= true;
539
540                                err.format("%s %s\n%s\n",
541                                           event.status(),
542                                           event.snippet().kind(),
543                                           event.snippet().source());
544                                jshell.diagnostics(event.snippet())
545                                    .map(t -> t.getMessage(locale))
546                                    .forEach(err::println);
547                            }
548                        }
549
550                        if (! iterator.hasNext()) {
551                            if (! events.isEmpty()) {
552                                var event = events.get(events.size() - 1);
553
554                                switch (event.snippet().subKind()) {
555                                case TEMP_VAR_EXPRESSION_SUBKIND:
556                                case VAR_VALUE_SUBKIND:
557                                    kernel.print(Message.mime_bundle(unescape(event.value())));
558                                    break;
559
560                                default:
561                                    break;
562                                }
563                            }
564                        }
565                        break;
566
567                    default:
568                        throw new IllegalStateException(String.valueOf(info.completeness()));
569                        /* break; */
570                    }
571
572                    if (errored) {
573                        break;
574                    }
575                }
576            } catch (IncompleteParseException exception) {
577                err.format("Incomplete input:\n%s\n",
578                           code.substring(exception.getErrorOffset()));
579            } catch (UnknownParseException exception) {
580                err.format("Unknown input:\n%s\n",
581                           code.substring(exception.getErrorOffset()));
582            } catch (Exception exception) {
583                exception.printStackTrace(err);
584            } finally {
585                out.flush();
586                err.flush();
587            }
588        }
589    }
