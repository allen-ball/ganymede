093    /**
094     * Method to start a {@link Shell}.
095     *
096     * @param   in              The {@code in} {@link InputStream}.
097     * @param   out             The {@code out} {@link PrintStream}.
098     * @param   err             The {@code err} {@link PrintStream}.
099     */
100    @Synchronized
101    public void start(InputStream in, PrintStream out, PrintStream err) {
102        this.in = in;
103        this.out = out;
104        this.err = err;
105
106        magics.clear();
107        magics.reload();
108
109        Stream.of(java.getMagicNames()).forEach(t -> magics.put(t, java));
110    }

112    /**
113     * Method to restart a {@link Shell}.
114     *
115     * @param   in              The {@code in} {@link InputStream}.
116     * @param   out             The {@code out} {@link PrintStream}.
117     * @param   err             The {@code err} {@link PrintStream}.
118     */
119    @Synchronized
120    public void restart(InputStream in, PrintStream out, PrintStream err) {
121        /* stop(); */
122        close();
123
124        if (! kernel().isTerminating()) {
125            start(in, out, err);
126        }
127    }

129    /**
130     * Method to close (terminate) a {@link Shell}.
131     */
132    @Override
133    @Synchronized
134    public void close() {
135        Stream.of(java.getMagicNames()).forEach(t -> magics.remove(t));
136
137        try (var jshell = this.jshell) {
138            this.jshell = null;
139
140            if (jshell != null) {
141                restarts.incrementAndGet();
142            }
143        }
144    }

173    public void resolve(POM pom) {
174        for (var file : resolver().resolve(this, out, err, pom)) {
175            var jshell = this.jshell;
176
177            if (jshell != null) {
178                jshell.addToClasspath(file.toString());
179            } else {
180                break;
181            }
182        }
183    }

232    /**
233     * Accessor to the {@link JShell} instance (created and initialized on
234     * first call).
235     *
236     * @return  The {@link JShell} instance.
237     */
238    @Synchronized
239    public JShell jshell() {
240        if (jshell == null) {
241            var options = new ArrayList<String>();
242            var definitions =
243                Stream.of(ProcessHandle.current().info().arguments())
244                .flatMap(Optional::stream)
245                .flatMap(Stream::of)
246                .takeWhile(t -> (! Objects.equals(t, "-jar")))
247                .filter(t -> t.startsWith("-D"))
248                .toArray(String[]::new);
249
250            Collections.addAll(options, definitions);
251            Collections.addAll(options, VMOPTIONS);
252            options.add("-D" + Map.entry(Kernel.PORT_PROPERTY, kernel.getPort()));
253
254            if (! kernel.isTerminating()) {
255                try {
256                    jshell =
257                        JShell.builder()
258                        .remoteVMOptions(options.toArray(new String[] { }))
259                        .in(in).out(out).err(err).build();
260
261                    resolver().classpath()
262                        .forEach(t -> jshell.addToClasspath(t.toString()));
263
264                    var logIn =
265                        IoBuilder.forLogger(log)
266                        .setLevel(WARN)
267                        .filter(InputStream.nullInputStream())
268                        .buildInputStream();
269                    var logOut =
270                        IoBuilder.forLogger(log)
271                        .setLevel(WARN)
272                        .buildPrintStream();
273
274                    java.execute(jshell, logIn, logOut, logOut, Notebook.bootstrap());
275                } catch (Exception exception) {
276                    log.warn("{}", exception, exception);
277                }
278            }
279        }
280
281        return jshell;
282    }

284    /**
285     * Method to execute code (typically a cell's contents).
286     *
287     * @param   code            The code to execute.
288     */
289    public void execute(String code) throws Exception {
290        try {
291            var jshell = this.jshell;
292
293            if (jshell != null) {
294                NotebookContext.update(jshell);
295            }
296
297            var application = new Magic.Application(code);
298            var name = application.getMagicName();
299
300            if (name != null && (! magics.containsKey(name))) {
301                throw new IllegalArgumentException(application.getLine0());
302            }
303
304            var magic = (name != null) ? magics.get(name) : java;
305
306            application.apply(magic, this, in, out, err);
307        } catch (JShellException exception) {
308            exception.printStackTrace(err);
309            throw exception;
310        } catch (Exception exception) {
311            err.println(exception);
312            throw exception;
313        }
314    }

316    /**
317     * Method to evaluate an expression.
318     *
319     * @param   expression      The code to evaluate.
320     *
321     * @return  The result of evaluating the expression.
322     */
323    @Synchronized
324    public String evaluate(String expression) throws Exception {
325        var result = "";
326        var jshell = jshell();
327
328        if (jshell != null) {
329            var analyzer = jshell.sourceCodeAnalysis();
330            var info = analyzer.analyzeCompletion(expression);
331
332            if (! info.completeness().isComplete()) {
333                throw new IllegalArgumentException(expression);
334            }
335
336            result = unescape(jshell.eval(info.source()).get(0).value());
337        }
338
339        return result;
340    }

342    /**
343     * Method to determine code's {@link Message.completeness completeness}.
344     *
345     * @param   code            The code to execute.
346     *
347     * @return  The code's {@link Message.completeness completeness}.
348     */
349    public Message.completeness isComplete(String code) {
350        var completeness = Message.completeness.unknown;
351        var application = new Magic.Application(code);
352        var name = application.getMagicName();
353
354        if (name == null || magics.containsKey(name)) {
355            var magic = (name != null) ? magics.get(name) : java;
356
357            completeness =
358                magic.isComplete(application.getLine0(), application.getCode());
359        } else {
360            completeness = Message.completeness.invalid;
361        }
362
363        return completeness;
364    }

386    private SortedMap<Integer,SourceCodeAnalysis.CompletionInfo> parse(JShell jshell, String code) {
387        var map = new TreeMap<Integer,SourceCodeAnalysis.CompletionInfo>();
388        var analyzer = jshell.sourceCodeAnalysis();
389        var offset = 0;
390        var remaining = code;
391
392        while (! remaining.isEmpty()) {
393            var value = analyzer.analyzeCompletion(remaining);
394
395            map.put(offset, value);
396
397            offset += value.source().length();
398            remaining = remaining.substring(value.source().length());
399        }
400
401        return map;
402    }

404    @MagicNames({ "java" })
405    @Description("Execute code in Java REPL")
406    @NoArgsConstructor @ToString
407    private class Java implements AnnotatedMagic {
408        @Override
409        public void execute(Shell shell,
410                            InputStream in, PrintStream out, PrintStream err,
411                            String line0, String code) throws Exception {
412            if (! code.isBlank()) {
413                execute(shell.jshell(), in, out, err, code);
414            } else {
415                if (line0 != null) {
416                    Magic.sendTo(shell, getMagicNames()[0], line0, code);
417                }
418            }
419        }
420
421        @Override
422        public Message.completeness isComplete(String line0, String code) {
423            var completeness = Message.completeness.unknown;
424
425            if (! code.isBlank()) {
426                var jshell = Shell.this.jshell;
427
428                if (jshell != null) {
429                    var map = parse(jshell, code);
430
431                    if (! map.isEmpty()) {
432                        var info = map.get(map.lastKey());
433
434                        if (info.completeness().isComplete()) {
435                            completeness = Message.completeness.complete;
436                        } else {
437                            switch (info.completeness()) {
438                            case CONSIDERED_INCOMPLETE:
439                            case DEFINITELY_INCOMPLETE:
440                                completeness = Message.completeness.incomplete;
441                                break;
442
443                            case UNKNOWN:
444                                completeness = Message.completeness.invalid;
445                                break;
446
447                            default:
448                                break;
449                            }
450                        }
451                    } else {
452                        completeness = Message.completeness.invalid;
453                    }
454                }
455            } else {
456                completeness = Message.completeness.incomplete;
457            }
458
459            return completeness;
460        }
461
462        @Override
463        public void execute(NotebookContext __, String line0, String code) throws Exception {
464            throw new IllegalStateException();
465        }
466
467        protected void execute(JShell jshell,
468                               InputStream in, PrintStream out, PrintStream err,
469                               String code) throws Exception {
470            try {
471                var iterator = parse(jshell, code).entrySet().iterator();
472
473                while (iterator.hasNext()) {
474                    var entry = iterator.next();
475                    var info = entry.getValue();
476                    var events = jshell.eval(info.source());
477                    var exception =
478                        events.stream()
479                        .map(t -> t.exception())
480                        .filter(Objects::nonNull)
481                        .findFirst().orElse(null);
482
483                    if (exception != null) {
484                        exception.printStackTrace(err);
485                        break;
486                    }
487
488                    var reason =
489                        events.stream()
490                        .filter(t -> t.status().equals(REJECTED))
491                        .map(t -> jshell.diagnostics(t.snippet())
492                             .map(u -> u.getMessage(locale))
493                             .collect(joining("\n",
494                                              String.format("%s %s\n%s\n",
495                                                            t.status(),
496                                                            t.snippet().kind(),
497                                                            t.snippet().source()),
498                                              "")))
499                        .findFirst().orElse(null);
500
501                    if (reason != null) {
502                        err.println(reason);
503                        break;
504                    }
505
506                    if (! iterator.hasNext()) {
507                        if (! events.isEmpty()) {
508                            var event = events.get(events.size() - 1);
509
510                            switch (event.snippet().kind()) {
511                            case EXPRESSION:
512                                out.println(event.value());
513                                break;
514
515                            default:
516                                break;
517                            }
518                        }
519                    }
520                }
521            } finally {
522                out.flush();
523                err.flush();
524            }
525        }
526    }
