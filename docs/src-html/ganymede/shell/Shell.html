117    public void restart(InputStream in, PrintStream out, PrintStream err) {
118        /* stop(); */
119        close();
120

122            start(in, out, err);
123        }

126    /**
127     * Method to close (terminate) a {@link Shell}.
128     */
129    @Override
130    @Synchronized
131    public void close() {
132        try (var jshell = this.jshell) {
133            this.jshell = null;
134
135            if (jshell != null) {
136                restarts.incrementAndGet();
137            }
138        }
139    }

168                break;
169            }
170        }

254    /**
255     * Accessor to the {@link JShell} instance (created and initialized on
256     * first call).
257     *
258     * @return  The {@link JShell} instance.
259     */
260    @Synchronized
261    public JShell jshell() {
262        if (jshell == null) {
263            var options = new ArrayList<String>();
264            var definitions =
265                Stream.of(ProcessHandle.current().info().arguments())
266                .flatMap(Optional::stream)
267                .flatMap(Stream::of)
268                .takeWhile(t -> (! Objects.equals(t, "-jar")))
269                .filter(t -> t.startsWith("-D"))
270                .toArray(String[]::new);
271
272            Collections.addAll(options, definitions);
273            Collections.addAll(options, VMOPTIONS);
274            options.add("-D" + Map.entry(PORT_PROPERTY, kernel.getPort()));
275
276            if (! kernel.isTerminating()) {
277                try {
278                    jshell =
279                        JShell.builder()
280                        .remoteVMOptions(options.toArray(new String[] { }))
281                        .in(in).out(out).err(err).build();
282
283                    resolver().classpath()
284                        .forEach(t -> jshell.addToClasspath(t.toString()));
285
286                    var logIn =
287                        IoBuilder.forLogger(log)
288                        .setLevel(WARN)
289                        .filter(InputStream.nullInputStream())
290                        .buildInputStream();
291                    var logOut =
292                        IoBuilder.forLogger(log)
293                        .setLevel(WARN)
294                        .buildPrintStream();
295
296                    java.execute(jshell, logIn, logOut, logOut, NotebookContext.bootstrap());
297                } catch (Exception exception) {
298                    log.warn("{}", exception, exception);
299                }
300            }
301        }
302
303        return jshell;
304    }

306    /**
307     * Method to execute code (typically a cell's contents).
308     *
309     * @param   code            The code to execute.
310     */
311    public void execute(String code) {
312        var jshell = jshell();
313
314        try {
315            NotebookContext.preExecute(jshell);
316
317            var application = new Magic.Application(code);
318            var name = application.getMagicName();
319            var builtin = (name != null) ? builtins.get(name) : java;
320
321            if (builtin != null) {
322                builtin.execute(this, in, out, err, application);
323            } else {
324                NotebookContext.magic(jshell, name, application);
325            }
326        } catch (Exception exception) {
327            exception.printStackTrace(err);
328        } finally {
329            NotebookContext.postExecute(jshell);
330        }
331    }

333    /**
334     * Method to evaluate an expression.
335     *
336     * @param   expression      The code to evaluate.
337     *
338     * @return  The result of evaluating the expression.
339     */
340    @Synchronized
341    public String evaluate(String expression) throws Exception {
342        var result = "";
343        var jshell = jshell();
344
345        if (jshell != null) {
346            var analyzer = jshell.sourceCodeAnalysis();
347            var info = analyzer.analyzeCompletion(expression);
348
349            if (! info.completeness().isComplete()) {
350                throw new IllegalArgumentException(expression);
351            }
352
353            result = unescape(jshell.eval(info.source()).get(0).value());
354        }
355
356        return result;
357    }

359    /**
360     * Method to determine code's {@link Magic.completeness completeness}.
361     *
362     * @param   code            The code to execute.
363     *
364     * @return  The code's {@link Magic.completeness completeness}.
365     */
366    public Magic.completeness isComplete(String code) {
367        var completeness = Magic.completeness.unknown;
368        var application = new Magic.Application(code);
369        var name = application.getMagicName();
370
371        if (name == null || builtins.containsKey(name)) {
372            var builtin = (name != null) ? builtins.get(name) : java;
373
374            completeness = builtin.isComplete(application.getLine0(), application.getCode());
375        } else {
376            completeness = Magic.completeness.invalid;
377        }
378
379        return completeness;
380    }

402    @MagicNames({ "java" })
403    @Description("Execute code in Java REPL")
404    @NoArgsConstructor @ToString
405    private class Java extends Builtin {
406        @Override
407        public void execute(Shell shell,
408                            InputStream in, PrintStream out, PrintStream err,
409                            Application application) throws Exception {
410            var code = application.getCode();
411
412            if (! code.isBlank()) {
413                execute(shell.jshell(), in, out, err, code);
414            } else {
415                var line0 = application.getLine0();
416
417                if (line0 != null) {
418                    NotebookContext.magic(jshell(), getMagicNames()[0], application);
419                }
420            }
421        }
422
423        @Override
424        public completeness isComplete(String line0, String code) {
425            var completeness = Magic.completeness.unknown;
426
427            if (! code.isBlank()) {
428                var jshell = Shell.this.jshell;
429
430                if (jshell != null) {
431                    try {
432                        if (! parse(jshell, code).isEmpty()) {
433                            completeness = Magic.completeness.complete;
434                        } else {
435                            completeness = Magic.completeness.incomplete;
436                        }
437                    } catch (ParseException exception) {
438                        completeness = Magic.completeness.incomplete;
439                    }
440                }
441            } else {
442                completeness = Magic.completeness.incomplete;
443            }
444
445            return completeness;
446        }
447
448        private SortedMap<Integer,SourceCodeAnalysis.CompletionInfo> parse(JShell jshell, String code) throws ParseException {
449            var map = new TreeMap<Integer,SourceCodeAnalysis.CompletionInfo>();
450            var analyzer = jshell.sourceCodeAnalysis();
451            var offset = 0;
452            var remaining = code;
453
454            while (! remaining.isEmpty()) {
455                var value = analyzer.analyzeCompletion(remaining);
456
457                switch (value.completeness()) {
458                case DEFINITELY_INCOMPLETE:
459                    throw new IncompleteParseException(offset);
460                    /* break; */
461
462                case UNKNOWN:
463                    throw new UnknownParseException(offset);
464                    /* break; */
465
466                default:
467                    map.put(offset, value);
468
469                    offset += value.source().length();
470                    remaining = remaining.substring(value.source().length());
471                    break;
472                }
473
474                if (remaining.isBlank()) {
475                    break;
476                }
477            }
478
479            return map;
480        }

482        @Override
483        public void configure(NotebookContext context) {
484            throw new IllegalStateException();
485        }
486
487        @Override
488        public void execute(String line0, String code) throws Exception {
489            throw new IllegalStateException();
490        }
491
492        protected void execute(JShell jshell,
493                               InputStream in, PrintStream out, PrintStream err,
494                               String code) {
495            try {
496                var iterator = parse(jshell, code).entrySet().iterator();
497
498                while (iterator.hasNext()) {
499                    var entry = iterator.next();
500                    var errored = false;
501                    var info = entry.getValue();
502
503                    switch (info.completeness()) {
504                    case EMPTY:
505                        break;
506
507                    case COMPLETE:
508                    case COMPLETE_WITH_SEMI:
509                    case CONSIDERED_INCOMPLETE:
510                        var events = jshell.eval(info.source());
511
512                        for (var event : events) {
513                            if (! event.status().equals(REJECTED)) {
514                                if (event.exception() != null) {
515                                    errored |= true;
516
517                                    event.exception().printStackTrace(err);
518                                }
519                            } else {
520                                errored |= true;
521
522                                err.format("%s %s\n%s\n",
523                                           event.status(),
524                                           event.snippet().kind(),
525                                           event.snippet().source());
526                                jshell.diagnostics(event.snippet())
527                                    .map(t -> t.getMessage(locale))
528                                    .forEach(err::println);
529                            }
530                        }
531
532                        if (! iterator.hasNext()) {
533                            if (! events.isEmpty()) {
534                                var event = events.get(events.size() - 1);
535
536                                switch (event.snippet().subKind()) {
537                                case TEMP_VAR_EXPRESSION_SUBKIND:
538                                case VAR_VALUE_SUBKIND:
539                                    kernel.print(Message.mime_bundle(unescape(event.value())));
540                                    break;
541
542                                default:
543                                    break;
544                                }
545                            }
546                        }
547                        break;
548
549                    default:
550                        throw new IllegalStateException(String.valueOf(info.completeness()));
551                        /* break; */
552                    }
553
554                    if (errored) {
555                        break;
556                    }
557                }
558            } catch (IncompleteParseException exception) {
559                err.format("Incomplete input:\n%s\n",
560                           code.substring(exception.getErrorOffset()));
561            } catch (UnknownParseException exception) {
562                err.format("Unknown input:\n%s\n",
563                           code.substring(exception.getErrorOffset()));
564            } catch (Exception exception) {
565                exception.printStackTrace(err);
566            } finally {
567                out.flush();
568                err.flush();
569            }
570        }
571    }
