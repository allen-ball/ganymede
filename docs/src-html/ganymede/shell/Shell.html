118    public void restart(InputStream in, PrintStream out, PrintStream err) {

120        close();
121

127    /**
128     * Method to close (terminate) a {@link Shell}.
129     */
130    @Override
131    @Synchronized
132    public void close() {
133        try (var jshell = this.jshell) {
134            this.jshell = null;
135
136            if (jshell != null) {
137                restarts.incrementAndGet();
138            }
139        }
140    }

171        }

255    /**
256     * Accessor to the {@link JShell} instance (created and initialized on
257     * first call).
258     *
259     * @return  The {@link JShell} instance.
260     */
261    @Synchronized
262    public JShell jshell() {
263        if (jshell == null) {
264            var options = new ArrayList<String>();
265            var definitions =
266                Stream.of(ProcessHandle.current().info().arguments())
267                .flatMap(Optional::stream)
268                .flatMap(Stream::of)
269                .takeWhile(t -> (! Objects.equals(t, "-jar")))
270                .filter(t -> t.startsWith("-D"))
271                .toArray(String[]::new);
272
273            Collections.addAll(options, definitions);
274            Collections.addAll(options, VMOPTIONS);
275            options.add("-D" + Map.entry(PORT_PROPERTY, kernel.getPort()));
276
277            if (! kernel.isTerminating()) {
278                try {
279                    jshell =
280                        JShell.builder()
281                        .remoteVMOptions(options.toArray(new String[] { }))
282                        .in(in).out(out).err(err).build();
283
284                    resolver().classpath()
285                        .forEach(t -> jshell.addToClasspath(t.toString()));
286
287                    var logIn =
288                        IoBuilder.forLogger(log)
289                        .setLevel(WARN)
290                        .filter(InputStream.nullInputStream())
291                        .buildInputStream();
292                    var logOut =
293                        IoBuilder.forLogger(log)
294                        .setLevel(WARN)
295                        .buildPrintStream();
296
297                    java.execute(jshell, logIn, logOut, logOut, NotebookContext.bootstrap());
298                } catch (Exception exception) {
299                    log.warn("{}", exception, exception);
300                }
301            }
302        }
303
304        return jshell;
305    }

307    /**
308     * Method to execute code (typically a cell's contents).
309     *
310     * @param   code            The code to execute.
311     */
312    public void execute(String code) {
313        var jshell = jshell();
314
315        try {
316            NotebookContext.preExecute(jshell);
317
318            var application = new Magic.Application(code);
319            var name = application.getMagicName();
320            var builtin = (name != null) ? builtins.get(name) : java;
321
322            if (builtin != null) {
323                builtin.execute(this, in, out, err, application);
324            } else {
325                NotebookContext.invoke(jshell, name);
326            }
327        } catch (Exception exception) {
328            exception.printStackTrace(err);
329        } finally {
330            NotebookContext.postExecute(jshell);
331        }
332    }

334    /**
335     * Method to evaluate an expression.
336     *
337     * @param   expression      The code to evaluate.
338     *
339     * @return  The result of evaluating the expression.
340     */
341    @Synchronized
342    public String evaluate(String expression) throws Exception {
343        var result = "";
344        var jshell = jshell();
345
346        if (jshell != null) {
347            var analyzer = jshell.sourceCodeAnalysis();
348            var info = analyzer.analyzeCompletion(expression);
349
350            if (! info.completeness().isComplete()) {
351                throw new IllegalArgumentException(expression);
352            }
353
354            result = unescape(jshell.eval(info.source()).get(0).value());
355        }
356
357        return result;
358    }

360    /**
361     * Method to determine code's {@link Magic.completeness completeness}.
362     *
363     * @param   code            The code to execute.
364     *
365     * @return  The code's {@link Magic.completeness completeness}.
366     */
367    public Magic.completeness isComplete(String code) {
368        var completeness = Magic.completeness.unknown;
369        var application = new Magic.Application(code);
370        var name = application.getMagicName();
371
372        if (name == null || builtins.containsKey(name)) {
373            var builtin = (name != null) ? builtins.get(name) : java;
374
375            completeness = builtin.isComplete(application.getLine0(), application.getCode());
376        } else {
377            completeness = Magic.completeness.invalid;
378        }
379
380        return completeness;
381    }

403    @MagicNames({ "java" })
404    @Description("Execute code in Java REPL")
405    @NoArgsConstructor @ToString
406    private class Java extends Builtin {
407        @Override
408        public void execute(Shell shell,
409                            InputStream in, PrintStream out, PrintStream err,
410                            Application application) throws Exception {
411            var code = application.getCode();
412
413            if (! code.isBlank()) {
414                execute(shell.jshell(), in, out, err, code);
415            } else {
416                if (application.getLine0() != null) {
417                    NotebookContext.invoke(jshell(), getMagicNames()[0]);
418                }
419            }
420        }
421
422        @Override
423        public completeness isComplete(String line0, String code) {
424            var completeness = Magic.completeness.unknown;
425
426            if (! code.isBlank()) {
427                var jshell = Shell.this.jshell;
428
429                if (jshell != null) {
430                    try {
431                        if (! parse(jshell, code).isEmpty()) {
432                            completeness = Magic.completeness.complete;
433                        } else {
434                            completeness = Magic.completeness.incomplete;
435                        }
436                    } catch (ParseException exception) {
437                        completeness = Magic.completeness.incomplete;
438                    }
439                }
440            } else {
441                completeness = Magic.completeness.incomplete;
442            }
443
444            return completeness;
445        }
446
447        private SortedMap<Integer,SourceCodeAnalysis.CompletionInfo> parse(JShell jshell, String code) throws ParseException {
448            var map = new TreeMap<Integer,SourceCodeAnalysis.CompletionInfo>();
449            var analyzer = jshell.sourceCodeAnalysis();
450            var offset = 0;
451            var remaining = code;
452
453            while (! remaining.isEmpty()) {
454                var value = analyzer.analyzeCompletion(remaining);
455
456                switch (value.completeness()) {
457                case DEFINITELY_INCOMPLETE:
458                    throw new IncompleteParseException(offset);
459                    /* break; */
460
461                case UNKNOWN:
462                    throw new UnknownParseException(offset);
463                    /* break; */
464
465                default:
466                    map.put(offset, value);
467
468                    offset += value.source().length();
469                    remaining = remaining.substring(value.source().length());
470                    break;
471                }
472
473                if (remaining.isBlank()) {
474                    break;
475                }
476            }
477
478            return map;
479        }

481        @Override
482        public void configure(NotebookContext context) {
483            throw new IllegalStateException();
484        }

486        @Override
487        public void execute(String line0, String code, JsonNode metadata) throws Exception {
488            throw new IllegalStateException();
489        }
490
491        protected void execute(JShell jshell,
492                               InputStream in, PrintStream out, PrintStream err,
493                               String code) {
494            try {
495                var iterator = parse(jshell, code).entrySet().iterator();
496
497                while (iterator.hasNext()) {
498                    var entry = iterator.next();
499                    var errored = false;
500                    var info = entry.getValue();
501
502                    switch (info.completeness()) {
503                    case EMPTY:
504                        break;
505
506                    case COMPLETE:
507                    case COMPLETE_WITH_SEMI:
508                    case CONSIDERED_INCOMPLETE:
509                        var events = jshell.eval(info.source());
510
511                        for (var event : events) {
512                            if (! event.status().equals(REJECTED)) {
513                                if (event.exception() != null) {
514                                    errored |= true;
515
516                                    event.exception().printStackTrace(err);
517                                }
518                            } else {
519                                errored |= true;
520
521                                err.format("%s %s\n%s\n",
522                                           event.status(),
523                                           event.snippet().kind(),
524                                           event.snippet().source());
525                                jshell.diagnostics(event.snippet())
526                                    .map(t -> t.getMessage(locale))
527                                    .forEach(err::println);
528                            }
529                        }
530
531                        if (! iterator.hasNext()) {
532                            if (! events.isEmpty()) {
533                                var event = events.get(events.size() - 1);
534
535                                switch (event.snippet().subKind()) {
536                                case TEMP_VAR_EXPRESSION_SUBKIND:
537                                case VAR_VALUE_SUBKIND:
538                                    kernel.print(Message.mime_bundle(unescape(event.value())));
539                                    break;
540
541                                default:
542                                    break;
543                                }
544                            }
545                        }
546                        break;
547
548                    default:
549                        throw new IllegalStateException(String.valueOf(info.completeness()));
550                        /* break; */
551                    }
552
553                    if (errored) {
554                        break;
555                    }
556                }
557            } catch (IncompleteParseException exception) {
558                err.format("Incomplete input:\n%s\n",
559                           code.substring(exception.getErrorOffset()));
560            } catch (UnknownParseException exception) {
561                err.format("Unknown input:\n%s\n",
562                           code.substring(exception.getErrorOffset()));
563            } catch (Exception exception) {
564                exception.printStackTrace(err);
565            } finally {
566                out.flush();
567                err.flush();
568            }
569        }
570    }
