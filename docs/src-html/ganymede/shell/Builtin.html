021import com.fasterxml.jackson.databind.JsonNode;
022import ganymede.notebook.AbstractMagic;
023import java.io.InputStream;
024import java.io.PrintStream;
025import lombok.NoArgsConstructor;
026
027import static lombok.AccessLevel.PROTECTED;
028
029/**
030 * Abstract base class for built-in {@link ganymede.notebook.Magic}s
031 * (executed locally in the {@link ganymede.shell.Shell}).
032 *
033 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
034 */
035@NoArgsConstructor(access = PROTECTED)
036public abstract class Builtin extends AbstractMagic {
037
038    /**
039     * {@link Shell} execution method.
040     *
041     * @param   shell           The {@link Shell}.
042     * @param   in              The {@code in} {@link InputStream}.
043     * @param   out             The {@code out} {@link PrintStream}.
044     * @param   err             The {@code err} {@link PrintStream}.
045     * @param   application     The {@link Application} instance.
046     */
047    public abstract void execute(Shell shell,
048                                 InputStream in, PrintStream out, PrintStream err,
049                                 Application application) throws Exception;
050
051    @Override
052    public void execute(String line0, String code, JsonNode metadata) throws Exception {
053        throw new IllegalStateException(line0);
054    }
055}
