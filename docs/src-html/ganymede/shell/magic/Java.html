021import ball.annotation.ServiceProviderFor;
022import ganymede.server.renderer.ThymeleafRenderer;
023import ganymede.shell.Magic;
024import java.util.Map;
025import lombok.NoArgsConstructor;
026import lombok.ToString;
027import lombok.extern.log4j.Log4j2;
028
029import static javax.script.ScriptContext.ENGINE_SCOPE;
030
031/**
032 * {@link Java} {@link Magic}.
033 *
034 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
035 */
036@ServiceProviderFor({ Magic.class })
037@MagicNames({ "java" })
038@Description("Execute code in Java REPL")
039@NoArgsConstructor @ToString @Log4j2
040public class Java extends AbstractMagic {
041    @Override
042    public void execute(String line0, String code) throws Exception {
043        if (code.isBlank()) {
044            var resource = getClass().getSimpleName();
045            var map =
046                Map.<String,Object>of("bindings", context.context.getBindings(ENGINE_SCOPE),
047                                      "types", context.types);
048            var html = ThymeleafRenderer.process(getClass(), resource + ".html", "html", map);
049
050            context.print(html);
051        } else {
052            throw new IllegalStateException();
053        }
054    }
055}
