021import ball.annotation.ServiceProviderFor;
022import com.fasterxml.jackson.databind.node.ObjectNode;
023import ganymede.kernel.KernelRestClient;
024import ganymede.server.Message;
025import ganymede.server.Renderer;
026import ganymede.server.renderer.ForClass;
027import ganymede.shell.Magic;
028import java.util.ArrayList;
029import java.util.List;
030import lombok.NoArgsConstructor;
031import lombok.ToString;
032import lombok.extern.log4j.Log4j2;
033import org.commonmark.Extension;
034import org.commonmark.node.Node;
035import org.commonmark.parser.Parser;
036import org.commonmark.renderer.html.HtmlRenderer;
037import org.commonmark.renderer.text.TextContentRenderer;
038
039import static org.springframework.util.MimeTypeUtils.TEXT_HTML_VALUE;
040
041/**
042 * {@link Markdown} {@link Magic}.
043 *
044 * @see Parser
045 * @see HtmlRenderer
046 * @see TextContentRenderer
047 *
048 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
049 */
050@ServiceProviderFor({ Magic.class })
051@Description("Markdown template evaluator")
052@ScriptEngineName("handlebars")
053@NoArgsConstructor @ToString @Log4j2
054public class Markdown extends Handlebars {
055    private static final List<Extension> EXTENSIONS = new ArrayList<>();
056
057    /**
058     * Installed {@link Extension}s.
059     *
060     * {@include #TYPES}
061     */
062    public static final List<Class<? extends Extension>> TYPES =
063        List.of(org.commonmark.ext.autolink.AutolinkExtension.class,
064                org.commonmark.ext.gfm.strikethrough.StrikethroughExtension.class,
065                org.commonmark.ext.gfm.tables.TablesExtension.class,
066                org.commonmark.ext.heading.anchor.HeadingAnchorExtension.class,
067                org.commonmark.ext.image.attributes.ImageAttributesExtension.class,
068                org.commonmark.ext.ins.InsExtension.class,
069                org.commonmark.ext.task.list.items.TaskListItemsExtension.class,
070                org.commonmark.ext.front.matter.YamlFrontMatterExtension.class);
071    static {
072        try {
073            for (Class<? extends Extension> type : TYPES) {
074                var extension = (Extension) type.getMethod("create").invoke(null);
075
076                EXTENSIONS.add(extension);
077            }
078        } catch (Exception exception) {
079            throw new ExceptionInInitializerError(exception);
080        }
081    }
082
083    private final Parser parser = Parser.builder().extensions(EXTENSIONS).build();
084
085    @Override
086    protected void render(Object object) {
087        try {
088            var markdown = (String) object;
089            var node = parser.parse(markdown);
090
091            new KernelRestClient().print(Message.mime_bundle(node));
092        } catch (Exception exception) {
093            exception.printStackTrace(System.err);
094        }
095    }
096
097    /**
098     * CommonMark {@link Node} {@link Renderer} service provider.
099     */
100    @ServiceProviderFor({ Renderer.class })
101    @ForClass(Node.class)
102    @NoArgsConstructor @ToString
103    public static class CommonMarkNodeRenderer implements Renderer {
104        private HtmlRenderer html = HtmlRenderer.builder().extensions(EXTENSIONS).build();
105        private TextContentRenderer text = TextContentRenderer.builder().extensions(EXTENSIONS).build();
106
107        @Override
108        public void renderTo(ObjectNode bundle, Object object) {
109            var node = (Node) object;
110
111            if (! bundle.with(DATA).has(TEXT_HTML_VALUE)) {
112                bundle.with(DATA).put(TEXT_HTML_VALUE, html.render(node));
113            }
114
115            MAP.renderTo(bundle, text.render(node));
116        }
117    }
118}
