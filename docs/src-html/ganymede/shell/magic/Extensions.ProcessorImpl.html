021import ball.annotation.ServiceProviderFor;
022import ball.annotation.processing.AnnotatedProcessor;
023import ball.annotation.processing.For;
024import ball.annotation.processing.TargetMustExtend;
025import java.lang.annotation.Documented;
026import java.lang.annotation.Retention;
027import java.lang.annotation.Target;
028import javax.annotation.processing.Processor;
029import javax.annotation.processing.RoundEnvironment;
030import javax.lang.model.element.AnnotationMirror;
031import javax.lang.model.element.AnnotationValue;
032import javax.lang.model.element.Element;
033import javax.lang.model.element.TypeElement;
034import lombok.NoArgsConstructor;
035import lombok.ToString;
036
037import static java.lang.annotation.ElementType.TYPE;
038import static java.lang.annotation.RetentionPolicy.RUNTIME;
039import static javax.tools.Diagnostic.Kind.ERROR;
040
041/**
042 * {@link AnnotatedScriptEngineMagic#getExtensions()} source
043 * {@link java.lang.annotation.Annotation}.
044 *
045 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
046 */
047@Documented
048@Retention(RUNTIME)
049@Target({ TYPE })
050@TargetMustExtend(AnnotatedScriptEngineMagic.class)
051public @interface Extensions {
052    String[] value();
053
054    /**
055     * {@link Processor} implementation.
056     */
057    @ServiceProviderFor({ Processor.class })
058    @For({ Extensions.class })
059    @NoArgsConstructor @ToString
060    public static class ProcessorImpl extends AnnotatedProcessor {
061        @Override
062        public void process(RoundEnvironment roundEnv,
063                            TypeElement annotation, Element element) {
064            super.process(roundEnv, annotation, element);
065
066            AnnotationMirror mirror = getAnnotationMirror(element, annotation);
067            AnnotationValue value = getAnnotationValue(mirror, "value");
068
069            if (isEmptyArray(value)) {
070                print(ERROR, element, mirror, value, "value() is empty");
071            }
072        }
073    }
074}
