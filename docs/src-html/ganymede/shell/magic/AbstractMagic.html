021import ganymede.notebook.NotebookContext;
022import ganymede.server.Message;
023import ganymede.shell.Shell;
024import java.io.InputStream;
025import java.io.PrintStream;
026import java.util.stream.Stream;
027import lombok.NoArgsConstructor;
028import org.springframework.util.PropertyPlaceholderHelper;
029import picocli.CommandLine;
030import picocli.CommandLine.ParseResult;
031
032import static lombok.AccessLevel.PROTECTED;
033
034/**
035 * Abstract base class for {@link ganymede.shell.Magic}s.
036 *
037 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
038 */
039@NoArgsConstructor(access = PROTECTED)
040public abstract class AbstractMagic implements AnnotatedMagic {
041    /** Value received from {@link #configure(NotebookContext)}. */
042    protected NotebookContext context = null;
043
044    /**
045     * Common static {@link PropertyPlaceholderHelper}.
046     */
047    protected static final PropertyPlaceholderHelper HELPER =
048        new PropertyPlaceholderHelper("${", "}", ":", true);
049
050    @Override
051    public void execute(Shell shell,
052                        InputStream in, PrintStream out, PrintStream err,
053                        String line0, String code) throws Exception {
054        AnnotatedMagic.super.execute(shell, in, out, err, line0, code);
055    }
056
057    @Override
058    public Message.completeness isComplete(String line0, String code) {
059        return Message.completeness.complete;
060    }
061
062    @Override
063    public void configure(NotebookContext context) { this.context = context; }
064
065    /**
066     * Method to parse {@code argv} value to an annotated command
067     * {@link Object}.
068     *
069     * @param   argv            The argument vector (array of
070     *                          {@link String}s).
071     * @param   command         The annotated command {@link Object}.
072     *
073     * @return  The {@link ParseResult}.
074     *
075     * @see CommandLine
076     * @see CommandLine.Command
077     * @see CommandLine.Option
078     * @see CommandLine.Parameters
079     */
080    protected <T> ParseResult parse(String[] argv, T command) {
081        var name = (String) null;
082
083        if (argv.length > 0) {
084            name = argv[0];
085            argv = Stream.of(argv).skip(1).toArray(String[]::new);
086        }
087
088        var line =
089            new CommandLine(command)
090            .setCommandName(name)
091            .setCaseInsensitiveEnumValuesAllowed(true)
092            .setUnmatchedArgumentsAllowed(false);
093        var result = line.parseArgs(argv);
094
095        return result;
096    }
097
098    /**
099     * Method to get usage message from an annotated command {@link Object}.
100     *
101     * @param   command         The annotated command {@link Object}.
102     *
103     * @return  The usage message.
104     *
105     * @see CommandLine
106     * @see CommandLine.Command
107     * @see CommandLine.Option
108     * @see CommandLine.Parameters
109     */
110    protected String getUsage(Object command) {
111        var line =
112            new CommandLine(command)
113            .setCommandName(getMagicNames()[0])
114            .setCaseInsensitiveEnumValuesAllowed(true)
115            .setUnmatchedArgumentsAllowed(false);
116
117        return line.getUsageMessage();
118    }
119}
