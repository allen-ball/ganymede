021import ganymede.shell.Magic;
022import java.util.Objects;
023import javax.script.ScriptEngine;
024import javax.script.ScriptEngineManager;
025import lombok.NoArgsConstructor;
026import lombok.ToString;
027import lombok.extern.log4j.Log4j2;
028
029import static javax.script.ScriptContext.ENGINE_SCOPE;
030import static lombok.AccessLevel.PROTECTED;
031
032/**
033 * Abstract
034 * {@link.uri https://www.jcp.org/en/jsr/detail?id=223 target=newtab JSR 223}
035 * {@link ScriptEngine} {@link Magic} base class.
036 *
037 * @see ScriptEngineManager
038 *
039 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
040 */
041@NoArgsConstructor(access = PROTECTED) @ToString @Log4j2
042public abstract class AbstractScriptEngineMagic extends AbstractMagic
043                                                implements AnnotatedScriptEngineMagic {
044    protected ScriptEngine engine = null;
045
046    /**
047     * Method to get the {@link ScriptEngine}.
048     *
049     * @return  The {@link ScriptEngine} if it can be instantiated;
050     *          {@code null} otherwise.
051     */
052    protected ScriptEngine engine() {
053        if (engine == null) {
054            var manager = new ScriptEngineManager(getClass().getClassLoader());
055
056            engine = manager.getEngineByName(getScriptEngineName());
057
058            if (engine == null) {
059                engine =
060                    getExtensions().stream()
061                    .map(t -> manager.getEngineByExtension(t))
062                    .filter(Objects::nonNull)
063                    .findFirst().orElse(null);
064            }
065        }
066
067        return engine;
068    }
069
070    @Override
071    public void execute(String line0, String code) throws Exception {
072        execute(Magic.getCellMagicCommand(line0), code);
073    }
074
075    /**
076     * Method provided for subclass implementations to intercept calculation
077     * of {@link ScriptEngine#ARGV} binding.
078     *
079     * @param   argv            The first line parsed as an array of
080     *                          {@link String}s.
081     * @param   code            The remainder of the cell.
082     */
083    protected void execute(String[] argv, String code) throws Exception {
084        var bindings = context.context.getBindings(ENGINE_SCOPE);
085
086        try {
087            bindings.put(ScriptEngine.ARGV, argv);
088            execute(code);
089        } finally {
090            bindings.remove(ScriptEngine.ARGV);
091        }
092    }
093
094    /**
095     * Target of {@link #execute(String,String)}.  The
096     * {@code argv} is available in the {@link javax.script.ScriptContext}
097     * {@code ENGINE_SCOPE} {@link javax.script.Bindings} as
098     * {@link ScriptEngine#ARGV ScriptEngine.ARGV}.
099     *
100     * @param   code            The remainder of the cell.
101     */
102    protected void execute(String code) throws Exception {
103        var engine = engine();
104
105        if (engine != null) {
106            render(engine.eval(code, context.context));
107        } else {
108            System.err.format("No %s REPL available\n", getMagicNames()[0]);
109        }
110    }
111
112    /**
113     * Callback to render the results of
114     * {@link ScriptEngine#eval(String,ScriptContext)}.  Made available for
115     * template engines.  Default implementation does nothing.
116     *
117     * @param   object          The result of
118     *                          {@link ScriptEngine#eval(String,ScriptContext)}.
119     */
120    protected void render(Object object) { }
121}
