021import ganymede.notebook.NotebookContext;
022import ganymede.shell.Magic;
023import java.util.Objects;
024import javax.script.ScriptEngine;
025import javax.script.ScriptEngineManager;
026import lombok.NoArgsConstructor;
027import lombok.ToString;
028import lombok.extern.log4j.Log4j2;
029
030import static javax.script.ScriptContext.ENGINE_SCOPE;
031import static lombok.AccessLevel.PROTECTED;
032
033/**
034 * Abstract
035 * {@link.uri https://www.jcp.org/en/jsr/detail?id=223 target=newtab JSR 223}
036 * {@link ScriptEngine} {@link Magic} base class.
037 *
038 * @see ScriptEngineManager
039 *
040 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
041 */
042@NoArgsConstructor(access = PROTECTED) @ToString @Log4j2
043public abstract class AbstractScriptEngineMagic extends AbstractMagic
044                                                implements AnnotatedScriptEngineMagic {
045    protected ScriptEngine engine = null;
046
047    /**
048     * Method to get the {@link ScriptEngine}.
049     *
050     * @return  The {@link ScriptEngine} if it can be instantiated;
051     *          {@code null} otherwise.
052     */
053    protected ScriptEngine engine() {
054        if (engine == null) {
055            var manager = new ScriptEngineManager(getClass().getClassLoader());
056
057            engine = manager.getEngineByName(getScriptEngineName());
058
059            if (engine == null) {
060                engine =
061                    getExtensions().stream()
062                    .map(t -> manager.getEngineByExtension(t))
063                    .filter(Objects::nonNull)
064                    .findFirst().orElse(null);
065            }
066        }
067
068        return engine;
069    }
070
071    @Override
072    public void execute(NotebookContext __, String line0, String code) throws Exception {
073        execute(__, Magic.getCellMagicCommand(line0), code);
074    }
075
076    /**
077     * Method provided for subclass implementations to intercept calculation
078     * of {@link ScriptEngine#ARGV} binding.
079     *
080     * @param   __              The {@link NotebookContext}.
081     * @param   argv            The first line parsed as an array of
082     *                          {@link String}s.
083     * @param   code            The remainder of the cell.
084     */
085    protected void execute(NotebookContext __, String[] argv, String code) throws Exception {
086        var bindings = __.context.getBindings(ENGINE_SCOPE);
087
088        try {
089            bindings.put(ScriptEngine.ARGV, argv);
090            execute(__, code);
091        } finally {
092            bindings.remove(ScriptEngine.ARGV);
093        }
094    }
095
096    /**
097     * Target of {@link #execute(NotebookContext,String,String)}.  The
098     * {@code argv} is available in the {@link javax.script.ScriptContext}
099     * {@code ENGINE_SCOPE} {@link javax.script.Bindings} as
100     * {@link ScriptEngine#ARGV ScriptEngine.ARGV}.
101     *
102     * @param   __              The {@link NotebookContext}.
103     * @param   code            The remainder of the cell.
104     */
105    protected void execute(NotebookContext __, String code) throws Exception {
106        var engine = engine();
107
108        if (engine != null) {
109            render(__, engine.eval(code, __.context));
110        } else {
111            System.err.format("No %s REPL available\n", getMagicNames()[0]);
112        }
113    }
114
115    /**
116     * Callback to render the results of
117     * {@link ScriptEngine#eval(String,ScriptContext)}.  Made available for
118     * template engines.  Default implementation does nothing.
119     *
120     * @param   __              The {@link NotebookContext}.
121     * @param   object          The result of
122     *                          {@link ScriptEngine#eval(String,ScriptContext)}.
123     */
124    protected void render(NotebookContext __, Object object) { }
125}
