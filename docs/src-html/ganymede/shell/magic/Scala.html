021import ball.annotation.ServiceProviderFor;
022import ganymede.notebook.NotebookContext;
023import ganymede.shell.Magic;
024import java.io.InputStreamReader;
025import javax.script.Bindings;
026import javax.script.ScriptEngine;
027import javax.script.ScriptEngineManager;
028import lombok.NoArgsConstructor;
029import lombok.ToString;
030import lombok.extern.log4j.Log4j2;
031import org.springframework.core.io.ClassPathResource;
032import org.springframework.util.StreamUtils;
033import scala.tools.nsc.Settings;
034import scala.tools.nsc.interpreter.Scripted;
035
036import static java.nio.charset.StandardCharsets.UTF_8;
037import static java.util.Collections.disjoint;
038import static javax.script.ScriptContext.ENGINE_SCOPE;
039
040/**
041 * {@link Scala} {@link Magic}.
042 *
043 * @see Scripted
044 *
045 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
046 */
047@ServiceProviderFor({ Magic.class })
048@Description("Execute code in scala REPL")
049@NoArgsConstructor @ToString @Log4j2
050public class Scala extends AbstractScriptEngineMagic {
051    @Override
052    protected Scripted engine() {
053        if (engine == null) {
054            /*
055             * Can't simply extend AbstractScriptEngineMagic because:
056             *
057             *     new ScriptEngineManager().getEngineByName("scala");
058             *
059             * tickles https://github.com/scala/bug/issues/11754.
060             *
061             * The following workaround is required to use println():
062             *
063             *     Console.withOut(System.out) { println($ctx.greeting) }
064             */
065            try {
066                var manager = new ScriptEngineManager(getClass().getClassLoader());
067                var factory =
068                    (Scripted.Factory)
069                    manager.getEngineFactories().stream()
070                    .filter(t -> (! disjoint(t.getExtensions(), getExtensions())))
071                    .findFirst().orElse(null);
072                var settings = new Settings();
073                /*
074                 * settings.Yreplclassbased().value_$eq(true);
075                 * settings.usejavacp().value_$eq(true);
076                 */
077                context.classpath.stream()
078                    .forEach(t -> settings.classpath().append(t));
079
080                var scripted = Scripted.apply(factory, settings, Scripted.apply$default$3());
081                var bindings = context.context.getBindings(ENGINE_SCOPE);
082
083                scripted.setBindings(bindings, ENGINE_SCOPE);
084
085                if (initialize(scripted)) {
086                    engine = scripted;
087                }
088            } catch (NoClassDefFoundError exception) {
089            } catch (Exception exception) {
090                exception.printStackTrace(System.err);
091            }
092        }
093
094        return (Scripted) engine;
095    }
096
097    @Override
098    protected boolean initialize(ScriptEngine engine) {
099        var initialized = true;
100        var scripted = (Scripted) engine;
101        var name = getClass().getSimpleName() + ".sc";
102        var resource = new ClassPathResource(name, getClass());
103
104        if (initialized && resource.exists()) {
105            try (var in = new InputStreamReader(resource.getInputStream(), UTF_8)) {
106                scripted.compile(in).eval(context.context);
107            } catch (Exception exception) {
108                initialized = false;
109                exception.printStackTrace(System.err);
110            }
111        }
112
113        if (initialized) {
114            for (var method : NotebookContext.getNotebookFunctions()) {
115                // TBD: Generate Scala wrappers through reflection
116            }
117        }
118
119        return initialized;
120    }
121
122    @Override
123    protected void execute(String code) throws Exception {
124        var scripted = (Scripted) engine;
125
126        if (engine != null) {
127            try {
128                if (! code.isBlank()) {
129                    render(scripted.compile(code).eval(context.context));
130                } else {
131                    show();
132                }
133            } catch (Exception exception) {
134                exception.printStackTrace(System.err);
135            }
136        } else {
137            super.execute(code);
138        }
139    }
140}
