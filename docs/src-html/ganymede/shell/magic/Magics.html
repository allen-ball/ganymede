021import ball.annotation.ServiceProviderFor;
022import ganymede.kernel.KernelRestClient;
023import ganymede.server.Message;
024import ganymede.server.renderer.ThymeleafRenderer;
025import ganymede.shell.Magic;
026import java.util.Map;
027import java.util.TreeMap;
028import lombok.NoArgsConstructor;
029import lombok.ToString;
030import lombok.extern.log4j.Log4j2;
031
032import static java.util.stream.Collectors.toMap;
033
034/**
035 * {@link Magics} {@link Magic}: List configured {@link Magic}s.
036 *
037 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
038 */
039@ServiceProviderFor({ Magic.class })
040@Description("Lists available cell magics")
041@NoArgsConstructor @ToString @Log4j2
042public class Magics extends AbstractMagic {
043    @Override
044    public void execute(String line0, String code) throws Exception {
045        var resource = getClass().getSimpleName();
046        var magics =
047            context.magics().values().stream()
048            .distinct()
049            .collect(toMap(k -> String.join(", ", k.getMagicNames()), v -> v,
050                           (t, u) -> t, () -> new TreeMap<>()));
051        var map = Map.<String,Object>of("magics", magics);
052        var html = ThymeleafRenderer.process(getClass(), resource + ".html", "html", map);
053        var text = ThymeleafRenderer.process(getClass(), resource + ".text", "text", map);
054
055        new KernelRestClient().print(Message.mime_bundle(html, text));
056    }
057}
