021import ball.annotation.ServiceProviderFor;
022import ganymede.shell.Magic;
023import java.util.stream.Stream;
024import lombok.NoArgsConstructor;
025import lombok.ToString;
026import lombok.extern.log4j.Log4j2;
027
028import static java.lang.ProcessBuilder.Redirect.PIPE;
029import static java.nio.charset.StandardCharsets.UTF_8;
030
031/**
032 * {@link Script} {@link Magic}.
033 *
034 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
035 */
036@ServiceProviderFor({ Magic.class })
037@MagicNames({ "!", "script" })
038@Description("Execute script with the argument command")
039@NoArgsConstructor @ToString @Log4j2
040public class Script extends AbstractMagic {
041    @Override
042    public void execute(String line0, String code) throws Exception {
043        var argv =
044            Stream.of(Magic.getCellMagicCommand(line0))
045            .skip(1)
046            .toArray(String[]::new);
047        var process =
048            new ProcessBuilder(argv)
049            .redirectInput(PIPE)
050            .redirectErrorStream(true)
051            .redirectOutput(PIPE)
052            .start();
053
054        try (var in = process.getInputStream()) {
055            try (var out = process.getOutputStream()) {
056                out.write(code.getBytes(UTF_8));
057            }
058
059            in.transferTo(System.out);
060
061            int status = process.waitFor();
062        }
063    }
064}
