021import ball.annotation.ServiceProviderFor;
022import ganymede.shell.Magic;
023import ganymede.shell.Shell;
024import ganymede.util.PathPropertyMap;
025import java.io.File;
026import java.io.InputStream;
027import java.io.PrintStream;
028import java.util.stream.Stream;
029import lombok.NoArgsConstructor;
030import lombok.ToString;
031import lombok.extern.log4j.Log4j2;
032
033/**
034 * {@link Classpath} {@link Magic}.  See
035 * {@link jdk.jshell.JShell#addToClasspath(String)}.
036 *
037 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
038 */
039@ServiceProviderFor({ Magic.class })
040@Description("Add to or print JShell classpath")
041@NoArgsConstructor @ToString @Log4j2
042public class Classpath extends JShell {
043    private static final String SEPARATOR = System.getProperty("path.separator");
044
045    @Override
046    public void execute(Shell shell,
047                        InputStream in, PrintStream out, PrintStream err,
048                        String line0, String code) throws Exception {
049        if (! code.isBlank()) {
050            var files =
051                HELPER.replacePlaceholders(code, System.getProperties())
052                .lines()
053                .filter(t -> (! t.isBlank()))
054                .map(String::strip)
055                .flatMap(t -> Stream.of(t.split(SEPARATOR)))
056                .filter(t -> (! t.isBlank()))
057                .map(String::strip)
058                .map(File::new)
059                .toArray(File[]::new);
060
061            shell.addToClasspath(files);
062        } else {
063            var map = new PathPropertyMap();
064
065            shell.resolver().classpath().stream()
066                .map(Object::toString)
067                .map(map::shorten)
068                .forEach(out::println);
069        }
070    }
071}
