021import ganymede.util.ServiceProviderMap;
022import java.util.Map;
023import java.util.Set;
024import java.util.TreeMap;
025import java.util.function.Consumer;
026import java.util.stream.Stream;
027
028/**
029 * {@link Magic} {@link Map}.
030 *
031 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
032 */
033public class MagicMap extends TreeMap<String,Magic> {
034    private static final long serialVersionUID = 6930042409603983399L;
035
036    /** @serial */ private final ServiceProviderMap<Magic> map = new ServiceProviderMap<>(Magic.class);
037    /** @serial */ private final Consumer<Magic> initializer;
038
039    /**
040     * Sole constructor.
041     *
042     * @param   initializer     The {@link Consumer} to apply to newly
043     *                          allocated {@link Magic} providers.
044     */
045    public MagicMap(Consumer<Magic> initializer) {
046        this.initializer = initializer;
047    }
048
049    /**
050     * Reload the underlying {@link ServiceProviderMap} and add the
051     * corresponding entries.
052     *
053     * @return  {@link.this}
054     */
055    public MagicMap reload() {
056        var iterator = map.reload().values().iterator();
057
058        while (iterator.hasNext()) {
059            var magic = iterator.next();
060
061            try {
062                var keys = Set.of(magic.getMagicNames());
063
064                if (! keySet().containsAll(keys)) {
065                    if (initializer != null) {
066                        initializer.accept(magic);
067                    }
068
069                    for (var key : keys) {
070                        computeIfAbsent(key, k -> magic);
071                    }
072                }
073            } catch (Exception exception) {
074            }
075        }
076
077        return this;
078    }
079}
