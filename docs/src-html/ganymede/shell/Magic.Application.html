040/**
041 * {@link Magic} service interface.
042 *
043 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
044 */
045public interface Magic {
046
047    /**
048     * Cell {@link Magic} indicator.
049     */
050    public static final String CELL = "%%";
051
052    /**
053     * See {@link #decode(String)}.
054     */
055    public static final Base64.Decoder BASE64_DECODER = Base64.getDecoder();
056
057    /**
058     * See {@link #encode(String)}.
059     */
060    public static final Base64.Encoder BASE64_ENCODER = Base64.getEncoder();
061
062    /**
063     * Method to get the names associated with {@link.this} {@link Magic}.
064     *
065     * @return  The names.
066     */
067    public String[] getMagicNames();
068
069    /**
070     * Method to get the description of {@link.this} {@link Magic}.
071     *
072     * @return  The description.
073     */
074    public String getDescription();
075
076    /**
077     * Entry-point method.  Executed in the {@link ganymede.shell.Shell}.
078     * Default implementation sends to the
079     * {@link #execute(NotebookContext,String,String)} method in the
080     * {@link jdk.jshell.JShell} instance.
081     *
082     * @param   shell           The {@link Shell}.
083     * @param   in              The {@code in} {@link InputStream}.
084     * @param   out             The {@code out} {@link PrintStream}.
085     * @param   err             The {@code err} {@link PrintStream}.
086     * @param   line0           The initial magic line.
087     * @param   code            The remainder of the cell.
088     */
089    default void execute(Shell shell,
090                         InputStream in, PrintStream out, PrintStream err,
091                         String line0, String code) throws Exception {
092        sendTo(shell, getMagicNames()[0], line0, code);
093    }
094
095    /**
096     * Method to determine code's {@link Message.completeness completeness}.
097     *
098     * @param   line0           The initial magic line.
099     * @param   code            The code to execute.
100     *
101     * @return  The code's {@link Message.completeness completeness}.
102     */
103    public Message.completeness isComplete(String line0, String code);
104
105    /**
106     * Implementation method.  Executed in the {@link jdk.jshell.JShell}
107     * instance.
108     *
109     * @param   __              The {@link NotebookContext}.
110     * @param   line0           The initial magic line.
111     * @param   code            The remainder of the cell.
112     */
113    public void execute(NotebookContext __, String line0, String code) throws Exception;
114
115    /**
116     * Method to determine if the code is cell magic (starts with
117     * '{@code %%}').
118     *
119     * @param   code            The code to test.
120     *
121     * @return  {@code true} if cell magic; {@code false} otherwise.
122     */
123    public static boolean isCellMagic(String code) {
124        return code.startsWith(CELL);
125    }
126
127    /**
128     * Method to parse a cell magic command.  The code must start with
129     * "{@code %%}".  "{@code #}" to the EOL is a comment.  The command is
130     * either a word or a single punctuation character.  Words may be quoted
131     * with double quotes.
132     *
133     * @param   code            The code to parse.
134     *
135     * @return  The array of command arguments.
136     *
137     * @throws  IllegalArgumentException
138     *                          If the magic line can't be parsed.
139     */
140    public static String[] getCellMagicCommand(String code) {
141        var list = new LinkedList<String>();
142
143        try (var reader = new StringReader(code)) {
144            var tokenizer = new StreamTokenizer(reader);
145
146            tokenizer.eolIsSignificant(true);
147            tokenizer.lowerCaseMode(false);
148            tokenizer.commentChar('#');
149            tokenizer.wordChars('0', '9');
150
151            for (int i = 0; i < CELL.length(); i += 1) {
152                tokenizer.nextToken();
153
154                if (tokenizer.ttype != CELL.charAt(i)) {
155                    throw new IllegalArgumentException();
156                }
157            }
158
159            tokenizer.nextToken();
160
161            switch (tokenizer.ttype) {
162            case TT_WORD:
163                list.add(tokenizer.sval);
164                break;
165
166            default:
167                list.add(Character.toString(tokenizer.ttype));
168                break;
169
170            case TT_EOF:
171            case TT_EOL:
172                throw new IllegalArgumentException();
173                /* break; */
174            }
175
176            IntStream.range(0, 256)
177                .filter(t -> "\"\\#".indexOf(t) == -1)
178                .filter(t -> (! Character.isWhitespace(t)))
179                .forEach(t -> tokenizer.wordChars(t, t));
180
181            for (;;) {
182                tokenizer.nextToken();
183
184                if (tokenizer.ttype == TT_EOF || tokenizer.ttype == TT_EOL) {
185                    break;
186                }
187
188                switch (tokenizer.ttype) {
189                case TT_WORD:
190                    list.add(tokenizer.sval);
191                    break;
192
193                default:
194                    list.add(Character.toString(tokenizer.ttype));
195                    break;
196                }
197            }
198        } catch (IllegalStateException exception) {
199            throw exception;
200        } catch (Exception exception) {
201            throw new IllegalArgumentException(code.split("\\R", 2)[0]);
202        }
203
204        return list.toArray(new String[] { });
205    }
206
207    /**
208     * Static method to send a request to be executed in the
209     * {@link jdk.jshell.JShell} instance.  The
210     * {@link #sendTo(Shell,String,String,String)} method packs the
211     * arguments and creates a
212     * {@link #receive(String,NotebookContext,String,String)} expression which
213     * is evaluated in the {@link jdk.jshell.JShell}.
214     *
215     * @param   shell           The {@link Shell}.
216     * @param   name            The magic name.
217     * @param   line0           The initial magic line.
218     * @param   code            The remainder of the cell.
219     */
220    public static void sendTo(Shell shell, String name, String line0, String code) throws Exception {
221        var expression =
222            String.format("__.invokeStaticMethod(\"%s\", \"%s\", new Class<?>[] { String.class, ganymede.notebook.NotebookContext.class, String.class, String.class }, \"%s\", __, \"%s\", \"%s\")",
223                          Magic.class.getName(), "receive",
224                          name, encode(line0), encode(code));
225
226        shell.evaluate(expression);
227    }
228
229    /**
230     * Static {@link MagicMap} instance used by
231     * {@link #receive(String,NotebookContext,String,String)}.
232     */
233    public static MagicMap MAP = new MagicMap();
234
235    /**
236     * Static method to receive a request in the {@link jdk.jshell.JShell}
237     * instance.  The {@link #sendTo(Shell,String,String,String)} method
238     * packs the arguments and creates a
239     * {@link #receive(String,NotebookContext,String,String)} expression which
240     * is evaluated in the {@link jdk.jshell.JShell}.
241     *
242     * @param   name            The magic name.
243     * @param   __              The {@link NotebookContext}.
244     * @param   line0           The initial magic line.
245     * @param   code            The remainder of the cell.
246     */
247    public static void receive(String name, NotebookContext __,
248                               String line0, String code) throws Exception {
249        MAP.reload();
250
251        if (MAP.containsKey(name)) {
252            MAP.get(name).execute(__, decode(line0), decode(code));
253        } else {
254            throw new IllegalStateException("Magic '" + name + "' not found");
255        }
256    }
257
258    /**
259     * Convenience method to {@link Base64}-decode a {@link String}.
260     *
261     * @param   string          The encoded {@link String}.
262     *
263     * @return  The decoded {@link String}.
264     */
265    public static String decode(String string) {
266        return new String(BASE64_DECODER.decode(string), UTF_8);
267    }
268
269    /**
270     * Convenience method to {@link Base64}-encode a {@link String}.
271     *
272     * @param   string          The un-encoded {@link String}.
273     *
274     * @return  The encoded {@link String}.
275     */
276    public static String encode(String string) {
277        return BASE64_ENCODER.encodeToString(((string != null) ? string : "").getBytes(UTF_8));
278    }
279
280    /**
281     * {@link Magic} {@link Application Application} instance to parse code,
282     * find {@link Magic}, and execute.
283     *
284     * {@bean.info}
285     */
286    @Data
287    public static class Application {
288        private final String line0;
289        private final String code;
290        private final String[] argv;
291
292        /**
293         * Sole public constructor.
294         *
295         * @param       code    The complete cell code.
296         */
297        public Application(String code) {
298            this(isCellMagic(code) ? code.split("\\R", 2) : new String[] { null, code });
299        }
300
301        private Application(String[] lines) {
302            this(lines[0], (lines.length > 1) ? lines[1] : "");
303        }
304
305        private Application(String line0, String code) {
306            this.line0 = line0;
307            this.code = code;
308            this.argv = (line0 != null) ? getCellMagicCommand(line0) : new String[] { };
309        }
310
311        /**
312         * Method to determine if the code has a "magic" line.
313         *
314         * @return      {@code true} or {@code false}.
315         */
316        public boolean hasMagicLine() { return (line0 != null); }
317
318        /**
319         * Method to get the specified {@link Magic} name.
320         *
321         * @return      The specified {@link Magic} name; {@code null} if
322         *              none is specified.
323         */
324        public String getMagicName() {
325            return (argv.length > 0) ? argv[0] : null;
326        }
327
328        /**
329         * Method to invoke the {@link Magic}.
330         *
331         * @param       magic   The {@link Magic} to apply.
332         * @param       shell   The {@link Shell}.
333         * @param       in      The {@code in} {@link InputStream}.
334         * @param       out     The {@code out} {@link PrintStream}.
335         * @param       err     The {@code err} {@link PrintStream}.
336         */
337        public void apply(Magic magic, Shell shell, InputStream in, PrintStream out, PrintStream err) throws Exception {
338            magic.execute(shell, in, out, err, getLine0(), getCode());
339        }
340    }
