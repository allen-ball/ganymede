021import ganymede.notebook.NotebookContext;
022import ganymede.server.Message;
023import java.io.InputStream;
024import java.io.PrintStream;
025import java.io.StreamTokenizer;
026import java.io.StringReader;
027import java.util.LinkedList;
028import java.util.stream.IntStream;
029import lombok.Data;
030
031import static java.io.StreamTokenizer.TT_EOF;
032import static java.io.StreamTokenizer.TT_EOL;
033import static java.io.StreamTokenizer.TT_WORD;
034
035/**
036 * {@link Magic} service interface.
037 *
038 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
039 */
040public interface Magic {
041
042    /**
043     * Cell {@link Magic} indicator.
044     */
045    public static final String CELL = "%%";
046
047    /**
048     * Method to get the names associated with {@link.this} {@link Magic}.
049     *
050     * @return  The names.
051     */
052    public String[] getMagicNames();
053
054    /**
055     * Method to get the description of {@link.this} {@link Magic}.
056     *
057     * @return  The description.
058     */
059    public String getDescription();
060
061    /**
062     * Entry-point method.  Executed in the {@link ganymede.shell.Shell}.
063     * Default implementation sends to the
064     * {@link #execute(NotebookContext,String,String)} method in the
065     * {@link jdk.jshell.JShell} instance.
066     *
067     * @param   shell           The {@link Shell}.
068     * @param   in              The {@code in} {@link InputStream}.
069     * @param   out             The {@code out} {@link PrintStream}.
070     * @param   err             The {@code err} {@link PrintStream}.
071     * @param   line0           The initial magic line.
072     * @param   code            The remainder of the cell.
073     */
074    default void execute(Shell shell,
075                         InputStream in, PrintStream out, PrintStream err,
076                         String line0, String code) throws Exception {
077        NotebookContext.magic(shell, getMagicNames()[0], line0, code);
078    }
079
080    /**
081     * Method to determine code's {@link Message.completeness completeness}.
082     *
083     * @param   line0           The initial magic line.
084     * @param   code            The code to execute.
085     *
086     * @return  The code's {@link Message.completeness completeness}.
087     */
088    public Message.completeness isComplete(String line0, String code);
089
090    /**
091     * Implementation method.  Executed in the {@link jdk.jshell.JShell}
092     * instance.
093     *
094     * @param   __              The {@link NotebookContext}.
095     * @param   line0           The initial magic line.
096     * @param   code            The remainder of the cell.
097     */
098    public void execute(NotebookContext __, String line0, String code) throws Exception;
099
100    /**
101     * Method to determine if the code is cell magic (starts with
102     * '{@code %%}').
103     *
104     * @param   code            The code to test.
105     *
106     * @return  {@code true} if cell magic; {@code false} otherwise.
107     */
108    public static boolean isCellMagic(String code) {
109        return code.startsWith(CELL);
110    }
111
112    /**
113     * Method to parse a cell magic command.  The code must start with
114     * "{@code %%}".  "{@code #}" to the EOL is a comment.  The command is
115     * either a word or a single punctuation character.  Words may be quoted
116     * with double quotes.
117     *
118     * @param   code            The code to parse.
119     *
120     * @return  The array of command arguments.
121     *
122     * @throws  IllegalArgumentException
123     *                          If the magic line can't be parsed.
124     */
125    public static String[] getCellMagicCommand(String code) {
126        var list = new LinkedList<String>();
127
128        try (var reader = new StringReader(code)) {
129            var tokenizer = new StreamTokenizer(reader);
130
131            tokenizer.eolIsSignificant(true);
132            tokenizer.lowerCaseMode(false);
133            tokenizer.commentChar('#');
134            tokenizer.wordChars('0', '9');
135
136            for (int i = 0; i < CELL.length(); i += 1) {
137                tokenizer.nextToken();
138
139                if (tokenizer.ttype != CELL.charAt(i)) {
140                    throw new IllegalArgumentException();
141                }
142            }
143
144            tokenizer.nextToken();
145
146            switch (tokenizer.ttype) {
147            case TT_WORD:
148                list.add(tokenizer.sval);
149                break;
150
151            default:
152                list.add(Character.toString(tokenizer.ttype));
153                break;
154
155            case TT_EOF:
156            case TT_EOL:
157                throw new IllegalArgumentException();
158                /* break; */
159            }
160
161            IntStream.range(0, 256)
162                .filter(t -> "\"\\#".indexOf(t) == -1)
163                .filter(t -> (! Character.isWhitespace(t)))
164                .forEach(t -> tokenizer.wordChars(t, t));
165
166            for (;;) {
167                tokenizer.nextToken();
168
169                if (tokenizer.ttype == TT_EOF || tokenizer.ttype == TT_EOL) {
170                    break;
171                }
172
173                switch (tokenizer.ttype) {
174                case TT_WORD:
175                    list.add(tokenizer.sval);
176                    break;
177
178                default:
179                    list.add(Character.toString(tokenizer.ttype));
180                    break;
181                }
182            }
183        } catch (IllegalStateException exception) {
184            throw exception;
185        } catch (Exception exception) {
186            throw new IllegalArgumentException(code.split("\\R", 2)[0]);
187        }
188
189        return list.toArray(new String[] { });
190    }
191
192    /**
193     * {@link Magic} {@link Application Application} instance to parse code,
194     * find {@link Magic}, and execute.
195     *
196     * {@bean.info}
197     */
198    @Data
199    public static class Application {
200        private final String line0;
201        private final String code;
202        private final String[] argv;
203
204        /**
205         * Sole public constructor.
206         *
207         * @param       code    The complete cell code.
208         */
209        public Application(String code) {
210            this(isCellMagic(code) ? code.split("\\R", 2) : new String[] { null, code });
211        }
212
213        private Application(String[] lines) {
214            this(lines[0], (lines.length > 1) ? lines[1] : "");
215        }
216
217        private Application(String line0, String code) {
218            this.line0 = line0;
219            this.code = code;
220            this.argv = (line0 != null) ? getCellMagicCommand(line0) : new String[] { };
221        }
222
223        /**
224         * Method to determine if the code has a "magic" line.
225         *
226         * @return      {@code true} or {@code false}.
227         */
228        public boolean hasMagicLine() { return (line0 != null); }
229
230        /**
231         * Method to get the specified {@link Magic} name.
232         *
233         * @return      The specified {@link Magic} name; {@code null} if
234         *              none is specified.
235         */
236        public String getMagicName() {
237            return (argv.length > 0) ? argv[0] : null;
238        }
239
240        /**
241         * Method to invoke the {@link Magic}.
242         *
243         * @param       magic   The {@link Magic} to apply.
244         * @param       shell   The {@link Shell}.
245         * @param       in      The {@code in} {@link InputStream}.
246         * @param       out     The {@code out} {@link PrintStream}.
247         * @param       err     The {@code err} {@link PrintStream}.
248         */
249        public void apply(Magic magic, Shell shell, InputStream in, PrintStream out, PrintStream err) throws Exception {
250            magic.execute(shell, in, out, err, getLine0(), getCode());
251        }
252    }
253}
