021import ganymede.notebook.NotebookContext;
022import ganymede.server.Message;
023import java.io.InputStream;
024import java.io.PrintStream;
025import java.util.Optional;
026import java.util.stream.Stream;
027import lombok.Data;
028import org.codehaus.plexus.util.cli.CommandLineUtils;
029
030/**
031 * {@link Magic} service interface.
032 *
033 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
034 */
035public interface Magic {
036
037    /**
038     * Cell {@link Magic} indicator.
039     */
040    public static final String CELL = "%%";
041
042    /**
043     * Special-case {@link Magic} name.
044     */
045    public static final String BANG = "!";
046
047    /**
048     * Method to get a configured instance of this {@link Magic}.
049     * Subclasses may override to return an instance that requires
050     * additional resources and/or configuration.  Default implementaion
051     * returns {@link.this}.
052     *
053     * @return  An {@link Optional} containing the configured instance.
054     */
055    public default Optional<? extends Magic> instance() {
056        return Optional.of(this);
057    }
058
059    /**
060     * Method to get the names associated with {@link.this} {@link Magic}.
061     *
062     * @return  The names.
063     */
064    public String[] getMagicNames();
065
066    /**
067     * Method to get the description of {@link.this} {@link Magic}.
068     *
069     * @return  The description.
070     */
071    public String getDescription();
072
073    /**
074     * Method to get the usage of {@link.this} {@link Magic}.
075     *
076     * @return  The usage.
077     */
078    public String getUsage();
079
080    /**
081     * {@link ganymede.shell.Shell} execution method.  Default
082     * implementation sends to the {@link #execute(String,String)} method in
083     * the {@link jdk.jshell.JShell} instance.
084     *
085     * @param   shell           The {@link Shell}.
086     * @param   in              The {@code in} {@link InputStream}.
087     * @param   out             The {@code out} {@link PrintStream}.
088     * @param   err             The {@code err} {@link PrintStream}.
089     * @param   line0           The initial magic line.
090     * @param   code            The remainder of the cell.
091     */
092    default void execute(Shell shell,
093                         InputStream in, PrintStream out, PrintStream err,
094                         String line0, String code) throws Exception {
095        NotebookContext.magic(shell, getMagicNames()[0], line0, code);
096    }
097
098    /**
099     * Method to determine code's {@link Message.completeness completeness}.
100     *
101     * @param   line0           The initial magic line.
102     * @param   code            The code to execute.
103     *
104     * @return  The code's {@link Message.completeness completeness}.
105     */
106    public Message.completeness isComplete(String line0, String code);
107
108    /**
109     * {@link jdk.jshell.JShell} configuation method.
110     *
111     * @param   context         The {@link NotebookContext}.
112     */
113    public void configure(NotebookContext context);
114
115    /**
116     * {@link jdk.jshell.JShell} execution method.
117     *
118     * @param   line0           The initial magic line.
119     * @param   code            The remainder of the cell.
120     */
121    public void execute(String line0, String code) throws Exception;
122
123    /**
124     * Method to determine if the code is cell magic (starts with
125     * '{@code %%}').
126     *
127     * @param   code            The code to test.
128     *
129     * @return  {@code true} if cell magic; {@code false} otherwise.
130     */
131    public static boolean isCellMagic(String code) {
132        return code.startsWith(CELL);
133    }
134
135    /**
136     * Method to parse a cell magic command.  The code must start with
137     * "{@code %%}".  "{@code #}" to the EOL is a comment.  The command is
138     * either a word or a single punctuation character.  Words may be quoted
139     * with double quotes.
140     *
141     * @param   code            The code to parse.
142     *
143     * @return  The array of command arguments.
144     *
145     * @throws  IllegalArgumentException
146     *                          If the magic line can't be parsed.
147     */
148    public static String[] getCellMagicCommand(String code) {
149        var line0 = code.split("\\R", 2)[0];
150        var argv = new String[] { line0 };
151
152        try {
153            argv = CommandLineUtils.translateCommandline(argv[0]);
154
155            if (argv[0].startsWith(CELL)) {
156                argv[0] = argv[0].substring(CELL.length()).trim();
157
158                if (argv[0].isBlank()) {
159                    argv = Stream.of(argv).skip(1).toArray(String[]::new);
160                }
161
162                if (argv[0].startsWith(BANG) && argv[0].length() > BANG.length()) {
163                    argv =
164                        Stream.concat(Stream.of(BANG, argv[0].substring(BANG.length())),
165                                      Stream.of(argv).skip(1))
166                        .toArray(String[]::new);
167                }
168            } else {
169                throw new IllegalArgumentException();
170            }
171        } catch (IllegalStateException exception) {
172            throw exception;
173        } catch (Exception exception) {
174            throw new IllegalArgumentException(line0);
175        }
176
177        return argv;
178    }
179
180    /**
181     * {@link Magic} {@link Application Application} instance to parse code,
182     * find {@link Magic}, and execute.
183     *
184     * {@bean.info}
185     */
186    @Data
187    public static class Application {
188        private final String line0;
189        private final String code;
190        private final String[] argv;
191
192        /**
193         * Sole public constructor.
194         *
195         * @param       code    The complete cell code.
196         */
197        public Application(String code) {
198            this(isCellMagic(code) ? code.split("\\R", 2) : new String[] { null, code });
199        }
200
201        private Application(String[] lines) {
202            this(lines[0], (lines.length > 1) ? lines[1] : "");
203        }
204
205        private Application(String line0, String code) {
206            this.line0 = line0;
207            this.code = code;
208            this.argv = (line0 != null) ? getCellMagicCommand(line0) : new String[] { };
209        }
210
211        /**
212         * Method to determine if the code has a "magic" line.
213         *
214         * @return      {@code true} or {@code false}.
215         */
216        public boolean hasMagicLine() { return (line0 != null); }
217
218        /**
219         * Method to get the specified {@link Magic} name.
220         *
221         * @return      The specified {@link Magic} name; {@code null} if
222         *              none is specified.
223         */
224        public String getMagicName() {
225            return (argv.length > 0) ? argv[0] : null;
226        }
227
228        /**
229         * Method to invoke the {@link Magic} locally.
230         *
231         * @param       shell   The {@link Shell}.
232         * @param       magic   The {@link Magic} to apply.
233         * @param       in      The {@code in} {@link InputStream}.
234         * @param       out     The {@code out} {@link PrintStream}.
235         * @param       err     The {@code err} {@link PrintStream}.
236         */
237        public void apply(Shell shell, Magic magic, InputStream in, PrintStream out, PrintStream err) throws Exception {
238            magic.execute(shell, in, out, err, getLine0(), getCode());
239        }
240
241        /**
242         * Method to invoke the {@link Magic} remotely.
243         *
244         * @param       shell   The {@link Shell}.
245         */
246        public void apply(Shell shell) {
247            NotebookContext.magic(shell, getMagicName(), getLine0(), getCode());
248        }
249    }
250}
