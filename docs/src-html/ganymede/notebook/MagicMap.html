021import ganymede.util.ServiceProviderMap;
022import java.util.ServiceLoader;
023import java.util.Set;
024import java.util.TreeMap;
025import java.util.function.Consumer;
026import java.util.stream.Stream;
027
028/**
029 * {@link Magic} {@link java.util.Map}.
030 *
031 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
032 */
033public class MagicMap extends TreeMap<String,Magic> {
034    private static final long serialVersionUID = -5870351830510141326L;
035
036    /** @serial */ private final Consumer<Magic> initializer;
037    /** @serial */ private final ServiceProviderMap<Magic> map;
038
039    /**
040     * Sole constructor.
041     *
042     * @param   subtype         The {@link Magic} subtype.
043     * @param   initializer     The {@link Consumer} to apply to newly
044     *                          allocated {@link Magic} providers.
045     */
046    public MagicMap(Class<? extends Magic> subtype, Consumer<Magic> initializer) {
047        this.initializer = initializer;
048        this.map = new ServiceProviderMap<>(subtype.asSubclass(Magic.class), this::compute);
049    }
050
051    private Magic compute(ServiceLoader.Provider<Magic> provider) {
052        var value = provider.get().instance().orElse(null);
053
054        if (value != null) {
055            if (initializer != null) {
056                initializer.accept(value);
057            }
058
059            Set.of(value.getMagicNames()).stream()
060                .forEach(t -> computeIfAbsent(t, k -> value));
061        }
062
063        return value;
064    }
065
066    /**
067     * Reload the underlying {@link ServiceProviderMap} and add the
068     * corresponding entries.
069     *
070     * @return  {@link.this}
071     */
072    public MagicMap reload() {
073        map.reload();
074
075        return this;
076    }
077}
