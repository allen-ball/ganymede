021import com.fasterxml.jackson.databind.JsonNode;
022import ganymede.notebook.Magic;
023import java.io.InputStreamReader;
024import java.util.List;
025import java.util.Objects;
026import java.util.Optional;
027import javax.script.ScriptEngine;
028import javax.script.ScriptEngineManager;
029import lombok.NoArgsConstructor;
030import lombok.ToString;
031import lombok.extern.log4j.Log4j2;
032import org.springframework.core.io.ClassPathResource;
033
034import static java.nio.charset.StandardCharsets.UTF_8;
035import static javax.script.ScriptContext.ENGINE_SCOPE;
036import static lombok.AccessLevel.PROTECTED;
037
038/**
039 * Abstract
040 * {@link.uri https://www.jcp.org/en/jsr/detail?id=223 target=newtab JSR 223}
041 * {@link ScriptEngine} {@link Magic} base class.
042 *
043 * @see ScriptEngineManager
044 *
045 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
046 */
047@NoArgsConstructor(access = PROTECTED) @ToString @Log4j2
048public abstract class AbstractScriptEngineMagic extends AbstractMagic
049                                                implements AnnotatedScriptEngineMagic {
050    /** See {@link #engine()}. */
051    protected ScriptEngine engine = null;
052
053    /**
054     * Method to get the {@link ScriptEngine}.
055     *
056     * @return  The {@link ScriptEngine} if it can be instantiated;
057     *          {@code null} otherwise.
058     */
059    protected ScriptEngine engine() {
060        if (engine == null) {
061            var manager = new ScriptEngineManager(getClass().getClassLoader());
062
063            engine = manager.getEngineByName(getScriptEngineName());
064
065            if (engine == null || (! initialize(engine))) {
066                engine =
067                    getExtensions().stream()
068                    .map(t -> manager.getEngineByExtension(t))
069                    .filter(Objects::nonNull)
070                    .filter(this::initialize)
071                    .findFirst().orElse(null);
072            }
073        }
074
075        return engine;
076    }
077
078    /**
079     * Method to initialize the {@link ScriptEngine}.  Default
080     * implementation looks for a {@link Class} resource named with the
081     * {@link Class#getSimpleName()} and one of the {@link ScriptEngine}'s
082     * {@link javax.script.ScriptEngineFactory#getExtensions() extensions}.
083     * A {@link ScriptEngine} instance that cannot be initialized will be
084     * ignored.
085     *
086     * @return  {@code true} if the {@link ScriptEngine} can be initialized,
087     *          {@code false} otherwise.
088     */
089    protected boolean initialize(ScriptEngine engine) {
090        var initialized = true;
091        var resource =
092            Optional.ofNullable(engine.getFactory().getExtensions()).stream()
093            .flatMap(List::stream)
094            .map(t -> getClass().getSimpleName() + "." + t)
095            .map(t -> new ClassPathResource(t, getClass()))
096            .filter(ClassPathResource::exists)
097            .findFirst().orElse(null);
098
099        if (initialized && resource != null) {
100            try (var in = new InputStreamReader(resource.getInputStream(), UTF_8)) {
101                engine.eval(in, context.context);
102            } catch (Throwable throwable) {
103                initialized = false;
104                throwable.printStackTrace(System.err);
105            }
106        }
107
108        return initialized;
109    }
110
111    @Override
112    public void execute(String line0, String code, JsonNode metadata) throws Exception {
113        execute(Magic.getCellMagicCommand(line0), code);
114    }
115
116    /**
117     * Method provided for subclass implementations to intercept calculation
118     * of {@link ScriptEngine#ARGV} binding.
119     *
120     * @param   argv            The first line parsed as an array of
121     *                          {@link String}s.
122     * @param   code            The remainder of the cell.
123     */
124    protected void execute(String[] argv, String code) throws Exception {
125        var bindings = context.context.getBindings(ENGINE_SCOPE);
126
127        try {
128            bindings.put(ScriptEngine.ARGV, argv);
129            execute(code);
130        } finally {
131            bindings.remove(ScriptEngine.ARGV);
132        }
133    }
134
135    /**
136     * Target of {@link #execute(String[],String)}.  The {@code argv} is
137     * available in the {@link javax.script.ScriptContext}
138     * {@code ENGINE_SCOPE} {@link javax.script.Bindings} as
139     * {@link ScriptEngine#ARGV ScriptEngine.ARGV}.
140     *
141     * @param   code            The remainder of the cell.
142     */
143    protected void execute(String code) {
144        var engine = engine();
145
146        if (engine != null) {
147            try {
148                if (! code.isBlank()) {
149                    render(engine.eval(code, context.context));
150                } else {
151                    show();
152                }
153            } catch (Throwable throwable) {
154                throwable.printStackTrace(System.err);
155            }
156        } else {
157            System.err.format("No %s REPL available\n", getMagicNames()[0]);
158        }
159    }
160
161    /**
162     * Callback to render the results of
163     * {@link ScriptEngine#eval(String,ScriptContext)}.  Made available for
164     * template engines.  Default implementation does nothing.
165     *
166     * @param   object          The result of
167     *                          {@link ScriptEngine#eval(String,ScriptContext)}.
168     */
169    protected void render(Object object) { }
170
171    /**
172     * Callback to show information about the {@link ScriptEngine} instance.
173     * Default implementation does nothing.
174     */
175    protected void show() { }
176}
