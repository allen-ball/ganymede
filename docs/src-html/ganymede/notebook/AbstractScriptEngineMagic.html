021import com.fasterxml.jackson.databind.JsonNode;
022import java.io.InputStreamReader;
023import java.util.List;
024import java.util.Objects;
025import java.util.Optional;
026import javax.script.ScriptEngine;
027import javax.script.ScriptEngineManager;
028import lombok.NoArgsConstructor;
029import lombok.ToString;
030import lombok.extern.log4j.Log4j2;
031import org.springframework.core.io.ClassPathResource;
032
033import static java.nio.charset.StandardCharsets.UTF_8;
034import static javax.script.ScriptContext.ENGINE_SCOPE;
035import static lombok.AccessLevel.PROTECTED;
036
037/**
038 * Abstract
039 * {@link.uri https://www.jcp.org/en/jsr/detail?id=223 target=newtab JSR 223}
040 * {@link ScriptEngine} {@link Magic} base class.
041 *
042 * @see ScriptEngineManager
043 *
044 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
045 */
046@NoArgsConstructor(access = PROTECTED) @ToString @Log4j2
047public abstract class AbstractScriptEngineMagic extends AbstractMagic implements AnnotatedScriptEngineMagic {
048    /** See {@link #engine()}. */
049    protected ScriptEngine engine = null;
050
051    /**
052     * Method to get the {@link ScriptEngine}.
053     *
054     * @return  The {@link ScriptEngine} if it can be instantiated;
055     *          {@code null} otherwise.
056     */
057    protected ScriptEngine engine() {
058        if (engine == null) {
059            var manager = new ScriptEngineManager(context.getClassLoader());
060
061            engine = manager.getEngineByName(getScriptEngineName());
062
063            if (engine == null || (! initialize(engine))) {
064                engine =
065                    getExtensions().stream()
066                    .map(t -> manager.getEngineByExtension(t))
067                    .filter(Objects::nonNull)
068                    .filter(this::initialize)
069                    .findFirst().orElse(null);
070            }
071        }
072
073        return engine;
074    }
075
076    /**
077     * Method to initialize the {@link ScriptEngine}.  Default
078     * implementation looks for a {@link Class} resource named with the
079     * {@link Class#getSimpleName()} and one of the {@link ScriptEngine}'s
080     * {@link javax.script.ScriptEngineFactory#getExtensions() extensions}.
081     * A {@link ScriptEngine} instance that cannot be initialized will be
082     * ignored.
083     *
084     * @return  {@code true} if the {@link ScriptEngine} can be initialized,
085     *          {@code false} otherwise.
086     */
087    protected boolean initialize(ScriptEngine engine) {
088        var initialized = true;
089        var resource = getInitScript(engine.getFactory().getExtensions());
090
091        if (initialized && resource != null) {
092            try (var in = new InputStreamReader(resource.getInputStream(), UTF_8)) {
093                engine.eval(in, context.context);
094            } catch (Throwable throwable) {
095                initialized = false;
096                throwable.printStackTrace(System.err);
097            }
098        }
099
100        return initialized;
101    }
102
103    /**
104     * Method to locate {@link ScriptEngine} initialization script.
105     *
106     * @param   suffixes        The {@link List} of script suffixes.
107     *
108     * @return  Return the first resource that matches this class's simple
109     *          name and suffix (and exists).
110     */
111    protected ClassPathResource getInitScript(List<String> suffixes) {
112        var resource =
113            Optional.ofNullable(suffixes).stream()
114            .flatMap(List::stream)
115            .map(t -> getClass().getSimpleName() + "." + t)
116            .map(t -> new ClassPathResource(t, getClass()))
117            .filter(ClassPathResource::exists)
118            .findFirst().orElse(null);
119
120        return resource;
121    }
122
123    @Override
124    public void execute(String line0, String code, JsonNode metadata) throws Exception {
125        execute(Magic.getCellMagicCommand(line0), code);
126    }
127
128    /**
129     * Method provided for subclass implementations to intercept calculation
130     * of {@link ScriptEngine#ARGV} binding.
131     *
132     * @param   argv            The first line parsed as an array of
133     *                          {@link String}s.
134     * @param   code            The remainder of the cell.
135     */
136    protected void execute(String[] argv, String code) throws Exception {
137        var bindings = context.context.getBindings(ENGINE_SCOPE);
138
139        try {
140            bindings.put(ScriptEngine.ARGV, argv);
141            execute(code);
142        } finally {
143            bindings.remove(ScriptEngine.ARGV);
144        }
145    }
146
147    /**
148     * Target of {@link #execute(String[],String)}.  The {@code argv} is
149     * available in the {@link javax.script.ScriptContext}
150     * {@code ENGINE_SCOPE} {@link javax.script.Bindings} as
151     * {@link ScriptEngine#ARGV ScriptEngine.ARGV}.
152     *
153     * @param   code            The remainder of the cell.
154     */
155    protected void execute(String code) {
156        var engine = engine();
157
158        if (engine != null) {
159            try {
160                if (! code.isBlank()) {
161                    render(engine.eval(code, context.context));
162                } else {
163                    show();
164                }
165            } catch (Throwable throwable) {
166                throwable.printStackTrace(System.err);
167            }
168        } else {
169            System.err.format("No %s REPL available\n", getMagicNames()[0]);
170        }
171    }
172
173    /**
174     * Callback to render the results of
175     * {@link ScriptEngine#eval(String,ScriptContext)}.  Made available for
176     * template engines.  Default implementation does nothing.
177     *
178     * @param   object          The result of
179     *                          {@link ScriptEngine#eval(String,ScriptContext)}.
180     */
181    protected void render(Object object) { }
182
183    /**
184     * Callback to show information about the {@link ScriptEngine} instance.
185     * Default implementation does nothing.
186     */
187    protected void show() { }
188}
