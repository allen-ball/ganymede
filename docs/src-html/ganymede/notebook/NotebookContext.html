123     * Method to receive a {@link ganymede.shell.Magic} request in the
124     * {@link JShell} instance.  See
125     * {@link magic(Shell,String,String,String)}.
126     *
127     * @param   name            The magic name.
128     * @param   line0           The initial magic line.
129     * @param   code            The remainder of the cell.
130     */
131    public void magic(String name, String line0, String code) {
132        try {
133            var magic = magics.reload().get(name);
134
135            if (magic != null) {
136                magic.execute(decode(line0), decode(code));
137            } else {
138                System.err.format("Magic '%s' not found\n", name);
139            }
140        } catch (Throwable throwable) {
141            throwable.printStackTrace(System.err);
142        }
143    }

209    /**
210     * Method to generate the bootstrap code for a new {@link JShell}
211     * instance.
212     *
213     * @return  The boostrap code.
214     */
215    public static String bootstrap() {
216        var code =
217            String.format("var %1$s = %2$s.newNotebookContext();\n",
218                          NAME, Notebook.class.getCanonicalName());
219
220        for (var method : getNotebookFunctions()) {
221            code += makeWrapperFor(NAME, method);
222        }
223
224        return code;
225    }
226
227    private static String makeWrapperFor(String instance, Method method) {
228        var types = method.getGenericParameterTypes();
229        var arguments = new String[types.length];
230        var parameters = new String[types.length];
231
232        for (int i = 0; i < arguments.length; i += 1) {
233            arguments[i] = String.format("argument%1$d", i);
234        }
235
236        for (int i = 0; i < parameters.length; i += 1) {
237            parameters[i] =
238                String.format("%1$s %2$s",
239                              types[i].getTypeName(), arguments[i]);
240        }
241
242        var plist = String.join(", ", parameters);
243        var alist = String.join(", ", arguments);
244
245        return String.format("%1$s %2$s(%3$s) { %4$s%5$s.%2$s(%6$s); }\n",
246                             method.getGenericReturnType().getTypeName(),
247                             method.getName(), plist,
248                             Void.TYPE.equals(method.getReturnType()) ? "" : "return ",
249                             instance, alist);
250    }

252    /**
253     * Method to get the {@link NotebookContext} {@link Method}s annotated
254     * with {@link NotebookFunction} that should be linked into the
255     * {@link Notebook} environment.
256     *
257     * @return  The array of {@link Method}s.
258     */
259    public static Method[] getNotebookFunctions() {
260        var functions =
261            Stream.of(NotebookContext.class.getDeclaredMethods())
262            .filter(t -> t.isAnnotationPresent(NotebookFunction.class))
263            .filter(t -> isPublic(t.getModifiers()))
264            .toArray(Method[]::new);
265
266        return functions;
267    }

269    /**
270     * Static method used by the {@link Shell} REPL to update the
271     * {@link NotebookContext} instance before execution.
272     *
273     * @param   shell           The {@link Shell}.
274     */
275    public static void preExecute(Shell shell) {
276        var jshell = shell.jshell();
277        var classpath =
278            shell.classpath().stream()
279            .map(File::getAbsolutePath)
280            .collect(toList());
281
282        evaluate(jshell, "%1$s.classpath.clear()", NAME);
283
284        if (! classpath.isEmpty()) {
285            evaluate(jshell,
286                     "java.util.Collections.addAll(%1$s.classpath, \"%2$s\".split(\",\"))",
287                     NAME, String.join(",", classpath));
288        }
289
290        var imports =
291            jshell.imports()
292            .map(t -> t.source())
293            .map(String::strip)
294            .collect(toCollection(LinkedHashSet::new));
295
296        evaluate(jshell, "%1$s.imports.clear()", NAME);
297
298        if (! imports.isEmpty()) {
299            evaluate(jshell,
300                     "java.util.Collections.addAll(%1$s.imports, \"%2$s\".split(\",\"))",
301                     NAME, String.join(",", imports));
302        }
303
304        var types =
305            jshell.variables()
306            .filter(t -> (! t.subKind().equals(TEMP_VAR_EXPRESSION_SUBKIND)))
307            .collect(toMap(k -> k.name(), v -> v.typeName()));
308
309        evaluate(jshell, "%1$s.types.clear()", NAME);
310
311        for (var entry : types.entrySet()) {
312            evaluate(jshell,
313                     "%1$s.context.getBindings(%2$d).put(\"%3$s\", %3$s)",
314                     NAME, ScriptContext.ENGINE_SCOPE, entry.getKey());
315            evaluate(jshell,
316                     "%1$s.types.put(\"%2$s\", \"%3$s\")",
317                     NAME, entry.getKey(), entry.getValue());
318        }
319    }

331    private static String evaluate(JShell jshell, String expression, Object... argv) {
332        var analyzer = jshell.sourceCodeAnalysis();
333        var info = analyzer.analyzeCompletion(String.format(expression, argv));
334
335        return unescape(jshell.eval(info.source()).get(0).value());
336    }

352                 NAME, name, encode(line0), encode(code));

355    /**
356     * Method to unescape a Java-escaped string literal.
357     *
358     * @param   literal         The {@link String} to unescape.
359     *
360     * @return  The unescaped {@link String}.
361     */
362    public static String unescape(String literal) {
363        var string = literal;
364        /*
365         * https://stackoverflow.com/questions/3537706/how-to-unescape-a-java-string-literal-in-java
366         */
367        if (literal != null) {
368            try (var reader = new StringReader(literal)) {
369                var tokenizer = new StreamTokenizer(reader);
370
371                tokenizer.nextToken();
372
373                if (tokenizer.ttype == '"') {
374                    string = tokenizer.sval;
375                }
376            } catch (IOException exception) {
377            }
378        }
379
380        return string;
381    }
382
383    private static String decode(String string) {
384        return new String(DECODER.decode(string), UTF_8);
385    }
386
387    private static String encode(String string) {
388        return ENCODER.encodeToString(((string != null) ? string : "").getBytes(UTF_8));
389    }

391    /**
392     * {@link ganymede.shell.magic.SQL}-specific context.  Implements
393     * {@link Map} of JDBC URLs to {@link DSLContext}s.
394     */
395    @NoArgsConstructor
396    public class SQL extends LinkedHashMap<String,DSLContext> {
397        private static final long serialVersionUID = -4901551333824542142L;
398
399        /**
400         * {@link List} of most recent {@link ganymede.shell.magic.SQL}
401         * {@link Query Queries}.
402         *
403         * @serial
404         */
405        public final List<Query> queries = new ArrayList<>();
406
407        /**
408         * {@link List} of most recent {@link ganymede.shell.magic.SQL}
409         * {@link Result}s.
410         *
411         * @serial
412         */
413        public final List<Result<Record>> results = new ArrayList<>();
414
415        /**
416         * Target of the {@link ganymede.shell.magic.SQL}
417         * {@link ganymede.shell.Magic}.
418         *
419         * @param       url             The JDBC URL.
420         * @param       username        The JDBC Username.
421         * @param       password        The JDBC Password.
422         *
423         * @return      The {@link DSLContext} corresponding to the URL.
424         */
425        public DSLContext connect(String url, String username, String password) {
426            return computeIfAbsent(toKey(url), k -> DSL.using(url, username, password));
427        }
428
429        private String toKey(String url) {
430            var key = url;
431
432            if (key != null) {
433                try {
434                    var list = new LinkedList<URI>();
435
436                    list.add(URI.create(key));
437
438                    for (;;) {
439                        var last = list.getLast();
440                        var ssp = last.getSchemeSpecificPart();
441
442                        if (last.isOpaque() && ssp.contains(":")) {
443                            list.add(URI.create(ssp));
444                        } else {
445                            break;
446                        }
447                    }
448
449                    var uri = list.removeLast();
450
451                    if (uri.isOpaque()) {
452                        uri = new URI(uri.getScheme(), uri.getSchemeSpecificPart().split(";", 2)[0], null);
453                    } else {
454                        uri = new URI(uri.getScheme(), null, uri.getHost(), uri.getPort(), uri.getPath(), null, null);
455                    }
456
457                    while (! list.isEmpty()) {
458                        uri = new URI(list.removeLast().getScheme(), uri.toString(), null);
459                    }
460
461                    key = uri.toString();
462                } catch (URISyntaxException exception) {
463                    throw new IllegalArgumentException(exception);
464                }
465            }
466
467            return key;
468        }
469    }
