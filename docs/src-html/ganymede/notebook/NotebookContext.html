172    public void magic(String name, String line0, String code) {
173        try {
174            var magic = magics.reload().get(name);
175
176            if (magic != null) {
177                magic.execute(decode(line0), decode(code));
178            } else {
179                System.err.format("Magic '%s' not found\n", name);
180            }
181        } catch (Throwable throwable) {
182            throwable.printStackTrace(System.err);
183        }
184    }

250    /**
251     * Method to generate the bootstrap code for a new {@link JShell}
252     * instance.
253     *
254     * @return  The boostrap code.
255     */
256    public static String bootstrap() {
257        var code =
258            String.format("var %1$s = %2$s.newNotebookContext();\n",
259                          NAME, Notebook.class.getCanonicalName());
260
261        for (var method : getNotebookFunctions()) {
262            code += makeWrapperFor(NAME, method);
263        }
264
265        return code;
266    }
267
268    private static String makeWrapperFor(String instance, Method method) {
269        var types = method.getGenericParameterTypes();
270        var arguments = new String[types.length];
271        var parameters = new String[types.length];
272
273        for (int i = 0; i < arguments.length; i += 1) {
274            arguments[i] = String.format("argument%1$d", i);
275        }
276
277        for (int i = 0; i < parameters.length; i += 1) {
278            parameters[i] =
279                String.format("%1$s %2$s",
280                              types[i].getTypeName(), arguments[i]);
281        }
282
283        var plist = String.join(", ", parameters);
284        var alist = String.join(", ", arguments);
285
286        return String.format("%1$s %2$s(%3$s) { %4$s%5$s.%2$s(%6$s); }\n",
287                             method.getGenericReturnType().getTypeName(),
288                             method.getName(), plist,
289                             Void.TYPE.equals(method.getReturnType()) ? "" : "return ",
290                             instance, alist);
291    }

293    /**
294     * Method to get the {@link NotebookContext} {@link Method}s annotated
295     * with {@link NotebookFunction} that should be linked into the
296     * {@link Notebook} environment.
297     *
298     * @return  The array of {@link Method}s.
299     */
300    public static Method[] getNotebookFunctions() {
301        var functions =
302            Stream.of(NotebookContext.class.getDeclaredMethods())
303            .filter(t -> t.isAnnotationPresent(NotebookFunction.class))
304            .filter(t -> isPublic(t.getModifiers()))
305            .toArray(Method[]::new);
306
307        return functions;
308    }

310    /**
311     * Static method used by the {@link Shell} REPL to update the
312     * {@link NotebookContext} instance before execution.
313     *
314     * @param   shell           The {@link Shell}.
315     */
316    public static void preExecute(Shell shell) {
317        var jshell = shell.jshell();
318
319        evaluate(jshell, "%1$s.refresh()", NAME);
320
321        var classpath =
322            shell.classpath().stream()
323            .map(File::getAbsolutePath)
324            .collect(toList());
325
326        evaluate(jshell, "%1$s.classpath.clear()", NAME);
327
328        if (! classpath.isEmpty()) {
329            evaluate(jshell,
330                     "java.util.Collections.addAll(%1$s.classpath, \"%2$s\".split(\",\"))",
331                     NAME, String.join(",", classpath));
332        }
333
334        var imports =
335            jshell.imports()
336            .map(t -> t.source())
337            .map(String::strip)
338            .collect(toCollection(LinkedHashSet::new));
339
340        evaluate(jshell, "%1$s.imports.clear()", NAME);
341
342        if (! imports.isEmpty()) {
343            evaluate(jshell,
344                     "java.util.Collections.addAll(%1$s.imports, \"%2$s\".split(\",\"))",
345                     NAME, String.join(",", imports));
346        }
347
348        var types =
349            jshell.variables()
350            .filter(t -> (! t.subKind().equals(TEMP_VAR_EXPRESSION_SUBKIND)))
351            .collect(toMap(k -> k.name(), v -> v.typeName()));
352
353        evaluate(jshell, "%1$s.types.clear()", NAME);
354
355        for (var entry : types.entrySet()) {
356            evaluate(jshell,
357                     "%1$s.context.getBindings(%2$d).put(\"%3$s\", %3$s)",
358                     NAME, ScriptContext.ENGINE_SCOPE, entry.getKey());
359            evaluate(jshell,
360                     "%1$s.types.put(\"%2$s\", \"%3$s\")",
361                     NAME, entry.getKey(), entry.getValue());
362        }
363    }

375    private static String evaluate(JShell jshell, String expression, Object... argv) {
376        var analyzer = jshell.sourceCodeAnalysis();
377        var info = analyzer.analyzeCompletion(String.format(expression, argv));
378
379        return unescape(jshell.eval(info.source()).get(0).value());
380    }

399    /**
400     * Method to unescape a Java-escaped string literal.
401     *
402     * @param   literal         The {@link String} to unescape.
403     *
404     * @return  The unescaped {@link String}.
405     */
406    public static String unescape(String literal) {
407        var string = literal;
408        /*
409         * https://stackoverflow.com/questions/3537706/how-to-unescape-a-java-string-literal-in-java
410         */
411        if (literal != null) {
412            try (var reader = new StringReader(literal)) {
413                var tokenizer = new StreamTokenizer(reader);
414
415                tokenizer.nextToken();
416
417                if (tokenizer.ttype == '"') {
418                    string = tokenizer.sval;
419                }
420            } catch (IOException exception) {
421            }
422        }
423
424        return string;
425    }

427    private static String decode(String string) {
428        return new String(DECODER.decode(string), UTF_8);
429    }
430
431    private static String encode(String string) {
432        return ENCODER.encodeToString(((string != null) ? string : "").getBytes(UTF_8));
433    }

435    /**
436     * {@link ganymede.shell.magic.SQL}-specific context.  Implements
437     * {@link Map} of JDBC URLs to {@link DSLContext}s.
438     */
439    @NoArgsConstructor
440    public class SQL extends LinkedHashMap<String,DSLContext> {
441        private static final long serialVersionUID = -4901551333824542142L;
442
443        /**
444         * {@link List} of most recent {@link ganymede.shell.magic.SQL}
445         * {@link Query Queries}.
446         *
447         * @serial
448         */
449        public final List<Query> queries = new ArrayList<>();
450
451        /**
452         * {@link List} of most recent {@link ganymede.shell.magic.SQL}
453         * {@link Result}s.
454         *
455         * @serial
456         */
457        public final List<Result<Record>> results = new ArrayList<>();
458
459        /**
460         * Target of the {@link ganymede.shell.magic.SQL}
461         * {@link ganymede.shell.Magic}.
462         *
463         * @param       url             The JDBC URL.
464         * @param       username        The JDBC Username.
465         * @param       password        The JDBC Password.
466         *
467         * @return      The {@link DSLContext} corresponding to the URL.
468         */
469        public DSLContext connect(String url, String username, String password) {
470            return computeIfAbsent(toKey(url), k -> DSL.using(url, username, password));
471        }
472
473        private String toKey(String url) {
474            var key = url;
475
476            if (key != null) {
477                try {
478                    var list = new LinkedList<URI>();
479
480                    list.add(URI.create(key));
481
482                    for (;;) {
483                        var last = list.getLast();
484                        var ssp = last.getSchemeSpecificPart();
485
486                        if (last.isOpaque() && ssp.contains(":")) {
487                            list.add(URI.create(ssp));
488                        } else {
489                            break;
490                        }
491                    }
492
493                    var uri = list.removeLast();
494
495                    if (uri.isOpaque()) {
496                        uri = new URI(uri.getScheme(), uri.getSchemeSpecificPart().split(";", 2)[0], null);
497                    } else {
498                        uri = new URI(uri.getScheme(), null, uri.getHost(), uri.getPort(), uri.getPath(), null, null);
499                    }
500
501                    while (! list.isEmpty()) {
502                        uri = new URI(list.removeLast().getScheme(), uri.toString(), null);
503                    }
504
505                    key = uri.toString();
506                } catch (URISyntaxException exception) {
507                    throw new IllegalArgumentException(exception);
508                }
509            }
510
511            return key;
512        }
513    }
