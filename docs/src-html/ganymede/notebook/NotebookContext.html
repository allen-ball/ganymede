052/**

093    /**
094     * Method to receive a {@link ganymede.shell.Magic} request in the
095     * {@link JShell} instance.  See
096     * {@link magic(Shell,String,String,String)}.
097     *
098     * @param   name            The magic name.
099     * @param   line0           The initial magic line.
100     * @param   code            The remainder of the cell.
101     */
102    public void magic(String name, String line0, String code) throws Exception {
103        magics.reload();
104
105        if (magics.containsKey(name)) {
106            magics.get(name).execute(this, decode(line0), decode(code));
107        } else {
108            throw new IllegalStateException("Magic '" + name + "' not found");
109        }
110    }

155    /**
156     * Method to generate the bootstrap code for a new {@link JShell}
157     * instance.
158     *
159     * @return  The boostrap code.
160     */
161    public static String bootstrap() {
162        var code =
163            String.format("var __ = %s.newNotebookContext();\n",
164                          Notebook.class.getCanonicalName());
165
166        for (var method : NotebookContext.class.getDeclaredMethods()) {
167            if (method.isAnnotationPresent(NotebookFunction.class)) {
168                if (isPublic(method.getModifiers())) {
169                    code += makeWrapperFor("__", method);
170                }
171            }
172        }
173
174        return code;
175    }
176
177    private static String makeWrapperFor(String instance, Method method) {
178        var types = method.getGenericParameterTypes();
179        var arguments = new String[types.length];
180        var parameters = new String[types.length];
181
182        for (int i = 0; i < arguments.length; i += 1) {
183            arguments[i] = String.format("argument%d", i);
184        }
185
186        for (int i = 0; i < parameters.length; i += 1) {
187            parameters[i] =
188                String.format("%s %s", types[i].getTypeName(), arguments[i]);
189        }
190
191        var plist = String.join(", ", parameters);
192        var alist = String.join(", ", arguments);
193
194        return String.format("%1$s %2$s(%3$s) { %4$s%5$s.%2$s(%6$s); }\n",
195                             method.getGenericReturnType().getTypeName(),
196                             method.getName(), plist,
197                             Void.TYPE.equals(method.getReturnType()) ? "" : "return ",
198                             instance, alist);
199    }

201    /**
202     * Static method used by the {@link Shell} to update the
203     * {@link NotebookContext} members.
204     *
205     * @param   shell           The {@link Shell}.
206     */
207    public static void update(Shell shell) {
208        var jshell = shell.jshell();
209        var classpath =
210            shell.classpath().stream()
211            .map(File::getAbsolutePath)
212            .collect(toList());
213
214        evaluate(jshell, "__.classpath.clear()");
215
216        if (! classpath.isEmpty()) {
217            evaluate(jshell,
218                     "java.util.Collections.addAll(__.classpath, \"%1$s\".split(\",\"))",
219                     String.join(",", classpath));
220        }
221
222        var imports =
223            jshell.imports()
224            .map(t -> t.source())
225            .map(String::strip)
226            .collect(toCollection(LinkedHashSet::new));
227
228        evaluate(jshell, "__.imports.clear()");
229
230        if (! imports.isEmpty()) {
231            evaluate(jshell,
232                     "java.util.Collections.addAll(__.imports, \"%1$s\".split(\",\"))",
233                     String.join(",", imports));
234        }
235
236        var types =
237            jshell.variables()
238            .filter(t -> (! t.subKind().equals(TEMP_VAR_EXPRESSION_SUBKIND)))
239            .filter(t -> (! t.name().equals("__")))
240            .collect(toMap(k -> k.name(), v -> v.typeName()));
241
242        evaluate(jshell, "__.types.clear()");
243
244        for (var entry : types.entrySet()) {
245            evaluate(jshell,
246                     "__.context.getBindings(%1d).put(\"%2$s\", %2$s)",
247                     ScriptContext.ENGINE_SCOPE, entry.getKey());
248            evaluate(jshell,
249                     "__.types.put(\"%1$s\", \"%2$s\")",
250                     entry.getKey(), entry.getValue());
251        }
252    }
253
254    private static String evaluate(JShell jshell, String expression, Object... argv) {
255        var analyzer = jshell.sourceCodeAnalysis();
256        var info = analyzer.analyzeCompletion(String.format(expression, argv));
257
258        return unescape(jshell.eval(info.source()).get(0).value());
259    }

261    /**
262     * Static method invoke a {@link ganymede.shell.Magic} in a
263     * {@link JShell} instance.  See {@link #magic(String,String,String)}.
264     *
265     * @param   shell           The {@link Shell}.
266     * @param   name            The magic name.
267     * @param   line0           The initial magic line.
268     * @param   code            The remainder of the cell.
269     */
270    public static void magic(Shell shell, String name, String line0, String code) {
271        var jshell = shell.jshell();
272
273        evaluate(jshell,
274                 "__.magic(\"%s\", \"%s\", \"%s\")",
275                 name, encode(line0), encode(code));
276    }

278    /**
279     * Method to unescape a Java-escaped spring literal.
280     *
281     * @param   literal         The {@link String} to unescape.
282     *
283     * @return  The unescaped {@link String}.
284     */
285    public static String unescape(String literal) {
286        var string = literal;
287        /*
288         * https://stackoverflow.com/questions/3537706/how-to-unescape-a-java-string-literal-in-java
289         */
290        if (literal != null) {
291            try (var reader = new StringReader(literal)) {
292                var tokenizer = new StreamTokenizer(reader);
293
294                tokenizer.nextToken();
295
296                if (tokenizer.ttype == '"') {
297                    string = tokenizer.sval;
298                }
299            } catch (IOException exception) {
300            }
301        }
302
303        return string;
304    }
305
306    private static String decode(String string) {
307        return new String(DECODER.decode(string), UTF_8);
308    }
309
310    private static String encode(String string) {
311        return ENCODER.encodeToString(((string != null) ? string : "").getBytes(UTF_8));
312    }
