021import ganymede.shell.Shell;
022import java.io.File;
023import java.io.IOException;
024import java.io.StreamTokenizer;
025import java.io.StringReader;
026import java.util.LinkedHashSet;
027import java.util.LinkedList;
028import java.util.List;
029import java.util.Map;
030import java.util.concurrent.ConcurrentSkipListMap;
031import javax.script.ScriptContext;
032import javax.script.SimpleBindings;
033import javax.script.SimpleScriptContext;
034import jdk.jshell.JShell;
035import lombok.NoArgsConstructor;
036import lombok.ToString;
037
038import static java.util.stream.Collectors.toCollection;
039import static java.util.stream.Collectors.toList;
040import static java.util.stream.Collectors.toMap;
041import static javax.script.ScriptContext.ENGINE_SCOPE;
042import static javax.script.ScriptContext.GLOBAL_SCOPE;
043import static jdk.jshell.Snippet.SubKind.TEMP_VAR_EXPRESSION_SUBKIND;
044
045/**
046 * {@link NotebookContext} for {@link Notebook} {@link ganymede.shell.Shell}
047 * {@link JShell} instance.  Bound to {@code __} in the {@link JShell}
048 * instance.
049 *
050 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
051 */
052@NoArgsConstructor @ToString
053public class NotebookContext {
054
055    /**
056     * Common {@link ScriptContext} supplied to
057     * {@link ganymede.shell.Magic#execute(NotebookContext,String,String)}.
058     */
059    public final ScriptContext context =
060        new SimpleScriptContext() {
061            {
062                setBindings(new SimpleBindings(new ConcurrentSkipListMap<>()), GLOBAL_SCOPE);
063                setBindings(new SimpleBindings(new ConcurrentSkipListMap<>()), ENGINE_SCOPE);
064            }
065        };
066
067    /**
068     * {@link List} of {@code classpath} entries.
069     */
070    public final List<String> classpath = new LinkedList<>();
071
072    /**
073     * {@link List} of accumulated {@code import}s.
074     */
075    public final List<String> imports = new LinkedList<>();
076
077    /**
078     * {@link Map} of known bindings' {@link Class types}.
079     */
080    public final Map<String,String> types = new ConcurrentSkipListMap<>();
081
082    /**
083     * Method to construct a call to a static
084     * {@link java.lang.reflect.Method}.
085     *
086     * @param   type            The containing {@link Class}.
087     * @param   method          The {@link java.lang.reflect.Method} name.
088     * @param   parameters      The parameter types ({@link Class}es).
089     * @param   arguments       The argument {@link Object} array.
090     *
091     * @return  The {@link Object result}.
092     */
093    public Object invokeStaticMethod(String type, String method,
094                                     Class<?>[] parameters, Object... arguments) {
095        Object object = null;
096
097        try {
098            object =
099                Class.forName(type)
100                .getDeclaredMethod(method, parameters)
101                .invoke(null, arguments);
102        } catch (Throwable throwable) {
103            throwable.printStackTrace(System.err);
104        }
105
106        return object;
107    }
108
109    /**
110     * Method to construct a call to a static
111     * {@link java.lang.reflect.Method} with an empty argument list.
112     *
113     * @param   type            The containing {@link Class}.
114     * @param   method          The {@link java.lang.reflect.Method} name.
115     *
116     * @return  The {@link Object result}.
117     */
118    public Object invokeStaticMethod(String type, String method) {
119        return invokeStaticMethod(type, method,
120                                  new Class<?>[] { }, new Object[] { });
121    }
122
123    /**
124     * Static method used by the {@link ganymede.shell.Shell} to update the
125     * {@link NotebookContext} members.
126     *
127     * @param   shell           The {@link Shell}.
128     */
129    public static void update(Shell shell) {
130        var jshell = shell.jshell();
131        var classpath =
132            shell.classpath().stream()
133            .map(File::getAbsolutePath)
134            .collect(toList());
135
136        evaluate(jshell, "__.classpath.clear()");
137
138        if (! classpath.isEmpty()) {
139            evaluate(jshell,
140                     "java.util.Collections.addAll(__.classpath, \"%1$s\".split(\",\"))",
141                     String.join(",", classpath));
142        }
143
144        var imports =
145            jshell.imports()
146            .map(t -> t.source())
147            .map(String::strip)
148            .collect(toCollection(LinkedHashSet::new));
149
150        evaluate(jshell, "__.imports.clear()");
151
152        if (! imports.isEmpty()) {
153            evaluate(jshell,
154                     "java.util.Collections.addAll(__.imports, \"%1$s\".split(\",\"))",
155                     String.join(",", imports));
156        }
157
158        var types =
159            jshell.variables()
160            .filter(t -> (! t.subKind().equals(TEMP_VAR_EXPRESSION_SUBKIND)))
161            .filter(t -> (! t.name().equals("__")))
162            .collect(toMap(k -> k.name(), v -> v.typeName()));
163
164        evaluate(jshell, "__.types.clear()");
165
166        for (var entry : types.entrySet()) {
167            evaluate(jshell,
168                     "__.context.getBindings(%1d).put(\"%2$s\", %2$s)",
169                     ENGINE_SCOPE, entry.getKey());
170            evaluate(jshell,
171                     "__.types.put(\"%1$s\", \"%2$s\")",
172                     entry.getKey(), entry.getValue());
173        }
174    }
175
176    private static String evaluate(JShell jshell, String expression, Object... argv) {
177        var analyzer = jshell.sourceCodeAnalysis();
178        var info = analyzer.analyzeCompletion(String.format(expression, argv));
179
180        return unescape(jshell.eval(info.source()).get(0).value());
181    }
182
183    /**
184     * Method to unescape a Java-escaped spring literal.
185     *
186     * @param   literal         The {@link String} to unescape.
187     *
188     * @return  The unescaped {@link String}.
189     */
190    public static String unescape(String literal) {
191        var string = literal;
192        /*
193         * https://stackoverflow.com/questions/3537706/how-to-unescape-a-java-string-literal-in-java
194         */
195        if (literal != null) {
196            try (var reader = new StringReader(literal)) {
197                var tokenizer = new StreamTokenizer(reader);
198
199                tokenizer.nextToken();
200
201                if (tokenizer.ttype == '"') {
202                    string = tokenizer.sval;
203                }
204            } catch (IOException exception) {
205            }
206        }
207
208        return string;
209    }
210}
