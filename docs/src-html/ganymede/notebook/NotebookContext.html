136    /**
137     * Method to update notebook context.
138     *
139     * @see #kernel
140     */
141    public void refresh() {
142        try {
143            kernel = nsc.getKernel(kernelId);
144
145            session =
146                nsc.getSessionList().stream()
147                .filter(t -> kernelId.equals(t.getKernel().getId()))
148                .findFirst().orElse(null);
149
150            classpath.clear();
151            classpath.addAll(krc.classpath());
152
153            imports.clear();
154            imports.addAll(krc.imports());
155
156            types.clear();
157            types.putAll(krc.variables());
158        } catch (Throwable throwable) {
159            throwable.printStackTrace(System.err);
160        }
161    }

170    /**
171     * Method to receive a {@link Magic} request in the {@link JShell}
172     * instance.  See {@link #invoke(JShell,String)}.
173     *
174     * @param   name            The magic name.
175     */
176    public void invoke(String name) {
177        try {
178            var magic = magics.reload().get(name);
179
180            if (magic != null) {
181                var request = krc.getExecuteRequest();
182                var code = request.at("/content/code").asText();
183                var metadata = request.at("/metadata");
184                var application = new Magic.Application(code);
185
186                magic.execute(application.getLine0(), application.getCode(), metadata);
187            } else {
188                System.err.format("Magic '%s' not found\n", name);
189            }
190        } catch (Throwable throwable) {
191            throwable.printStackTrace(System.err);
192        }
193    }

259    /**
260     * Method to generate the bootstrap code for a new {@link JShell}
261     * instance.
262     *
263     * @return  The boostrap code.
264     */
265    public static String bootstrap() {
266        var code =
267            String.format("var %1$s = %2$s.newNotebookContext();\n",
268                          NAME, Notebook.class.getCanonicalName());
269
270        for (var method : getNotebookFunctions()) {
271            code += makeWrapperFor(NAME, method);
272        }
273
274        return code;
275    }
276
277    private static String makeWrapperFor(String instance, Method method) {
278        var types = method.getGenericParameterTypes();
279        var arguments = new String[types.length];
280        var parameters = new String[types.length];
281
282        for (int i = 0; i < arguments.length; i += 1) {
283            arguments[i] = String.format("argument%1$d", i);
284        }
285
286        for (int i = 0; i < parameters.length; i += 1) {
287            parameters[i] =
288                String.format("%1$s %2$s",
289                              types[i].getTypeName(), arguments[i]);
290        }
291
292        var plist = String.join(", ", parameters);
293        var alist = String.join(", ", arguments);
294
295        return String.format("%1$s %2$s(%3$s) { %4$s%5$s.%2$s(%6$s); }\n",
296                             method.getGenericReturnType().getTypeName(),
297                             method.getName(), plist,
298                             Void.TYPE.equals(method.getReturnType()) ? "" : "return ",
299                             instance, alist);
300    }

302    /**
303     * Method to get the {@link NotebookContext} {@link Method}s annotated
304     * with {@link NotebookFunction} that should be linked into the
305     * {@link Notebook} environment.
306     *
307     * @return  The array of {@link Method}s.
308     */
309    public static Method[] getNotebookFunctions() {
310        var functions =
311            Stream.of(NotebookContext.class.getDeclaredMethods())
312            .filter(t -> t.isAnnotationPresent(NotebookFunction.class))
313            .filter(t -> isPublic(t.getModifiers()))
314            .toArray(Method[]::new);
315
316        return functions;
317    }

319    /**
320     * Static method to get the current imports.
321     *
322     * @param   jshell          The {@link JShell}.
323     *
324     * @return  The {@link Set} of imports as {@link String}s.
325     */
326    public static Set<String> imports(JShell jshell) {
327        var imports = Set.<String>of();
328
329        if (jshell != null) {
330            imports =
331                jshell.imports()
332                .map(t -> t.source())
333                .map(String::strip)
334                .collect(toCollection(LinkedHashSet::new));
335        }
336
337        return imports;
338    }

340    /**
341     * Static method to get the current {@link Map} of defined variables to
342     * their type definitions.
343     *
344     * @param   jshell          The {@link JShell}.
345     *
346     * @return  The {@link Map} of defined variables and their types as
347     *          {@link String}s.
348     */
349    public static Map<String,String> variables(JShell jshell) {
350        var variables = Map.<String,String>of();
351
352        if (jshell != null) {
353            variables =
354                jshell.variables()
355                .filter(t -> (! t.subKind().equals(TEMP_VAR_EXPRESSION_SUBKIND)))
356                .collect(toMap(k -> k.name(), v -> v.typeName()));
357        }
358
359        return variables;
360    }

362    /**
363     * Static method used by the {@link ganymede.shell.Shell} REPL to update
364     * the {@link NotebookContext} instance before execution.
365     *
366     * @param   jshell          The {@link JShell}.
367     */
368    public static void preExecute(JShell jshell) {
369        evaluate(jshell, "%1$s.refresh()", NAME);
370
371        var variables = variables(jshell);
372
373        for (var entry : variables.entrySet()) {
374            evaluate(jshell,
375                     "%1$s.context.getBindings(%2$d).put(\"%3$s\", %3$s)",
376                     NAME, ScriptContext.ENGINE_SCOPE, entry.getKey());
377        }
378    }

389    private static String evaluate(JShell jshell, String expression, Object... argv) {
390        var analyzer = jshell.sourceCodeAnalysis();
391        var info = analyzer.analyzeCompletion(String.format(expression, argv));
392
393        return unescape(jshell.eval(info.source()).get(0).value());
394    }

407    /**
408     * Method to unescape a Java-escaped string literal.
409     *
410     * @param   literal         The {@link String} to unescape.
411     *
412     * @return  The unescaped {@link String}.
413     */
414    public static String unescape(String literal) {
415        var string = literal;
416        /*
417         * https://stackoverflow.com/questions/3537706/how-to-unescape-a-java-string-literal-in-java
418         */
419        if (literal != null) {
420            try (var reader = new StringReader(literal)) {
421                var tokenizer = new StreamTokenizer(reader);
422
423                tokenizer.nextToken();
424
425                if (tokenizer.ttype == '"') {
426                    string = tokenizer.sval;
427                }
428            } catch (IOException exception) {
429            }
430        }
431
432        return string;
433    }

435    /**
436     * {@link ganymede.kernel.magic.SQL}-specific context.  Implements
437     * {@link Map} of JDBC URLs to {@link DSLContext}s.
438     */
439    @NoArgsConstructor
440    public class SQL extends LinkedHashMap<String,DSLContext> {
441        private static final long serialVersionUID = -4901551333824542142L;
442
443        /**
444         * {@link List} of most recent {@link ganymede.kernel.magic.SQL}
445         * {@link Query Queries}.
446         *
447         * @serial
448         */
449        public final List<Query> queries = new ArrayList<>();
450
451        /**
452         * {@link List} of most recent {@link ganymede.kernel.magic.SQL}
453         * {@link Result}s.
454         *
455         * @serial
456         */
457        public final List<Result<Record>> results = new ArrayList<>();
458
459        /**
460         * Target of the {@link ganymede.kernel.magic.SQL} {@link Magic}.
461         *
462         * @param       url             The JDBC URL.
463         * @param       username        The JDBC Username.
464         * @param       password        The JDBC Password.
465         *
466         * @return      The {@link DSLContext} corresponding to the URL.
467         */
468        public DSLContext connect(String url, String username, String password) {
469            return computeIfAbsent(toKey(url), k -> DSL.using(url, username, password));
470        }
471
472        private String toKey(String url) {
473            var key = url;
474
475            if (key != null) {
476                try {
477                    var list = new LinkedList<URI>();
478
479                    list.add(URI.create(key));
480
481                    for (;;) {
482                        var last = list.getLast();
483                        var ssp = last.getSchemeSpecificPart();
484
485                        if (last.isOpaque() && ssp.contains(":")) {
486                            list.add(URI.create(ssp));
487                        } else {
488                            break;
489                        }
490                    }
491
492                    var uri = list.removeLast();
493
494                    if (uri.isOpaque()) {
495                        uri = new URI(uri.getScheme(), uri.getSchemeSpecificPart().split(";", 2)[0], null);
496                    } else {
497                        uri = new URI(uri.getScheme(), null, uri.getHost(), uri.getPort(), uri.getPath(), null, null);
498                    }
499
500                    while (! list.isEmpty()) {
501                        uri = new URI(list.removeLast().getScheme(), uri.toString(), null);
502                    }
503
504                    key = uri.toString();
505                } catch (URISyntaxException exception) {
506                    throw new IllegalArgumentException(exception);
507                }
508            }
509
510            return key;
511        }
512    }
