021import java.lang.reflect.Method;
022import lombok.NoArgsConstructor;
023import lombok.ToString;
024import lombok.extern.log4j.Log4j2;
025import org.springframework.boot.autoconfigure.SpringBootApplication;
026import org.springframework.boot.builder.SpringApplicationBuilder;
027
028import static java.lang.reflect.Modifier.isPublic;
029
030/**
031 * {@link Notebook} Spring launcher.
032 *
033 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
034 */
035@SpringBootApplication
036@NoArgsConstructor @ToString
037public class Notebook {
038
039    /**
040     * Static factory method to create an {@link NotebookContext}.  Also
041     * initializes the Spring environment.
042     *
043     * @see SpringApplicationBuilder
044     *
045     * @return  An initialized {@link NotebookContext}.
046     */
047    public static NotebookContext newNotebookContext() {
048        var __ = new NotebookContext();
049
050        try {
051            var type = Notebook.class;
052            var profile = type.getSimpleName().toLowerCase();
053
054            new SpringApplicationBuilder(type)
055                .profiles(profile)
056                .run();
057        } catch (Exception exception) {
058            exception.printStackTrace(System.err);
059        }
060
061        return __;
062    }
063
064    /**
065     * Method to generate the bootstrap code for a new
066     * {@link jdk.jshell.JShell} instance.
067     *
068     * @return  The boostrap code.
069     */
070    public static String bootstrap() {
071        var code =
072            String.format("var __ = %s.newNotebookContext();\n",
073                          Notebook.class.getCanonicalName());
074
075        for (var method : NotebookContext.class.getDeclaredMethods()) {
076            if (method.isAnnotationPresent(NotebookFunction.class)) {
077                if (isPublic(method.getModifiers())) {
078                    code += makeWrapperFor("__", method);
079                }
080            }
081        }
082
083        return code;
084    }
085
086    private static String makeWrapperFor(String instance, Method method) {
087        var types = method.getGenericParameterTypes();
088        var arguments = new String[types.length];
089        var parameters = new String[types.length];
090
091        for (int i = 0; i < arguments.length; i += 1) {
092            arguments[i] = String.format("argument%d", i);
093        }
094
095        for (int i = 0; i < parameters.length; i += 1) {
096            parameters[i] =
097                String.format("%s %s", types[i].getTypeName(), arguments[i]);
098        }
099
100        var plist = String.join(", ", parameters);
101        var alist = String.join(", ", arguments);
102
103        return String.format("%1$s %2$s(%3$s) { %4$s%5$s.%2$s(%6$s); }\n",
104                             method.getGenericReturnType().getTypeName(),
105                             method.getName(), plist,
106                             Void.TYPE.equals(method.getReturnType()) ? "" : "return ",
107                             instance, alist);
108    }
109}
