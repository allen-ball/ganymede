021import com.fasterxml.jackson.databind.JsonNode;
022import java.util.Optional;
023import java.util.stream.Stream;
024import jdk.jshell.JShell;
025import lombok.Data;
026import org.codehaus.plexus.util.cli.CommandLineUtils;
027
028/**
029 * {@link Magic} service interface.
030 *
031 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
032 */
033public interface Magic {
034
035    /**
036     * Cell {@link Magic} indicator.
037     */
038    public static final String CELL = "%%";
039
040    /**
041     * Special-case {@link Magic} name.
042     */
043    public static final String BANG = "!";
044
045    /**
046     * Method to get a configured instance of this {@link Magic}.
047     * Subclasses may override to return an instance that requires
048     * additional resources and/or configuration.  Default implementaion
049     * returns {@link.this}.
050     *
051     * @return  An {@link Optional} containing the configured instance.
052     */
053    public default Optional<? extends Magic> instance() {
054        return Optional.of(this);
055    }
056
057    /**
058     * Method to get the names associated with {@link.this} {@link Magic}.
059     *
060     * @return  The names.
061     */
062    public String[] getMagicNames();
063
064    /**
065     * Method to get the description of {@link.this} {@link Magic}.
066     *
067     * @return  The description.
068     */
069    public String getDescription();
070
071    /**
072     * Method to get the usage of {@link.this} {@link Magic}.
073     *
074     * @return  The usage.
075     */
076    public String getUsage();
077
078    /**
079     * Return result of {@link #isComplete(String,String)}.
080     */
081    public enum completeness { complete, incomplete, invalid, unknown };
082
083    /**
084     * Method to determine code's {@link completeness}.
085     *
086     * @param   line0           The initial magic line.
087     * @param   code            The code to execute.
088     *
089     * @return  The code's {@link completeness}.
090     */
091    public completeness isComplete(String line0, String code);
092
093    /**
094     * {@link JShell} configuration method.
095     *
096     * @param   context         The {@link NotebookContext}.
097     */
098    public void configure(NotebookContext context);
099
100    /**
101     * {@link JShell} execution method.
102     *
103     * @param   line0           The initial magic line.
104     * @param   code            The remainder of the cell.
105     * @param   metadata        The metadata from the Jupyter
106     *                          {@code execute_request}.
107     */
108    public void execute(String line0, String code, JsonNode metadata) throws Exception;
109
110    /**
111     * Method to determine if the code is cell magic (starts with
112     * '{@code %%}').
113     *
114     * @param   code            The code to test.
115     *
116     * @return  {@code true} if cell magic; {@code false} otherwise.
117     */
118    public static boolean isCellMagic(String code) {
119        return code.startsWith(CELL);
120    }
121
122    /**
123     * Method to parse a cell magic command.  The code must start with
124     * "{@code %%}".  "{@code #}" to the EOL is a comment.  The command is
125     * either a word or a single punctuation character.  Words may be quoted
126     * with double quotes.
127     *
128     * @param   code            The code to parse.
129     *
130     * @return  The array of command arguments.
131     *
132     * @throws  IllegalArgumentException
133     *                          If the magic line can't be parsed.
134     */
135    public static String[] getCellMagicCommand(String code) {
136        var line0 = code.split("\\R", 2)[0];
137        var argv = new String[] { line0 };
138
139        try {
140            argv = CommandLineUtils.translateCommandline(argv[0]);
141
142            if (argv[0].startsWith(CELL)) {
143                argv[0] = argv[0].substring(CELL.length()).trim();
144
145                if (argv[0].isBlank()) {
146                    argv = Stream.of(argv).skip(1).toArray(String[]::new);
147                }
148
149                if (argv[0].startsWith(BANG) && argv[0].length() > BANG.length()) {
150                    argv =
151                        Stream.concat(Stream.of(BANG, argv[0].substring(BANG.length())),
152                                      Stream.of(argv).skip(1))
153                        .toArray(String[]::new);
154                }
155            } else {
156                throw new IllegalArgumentException();
157            }
158        } catch (IllegalStateException exception) {
159            throw exception;
160        } catch (Exception exception) {
161            throw new IllegalArgumentException(line0);
162        }
163
164        return argv;
165    }
166
167    /**
168     * {@link Magic} {@link Application Application} instance to parse code,
169     * find {@link Magic}, and execute.
170     *
171     * {@bean.info}
172     */
173    @Data
174    public static class Application {
175        private final String line0;
176        private final String code;
177        private final String[] argv;
178
179        /**
180         * Sole public constructor.
181         *
182         * @param       code    The complete cell code.
183         */
184        public Application(String code) {
185            this(isCellMagic(code) ? code.split("\\R", 2) : new String[] { null, code });
186        }
187
188        private Application(String[] lines) {
189            this(lines[0], (lines.length > 1) ? lines[1] : "");
190        }
191
192        private Application(String line0, String code) {
193            this.line0 = line0;
194            this.code = code;
195            this.argv = (line0 != null) ? getCellMagicCommand(line0) : new String[] { };
196        }
197
198        /**
199         * Method to determine if the code has a "magic" line.
200         *
201         * @return      {@code true} or {@code false}.
202         */
203        public boolean hasMagicLine() { return (line0 != null); }
204
205        /**
206         * Method to get the specified {@link Magic} name.
207         *
208         * @return      The specified {@link Magic} name; {@code null} if
209         *              none is specified.
210         */
211        public String getMagicName() {
212            return (argv.length > 0) ? argv[0] : null;
213        }
214    }
215}
