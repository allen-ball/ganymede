021import com.fasterxml.jackson.databind.node.ObjectNode;
022import java.util.Base64;
023import java.util.Optional;
024
025/**
026 * {@link ganymede.server.Message#mime_bundle(Object,Object...)} output
027 * {@link Renderer}.
028 *
029 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
030 */
031public interface Renderer {
032    public static final String DATA = "data";
033    public static final String METADATA = "metadata";
034
035    /**
036     * Singleton {@link RendererMap} instance.
037     */
038    public static final RendererMap MAP = new RendererMap();
039
040    /**
041     * {@link Base64.Encoder} instance.
042     */
043    public static final Base64.Encoder BASE64_ENCODER = Base64.getEncoder();
044
045    /**
046     * Method to get a configured instance of this {@link Renderer}.
047     * Subclasses may override to return an instance that requires
048     * additional resources and/or configuration.  Default implementaion
049     * returns {@link.this}.
050     *
051     * @return  An {@link Optional} containing the configured instance.
052     */
053    public default Optional<? extends Renderer> instance() {
054        return Optional.of(this);
055    }
056
057    /**
058     * Default method to analyze {@link.this} instance for {@link ForClass}
059     * and {@link ForClassName} annotations.  May return {@code null} if a
060     * {@link Class} cannot be loaded.
061     *
062     * @return  The {@link Class type} the argument renders (may be
063     *          {@code null}).
064     */
065    public default Class<?> getRenderType() {
066        return getRenderType(getClass());
067    }
068
069    /**
070     * {@link RendererMap} configuration method.
071     *
072     * @param   renderers       The {@link RendererMap}.
073     */
074    public void configure(RendererMap renderers);
075
076    /**
077     * Method to render an {@link Object} to a {@code mime-bundle}.
078     *
079     * @param   bundle          The {@link ganymede.server.Message}
080     *                          {@code mime-bundle}.
081     * @param   object          The {@link Object} to render.
082     */
083    public void renderTo(ObjectNode bundle, Object object);
084
085    /**
086     * Static method to analyze a class for {@link ForClass} and
087     * {@link ForClassName} annotations.  May return {@code null} if a
088     * {@link Class} cannot be loaded.
089     *
090     * @param   type            The {@link Class} to analyze.
091     *
092     * @return  The {@link Class type} the argument renders (may be
093     *          {@code null}).
094     */
095    public static Class<?> getRenderType(Class<?> type) {
096        Class<?> renderType = null;
097
098        if (type != null) {
099            try {
100                if (type.isAnnotationPresent(ForClassName.class)) {
101                    var name = type.getAnnotation(ForClassName.class).value();
102
103                    renderType = Class.forName(name, false, type.getClassLoader());
104                } else {
105                    renderType = type.getAnnotation(ForClass.class).value();
106                }
107            } catch (Throwable throwable) {
108            }
109
110            if (renderType == null) {
111                var superType = type.getSuperclass();
112
113                if (superType != null && Renderer.class.isAssignableFrom(superType)) {
114                    renderType = getRenderType(superType);
115                }
116            }
117        }
118
119        return renderType;
120    }
121}
