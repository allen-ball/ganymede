021import ganymede.kernel.Kernel;
022import java.io.File;
023import java.util.Optional;
024import lombok.NoArgsConstructor;
025import lombok.ToString;
026import lombok.extern.log4j.Log4j2;
027import org.springframework.beans.factory.annotation.Value;
028import org.springframework.boot.ApplicationArguments;
029import org.springframework.boot.ApplicationRunner;
030import org.springframework.boot.autoconfigure.SpringBootApplication;
031import org.springframework.boot.builder.SpringApplicationBuilder;
032
033import static ganymede.server.Server.JSON_OBJECT_MAPPER;
034
035/**
036 * Ganymede Jupyter {@link ganymede.kernel.Kernel} {@link Connect} CLI.
037 * Parses {@code --connection-file} to look for a running
038 * {@link Kernel} and connects if one is found; otherwise starts a new
039 * {@link Kernel} instance.
040 *
041 * {@injected.fields}
042 *
043 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
044 */
045@SpringBootApplication
046@NoArgsConstructor @ToString @Log4j2
047public class Connect implements ApplicationRunner {
048    @Value("${JPY_PARENT_PID:#{-1}}")
049    private long jpy_parent_pid = -1;
050
051    @Value("${connection-file:#{null}}")
052    private String connection_file = null;
053
054    @Override
055    public void run(ApplicationArguments arguments) throws Exception {
056        try {
057            var file = new File(connection_file);
058            var node = JSON_OBJECT_MAPPER.readTree(file);
059            var isAlive =
060                Optional.of("pid")
061                .filter(t -> node.hasNonNull(t))
062                .map(t -> node.get(t).asLong())
063                .flatMap(ProcessHandle::of)
064                .map(ProcessHandle::isAlive)
065                .orElse(false);
066
067            if (! isAlive) {
068                var type = Kernel.class;
069                var profile = type.getSimpleName().toLowerCase();
070
071                new SpringApplicationBuilder(type)
072                    .profiles(profile)
073                    .run(arguments.getSourceArgs());
074            } else {
075                log.warn("Kernel already running");
076            }
077        } catch (Exception exception) {
078            log.warn("{}", connection_file, exception);
079        }
080    }
081}
